## 1. Memory Store

```javascript
// src/store/memoryStore.js

class MemoryStore {
  constructor() {
    this.store = new Map();
    this.cleanupInterval = setInterval(() => this.cleanup(), 60_000);
    this.cleanupInterval.unref(); // Don't block process exit
  }

  async get(key) {
    const entry = this.store.get(key);
    if (!entry) return null;
    if (Date.now() > entry.expiresAt) {
      this.store.delete(key);
      return null;
    }
    return entry.value;
  }

  async set(key, value, ttlMs) {
    this.store.set(key, {
      value,
      expiresAt: Date.now() + ttlMs,
    });
  }

  async increment(key, ttlMs) {
    const entry = this.store.get(key);
    const now = Date.now();

    if (!entry || now > entry.expiresAt) {
      this.store.set(key, { value: 1, expiresAt: now + ttlMs });
      return 1;
    }

    entry.value += 1;
    return entry.value;
  }

  async delete(key) {
    this.store.delete(key);
  }

  cleanup() {
    const now = Date.now();
    for (const [key, entry] of this.store.entries()) {
      if (now > entry.expiresAt) {
        this.store.delete(key);
      }
    }
  }

  destroy() {
    clearInterval(this.cleanupInterval);
    this.store.clear();
  }
}

module.exports = MemoryStore;
```

class RedisStore {
  /**
   * @param {import('ioredis').Redis} client - ioredis client instance
   */
  constructor(client) {
    if (!client) throw new Error('Redis client is required');
    this.client = client;
  }

  async get(key) {
    const value = await this.client.get(key);
    return value ? JSON.parse(value) : null;
  }

  async set(key, value, ttlMs) {
    await this.client.set(key, JSON.stringify(value), 'PX', ttlMs);
  }

  async increment(key, ttlMs) {
    // Atomic increment using Lua script
    const script = `
      local current = redis.call('GET', KEYS[1])
      if current == false then
        redis.call('SET', KEYS[1], 1, 'PX', ARGV[1])
        return 1
      else
        return redis.call('INCR', KEYS[1])
      end
    `;
    return this.client.eval(script, 1, key, ttlMs);
  }

  async delete(key) {
    await this.client.del(key);
  }
}

module.exports = RedisStore;

## 3. Fixed Window Strategy

```javascript
// src/strategies/fixedWindow.js

class FixedWindowStrategy {
  /**
   * @param {object} options
   * @param {number} options.windowMs   - Window size in milliseconds
   * @param {number} options.maxRequests - Max requests per window
   * @param {object} options.store      - Store instance
   */
  constructor({ windowMs, maxRequests, store }) {
    this.windowMs = windowMs;
    this.maxRequests = maxRequests;
    this.store = store;
  }

  _getWindowKey(key) {
    const windowId = Math.floor(Date.now() / this.windowMs);
    return `fw:${key}:${windowId}`;
  }

  async consume(key) {
    const windowKey = this._getWindowKey(key);
    const count = await this.store.increment(windowKey, this.windowMs);

    const remaining = Math.max(0, this.maxRequests - count);
    const resetTime = Math.ceil(Date.now() / this.windowMs) * this.windowMs;

    return {
      allowed: count <= this.maxRequests,
      count,
      remaining,
      resetTime,
      retryAfter: count > this.maxRequests
        ? Math.ceil((resetTime - Date.now()) / 1000)
        : 0,
    };
  }

  async reset(key) {
    const windowKey = this._getWindowKey(key);
    await this.store.delete(windowKey);
  }
}

module.exports = FixedWindowStrategy;
```

## 4. Sliding Window Strategy

```javascript
// src/strategies/slidingWindow.js

class SlidingWindowStrategy {
  /**
   * Uses a sliding log of timestamps for precise rate limiting.
   * More memory-intensive but avoids boundary spikes.
   *
   * @param {object} options
   * @param {number} options.windowMs    - Window size in milliseconds
   * @param {number} options.maxRequests - Max requests per window
   * @param {object} options.store       - Store instance
   */
  constructor({ windowMs, maxRequests, store }) {
    this.windowMs = windowMs;
    this.maxRequests = maxRequests;
    this.store = store;
  }

  async consume(key) {
    const storeKey = `sw:${key}`;
    const now = Date.now();
    const windowStart = now - this.windowMs;

    // Get existing timestamps
    let timestamps = (await this.store.get(storeKey)) || [];

    // Filter out expired timestamps
    timestamps = timestamps.filter((ts) => ts > windowStart);

    const count = timestamps.length;
    const allowed = count < this.maxRequests;

    if (allowed) {
      timestamps.push(now);
      await this.store.set(storeKey, timestamps, this.windowMs);
    }

    const resetTime = timestamps.length > 0
      ? timestamps[0] + this.windowMs
      : now + this.windowMs;

    return {
      allowed,
      count: allowed ? count + 1 : count,
      remaining: Math.max(0, this.maxRequests - count - (allowed ? 1 : 0)),
      resetTime,
      retryAfter: !allowed
        ? Math.ceil((timestamps[0] + this.windowMs - now) / 1000)
        : 0,
    };
  }

  async reset(key) {
    await this.store.delete(`sw:${key}`);
  }
}

module.exports = SlidingWindowStrategy;
```

## 5. Token Bucket Strategy

```javascript
// src/strategies/tokenBucket.js

class TokenBucketStrategy {
  /**
   * Allows bursting up to `capacity` requests, refilling at `refillRate`
   * tokens per second.
   *
   * @param {object} options
   * @param {number} options.capacity   - Max tokens (burst size)
   * @param {number} options.refillRate - Tokens added per second
   * @param {object} options.store      - Store instance
   */
  constructor({ capacity, refillRate, store }) {
    this.capacity = capacity;
    this.refillRate = refillRate; // tokens per second
    this.store = store;
  }

  async consume(key, tokensRequested = 1) {
    const storeKey = `tb:${key}`;
    const now = Date.now();

    let bucket = await this.store.get(storeKey);

    if (!bucket) {
      bucket = { tokens: this.capacity, lastRefill: now };
    }

    // Calculate tokens to add since last refill
    const elapsed = (now - bucket.lastRefill) / 1000; // seconds
    const tokensToAdd = elapsed * this.refillRate;
    bucket.tokens = Math.min(this.capacity, bucket.tokens + tokensToAdd);
    bucket.lastRefill = now;

    const allowed = bucket.tokens >= tokensRequested;

    if (allowed) {
      bucket.tokens -= tokensRequested;
    }

    // TTL: time to refill from 0 to capacity
    const ttlMs = (this.capacity / this.refillRate) * 1000 * 2;
    await this.store.set(storeKey, bucket, ttlMs);

    const timeToRefill = allowed
      ? 0
      : Math.ceil((tokensRequested - bucket.tokens) / this.refillRate);

    return {
      allowed,
      tokens: Math.floor(bucket.tokens),
      remaining: Math.floor(bucket.tokens),
      resetTime: now + timeToRefill * 1000,
      retryAfter: allowed ? 0 : timeToRefill,
    };
  }

  async reset(key) {
    await this.store.delete(`tb:${key}`);
  }
}

module.exports = TokenBucketStrategy;
```

## 6. Core Rate Limiter

```javascript
// src/rateLimiter.js

const MemoryStore = require('./store/memoryStore');
const FixedWindowStrategy = require('./strategies/fixedWindow');
const SlidingWindowStrategy = require('./strategies/slidingWindow');
const TokenBucketStrategy = require('./strategies/tokenBucket');

const STRATEGIES = {
  fixedWindow: FixedWindowStrategy,
  slidingWindow: SlidingWindowStrategy,
  tokenBucket: TokenBucketStrategy,
};

class RateLimiter {
  /**
   * @param {object}   options
   * @param {string}   [options.strategy='fixedWindow'] - 'fixedWindow' | 'slidingWindow' | 'tokenBucket'
   * @param {number}   [options.windowMs=60000]         - Window duration in ms
   * @param {number}   [options.maxRequests=100]        - Max requests per window
   * @param {number}   [options.capacity=10]            - Token bucket capacity
   * @param {number}   [options.refillRate=1]           - Token bucket refill rate (tokens/sec)
   * @param {object}   [options.store]                  - Custom store (default: MemoryStore)
   * @param {Function} [options.keyGenerator]           - Custom key function (req) => string
   * @param {Function} [options.onLimitReached]         - Callback when limit is hit
   * @param {boolean}  [options.skipFailedRequests]     - Don't count failed responses
   * @param {boolean}  [options.skipSuccessfulRequests] - Don't count successful responses
   */
  constructor(options = {}) {
    const {
      strategy = 'fixedWindow',
      windowMs = 60_000,
      maxRequests = 100,
      capacity = 10,
      refillRate = 1,
      store = new MemoryStore(),
      keyGenerator = (req) => req.ip || req.connection?.remoteAddress || 'global',
      onLimitReached = null,
      skipFailedRequests = false,
      skipSuccessfulRequests = false,
    } = options;

    if (!STRATEGIES[strategy]) {
      throw new Error(`Unknown strategy: "${strategy}". Use: ${Object.keys(STRATEGIES).join(', ')}`);
    }

    this.keyGenerator = keyGenerator;
    this.onLimitReached = onLimitReached;
    this.skipFailedRequests = skipFailedRequests;
    this.skipSuccessfulRequests = skipSuccessfulRequests;

    const StrategyClass = STRATEGIES[strategy];

    const strategyOptions =
      strategy === 'tokenBucket'
        ? { capacity, refillRate, store }
        : { windowMs, maxRequests, store };

    this.strategy = new StrategyClass(strategyOptions);
  }

  async check(key) {
    return this.strategy.consume(key);
  }

  async reset(key) {
    return this.strategy.reset(key);
  }
}

module.exports = RateLimiter;
```

## 7. Express Middleware

```javascript
// src/middleware.js

const RateLimiter = require('./rateLimiter');

/**
 * Creates an Express rate-limiting middleware.
 * @param {object} options - RateLimiter options + middleware-specific options
 * @param {string} [options.message] - Custom error message
 * @param {number} [options.statusCode=429] - HTTP status code
 * @param {boolean} [options.headers=true] - Send rate limit headers
 */
function createRateLimitMiddleware(options = {}) {
  const {
    message = 'Too many requests, please try again later.',
    statusCode = 429,
    headers = true,
    ...limiterOptions
  } = options;

  const limiter = new RateLimiter(limiterOptions);

  return async function rateLimitMiddleware(req, res, next) {
    try {
      const key = limiter.keyGenerator(req);
      const result = await limiter.check(key);

      // Set standard rate limit headers
      if (headers) {
        res.setHeader('X-RateLimit-Limit',
          limiterOptions.maxRequests || limiterOptions.capacity || 100);
        res.setHeader('X-RateLimit-Remaining', result.remaining);
        res.setHeader('X-RateLimit-Reset', Math.ceil(result.resetTime / 1000));

        if (!result.allowed) {
          res.setHeader('Retry-After', result.retryAfter);
        }
      }

      if (!result.allowed) {
        // Fire optional callback
        if (typeof limiter.onLimitReached === 'function') {
          limiter.onLimitReached(req, res, key, result);
        }

        return res.status(statusCode).json({
          error: message,
          retryAfter: result.retryAfter,
        });
      }

      // Handle skip options by patching res.end
      if (limiterOptions.skipFailedRequests || limiterOptions.skipSuccessfulRequests) {
        const originalEnd = res.end.bind(res);
        res.end = async (...args) => {
          const shouldRollback =
            (limiterOptions.skipFailedRequests && res.statusCode >= 400) ||
            (limiterOptions.skipSuccessfulRequests && res.statusCode < 400);

          if (shouldRollback) {
            await limiter.reset(key);
          }

          res.end = originalEnd;
          return originalEnd(...args);
        };
      }

      next();
    } catch (err) {
      // Fail open — don't block requests if the limiter errors
      console.error('[RateLimiter] Error:', err);
      next();
    }
  };
}

module.exports = { createRateLimitMiddleware };
```

## 8. Express Server Example

```javascript
// server.js

const express = require('express');
const { createRateLimitMiddleware } = require('./src/middleware');

const app = express();
app.use(express.json());

// ── Global limiter (fixed window) ──────────────────────────────────────────
const globalLimiter = createRateLimitMiddleware({
  strategy: 'fixedWindow',
  windowMs: 60_000,   // 1 minute
  maxRequests: 100,
  message: 'Global rate limit exceeded.',
});

// ── Strict API limiter (sliding window) ────────────────────────────────────
const apiLimiter = createRateLimitMiddleware({
  strategy: 'slidingWindow',
  windowMs: 15 * 60_000, // 15 minutes
  maxRequests: 50,
  keyGenerator: (req) => `api:${req.ip}`,
});

// ── Auth limiter (token bucket — allows small bursts) ──────────────────────
const authLimiter = createRateLimitMiddleware({
  strategy: 'tokenBucket',
  capacity: 5,       // max 5 attempts at once
  refillRate: 0.1,   // 1 token every 10 seconds
  keyGenerator: (req) => `auth:${req.ip}`,
  message: 'Too many login attempts.',
  onLimitReached: (req, res, key) => {
    console.warn(`[Security] Rate limit hit for key: ${key}`);
  },
});

// ── Per-user limiter ────────────────────────────────────────────────────────
const userLimiter = createRateLimitMiddleware({
  strategy: 'slidingWindow',
  windowMs: 60_000,
  maxRequests: 30,
  keyGenerator: (req) => `user:${req.headers['x-user-id'] || req.ip}`,
  skipFailedRequests: true, // don't penalize for 4xx/5xx
});

// Apply global limiter to all routes
app.use(globalLimiter);

// Public routes
app.get('/', (req, res) => {
  res.json({ message: 'Welcome!' });
});

// API routes with stricter limiting
app.use('/api', apiLimiter);

app.get('/api/data', userLimiter, (req, res) => {
  res.json({ data: 'Here is your data', timestamp: Date.now() });
});

// Auth routes
app.post('/auth/login', authLimiter, (req, res) => {
  const { username, password } = req.body;
  if (username === 'admin' && password === 'secret') {
    return res.json({ token: 'fake-jwt-token' });
  }
  res.status(401).json({ error: 'Invalid credentials' });
});

// Rate limit info endpoint
app.get('/api/rate-limit-status', (req, res) => {
  res.json({
    headers: {
      limit: res.getHeader('X-RateLimit-Limit'),
      remaining: res.getHeader('X-RateLimit-Remaining'),
      reset: res.getHeader('X-RateLimit-Reset'),
    },
  });
});

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => {
  console.log(`Server running on http://localhost:${PORT}`);
});

module.exports = app;
```

const RateLimiter = require('../src/rateLimiter');

// Helper to wait
const sleep = (ms) => new Promise((r) => setTimeout(r, ms));

describe('FixedWindow Strategy', () => {
  let limiter;

    limiter = new RateLimiter({
      strategy: 'fixedWindow',
      windowMs: 1000,
      maxRequests: 3,
      store: new MemoryStore(),

    const r1 = await limiter.check('user1');

## Strategy Comparison

| Strategy | Accuracy | Memory | Burst Handling | Best For |
|---|---|---|---|---|
| **Fixed Window** | Medium | Low | Poor (boundary spikes) | Simple APIs |
| **Sliding Window** | High | Medium | Good | REST APIs |
| **Token Bucket** | High | Low | Excellent | Streaming / bursty traffic |

## Key Features

- ✅ **3 strategies** — Fixed Window, Sliding Window, Token Bucket
- ✅ **Pluggable stores** — In-memory (default) or Redis (distributed)
- ✅ **Express middleware** with standard `X-RateLimit-*` headers
- ✅ **Per-route & per-user** limiting with custom key generators
- ✅ **Fail-open** — limiter errors don't block requests
- ✅ **Skip options** — ignore failed or successful requests
- ✅ **Atomic Redis operations** via Lua scripts (no race conditions)
- ✅ **Full test suite** with timing-based tests
