# Production Rate Limiter in Node.js

Let me work through this properly — requirements first, then implementation, then operational gaps.

---

## Phase 1: Requirements Design

### Functional Requirements
- Limit requests per identifier (IP, user ID, API key) within a time window
- Support multiple rate limit strategies
- Return standard rate limit headers (`X-RateLimit-*`, `Retry-After`)
- Allow per-route and per-tier limit configuration
- Distinguish between limit types (sliding window vs fixed window)

### Non-Functional Requirements
- Sub-millisecond overhead on the hot path
- Survive backing store failures without taking down the API (fail-open vs fail-closed decision)
- Consistent behavior across multiple Node.js instances (distributed correctness)
- No race conditions under concurrent requests

### Explicit Design Decisions Made Upfront

| Decision | Choice | Reason |
|---|---|---|
| Algorithm | Sliding window log + fixed window counter | Sliding is accurate, fixed is cheap — both needed |
| Backing store | Redis with in-memory fallback | Distributed correctness; local fallback for resilience |
| Atomicity | Lua scripts in Redis | Prevents TOCTOU race conditions |
| Failure mode | Fail-open by default, configurable | Availability over perfect limiting |
| Identifier | Composable key function | IP alone is insufficient (NAT, proxies) |

/**
 * Central rate limit configuration.
 *
 * OPERATIONAL NOTE: In production, these values should be loaded from
 * environment variables or a config service (Consul, AWS Parameter Store)
 * so limits can be adjusted without redeployment.
 */

'use strict';

const DEFAULT_CONFIG = {
  // Default limits applied when no route-specific config matches
  default: {
    algorithm: 'slidingWindow',
    windowMs: 60_000,       // 1 minute
    maxRequests: 100,
    keyPrefix: 'rl:default',
  },

  // Route-specific overrides — matched in order, first match wins
  routes: [
    {
      // Authentication endpoints — tighter limits to slow brute force
      pattern: /^\/auth\//,
      algorithm: 'fixedWindow',
      windowMs: 15 * 60_000,  // 15 minutes
      maxRequests: 10,
      keyPrefix: 'rl:auth',
    },
    {
      // Public read endpoints — generous limits
      pattern: /^\/api\/public\//,
      algorithm: 'slidingWindow',
      windowMs: 60_000,
      maxRequests: 300,
      keyPrefix: 'rl:public',
    },
    {
      // Write endpoints — moderate limits
      pattern: /^\/api\/v[0-9]+\/(create|update|delete)/,
      algorithm: 'tokenBucket',
      windowMs: 60_000,
      maxRequests: 30,
      keyPrefix: 'rl:write',
    },
  ],

  // Per-tier overrides — applied after route match, can raise or lower limits
  tiers: {
    free:       { multiplier: 1.0 },
    pro:        { multiplier: 5.0 },
    enterprise: { multiplier: 20.0 },
  },

  // Behavior when the backing store is unavailable
  // 'open'   = allow request through (availability wins)
  // 'closed' = reject request (security wins)
  failMode: process.env.RATE_LIMIT_FAIL_MODE || 'open',

  redis: {
    host:           process.env.REDIS_HOST || '127.0.0.1',
    port:           parseInt(process.env.REDIS_PORT || '6379', 10),
    password:       process.env.REDIS_PASSWORD || undefined,
    db:             parseInt(process.env.REDIS_DB || '0', 10),
    connectTimeout: 2000,   // ms — fail fast, don't block requests
    commandTimeout: 500,    // ms — individual command timeout
    maxRetriesPerRequest: 1,
    enableOfflineQueue: false, // CRITICAL: don't queue commands when disconnected
  },
};

/**
 * Resolve the rate limit config for a given request path and user tier.
 * Returns a merged config object with effective limits.
 */
function resolveConfig(path, tier = 'free') {
  // Find first matching route config
  const routeConfig = DEFAULT_CONFIG.routes.find(
    (r) => r.pattern.test(path)
  ) || DEFAULT_CONFIG.default;

  // Apply tier multiplier
  const tierConfig = DEFAULT_CONFIG.tiers[tier] || DEFAULT_CONFIG.tiers.free;
  const effectiveMax = Math.floor(routeConfig.maxRequests * tierConfig.multiplier);

  return {
    algorithm:   routeConfig.algorithm,
    windowMs:    routeConfig.windowMs,
    maxRequests: effectiveMax,
    keyPrefix:   routeConfig.keyPrefix,
    failMode:    DEFAULT_CONFIG.failMode,
  };
}

module.exports = { DEFAULT_CONFIG, resolveConfig };

### `src/utils/keyBuilder.js`

```javascript
/**
 * Builds the Redis key used to track a rate limit bucket.
 *
 * SECURITY NOTE: The identifier must be validated/sanitized before
 * reaching this function. A malicious client controlling their own
 * key could cause key collisions or inflate key size unboundedly.
 *
 * OPERATIONAL NOTE: Key format changes are breaking changes.
 * If you change the format, you must flush old keys or use a new prefix,
 * otherwise old counters will be orphaned and new counters will start
 * fresh — effectively resetting limits for all users during a deploy.
 */

'use strict';

const crypto = require('crypto');

/**
 * Extract the best available identifier from a request.
 * Priority: authenticated user ID > API key > IP address
 *
 * KNOWN LIMITATION: IP-based identification breaks behind NAT or
 * shared proxies (corporate networks, mobile carriers). This is
 * acceptable as a fallback but should not be the primary strategy
 * for authenticated APIs.
 */
function extractIdentifier(req) {
  // Authenticated user — most reliable
  if (req.user?.id) {
    return `user:${req.user.id}`;
  }

  // API key from header or query param
  const apiKey = req.headers['x-api-key'] || req.query?.api_key;
  if (apiKey) {
    // Hash the API key — don't store raw credentials in Redis keys
    const hashed = crypto
      .createHash('sha256')
      .update(apiKey)
      .digest('hex')
      .slice(0, 16); // 16 hex chars = 64 bits, sufficient for key uniqueness
    return `apikey:${hashed}`;
  }

  // IP address — least reliable, last resort
  const ip = extractIp(req);
  return `ip:${ip}`;
}

/**
 * Extract real IP, handling common proxy headers.
 *
 * SECURITY NOTE: X-Forwarded-For can be spoofed by clients unless
 * your load balancer strips and re-sets it. Only trust these headers
 * if you control the network path. In production, configure your
 * load balancer (nginx, ALB, Cloudflare) to set a trusted header.
 */
function extractIp(req) {
  const forwarded = req.headers['x-forwarded-for'];
  if (forwarded) {
    // X-Forwarded-For: client, proxy1, proxy2
    // Take the leftmost IP — that's the original client
    // ONLY safe if your LB strips client-supplied X-Forwarded-For
    return forwarded.split(',')[0].trim();
  }

  return req.socket?.remoteAddress || 'unknown';
}

/**
 * Build the full Redis key for a rate limit check.
 *
 * Format: {prefix}:{identifier}:{windowBucket}
 * Example: rl:auth:ip:1.2.3.4:1699900000
 *
 * The windowBucket is included for fixed-window algorithms.
 * Sliding window algorithms manage their own time bucketing.
 */
function buildKey(prefix, identifier, windowMs) {
  // Floor to current window bucket for fixed-window algorithms
  const windowBucket = Math.floor(Date.now() / windowMs);
  return `${prefix}:${identifier}:${windowBucket}`;
}

/**
 * Build a key without a time bucket — used by sliding window
 * which manages its own time-based cleanup internally.
 */
function buildSlidingKey(prefix, identifier) {
  return `${prefix}:${identifier}`;
}

module.exports = { extractIdentifier, extractIp, buildKey, buildSlidingKey };

/**
 * Writes standard rate limit response headers.
 *
 * We follow the IETF draft standard for RateLimit headers:
 * https://datatracker.ietf.org/doc/draft-ietf-httpapi-ratelimit-headers/
 *
 * Plus the widely-supported X-RateLimit-* headers for compatibility
 * with clients that expect them (most API clients do).
 */

'use strict';

/**
 * Write rate limit headers to a response.
 *
 * @param {object} res - Express response object
 * @param {object} limitResult - Result from rate limit check
 * @param {number} limitResult.limit - Max requests allowed in window
 * @param {number} limitResult.remaining - Requests remaining in window
 * @param {number} limitResult.resetAt - Unix timestamp (ms) when window resets
 * @param {boolean} limitResult.allowed - Whether this request is allowed
 */
function writeHeaders(res, limitResult) {
  const { limit, remaining, resetAt, allowed } = limitResult;

  const resetSeconds = Math.ceil(resetAt / 1000);
  const retryAfterSeconds = Math.ceil((resetAt - Date.now()) / 1000);

  // Standard headers (IETF draft)
  res.setHeader('RateLimit-Limit',     limit);
  res.setHeader('RateLimit-Remaining', Math.max(0, remaining));
  res.setHeader('RateLimit-Reset',     resetSeconds);

  // X-prefixed headers — widely expected by API clients
  res.setHeader('X-RateLimit-Limit',     limit);
  res.setHeader('X-RateLimit-Remaining', Math.max(0, remaining));
  res.setHeader('X-RateLimit-Reset',     resetSeconds);

  // Only write Retry-After when the request is being rejected
  // Writing it on allowed requests confuses some clients
  if (!allowed && retryAfterSeconds > 0) {
    res.setHeader('Retry-After', retryAfterSeconds);
  }
}

module.exports = { writeHeaders };

### `src/stores/redisStore.js`

```javascript
/**
 * Redis-backed store for rate limit state.
 *
 * All counter operations use Lua scripts to guarantee atomicity.
 * Without Lua, a check-then-increment sequence has a race condition:
 *
 *   Client A reads count=99  (limit=100)
 *   Client B reads count=99  (limit=100)
 *   Client A increments → 100 (allowed)
 *   Client B increments → 101 (should be denied, but wasn't checked again)
 *
 * Lua scripts run atomically inside Redis — no other command executes
 * between the check and the increment.
 */

'use strict';

const Redis = require('ioredis');

// ---------------------------------------------------------------------------
// Lua Scripts
// ---------------------------------------------------------------------------

/**
 * Fixed window counter script.
 *
 * Atomically:
 * 1. Increment the counter for this window
 * 2. Set TTL on first increment (so keys self-expire)
 * 3. Return [current_count, ttl_ms]
 *
 * KEYS[1] = rate limit key
 * ARGV[1] = window size in milliseconds
 * ARGV[2] = max requests allowed
 */
const FIXED_WINDOW_SCRIPT = `
local key      = KEYS[1]
local windowMs = tonumber(ARGV[1])
local limit    = tonumber(ARGV[2])

local count = redis.call('INCR', key)

if count == 1 then
  -- First request in this window — set expiry
  -- PEXPIRE uses milliseconds
  redis.call('PEXPIRE', key, windowMs)
end

local ttl = redis.call('PTTL', key)

-- Return: count, ttl_remaining_ms, limit
return {count, ttl, limit}
`;

/**
 * Sliding window log script using a Redis Sorted Set.
 *
 * Each request is stored as a member with score = timestamp.
 * Old entries outside the window are pruned on each request.
 *
 * This is accurate but uses more memory than fixed window.
 * Memory per key = O(requests_in_window).
 * For 100 req/min limit, worst case ~100 entries per user.
 *
 * KEYS[1] = rate limit key (sorted set)
 * ARGV[1] = current timestamp in ms
 * ARGV[2] = window size in ms
 * ARGV[3] = max requests
 * ARGV[4] = unique member ID for this request (to avoid collisions)
 */
const SLIDING_WINDOW_SCRIPT = `
local key       = KEYS[1]
local now       = tonumber(ARGV[1])
local windowMs  = tonumber(ARGV[2])
local limit     = tonumber(ARGV[3])
local member    = ARGV[4]
local windowStart = now - windowMs

-- Remove entries older than the window
redis.call('ZREMRANGEBYSCORE', key, '-inf', windowStart)

-- Count current entries in window
local count = redis.call('ZCARD', key)

if count < limit then
  -- Add this request to the log
  redis.call('ZADD', key, now, member)
  -- Expire the whole key after one window of inactivity
  redis.call('PEXPIRE', key, windowMs)
  -- Return: allowed=1, new_count, limit, reset_at
  return {1, count + 1, limit, now + windowMs}
else
  -- Get the oldest entry to calculate when a slot opens up
  local oldest = redis.call('ZRANGE', key, 0, 0, 'WITHSCORES')
  local resetAt = now + windowMs
  if oldest and oldest[2] then
    resetAt = tonumber(oldest[2]) + windowMs
  end
  -- Return: allowed=0, count, limit, reset_at
  return {0, count, limit, resetAt}
end
`;

/**
 * Token bucket script.
 *
 * Tokens refill continuously at rate = maxRequests / windowMs.
 * Each request consumes one token.
 * Allows short bursts up to the bucket capacity.
 *
 * State stored as a Redis Hash:
 *   tokens    = current token count (float, stored as string)
 *   lastRefill = timestamp of last refill calculation
 *
 * KEYS[1] = rate limit key (hash)
 * ARGV[1] = current timestamp in ms
 * ARGV[2] = max tokens (bucket capacity)
 * ARGV[3] = refill rate in tokens per ms
 * ARGV[4] = window TTL in ms (for key expiry)
 */
const TOKEN_BUCKET_SCRIPT = `
local key        = KEYS[1]
local now        = tonumber(ARGV[1])
local capacity   = tonumber(ARGV[2])
local refillRate = tonumber(ARGV[3])
local ttl        = tonumber(ARGV[4])

local data = redis.call('HMGET', key, 'tokens', 'lastRefill')
local tokens    = tonumber(data[1]) or capacity
local lastRefill = tonumber(data[2]) or now

-- Calculate tokens to add based on elapsed time
local elapsed = now - lastRefill
local newTokens = math.min(capacity, tokens + (elapsed * refillRate))

local allowed = 0
local remaining = math.floor(newTokens)

if newTokens >= 1 then
  newTokens = newTokens - 1
  allowed = 1
  remaining = math.floor(newTokens)
end

-- Persist updated state
redis.call('HMSET', key, 'tokens', tostring(newTokens), 'lastRefill', tostring(now))
redis.call('PEXPIRE', key, ttl)

-- Time until next token available (ms)
local msUntilToken = 0
if allowed == 0 then
  msUntilToken = math.ceil((1 - newTokens) / refillRate)
end

-- Return: allowed, remaining_tokens, capacity, ms_until_next_token
return {allowed, remaining, capacity, msUntilToken}
`;

// ---------------------------------------------------------------------------
// RedisStore class
// ---------------------------------------------------------------------------

class RedisStore {
  constructor(config) {
    this.config = config;
    this.client = null;
    this.isConnected = false;
    this._scripts = {};
  }

  async connect() {
    this.client = new Redis({
      host:                 this.config.host,
      port:                 this.config.port,
      password:             this.config.password,
      db:                   this.config.db,
      connectTimeout:       this.config.connectTimeout,
      commandTimeout:       this.config.commandTimeout,
      maxRetriesPerRequest: this.config.maxRetriesPerRequest,
      enableOfflineQueue:   this.config.enableOfflineQueue,
      lazyConnect:          true,
    });

    this.client.on('connect',           () => { this.isConnected = true;  });
    this.client.on('close',             () => { this.isConnected = false; });
    this.client.on('error',             (err) => {
      // Log but don't crash — the middleware handles store unavailability
      console.error('[RedisStore] Connection error:', err.message);
      this.isConnected = false;
    });

    await this.client.connect();

    // Pre-load scripts using SCRIPT LOAD for slightly faster execution
    // (avoids sending script body on every call after first)
    this._scripts.fixedWindow   = await this.client.script('LOAD', FIXED_WINDOW_SCRIPT);
    this._scripts.slidingWindow = await this.client.script('LOAD', SLIDING_WINDOW_SCRIPT);
    this._scripts.tokenBucket   = await this.client.script('LOAD', TOKEN_BUCKET_SCRIPT);
  }

  async disconnect() {
    if (this.client) {
      await this.client.quit();
    }
  }

  /**
   * Fixed window rate limit check.
   * Returns standardized result object.
   */
  async fixedWindow(key, windowMs, maxRequests) {
    const result = await this.client.evalsha(
      this._scripts.fixedWindow,
      1,           // number of keys
      key,         // KEYS[1]
      windowMs,    // ARGV[1]
      maxRequests  // ARGV[2]
    );

    const [count, ttlMs] = result;
    const allowed = count <= maxRequests;
    const resetAt = Date.now() + (ttlMs > 0 ? ttlMs : windowMs);

    return {
      allowed,
      limit:     maxRequests,
      remaining: Math.max(0, maxRequests - count),
      resetAt,
      count,
    };
  }

  /**
   * Sliding window log rate limit check.
   */
  async slidingWindow(key, windowMs, maxRequests) {
    const now    = Date.now();
    // Unique member: timestamp + random suffix to handle concurrent requests
    // at the same millisecond from the same client
    const member = `${now}:${Math.random().toString(36).slice(2, 9)}`;

    const result = await this.client.evalsha(
      this._scripts.slidingWindow,
      1,           // number of keys
      key,         // KEYS[1]
      now,         // ARGV[1]
      windowMs,    // ARGV[2]
      maxRequests, // ARGV[3]
      member       // ARGV[4]
    );

    const [allowed, count, limit, resetAt] = result;

    return {
      allowed:   allowed === 1,
      limit,
      remaining: Math.max(0, limit - count),
      resetAt:   Number(resetAt),
      count,
    };
  }

  /**
   * Token bucket rate limit check.
   */
  async tokenBucket(key, windowMs, maxRequests) {
    const now        = Date.now();
    const refillRate = maxRequests / windowMs; // tokens per ms

    const result = await this.client.evalsha(
      this._scripts.tokenBucket,
      1,           // number of keys
      key,         // KEYS[1]
      now,         // ARGV[1]
      maxRequests, // ARGV[2]
      refillRate,  // ARGV[3]
      windowMs * 2 // ARGV[4] — TTL: keep key for 2 windows of inactivity
    );

    const [allowed, remaining, capacity, msUntilToken] = result;
    const resetAt = now + Number(msUntilToken);

    return {
      allowed:   allowed === 1,
      limit:     capacity,
      remaining: Number(remaining),
      resetAt,
      count:     capacity - remaining,
    };
  }
}

module.exports = { RedisStore };
```

### `src/stores/memoryStore.js`

```javascript
/**
 * In-memory rate limit store.
 *
 * Used as a fallback when Redis is unavailable.
 *
 * CRITICAL LIMITATIONS — must be understood before deploying:
 *
 * 1. NOT DISTRIBUTED: Each Node.js process has its own counters.
 *    With 4 worker processes, a user can make 4x the intended limit.
 *    This is acceptable as a degraded-mode fallback, not a primary store.
 *
 * 2. NOT PERSISTENT: Counters reset on process restart.
 *    A restart during an attack effectively resets all limits.
 *
 * 3. MEMORY GROWTH: Without cleanup, the Map grows unboundedly.
 *    The cleanup interval below prevents this, but there's a window
 *    between cleanups where memory can spike under attack.
 *
 * These limitations are documented here and in the operational gaps section.
 * The in-memory store should only be used as a temporary fallback.
 */

'use strict';

class MemoryStore {
  constructor() {
    // Map<key, { count, resetAt, tokens, lastRefill }>
    this._store = new Map();

    // Periodic cleanup of expired entries
    // Prevents unbounded memory growth under sustained traffic
    this._cleanupInterval = setInterval(
      () => this._cleanup(),
      30_000 // every 30 seconds
    );

    // Don't let this interval prevent process exit
    this._cleanupInterval.unref();
  }

  _cleanup() {
    const now = Date.now();
    let cleaned = 0;
    for (const [key, entry] of this._store.entries()) {
      if (entry.resetAt && entry.resetAt < now) {
        this._store.delete(key);
        cleaned++;
      }
    }
    if (cleaned > 0) {
      console.debug(`[MemoryStore] Cleaned ${cleaned} expired entries`);
    }
  }

  destroy() {
    clearInterval(this._cleanupInterval);
    this._store.clear();
  }

  async fixedWindow(key, windowMs, maxRequests) {
    const now = Date.now();
    let entry = this._store.get(key);

    if (!entry || entry.resetAt <= now) {
      // New window
      entry = { count: 0, resetAt: now + windowMs };
    }

    entry.count++;
    this._store.set(key, entry);

    return {
      allowed:   entry.count <= maxRequests,
      limit:     maxRequests,
      remaining: Math.max(0, maxRequests - entry.count),
      resetAt:   entry.resetAt,
      count:     entry.count,
    };
  }

  async slidingWindow(key, windowMs, maxRequests) {
    const now         = Date.now();
    const windowStart = now - windowMs;
    let entry = this._store.get(key);

    if (!entry) {
      entry = { requests: [] };
    }

    // Prune old requests
    entry.requests = entry.requests.filter((ts) => ts > windowStart);

    const count   = entry.requests.length;
    const allowed = count < maxRequests;

    if (allowed) {
      entry.requests.push(now);
    }

    this._store.set(key, entry);

    // Reset time = when the oldest request in the window expires
    const oldestTs = entry.requests[0] || now;
    const resetAt  = oldestTs + windowMs;

    return {
      allowed,
      limit:     maxRequests,
      remaining: Math.max(0, maxRequests - entry.requests.length),
      resetAt,
      count:     entry.requests.length,
    };
  }

  async tokenBucket(key, windowMs, maxRequests) {
    const now        = Date.now();
    const refillRate = maxRequests / windowMs;
    let entry = this._store.get(key);

    if (!entry) {
      entry = { tokens: maxRequests, lastRefill: now };
    }

    // Refill tokens based on elapsed time
    const elapsed  = now - entry.lastRefill;
    entry.tokens   = Math.min(maxRequests, entry.tokens + elapsed * refillRate);
    entry.lastRefill = now;

    const allowed = entry.tokens >= 1;
    if (allowed) {
      entry.tokens -= 1;
    }

    // Set a resetAt so cleanup can remove this entry
    entry.resetAt = now + windowMs * 2;
    this._store.set(key, entry);

    const msUntilToken = allowed ? 0 : Math.ceil((1 - entry.tokens) / refillRate);

    return {
      allowed,
      limit:     maxRequests,
      remaining: Math.floor(entry.tokens),
      resetAt:   now + msUntilToken,
      count:     maxRequests - Math.floor(entry.tokens),
    };
  }
}

module.exports = { MemoryStore };
```

### `src/middleware/rateLimitMiddleware.js`

```javascript
/**
 * Express middleware that enforces rate limits.
 *
 * This is the hot path — every request goes through here.
 * Design priorities: fast, non-blocking, never crashes the app.
 */

'use strict';

const { resolveConfig }    = require('../config/rateLimitConfig');
const { RedisStore }       = require('../stores/redisStore');
const { MemoryStore }      = require('../stores/memoryStore');
const { extractIdentifier, buildKey, buildSlidingKey } = require('../utils/keyBuilder');
const { writeHeaders }     = require('../utils/headerWriter');

class RateLimiter {
  /**
   * @param {object} options
   * @param {object} options.redisConfig   - Redis connection config
   * @param {object} [options.logger]      - Logger with .warn/.error/.info methods
   * @param {boolean} [options.trustProxy] - Whether to trust X-Forwarded-For
   */
  constructor(options = {}) {
    this.logger       = options.logger || console;
    this.trustProxy   = options.trustProxy || false;
    this.redisStore   = new RedisStore(options.redisConfig);
    this.memoryStore  = new MemoryStore();
    this.activeStore  = null; // set after connect()
    this._initialized = false;
  }

  async initialize() {
    try {
      await this.redisStore.connect();
      this.activeStore  = this.redisStore;
      this._initialized = true;
      this.logger.info('[RateLimiter] Connected to Redis — using distributed store');
    } catch (err) {
      this.logger.error(
        '[RateLimiter] Failed to connect to Redis, falling back to memory store:',
        err.message
      );
      this.activeStore  = this.memoryStore;
      this._initialized = true;
    }
  }

  async destroy() {
    await this.redisStore.disconnect();
    this.memoryStore.destroy();
  }

  /**
   * Returns an Express middleware function.
   *
   * Usage:
   *   const limiter = new RateLimiter({ redisConfig });
   *   await limiter.initialize();
   *   app.use(limiter.middleware());
   */
  middleware() {
    return async (req, res, next) => {
      // If somehow called before initialize(), fail open
      if (!this._initialized) {
        this.logger.warn('[RateLimiter] Called before initialize() — allowing request');
        return next();
      }

      let limitResult;

      try {
        limitResult = await this._checkLimit(req);
      } catch (err) {
        // Store operation failed — apply fail mode
        this.logger.error('[RateLimiter] Store error during limit check:', err.message);
        limitResult = this._handleStoreFailure(req);
      }

      // Always write headers — even on allowed requests
      // Clients use these to implement backoff
      writeHeaders(res, limitResult);
