# Production Rate Limiter in Node.js

Let me work through this properly — requirements first, then implementation, then operational gaps.

---

## Phase 1: Requirements Design

### Functional Requirements
- Limit requests per identifier (IP, user ID, API key) within a time window
- Support multiple algorithms (Token Bucket for burst tolerance, Fixed Window for simplicity, Sliding Window Log for precision)
- Return standard rate limit headers (`X-RateLimit-Limit`, `X-RateLimit-Remaining`, `X-RateLimit-Reset`, `Retry-After`)
- Support per-route/per-tier limit configurations
- Allow requests through when the limiter itself fails (fail-open) OR block them (fail-closed) — configurable

### Non-Functional Requirements
- Sub-millisecond overhead on the hot path
- Atomic operations to prevent race conditions under concurrency
- Horizontally scalable (multiple Node.js instances must share state)
- Graceful degradation if Redis is unavailable

### Explicit Design Decisions Made Here
| Decision | Choice | Reason |
|---|---|---|
| Storage backend | Redis with Lua scripts | Atomic operations, shared state across instances |
| Primary algorithm | Sliding Window Counter | Better precision than fixed window, cheaper than sliding log |
| Fallback behavior | Fail-open with logging | Availability over strict enforcement |
| Clock source | Redis server time | Avoids clock skew across app instances |

### Lua Scripts — Atomicity is Non-Negotiable

**`scripts/lua/slidingWindow.lua`**
```lua
-- Sliding Window Counter Algorithm
-- Atomic: runs as a single Redis transaction
-- 
-- KEYS[1] = current window key   e.g. "rl:user:123:1700000060"
-- KEYS[2] = previous window key  e.g. "rl:user:123:1700000050"
-- ARGV[1] = current timestamp (ms)
-- ARGV[2] = window size (ms)
-- ARGV[3] = max requests allowed
-- ARGV[4] = TTL for keys (seconds)
--
-- Returns: { allowed (0|1), current_count, limit, reset_timestamp_ms }

local current_key    = KEYS[1]
local previous_key   = KEYS[2]
local now            = tonumber(ARGV[1])
local window_ms      = tonumber(ARGV[2])
local limit          = tonumber(ARGV[3])
local ttl_seconds    = tonumber(ARGV[4])

-- How far into the current window are we? (0.0 to 1.0)
local window_start   = now - (now % window_ms)
local elapsed_ratio  = (now - window_start) / window_ms

-- Weighted count: previous window contributes proportionally
-- to how much of it overlaps with our rolling window
local prev_count     = tonumber(redis.call('GET', previous_key) or 0)
local curr_count     = tonumber(redis.call('GET', current_key) or 0)

local weighted_count = (prev_count * (1 - elapsed_ratio)) + curr_count

if weighted_count >= limit then
    -- Rejected — do NOT increment
    local reset_at = window_start + window_ms
    return { 0, math.floor(weighted_count), limit, reset_at }
end

-- Allowed — increment current window counter
local new_count = redis.call('INCR', current_key)

-- Set TTL only on first write to avoid resetting it
if new_count == 1 then
    redis.call('EXPIRE', current_key, ttl_seconds)
end

local reset_at = window_start + window_ms
return { 1, math.floor(weighted_count + 1), limit, reset_at }
```

**`scripts/lua/tokenBucket.lua`**
```lua
-- Token Bucket Algorithm
-- Better for burst tolerance (e.g., allow 10 req burst, then 1/sec steady)
--
-- KEYS[1] = bucket key
-- ARGV[1] = current timestamp (ms)
-- ARGV[2] = bucket capacity (max tokens)
-- ARGV[3] = refill rate (tokens per ms)
-- ARGV[4] = TTL (seconds)
--
-- Returns: { allowed (0|1), tokens_remaining, capacity, retry_after_ms }

local key          = KEYS[1]
local now          = tonumber(ARGV[1])
local capacity     = tonumber(ARGV[2])
local refill_rate  = tonumber(ARGV[3])  -- tokens per millisecond
local ttl          = tonumber(ARGV[4])

local bucket = redis.call('HMGET', key, 'tokens', 'last_refill')
local tokens      = tonumber(bucket[1])
local last_refill = tonumber(bucket[2])

if tokens == nil then
    -- First request: initialize full bucket
    tokens      = capacity
    last_refill = now
end

-- Refill tokens based on elapsed time
local elapsed       = math.max(0, now - last_refill)
local refill_amount = elapsed * refill_rate
tokens = math.min(capacity, tokens + refill_amount)

if tokens < 1 then
    -- Calculate how long until 1 token is available
    local tokens_needed = 1 - tokens
    local retry_after   = math.ceil(tokens_needed / refill_rate)
    
    -- Still update last_refill so partial tokens accumulate correctly
    redis.call('HMSET', key, 'tokens', tokens, 'last_refill', now)
    redis.call('EXPIRE', key, ttl)
    
    return { 0, 0, capacity, retry_after }
end

-- Consume one token
tokens = tokens - 1
redis.call('HMSET', key, 'tokens', tokens, 'last_refill', now)
redis.call('EXPIRE', key, ttl)

return { 1, math.floor(tokens), capacity, 0 }
```

### Redis Store

**`src/store/redisStore.js`**
```javascript
'use strict';

const Redis = require('ioredis');
const fs    = require('fs');
const path  = require('path');

// Load Lua scripts at startup — not on every request
const SCRIPTS = {
  slidingWindow: fs.readFileSync(
    path.join(__dirname, '../../scripts/lua/slidingWindow.lua'),
    'utf8'
  ),
  tokenBucket: fs.readFileSync(
    path.join(__dirname, '../../scripts/lua/tokenBucket.lua'),
    'utf8'
  ),
};

class RedisStore {
  constructor(options = {}) {
    this.client = new Redis({
      host:              options.host     || process.env.REDIS_HOST || '127.0.0.1',
      port:              options.port     || process.env.REDIS_PORT || 6379,
      password:          options.password || process.env.REDIS_PASSWORD,
      db:                options.db       || 0,
      
      // Connection pool — critical for high throughput
      maxRetriesPerRequest: 1,          // fail fast; don't queue up retries
      enableReadyCheck:     true,
      lazyConnect:          false,
      
      // Reconnect with backoff — don't hammer a recovering Redis
      retryStrategy(times) {
        if (times > 10) return null;    // stop retrying after 10 attempts
        return Math.min(times * 100, 3000);
      },
    });

    this.isHealthy  = false;
    this.scriptShas = {};             // store SHA hashes after SCRIPT LOAD

    this._attachEventHandlers();
  }

  _attachEventHandlers() {
    this.client.on('ready', async () => {
      this.isHealthy = true;
      await this._loadScripts();
      console.log('[RedisStore] Connected and scripts loaded');
    });

    this.client.on('error', (err) => {
      // Log but don't crash — store will return unhealthy
      console.error('[RedisStore] Redis error:', err.message);
    });

    this.client.on('close', () => {
      this.isHealthy = false;
      console.warn('[RedisStore] Connection closed');
    });

    this.client.on('reconnecting', () => {
      console.warn('[RedisStore] Reconnecting to Redis...');
    });
  }

  /**
   * Pre-load Lua scripts using SCRIPT LOAD.
   * EVALSHA is faster than EVAL — Redis skips script parsing on every call.
   */
  async _loadScripts() {
    try {
      for (const [name, script] of Object.entries(SCRIPTS)) {
        this.scriptShas[name] = await this.client.script('LOAD', script);
      }
      console.log('[RedisStore] Lua scripts loaded:', Object.keys(this.scriptShas));
    } catch (err) {
      console.error('[RedisStore] Failed to load Lua scripts:', err.message);
      throw err;
    }
  }

  /**
   * Execute a pre-loaded Lua script by SHA.
   * Falls back to EVAL if SHA is missing (e.g., after Redis restart flushed scripts).
   */
  async evalScript(scriptName, keys, args) {
    const sha    = this.scriptShas[scriptName];
    const script = SCRIPTS[scriptName];

    if (!sha) {
      // SHA not loaded yet — use EVAL directly
      return this.client.eval(script, keys.length, ...keys, ...args);
    }

    try {
      return await this.client.evalsha(sha, keys.length, ...keys, ...args);
    } catch (err) {
      if (err.message.includes('NOSCRIPT')) {
        // Redis was restarted and flushed scripts — reload and retry
        console.warn('[RedisStore] Script cache miss, reloading scripts...');
        await this._loadScripts();
        return this.client.evalsha(
          this.scriptShas[scriptName],
          keys.length,
          ...keys,
          ...args
        );
      }
      throw err;
    }
  }

  /**
   * Returns current Redis server time in milliseconds.
   * Using server time avoids clock skew between app instances.
   */
  async getTime() {
    const [seconds, microseconds] = await this.client.time();
    return parseInt(seconds) * 1000 + Math.floor(parseInt(microseconds) / 1000);
  }

  healthy() {
    return this.isHealthy;
  }

  async quit() {
    await this.client.quit();
  }
}

module.exports = RedisStore;
```

### In-Memory Store (Fallback / Testing)

**`src/store/memoryStore.js`**
```javascript
'use strict';

/**
 * In-memory store for:
 *   1. Local development without Redis
 *   2. Fallback when Redis is unavailable (fail-open path)
 *   3. Unit testing without infrastructure dependencies
 *
 * WARNING: Not suitable for multi-instance deployments.
 * Each instance has its own counter — limits are per-process.
 * This is intentional for the fail-open degraded mode.
 */
class MemoryStore {
  constructor() {
    this.store = new Map();

    // Periodic cleanup to prevent unbounded memory growth
    // In production, tune this interval based on your traffic patterns
    this._cleanupInterval = setInterval(
      () => this._cleanup(),
      60_000  // every 60 seconds
    );

    // Don't let this interval prevent process exit
    this._cleanupInterval.unref();

    this.isHealthy = true;
  }

  /**
   * Sliding window counter — mirrors the Lua script logic
   * so behavior is consistent between Redis and memory modes.
   */
  slidingWindow(key, windowMs, limit, nowMs) {
    const windowStart    = nowMs - (nowMs % windowMs);
    const currentKey     = `${key}:${windowStart}`;
    const previousKey    = `${key}:${windowStart - windowMs}`;
    const elapsedRatio   = (nowMs - windowStart) / windowMs;

    const prevEntry      = this.store.get(previousKey);
    const currEntry      = this.store.get(currentKey);

    const prevCount      = prevEntry ? prevEntry.count : 0;
    const currCount      = currEntry ? currEntry.count : 0;
    const weightedCount  = (prevCount * (1 - elapsedRatio)) + currCount;

    const resetAt        = windowStart + windowMs;

    if (weightedCount >= limit) {
      return { allowed: false, count: Math.floor(weightedCount), limit, resetAt };
    }

    const newCount = currCount + 1;
    this.store.set(currentKey, {
      count:     newCount,
      expiresAt: resetAt + windowMs,  // keep previous window alive for one extra window
    });

    return { allowed: true, count: Math.floor(weightedCount + 1), limit, resetAt };
  }

  tokenBucket(key, capacity, refillRatePerMs, nowMs, ttlMs) {
    const entry = this.store.get(key);
    let tokens, lastRefill;

    if (!entry) {
      tokens     = capacity;
      lastRefill = nowMs;
    } else {
      tokens     = entry.tokens;
      lastRefill = entry.lastRefill;
    }

    const elapsed      = Math.max(0, nowMs - lastRefill);
    const refillAmount = elapsed * refillRatePerMs;
    tokens             = Math.min(capacity, tokens + refillAmount);

    if (tokens < 1) {
      const retryAfterMs = Math.ceil((1 - tokens) / refillRatePerMs);
      this.store.set(key, { tokens, lastRefill: nowMs, expiresAt: nowMs + ttlMs });
      return { allowed: false, tokens: 0, capacity, retryAfterMs };
    }

    tokens -= 1;
    this.store.set(key, { tokens, lastRefill: nowMs, expiresAt: nowMs + ttlMs });
    return { allowed: true, tokens: Math.floor(tokens), capacity, retryAfterMs: 0 };
  }

  _cleanup() {
    const now = Date.now();
    let removed = 0;
    for (const [key, entry] of this.store.entries()) {
      if (entry.expiresAt && entry.expiresAt < now) {
        this.store.delete(key);
        removed++;
      }
    }
    if (removed > 0) {
      console.debug(`[MemoryStore] Cleaned up ${removed} expired entries`);
    }
  }

  healthy() {
    return this.isHealthy;
  }
}

module.exports = MemoryStore;
```

### Limit Configuration

**`src/config/limits.js`**
```javascript
'use strict';

/**
 * Rate limit configurations per route/tier.
 *
 * Structure:
 *   key:         unique name for this limit config
 *   algorithm:   'slidingWindow' | 'tokenBucket'
 *   windowMs:    time window in milliseconds (slidingWindow)
 *   limit:       max requests per window (slidingWindow)
 *   capacity:    max tokens in bucket (tokenBucket)
 *   refillRate:  tokens per second (tokenBucket)
 *   keyBy:       function(req) => string — what to rate limit by
 *   skip:        function(req) => bool  — bypass condition
 */

const LIMIT_CONFIGS = {
  // Default: applied to all routes unless overridden
  default: {
    algorithm: 'slidingWindow',
    windowMs:  60_000,      // 1 minute
    limit:     100,
    keyBy:     (req) => req.ip,
    skip:      () => false,
  },

  // Strict limit for authentication endpoints
  // Prevents brute-force and credential stuffing
  auth: {
    algorithm: 'slidingWindow',
    windowMs:  15 * 60_000,  // 15 minutes
    limit:     10,
    keyBy:     (req) => `auth:${req.ip}`,
    skip:      () => false,
  },

  // API endpoints keyed by API key — allows per-customer limits
  api: {
    algorithm: 'tokenBucket',
    capacity:  50,            // burst: up to 50 requests
    refillRate: 10,           // steady state: 10 requests/second
    keyBy:     (req) => {
      const apiKey = req.headers['x-api-key'];
      if (!apiKey) return `anon:${req.ip}`;
      return `apikey:${apiKey}`;
    },
    skip: (req) => {
      // Internal health checks bypass rate limiting
      return req.headers['x-internal-token'] === process.env.INTERNAL_TOKEN;
    },
  },

  // Expensive operations — very tight limits
  expensive: {
    algorithm: 'slidingWindow',
    windowMs:  60_000,
    limit:     5,
    keyBy:     (req) => {
      const userId = req.user?.id;
      return userId ? `expensive:user:${userId}` : `expensive:ip:${req.ip}`;
    },
    skip: () => false,
  },
};

module.exports = LIMIT_CONFIGS;
```

### Headers Utility

**`src/utils/headers.js`**
```javascript
'use strict';

/**
 * Sets standard rate limit response headers.
 *
 * Headers follow the IETF draft spec:
 * https://datatracker.ietf.org/doc/html/draft-ietf-httpapi-ratelimit-headers
 *
 * Also sets the legacy X-RateLimit-* headers for client compatibility.
 */
function setRateLimitHeaders(res, { limit, remaining, resetAtMs, retryAfterMs }) {
  const resetAtSeconds = Math.ceil(resetAtMs / 1000);
  const nowSeconds     = Math.floor(Date.now() / 1000);

  // Standard headers (IETF draft)
  res.setHeader('RateLimit-Limit',     limit);
  res.setHeader('RateLimit-Remaining', Math.max(0, remaining));
  res.setHeader('RateLimit-Reset',     resetAtSeconds);

  // Legacy X- headers — many clients still expect these
  res.setHeader('X-RateLimit-Limit',     limit);
  res.setHeader('X-RateLimit-Remaining', Math.max(0, remaining));
  res.setHeader('X-RateLimit-Reset',     resetAtSeconds);

  if (retryAfterMs != null && retryAfterMs > 0) {
    // Retry-After in seconds, rounded up
    res.setHeader('Retry-After', Math.ceil(retryAfterMs / 1000));
  }
}

module.exports = { setRateLimitHeaders };
```

### Core Middleware

**`src/middleware/rateLimiter.js`**
```javascript
'use strict';

const RedisStore  = require('../store/redisStore');
const MemoryStore = require('../store/memoryStore');
const LIMIT_CONFIGS = require('../config/limits');
const { setRateLimitHeaders } = require('../utils/headers');

class RateLimiter {
  constructor(options = {}) {
    this.redisStore   = new RedisStore(options.redis || {});
    this.memoryStore  = new MemoryStore();
    this.failBehavior = options.failBehavior || 'open';  // 'open' | 'closed'
    this.onLimitReached = options.onLimitReached || null; // hook for metrics/alerting
    this.onError        = options.onError || null;        // hook for error reporting
  }

  /**
   * Returns an Express middleware function for a named limit config.
   *
   * Usage:
   *   app.post('/login', rateLimiter.middleware('auth'), loginHandler);
   *   app.use(rateLimiter.middleware('default'));
   */
  middleware(configName = 'default') {
    const config = LIMIT_CONFIGS[configName];

    if (!config) {
      throw new Error(`[RateLimiter] Unknown config: "${configName}". ` +
                      `Available: ${Object.keys(LIMIT_CONFIGS).join(', ')}`);
    }

    return async (req, res, next) => {
      // Allow certain requests to bypass rate limiting entirely
      if (config.skip && config.skip(req)) {
        return next();
      }

      const key = config.keyBy(req);

      try {
        const result = await this._check(key, config);
        
        setRateLimitHeaders(res, {
          limit:        result.limit,
          remaining:    result.remaining,
          resetAtMs:    result.resetAtMs,
          retryAfterMs: result.retryAfterMs,
        });

        if (!result.allowed) {
          // Fire the hook — caller can emit metrics, send alerts, etc.
          if (this.onLimitReached) {
            this.onLimitReached({ key, config: configName, req });
          }

          return res.status(429).json({
            error:   'Too Many Requests',
            message: `Rate limit exceeded. Try again in ${
              Math.ceil((result.retryAfterMs || 0) / 1000)
            } seconds.`,
          });
        }

        next();

      } catch (err) {
        // Redis or script error — decide fail-open vs fail-closed
        if (this.onError) {
          this.onError(err, { key, config: configName, req });
        }

        console.error('[RateLimiter] Error during rate limit check:', err.message);

        if (this.failBehavior === 'closed') {
          return res.status(503).json({
            error:   'Service Unavailable',
            message: 'Rate limiter is temporarily unavailable.',
          });
        }

        // Fail-open: allow the request through but log it
        // This trades strict enforcement for availability
        console.warn('[RateLimiter] Failing open for key:', key);
        next();
      }
    };
  }

  /**
   * Core check logic — dispatches to the correct algorithm
   * using Redis if healthy, falling back to memory store.
   */
  async _check(key, config) {
    const useRedis = this.redisStore.healthy();
    const now      = useRedis
      ? await this.redisStore.getTime()
      : Date.now();

    if (config.algorithm === 'slidingWindow') {
      return this._checkSlidingWindow(key, config, now, useRedis);
    }

    if (config.algorithm === 'tokenBucket') {
      return this._checkTokenBucket(key, config, now, useRedis);
    }

    throw new Error(`[RateLimiter] Unknown algorithm: "${config.algorithm}"`);
  }

  async _checkSlidingWindow(key, config, nowMs, useRedis) {
    const { windowMs, limit } = config;
    const windowStart         = nowMs - (nowMs % windowMs);
    const currentWindowKey    = `rl:${key}:${windowStart}`;
    const previousWindowKey   = `rl:${key}:${windowStart - windowMs}`;
    const ttlSeconds          = Math.ceil((windowMs * 2) / 1000);

    let result;

    if (useRedis) {
      const raw = await this.redisStore.evalScript(
        'slidingWindow',
        [currentWindowKey, previousWindowKey],
        [nowMs, windowMs, limit, ttlSeconds]
      );

      // Lua returns: { allowed, count, limit, reset_at_ms }
      result = {
        allowed:      raw[0] === 1,
        count:        raw[1],
        limit:        raw[2],
        resetAtMs:    raw[3],
        retryAfterMs: raw[0] === 1 ? 0 : raw[3] - nowMs,
      };
    } else {
      // Memory store fallback
      const mem = this.memoryStore.slidingWindow(key, windowMs, limit, nowMs);
      result = {
        allowed:      mem.allowed,
        count:        mem.count,
        limit:        mem.limit,
        resetAtMs:    mem.resetAt,
        retryAfterMs: mem.allowed ? 0 : mem.resetAt - nowMs,
      };
    }

    result.remaining = Math.max(0, result.limit - result.count);
    return result;
  }

  async _checkTokenBucket(key, config, nowMs, useRedis) {
    const { capacity, refillRate } = config;
    // Convert refillRate (tokens/sec) to tokens/ms for the Lua script
    const refillRatePerMs = refillRate / 1000;
    const ttlSeconds      = Math.ceil((capacity / refillRate) * 2);
    const bucketKey       = `rl:tb:${key}`;

    let result;

    if (useRedis) {
      const raw = await this.redisStore.evalScript(
        'tokenBucket',
        [bucketKey],
        [nowMs, capacity, refillRatePerMs, ttlSeconds]
      );

      // Lua returns: { allowed, tokens_remaining, capacity, retry_after_ms }
      result = {
        allowed:      raw[0] === 1,
        remaining:    raw[1],
        limit:        raw[2],
        resetAtMs:    nowMs + (raw[3] || 0),
        retryAfterMs: raw[3] || 0,
      };
    } else {
      const mem = this.memoryStore.tokenBucket(
        bucketKey, capacity, refillRatePerMs, nowMs, ttlSeconds * 1000
      );
      result = {
        allowed:      mem.allowed,
        remaining:    mem.tokens,
        limit:        mem.capacity,
        resetAtMs:    nowMs + (mem.retryAfterMs || 0),
        retryAfterMs: mem.retryAfterMs || 0,
      };
    }

    return result;
  }

  /**
   * Programmatic check — use this outside of Express middleware
   * (e.g., in a WebSocket handler, background job, gRPC service)
   */
  async check(identifier, configName = 'default') {
    const config = LIMIT_CONFIGS[configName];
    if (!config) throw new Error(`Unknown rate limit config: ${configName}`);
    return this._check(identifier, config);
  }

  async close() {
    await this.redisStore.quit();
  }
}

module.exports = RateLimiter;
```
