# Production Rate Limiter in Node.js

Let me work through this properly — requirements first, then implementation, then operational gaps.

---

## Phase 1: Requirements Design

### Functional Requirements
- Limit requests per identifier (IP, user ID, API key) within a time window
- Support multiple rate limit strategies
- Return standard rate limit headers (`X-RateLimit-*`, `Retry-After`)
- Allow per-route and per-tier limit configuration
- Graceful handling when the backing store is unavailable

### Non-Functional Requirements
- Sub-millisecond overhead on the hot path
- No single point of failure
- Correct behavior under concurrent requests (race-condition safe)
- Horizontally scalable across multiple Node.js processes

### Algorithm Selection

| Algorithm | Pros | Cons | Verdict |
|---|---|---|---|
| Fixed Window | Simple, fast | Burst at window boundary (2x spike) | ❌ Too naive for production |
| Sliding Window Log | Perfectly accurate | High memory per user | ❌ Memory cost at scale |
| Token Bucket | Smooth bursting, intuitive | State harder to reason about | ✅ Good for API quotas |
| Sliding Window Counter | Accurate, memory efficient | Approximate (acceptable) | ✅ **Primary choice** |

**Decision:** Sliding Window Counter as default, Token Bucket available for burst-sensitive routes.

### `src/utils/identifierExtractor.js`

```javascript
'use strict';

/**
 * Extracts a stable, trustworthy identifier from the request.
 *
 * Priority:
 *   1. Authenticated user ID (set by upstream auth middleware)
 *   2. API key from Authorization header
 *   3. IP address (with proxy trust handled explicitly)
 *
 * SECURITY NOTE:
 *   Never trust req.ip blindly in production. Express sets req.ip from
 *   X-Forwarded-For only when app.set('trust proxy', n) is configured.
 *   Without that setting, a client can spoof their IP and bypass limits.
 *   See: https://expressjs.com/en/guide/behind-proxies.html
 */

const BEARER_PREFIX = 'Bearer ';
const API_KEY_PREFIX = 'ApiKey ';

function extractIdentifier(req) {
  // Prefer authenticated identity — most accurate, hardest to spoof
  if (req.user?.id) {
    return { id: `user:${req.user.id}`, type: 'user' };
  }

  // API key from Authorization header
  const authHeader = req.headers['authorization'] || '';
  if (authHeader.startsWith(API_KEY_PREFIX)) {
    const apiKey = authHeader.slice(API_KEY_PREFIX.length).trim();
    if (apiKey) {
      // Hash the key so we don't store raw credentials in Redis
      return { id: `apikey:${hashKey(apiKey)}`, type: 'apikey' };
    }
  }

  // Fall back to IP
  const ip = req.ip || req.socket?.remoteAddress || 'unknown';
  return { id: `ip:${ip}`, type: 'ip' };
}

/**
 * Simple non-cryptographic hash for Redis key generation.
 * We don't need security here — just a fixed-length key.
 * For actual API key validation, use your auth middleware.
 */
function hashKey(str) {
  let hash = 5381;
  for (let i = 0; i < str.length; i++) {
    hash = ((hash << 5) + hash) ^ str.charCodeAt(i);
    hash = hash >>> 0; // Convert to unsigned 32-bit
  }
  return hash.toString(16);
}

module.exports = { extractIdentifier };
```

### `src/stores/redisStore.js`

```javascript
'use strict';

/**
 * Redis-backed store using atomic Lua scripts.
 *
 * WHY LUA SCRIPTS?
 * ----------------
 * A naive implementation does:
 *   1. GET counter
 *   2. Check if under limit
 *   3. INCR counter
 *
 * Between steps 1 and 3, another request can read the same counter value.
 * Both see "under limit", both increment — you've allowed 2 requests when
 * only 1 slot remained. Under load this compounds badly.
 *
 * Lua scripts execute atomically on the Redis server. No other command
 * runs between our GET and INCR. This is the correct solution.
 *
 * REDIS CLUSTER NOTE:
 * All keys accessed in one Lua script must live on the same Redis slot.
 * We use a single key per script call, so this is safe.
 */

const Redis = require('ioredis');

// Sliding window counter — atomic increment + TTL set in one round trip
const SLIDING_WINDOW_SCRIPT = `
local key        = KEYS[1]
local window_ms  = tonumber(ARGV[1])
local now        = tonumber(ARGV[2])
local max        = tonumber(ARGV[3])

-- Remove entries outside the current window
redis.call('ZREMRANGEBYSCORE', key, '-inf', now - window_ms)

-- Count remaining entries
local count = redis.call('ZCARD', key)

if count < max then
  -- Add current request with score = timestamp (microseconds for uniqueness)
  redis.call('ZADD', key, now, now .. '-' .. math.random(1, 1000000))
  redis.call('PEXPIRE', key, window_ms)
  return {count + 1, max - count - 1, 0}  -- {current, remaining, limited}
else
  -- Get the oldest entry to calculate retry-after
  local oldest = redis.call('ZRANGE', key, 0, 0, 'WITHSCORES')
  local retry_after_ms = 0
  if oldest[2] then
    retry_after_ms = window_ms - (now - tonumber(oldest[2]))
  end
  return {count, 0, 1, retry_after_ms}  -- {current, remaining, limited, retry_ms}
end
`;

// Token bucket — refill tokens based on elapsed time, then consume
const TOKEN_BUCKET_SCRIPT = `
local key            = KEYS[1]
local capacity       = tonumber(ARGV[1])
local refill_rate    = tonumber(ARGV[2])
local now            = tonumber(ARGV[3])
local window_ms      = tonumber(ARGV[4])

local data = redis.call('HMGET', key, 'tokens', 'last_refill')
local tokens      = tonumber(data[1]) or capacity
local last_refill = tonumber(data[2]) or now

-- Calculate tokens to add based on elapsed time
local elapsed = now - last_refill
local new_tokens = math.min(capacity, tokens + (elapsed * refill_rate / 1000))

if new_tokens >= 1 then
  new_tokens = new_tokens - 1
  redis.call('HMSET', key, 'tokens', new_tokens, 'last_refill', now)
  redis.call('PEXPIRE', key, window_ms)
  local remaining = math.floor(new_tokens)
  return {1, remaining, 0}  -- {consumed, remaining, limited}
else
  local retry_ms = math.ceil((1 - new_tokens) / refill_rate * 1000)
  redis.call('HMSET', key, 'tokens', new_tokens, 'last_refill', now)
  redis.call('PEXPIRE', key, window_ms)
  return {0, 0, 1, retry_ms}  -- {consumed, remaining, limited, retry_ms}
end
`;

class RedisStore {
  constructor(options = {}) {
    this.client = options.client || new Redis({
      host:           options.host || '127.0.0.1',
      port:           options.port || 6379,
      password:       options.password,
      db:             options.db || 0,
      // Fail fast — don't queue commands when Redis is down
      enableOfflineQueue: false,
      // Reconnect with backoff, but don't block the app
      retryStrategy: (times) => Math.min(times * 100, 3000),
      lazyConnect: true,
    });

    this._healthy = false;
    this._setupHealthMonitoring();

    // Pre-load scripts for performance (avoids sending script body each time)
    this._slidingWindowSha = null;
    this._tokenBucketSha   = null;
    this._scriptsLoaded    = false;
  }

  async connect() {
    await this.client.connect();
    await this._loadScripts();
    this._healthy = true;
  }

  async _loadScripts() {
    // SCRIPT LOAD returns a SHA1 we can use with EVALSHA (faster than EVAL)
    this._slidingWindowSha = await this.client.script('LOAD', SLIDING_WINDOW_SCRIPT);
    this._tokenBucketSha   = await this.client.script('LOAD', TOKEN_BUCKET_SCRIPT);
    this._scriptsLoaded = true;
  }

  _setupHealthMonitoring() {
    this.client.on('ready',        () => { this._healthy = true; });
    this.client.on('error',        () => { this._healthy = false; });
    this.client.on('close',        () => { this._healthy = false; });
    this.client.on('reconnecting', () => { this._healthy = false; });
  }

  isHealthy() {
    return this._healthy && this._scriptsLoaded;
  }

  /**
   * Sliding window counter check + increment.
   * Returns: { current, remaining, limited, retryAfterMs }
   */
  async slidingWindow(key, windowMs, max) {
    const now = Date.now();

    let result;
    try {
      // Try EVALSHA first (uses cached script)
      result = await this.client.evalsha(
        this._slidingWindowSha,
        1,          // number of keys
        key,        // KEYS[1]
        windowMs,   // ARGV[1]
        now,        // ARGV[2]
        max,        // ARGV[3]
      );
    } catch (err) {
      if (err.message?.includes('NOSCRIPT')) {
        // Script was flushed (e.g., Redis restart) — reload and retry
        await this._loadScripts();
        result = await this.client.evalsha(
          this._slidingWindowSha, 1, key, windowMs, now, max,
        );
      } else {
        throw err;
      }
    }

    return {
      current:      result[0],
      remaining:    result[1],
      limited:      result[2] === 1,
      retryAfterMs: result[3] || 0,
    };
  }

  /**
   * Token bucket check + consume.
   * Returns: { consumed, remaining, limited, retryAfterMs }
   */
  async tokenBucket(key, capacity, refillRatePerSecond, windowMs) {
    const now = Date.now();

    let result;
    try {
      result = await this.client.evalsha(
        this._tokenBucketSha,
        1,
        key,
        capacity,
        refillRatePerSecond,
        now,
        windowMs,
      );
    } catch (err) {
      if (err.message?.includes('NOSCRIPT')) {
        await this._loadScripts();
        result = await this.client.evalsha(
          this._tokenBucketSha, 1, key, capacity, refillRatePerSecond, now, windowMs,
        );
      } else {
        throw err;
      }
    }

    return {
      consumed:     result[0] === 1,
      remaining:    result[1],
      limited:      result[2] === 1,
      retryAfterMs: result[3] || 0,
    };
  }

  async close() {
    await this.client.quit();
  }
}

module.exports = { RedisStore };
```

### `src/stores/memoryStore.js`

```javascript
'use strict';

/**
 * In-process memory store.
 *
 * Used as:
 *   1. Primary store in development / single-process deployments
 *   2. Fallback when Redis is unavailable (fail-open strategy)
 *
 * CRITICAL LIMITATION:
 *   This store is NOT shared across processes or machines.
 *   In a multi-process deployment (PM2 cluster, Kubernetes pods),
 *   each process has its own counters. A user can multiply their
 *   effective limit by the number of processes.
 *
 *   This is an acceptable degraded mode, not a primary production strategy.
 *   Always log when this store is active so you know you're degraded.
 */

class MemoryStore {
  constructor() {
    // Map<key, { count, windowStart, tokens, lastRefill }>
    this._windows = new Map();

    // Prevent unbounded memory growth — clean up expired entries
    this._cleanupInterval = setInterval(() => this._cleanup(), 60_000);

    // Don't keep the process alive just for cleanup
    if (this._cleanupInterval.unref) {
      this._cleanupInterval.unref();
    }
  }

  isHealthy() {
    return true;
  }

  /**
   * Sliding window counter — approximate (uses fixed window internally).
   *
   * Full sliding window in memory requires storing per-request timestamps,
   * which is expensive. For the fallback store, fixed window is acceptable.
   * The degradation is bounded: at worst, 2x the limit at window boundaries.
   */
  slidingWindow(key, windowMs, max) {
    const now = Date.now();
    const entry = this._windows.get(key);

    if (!entry || now - entry.windowStart >= windowMs) {
      // New window
      this._windows.set(key, { count: 1, windowStart: now });
      return Promise.resolve({
        current:      1,
        remaining:    max - 1,
        limited:      false,
        retryAfterMs: 0,
      });
    }

    entry.count++;

    if (entry.count > max) {
      const retryAfterMs = windowMs - (now - entry.windowStart);
      return Promise.resolve({
        current:      entry.count,
        remaining:    0,
        limited:      true,
        retryAfterMs: Math.max(0, retryAfterMs),
      });
    }

    return Promise.resolve({
      current:      entry.count,
      remaining:    max - entry.count,
      limited:      false,
      retryAfterMs: 0,
    });
  }

  /**
   * Token bucket — in-memory implementation.
   */
  tokenBucket(key, capacity, refillRatePerSecond, windowMs) {
    const now = Date.now();
    let entry = this._windows.get(key);

    if (!entry) {
      entry = { tokens: capacity, lastRefill: now };
      this._windows.set(key, entry);
    }

    // Refill tokens based on elapsed time
    const elapsed = now - entry.lastRefill;
    entry.tokens = Math.min(capacity, entry.tokens + (elapsed * refillRatePerSecond / 1000));
    entry.lastRefill = now;

    if (entry.tokens >= 1) {
      entry.tokens -= 1;
      return Promise.resolve({
        consumed:     true,
        remaining:    Math.floor(entry.tokens),
        limited:      false,
        retryAfterMs: 0,
      });
    }

    const retryAfterMs = Math.ceil((1 - entry.tokens) / refillRatePerSecond * 1000);
    return Promise.resolve({
      consumed:     false,
      remaining:    0,
      limited:      true,
      retryAfterMs,
    });
  }

  _cleanup() {
    const now = Date.now();
    for (const [key, entry] of this._windows.entries()) {
      // Remove entries that haven't been touched in 10 minutes
      const age = now - (entry.windowStart || entry.lastRefill || 0);
      if (age > 600_000) {
        this._windows.delete(key);
      }
    }
  }

  close() {
    clearInterval(this._cleanupInterval);
    this._windows.clear();
  }
}

module.exports = { MemoryStore };
```

### `src/algorithms/slidingWindowCounter.js`

```javascript
'use strict';

/**
 * Sliding window counter algorithm.
 *
 * HOW IT WORKS:
 * Uses a sorted set in Redis where each member is a request timestamp.
 * On each request:
 *   1. Remove all entries older than (now - windowMs)
 *   2. Count remaining entries
 *   3. If count < max, add current timestamp and allow
 *   4. If count >= max, reject and return retry time
 *
 * This is exact (not approximate) because we store individual timestamps.
 * Memory cost: O(requests_per_window) per unique identifier.
 *
 * For 1000 req/min limit with 100k users: ~100M entries worst case.
 * Each sorted set entry ≈ 50 bytes → ~5GB. Monitor your Redis memory.
 */

async function slidingWindowCheck(store, key, windowMs, max) {
  return store.slidingWindow(key, windowMs, max);
}

module.exports = { slidingWindowCheck };
```

### `src/algorithms/tokenBucket.js`

```javascript
'use strict';

/**
 * Token bucket algorithm.
 *
 * HOW IT WORKS:
 * Each identifier has a bucket with a maximum capacity of tokens.
 * Tokens refill at a constant rate (refillRate per second).
 * Each request consumes one token.
 * If no tokens are available, the request is rejected.
 *
 * WHEN TO USE THIS OVER SLIDING WINDOW:
 * - You want to allow controlled bursting (e.g., 10 tokens burst, 1/sec refill)
 * - Your clients do batch operations and need burst headroom
 * - You're rate limiting downstream API calls (not inbound requests)
 *
 * PARAMETERS:
 * @param capacity         - Max tokens (burst size)
 * @param refillRate       - Tokens added per second
 *
 * Example: capacity=10, refillRate=1
 *   → Client can burst 10 requests immediately
 *   → Then 1 request per second sustained
 */

async function tokenBucketCheck(store, key, capacity, refillRatePerSecond, windowMs) {
  return store.tokenBucket(key, capacity, refillRatePerSecond, windowMs);
}

module.exports = { tokenBucketCheck };
```

### `src/utils/headers.js`

```javascript
'use strict';

/**
 * Rate limit response headers.
 *
 * We implement both the legacy de-facto standard and the new IETF draft:
 * https://datatracker.ietf.org/doc/draft-ietf-httpapi-ratelimit-headers/
 *
 * Legacy (widely supported by clients):
 *   X-RateLimit-Limit     - Max requests allowed in window
 *   X-RateLimit-Remaining - Requests remaining in current window
 *   X-RateLimit-Reset     - Unix timestamp when window resets
 *   Retry-After           - Seconds to wait before retrying (on 429)
 *
 * IETF Draft (forward-compatible):
 *   RateLimit-Limit       - Quota policy
 *   RateLimit-Remaining   - Remaining quota
 *   RateLimit-Reset       - Seconds until quota resets
 */

function setRateLimitHeaders(res, { limit, remaining, windowMs, retryAfterMs, limited }) {
  const resetTimestamp = Math.ceil((Date.now() + windowMs) / 1000);
  const resetSeconds   = Math.ceil(windowMs / 1000);

  // Legacy headers
  res.setHeader('X-RateLimit-Limit',     limit);
  res.setHeader('X-RateLimit-Remaining', Math.max(0, remaining));
  res.setHeader('X-RateLimit-Reset',     resetTimestamp);

  // IETF draft headers
  res.setHeader('RateLimit-Limit',     limit);
  res.setHeader('RateLimit-Remaining', Math.max(0, remaining));
  res.setHeader('RateLimit-Reset',     resetSeconds);

  if (limited && retryAfterMs > 0) {
    res.setHeader('Retry-After', Math.ceil(retryAfterMs / 1000));
  }
}

module.exports = { setRateLimitHeaders };
```

### `src/middleware/rateLimiter.js`

```javascript
'use strict';

const { extractIdentifier }    = require('../utils/identifierExtractor');
const { setRateLimitHeaders }  = require('../utils/headers');
const { slidingWindowCheck }   = require('../algorithms/slidingWindowCounter');
const { tokenBucketCheck }     = require('../algorithms/tokenBucket');
const { MemoryStore }          = require('../stores/memoryStore');
const limits                   = require('../config/limits');

/**
 * Core rate limiter middleware factory.
 *
 * @param {object} primaryStore  - Redis store (preferred)
 * @param {object} options
 * @param {boolean} options.failOpen  - Allow requests when store is down (default: true)
 *                                      Set to false for security-critical endpoints
 * @param {function} options.onLimited - Called when a request is rate limited (for metrics)
 * @param {function} options.onError   - Called when the store errors (for alerting)
 */
function createRateLimiter(primaryStore, options = {}) {
  const fallbackStore = new MemoryStore();
  const failOpen      = options.failOpen !== false; // Default: fail open
  const onLimited     = options.onLimited || (() => {});
  const onError       = options.onError   || ((err) => console.error('[RateLimiter] Store error:', err));

  return async function rateLimiterMiddleware(req, res, next) {
    // Resolve which config applies to this route
    const config = resolveConfig(req);

    // Extract identifier
    const { id: identifier, type: identifierType } = extractIdentifier(req);

    // Apply tier overrides if auth middleware has set req.user.tier
    const effectiveConfig = applyTierOverride(config, req.user?.tier);

    // Build Redis key: prefix:identifier
    const key = `${effectiveConfig.keyPrefix}:${identifier}`;

    // Choose store — fall back to memory if Redis is unhealthy
    const store = primaryStore?.isHealthy() ? primaryStore : fallbackStore;
    const usingFallback = store === fallbackStore;

    if (usingFallback) {
      // This should trigger an alert — see operational gaps section
      onError(new Error('Redis unavailable, using in-process fallback store'));
    }

    let result;
    try {
      result = await checkLimit(store, key, effectiveConfig);
    } catch (err) {
      onError(err);

      if (failOpen) {
        // Store is broken and we're configured to fail open — allow the request
        // Log this — every allowed request during an outage is a potential abuse window
        console.warn('[RateLimiter] Store error, failing open for:', identifier);
        return next();
      } else {
        // Fail closed — reject with 503, not 429 (this is our problem, not theirs)
        return res.status(503).json({
          error: 'Service temporarily unavailable',
          code:  'RATE_LIMITER_UNAVAILABLE',
        });
      }
    }

    // Always set headers — clients need this even on allowed requests
    setRateLimitHeaders(res, {
      limit:        effectiveConfig.max,
      remaining:    result.remaining,
      windowMs:     effectiveConfig.windowMs,
      retryAfterMs: result.retryAfterMs,
      limited:      result.limited,
    });

    if (result.limited) {
      // Emit metrics event — caller should hook this to their metrics system
      onLimited({
        identifier,
        identifierType,
        route:     req.path,
        method:    req.method,
        config:    effectiveConfig,
        timestamp: Date.now(),
      });

      return res.status(429).json({
        error:      effectiveConfig.message || 'Rate limit exceeded',
        code:       'RATE_LIMIT_EXCEEDED',
        retryAfter: Math.ceil((result.retryAfterMs || 0) / 1000),
      });
    }

    next();
  };
}

/**
 * Dispatch to the correct algorithm based on config.
 */
async function checkLimit(store, key, config) {
  if (config.algorithm === 'tokenBucket') {
    const capacity   = config.burst || config.max;
    const refillRate = config.max / (config.windowMs / 1000); // tokens per second
    return tokenBucketCheck(store, key, capacity, refillRate, config.windowMs);
  }

  // Default: sliding window
  return slidingWindowCheck(store, key, config.windowMs, config.max);
}

/**
 * Find the first route config that matches the request path.
 * Falls back to default config.
 */
function resolveConfig(req) {
  for (const routeConfig of limits.routes) {
    if (routeConfig.pattern.test(req.path)) {
      return routeConfig;
    }
  }
  return limits.default;
}

/**
 * Apply per-tier overrides on top of the resolved route config.
 * Tier config only overrides max and windowMs — not algorithm or prefix.
 */
function applyTierOverride(config, tier) {
  if (!tier || !limits.tiers[tier]) {
    return config;
  }

  return {
    ...config,
    ...limits.tiers[tier],
  };
}

module.exports = { createRateLimiter };
```
