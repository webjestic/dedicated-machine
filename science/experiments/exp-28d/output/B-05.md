# Production Rate Limiter in Node.js

Let me design this completely before writing a single line of code.

---

## Phase 1: Requirements Design

### What algorithm?

| Algorithm | Pros | Cons | Best For |
|-----------|------|------|----------|
| Fixed Window | Simple, fast | Boundary burst (2x traffic at window edges) | Internal tools |
| Sliding Window Log | Precise | Memory-heavy (stores every request timestamp) | Low-volume, high-precision |
| Sliding Window Counter | Precise + memory efficient | Slight approximation | **General APIs** |
| Token Bucket | Smooth bursting | State complexity | Streaming, SDKs |
| Leaky Bucket | Enforces steady rate | Drops valid bursts | Queue-based systems |

**Decision: Sliding Window Counter** — handles the boundary burst problem of fixed windows while remaining memory-efficient. The approximation error is bounded and acceptable for API rate limiting.

### What are we limiting by?

- Primary: IP address
- Secondary: API key (authenticated routes)
- Design must support composite keys (IP + route, API key + endpoint)

### What store?

- **Redis** — atomic operations via Lua scripts, TTL-native, horizontally scalable
- Must handle Redis failure gracefully (fail-open vs fail-closed — caller decides)

### What headers do we return?

Following the emerging [RateLimit header standard (RFC 6585 + draft-ietf-httpapi-ratelimit-headers)](https://datatracker.ietf.org/doc/draft-ietf-httpapi-ratelimit-headers/):

```
RateLimit-Limit: 100
RateLimit-Remaining: 45
RateLimit-Reset: 1693000060
Retry-After: 30          (only on 429)
X-RateLimit-Policy: sliding-window
```

### Behavioral requirements

- [ ] Per-route limit overrides (auth routes get stricter limits)
- [ ] Allowlist bypass (health checks, internal services)
- [ ] Denylist (immediately block known bad actors)
- [ ] Dry-run mode (count but never block — for testing new limits)
- [ ] Configurable fail behavior when Redis is down
- [ ] Request cost weights (some endpoints cost more than 1 token)

## Phase 2: Architecture

```
Incoming Request
      │
      ▼
┌─────────────────┐
│  Allowlist Check │──── PASS ──────────────────────────────► next()
└─────────────────┘
      │ not allowlisted
      ▼
┌─────────────────┐
│  Denylist Check  │──── MATCH ──────────────────────────► 403 Forbidden
└─────────────────┘
      │ not denylisted
      ▼
┌─────────────────┐
│   Key Builder    │  (ip, apiKey, route, composite)
└─────────────────┘
      │
      ▼
┌─────────────────────────────────┐
│   Redis Sliding Window Counter  │
│   (atomic Lua script)           │
│                                 │
│  current_window_count +         │
│  (prev_window_count *           │
│   overlap_ratio)                │
└─────────────────────────────────┘
      │
      ├── under limit ──────────────────────────────────► next() + headers
      │
      └── over limit ───────────────────────────────────► 429 + headers
                                                           + Retry-After
```

### Sliding Window Counter Math

```
weight = (window_size - (now % window_size)) / window_size
estimated_count = current_window_count + (previous_window_count * weight)

if estimated_count >= limit → REJECT
```

### `src/utils/keyBuilder.js`

```javascript
'use strict';

/**
 * Builds Redis keys for rate limiting.
 *
 * Key format: rl:{strategy}:{identifier}:{windowTimestamp}
 *
 * We store TWO windows at all times:
 *   - current window
 *   - previous window (for sliding calculation)
 */

const KEY_PREFIX = 'rl';

/**
 * Extract IP from request, respecting common proxy headers.
 * SECURITY NOTE: Only trust X-Forwarded-For if you control
 * the proxy layer. In production, configure `trust proxy` in Express
 * or validate the header source.
 */
function extractIp(req) {
  // Express sets this when `app.set('trust proxy', true)`
  const forwarded = req.ip;
  if (forwarded) return forwarded;

  // Fallback chain
  const xForwardedFor = req.headers['x-forwarded-for'];
  if (xForwardedFor) {
    // x-forwarded-for can be a comma-separated list; take the first
    return xForwardedFor.split(',')[0].trim();
  }

  return req.socket?.remoteAddress ?? 'unknown';
}

/**
 * Extract API key from request.
 * Checks Authorization header (Bearer token) and X-API-Key header.
 */
function extractApiKey(req) {
  const authHeader = req.headers['authorization'];
  if (authHeader?.startsWith('Bearer ')) {
    return authHeader.slice(7).trim();
  }

  const apiKeyHeader = req.headers['x-api-key'];
  if (apiKeyHeader) return apiKeyHeader.trim();

  return null;
}

/**
 * Compute the window timestamp bucket.
 * Returns the start-of-window Unix timestamp in seconds.
 */
function getWindowTimestamp(windowMs) {
  const windowSec = Math.floor(windowMs / 1000);
  return Math.floor(Date.now() / 1000 / windowSec) * windowSec;
}

/**
 * Build the Redis key for a given identifier and window.
 */
function buildKey(strategy, identifier, windowTimestamp) {
  return `${KEY_PREFIX}:${strategy}:${identifier}:${windowTimestamp}`;
}

/**
 * Sanitize an identifier to prevent key injection.
 * Strips characters that could interfere with Redis key parsing.
 */
function sanitize(value) {
  if (typeof value !== 'string') return 'unknown';
  // Allow alphanumeric, dots, dashes, underscores, colons (for IPv6)
  return value.replace(/[^a-zA-Z0-9.\-_:]/g, '_').slice(0, 256);
}

/**
 * Main key resolution function.
 * Returns { identifier, currentKey, previousKey, windowTimestamp }
 */
function resolveKeys(req, options) {
  const { keyStrategy, windowMs } = options;
  const windowSec = Math.floor(windowMs / 1000);
  const currentWindowTs = getWindowTimestamp(windowMs);
  const previousWindowTs = currentWindowTs - windowSec;

  let identifier;
  let strategy;

  if (typeof keyStrategy === 'function') {
    // Custom key function — caller is responsible for sanitization
    identifier = sanitize(keyStrategy(req));
    strategy = 'custom';
  } else if (keyStrategy === 'apiKey') {
    const apiKey = extractApiKey(req);
    identifier = sanitize(apiKey ?? extractIp(req)); // fallback to IP
    strategy = 'apiKey';
  } else if (keyStrategy === 'composite') {
    const ip = sanitize(extractIp(req));
    const route = sanitize(req.route?.path ?? req.path ?? 'unknown');
    identifier = `${ip}:${route}`;
    strategy = 'composite';
  } else {
    // Default: IP
    identifier = sanitize(extractIp(req));
    strategy = 'ip';
  }

  return {
    identifier,
    strategy,
    currentKey: buildKey(strategy, identifier, currentWindowTs),
    previousKey: buildKey(strategy, identifier, previousWindowTs),
    windowTimestamp: currentWindowTs,
    windowSec,
  };
}

module.exports = {
  resolveKeys,
  extractIp,
  extractApiKey,
  sanitize,
  getWindowTimestamp,
};
```

### `src/algorithms/slidingWindow.js`

```javascript
'use strict';

/**
 * Sliding Window Counter algorithm implemented as an atomic Lua script.
 *
 * WHY LUA?
 * --------
 * Redis executes Lua scripts atomically. Without this, a race condition
 * exists between GET (read count) and INCR (increment count) — two
 * requests could both read "99", both decide they're under the limit,
 * and both increment to 100, allowing 101 requests through.
 *
 * The Lua script runs as a single atomic unit on the Redis server,
 * eliminating this race condition entirely.
 *
 * ALGORITHM:
 * ----------
 * We maintain two fixed-window counters: current and previous.
 * The sliding window estimate is:
 *
 *   overlap = (window_size - time_elapsed_in_current_window) / window_size
 *   estimate = current_count + (previous_count * overlap)
 *
 * This approximates how many requests from the previous window still
 * "count" against the current sliding window. The maximum error is
 * bounded by the assumption that previous window requests were uniformly
 * distributed — in practice this is accurate enough for API rate limiting.
 *
 * KEYS[1] = current window key
 * KEYS[2] = previous window key
 * ARGV[1] = limit (max requests)
 * ARGV[2] = window size in seconds
 * ARGV[3] = current timestamp in seconds
 * ARGV[4] = cost (how many tokens this request consumes)
 * ARGV[5] = TTL for keys (2x window to keep previous window alive)
 */
const SLIDING_WINDOW_SCRIPT = `
local current_key  = KEYS[1]
local previous_key = KEYS[2]
local limit        = tonumber(ARGV[1])
local window_sec   = tonumber(ARGV[2])
local now          = tonumber(ARGV[3])
local cost         = tonumber(ARGV[4])
local ttl          = tonumber(ARGV[5])

-- How far into the current window are we? (0.0 to 1.0)
local current_window_start = math.floor(now / window_sec) * window_sec
local elapsed_in_window    = now - current_window_start
local overlap_ratio        = 1.0 - (elapsed_in_window / window_sec)

-- Read previous window count (may not exist)
local prev_count = tonumber(redis.call('GET', previous_key)) or 0

-- Sliding window estimate from previous window
local prev_weighted = prev_count * overlap_ratio

-- Read current window count
local curr_count = tonumber(redis.call('GET', current_key)) or 0

-- Would adding this request's cost exceed the limit?
local estimated = prev_weighted + curr_count + cost

if estimated > limit then
  -- REJECTED: return current state without incrementing
  -- Return: [allowed, current_count, prev_count, reset_time]
  return {0, curr_count, prev_count, current_window_start + window_sec}
end

-- ALLOWED: increment current window counter
local new_count = redis.call('INCRBY', current_key, cost)

-- Set TTL on first write (keep key alive for 2 windows)
-- We use 2x TTL so the previous window key stays readable
if new_count == cost then
  redis.call('EXPIRE', current_key, ttl)
end

-- Return: [allowed, new_count, prev_count, reset_time]
return {1, new_count, prev_count, current_window_start + window_sec}
`;

/**
 * Execute the sliding window check against Redis.
 *
 * @param {Object} redisClient - ioredis client instance
 * @param {string} currentKey  - Redis key for current window
 * @param {string} previousKey - Redis key for previous window
 * @param {Object} options     - { limit, windowMs, cost }
 * @returns {Promise<Object>}  - { allowed, count, prevCount, resetAt, remaining }
 */
async function checkSlidingWindow(redisClient, currentKey, previousKey, options) {
  const { limit, windowMs, cost = 1 } = options;
  const windowSec = Math.floor(windowMs / 1000);
  const nowSec = Math.floor(Date.now() / 1000);
  const ttl = windowSec * 2; // Keep previous window key alive

  const result = await redisClient.eval(
    SLIDING_WINDOW_SCRIPT,
    2,              // number of KEYS
    currentKey,
    previousKey,
    limit,          // ARGV[1]
    windowSec,      // ARGV[2]
    nowSec,         // ARGV[3]
    cost,           // ARGV[4]
    ttl,            // ARGV[5]
  );

  const [allowed, currentCount, prevCount, resetAt] = result;

  // Calculate remaining using the same sliding window math
  const currentWindowStart = Math.floor(nowSec / windowSec) * windowSec;
  const elapsedInWindow = nowSec - currentWindowStart;
  const overlapRatio = 1.0 - (elapsedInWindow / windowSec);
  const estimated = currentCount + (prevCount * overlapRatio);
  const remaining = Math.max(0, limit - Math.ceil(estimated));

  return {
    allowed: allowed === 1,
    count: currentCount,
    prevCount,
    resetAt,          // Unix timestamp (seconds) when current window resets
    remaining,
    limit,
  };
}

module.exports = { checkSlidingWindow, SLIDING_WINDOW_SCRIPT };
```

### `src/store/redisStore.js`

```javascript
'use strict';

const Redis = require('ioredis');

/**
 * Redis store wrapper.
 *
 * Handles:
 * - Connection lifecycle
 * - Health state tracking
 * - Error isolation (Redis errors must not crash the app)
 * - Reconnection (delegated to ioredis built-in retry)
 */
class RedisStore {
  constructor(redisOptions = {}, logger = null) {
    this._logger = logger;
    this._healthy = false;
    this._connectionAttempts = 0;

    // ioredis configuration with production-safe defaults
    const config = {
      // Connection
      host: redisOptions.host ?? '127.0.0.1',
      port: redisOptions.port ?? 6379,
      password: redisOptions.password,
      db: redisOptions.db ?? 0,
      tls: redisOptions.tls,

      // Timeouts — critical for rate limiter latency
      connectTimeout: redisOptions.connectTimeout ?? 2000,  // 2s to connect
      commandTimeout: redisOptions.commandTimeout ?? 500,   // 500ms per command

      // Retry strategy — exponential backoff capped at 10s
      retryStrategy: (times) => {
        this._connectionAttempts = times;
        if (times > 20) {
          this._log('error', 'Redis: max reconnection attempts reached');
          return null; // Stop retrying
        }
        const delay = Math.min(100 * Math.pow(2, times), 10_000);
        this._log('warn', `Redis: reconnecting in ${delay}ms (attempt ${times})`);
        return delay;
      },

      // Don't buffer commands when disconnected — fail fast
      enableOfflineQueue: redisOptions.enableOfflineQueue ?? false,

      // Lazy connect — don't connect until first command
      lazyConnect: true,

      ...redisOptions,
    };

    this.client = new Redis(config);

    this._bindEvents();
  }

  _bindEvents() {
    this.client.on('connect', () => {
      this._healthy = true;
      this._log('info', 'Redis: connected');
    });

    this.client.on('ready', () => {
      this._healthy = true;
      this._log('info', 'Redis: ready');
    });

    this.client.on('error', (err) => {
      this._healthy = false;
      // Log but DO NOT throw — Redis errors must be isolated
      this._log('error', `Redis error: ${err.message}`, { error: err });
    });

    this.client.on('close', () => {
      this._healthy = false;
      this._log('warn', 'Redis: connection closed');
    });

    this.client.on('reconnecting', () => {
      this._log('info', 'Redis: reconnecting...');
    });

    this.client.on('end', () => {
      this._healthy = false;
      this._log('warn', 'Redis: connection ended');
    });
  }

  async connect() {
    try {
      await this.client.connect();
    } catch (err) {
      this._log('error', `Redis: initial connection failed: ${err.message}`);
      // Don't throw — let the health state handle fail-open/closed behavior
    }
  }

  isHealthy() {
    return this._healthy && this.client.status === 'ready';
  }

  /**
   * Execute the rate limit Lua script.
   * Wraps eval with error isolation.
   */
  async eval(...args) {
    return this.client.eval(...args);
  }

  async disconnect() {
    await this.client.quit();
  }

  _log(level, message, meta = {}) {
    if (this._logger) {
      this._logger[level]?.(message, meta);
    }
  }
}

module.exports = RedisStore;
```

```javascript
'use strict';

/**
 * Set rate limit headers on the response.
 * Follows draft-ietf-httpapi-ratelimit-headers spec.
 */
function setRateLimitHeaders(res, result, options) {
  const { headers } = options;

  res.setHeader(headers.limit,     result.limit);
  res.setHeader(headers.remaining, result.remaining);
  res.setHeader(headers.reset,     result.resetAt);
  res.setHeader(headers.policy,    'sliding-window');
}

/**
 * Set Retry-After header on 429 responses.
 * Value is seconds until the window resets.
 */
function setRetryAfterHeader(res, result, options) {
  const { headers } = options;
  const nowSec = Math.floor(Date.now() / 1000);
  const retryAfter = Math.max(0, result.resetAt - nowSec);
  res.setHeader(headers.retryAfter, retryAfter);
}

module.exports = { setRateLimitHeaders, setRetryAfterHeader };
```

### `src/utils/logger.js`

```javascript
'use strict';

/**
 * Minimal structured logger adapter.
 *
 * Wraps whatever logger the caller provides (winston, pino, console)
 * into a consistent interface. The rate limiter itself only calls
 * logger.info(), logger.warn(), logger.error().
 *
 * In production, inject pino or winston. In tests, inject a spy.
 */
function createLogger(logger) {
  if (!logger) {
    return null; // Silent mode
  }

  // If it already has the right shape, use it directly
  if (typeof logger.info === 'function') {
    return logger;
  }

  // Wrap console as a fallback
  if (logger === 'console') {
    return {
      info:  (msg, meta) => console.log(JSON.stringify({ level: 'info',  msg, ...meta })),
      warn:  (msg, meta) => console.warn(JSON.stringify({ level: 'warn',  msg, ...meta })),
      error: (msg, meta) => console.error(JSON.stringify({ level: 'error', msg, ...meta })),
    };
  }

  return null;
}

module.exports = { createLogger };
```

### `src/middleware/rateLimiter.js`

```javascript
'use strict';

const { resolveKeys, extractIp, extractApiKey } = require('../utils/keyBuilder');
const { checkSlidingWindow }                     = require('../algorithms/slidingWindow');
const { setRateLimitHeaders, setRetryAfterHeader } = require('../utils/headers');

/**
 * Core rate limiter middleware factory.
 *
 * @param {Object} store   - RedisStore instance
 * @param {Object} options - Merged configuration (defaults + user options)
 * @returns {Function}     - Express middleware
 */
function createRateLimiterMiddleware(store, options) {
  const {
    limit,
    windowMs,
    cost,
    failMode,
    dryRun,
    allowlist,
    denylist,
    statusCode,
    message,
    logger,
  } = options;

  // Pre-compute sets for O(1) lookup
  const allowSet = new Set(allowlist);
  const denySet  = new Set(denylist);

  return async function rateLimiterMiddleware(req, res, next) {
    try {
      // ── 1. Extract identifiers for list checks ──────────────────────────
      const ip     = extractIp(req);
      const apiKey = extractApiKey(req);

      // ── 2. Denylist check ───────────────────────────────────────────────
      if (denySet.has(ip) || (apiKey && denySet.has(apiKey))) {
        logger?.warn('Rate limiter: denylist block', { ip, apiKey });
        return res.status(403).json({ error: 'Forbidden' });
      }

      // ── 3. Allowlist check ──────────────────────────────────────────────
      if (allowSet.has(ip) || (apiKey && allowSet.has(apiKey))) {
        logger?.info('Rate limiter: allowlist bypass', { ip, apiKey });
        return next();
      }

      // ── 4. Redis health check ───────────────────────────────────────────
      if (!store.isHealthy()) {
        logger?.warn('Rate limiter: Redis unhealthy', { failMode, ip });

        if (failMode === 'closed') {
          return res.status(503).json({ error: 'Service temporarily unavailable' });
        }
        // failMode === 'open': allow the request through
        return next();
      }

      // ── 5. Build Redis keys ─────────────────────────────────────────────
      const { currentKey, previousKey } = resolveKeys(req, options);

      // ── 6. Execute sliding window check ────────────────────────────────
      const result = await checkSlidingWindow(
        store.client,
        currentKey,
        previousKey,
        { limit, windowMs, cost },
      );

      // ── 7. Set standard rate limit headers (always) ─────────────────────
      setRateLimitHeaders(res, result, options);

      // ── 8. Evaluate decision ────────────────────────────────────────────
      if (!result.allowed) {
        setRetryAfterHeader(res, result, options);

        logger?.warn('Rate limiter: request blocked', {
          ip,
          apiKey,
          currentKey,
          count: result.count,
          limit: result.limit,
        });

        if (dryRun) {
          // Dry run: log the block but let it through
          logger?.info('Rate limiter: dry-run mode, allowing blocked request');
          return next();
        }

        return res.status(statusCode).json({ error: message });
      }

      // ── 9. Allowed ──────────────────────────────────────────────────────
      logger?.info('Rate limiter: request allowed', {
        ip,
        remaining: result.remaining,
      });

      return next();

    } catch (err) {
      // ── 10. Unexpected error handling ───────────────────────────────────
      // Redis eval errors, network errors, etc.
      logger?.error('Rate limiter: unexpected error', { error: err.message, stack: err.stack });

      if (failMode === 'closed') {
        return res.status(503).json({ error: 'Service temporarily unavailable' });
      }

      // failMode === 'open': fail gracefully
      return next();
    }
  };
}

module.exports = { createRateLimiterMiddleware };
```

### `src/index.js` — Public API

```javascript
'use strict';

const defaults                    = require('./config/defaults');
const RedisStore                  = require('./store/redisStore');
const { createRateLimiterMiddleware } = require('./middleware/rateLimiter');
const { createLogger }            = require('./utils/logger');

/**
 * Rate Limiter Factory
 *
 * Usage:
 *
 *   const { createRateLimiter } = require('./src');
 *
 *   const limiter = await createRateLimiter({
 *     windowMs: 60_000,
 *     limit: 100,
 *     redis: { host: 'localhost', port: 6379 },
 *     logger: pinoInstance,
 *   });
 *
 *   // Global
 *   app.use(limiter());
 *
 *   // Per-route override
 *   app.post('/auth/login', limiter({ limit: 5, windowMs: 60_000 }), handler);
 *
 * @param {Object} globalOptions - Global configuration
 * @returns {Function}           - Middleware factory function
 */
async function createRateLimiter(globalOptions = {}) {
  const { redis: redisOptions = {}, ...rest } = globalOptions;

  // Merge with defaults
  const resolvedOptions = {
    ...defaults,
    ...rest,
    headers: { ...defaults.headers, ...rest.headers },
    logger: createLogger(rest.logger ?? null),
  };

  // Initialize Redis store
  const store = new RedisStore(redisOptions, resolvedOptions.logger);
  await store.connect();

  /**
   * Returns an Express middleware.
   * Call with no args to use global options.
   * Call with options to override for a specific route.
   *
   * @param {Object} routeOptions - Per-route overrides
   * @returns {Function} Express middleware
   */
  function limiter(routeOptions = {}) {
    const mergedOptions = {
      ...resolvedOptions,
      ...routeOptions,
      headers: { ...resolvedOptions.headers, ...routeOptions.headers },
    };

    return createRateLimiterMiddleware(store, mergedOptions);
  }

  // Expose store for health checks and graceful shutdown
  limiter.store = store;
  limiter.disconnect = () => store.disconnect();

  return limiter;
}

module.exports = { createRateLimiter };
```
