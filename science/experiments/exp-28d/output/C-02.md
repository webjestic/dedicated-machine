## Strategy 1: Fixed Window — `src/strategies/fixedWindow.js`

```javascript
/**
 * Fixed Window Strategy
 * Counts requests in fixed time windows (e.g., 0-60s, 60-120s).
 * Simple but can allow burst traffic at window boundaries.
 */
class FixedWindow {
  constructor() {
    // store: { key -> { count, windowStart } }
    this.store = new Map();
  }

  /**
   * Check if a request is allowed.
   * @param {string} key       - Unique identifier (e.g., IP address)
   * @param {number} limit     - Max requests per window
   * @param {number} windowMs  - Window size in milliseconds
   * @returns {{ allowed: boolean, remaining: number, resetTime: number }}
   */
  check(key, limit, windowMs) {
    const now = Date.now();
    const record = this.store.get(key);

    // Start a new window if no record exists or the window has expired
    if (!record || now - record.windowStart >= windowMs) {
      this.store.set(key, { count: 1, windowStart: now });
      return {
        allowed: true,
        remaining: limit - 1,
        resetTime: now + windowMs,
      };
    }

    // Within the current window
    if (record.count < limit) {
      record.count++;
      return {
        allowed: true,
        remaining: limit - record.count,
        resetTime: record.windowStart + windowMs,
      };
    }

    // Limit exceeded
    return {
      allowed: false,
      remaining: 0,
      resetTime: record.windowStart + windowMs,
    };
  }

  reset(key) {
    this.store.delete(key);
  }

  cleanup() {
    this.store.clear();
  }
}

module.exports = FixedWindow;
```

## Strategy 2: Sliding Window — `src/strategies/slidingWindow.js`

```javascript
/**
 * Sliding Window Strategy
 * Tracks individual request timestamps within a rolling time window.
 * More accurate than Fixed Window — prevents boundary bursts.
 */
class SlidingWindow {
  constructor() {
    // store: { key -> number[] } (array of timestamps)
    this.store = new Map();
  }

  /**
   * Check if a request is allowed.
   * @param {string} key       - Unique identifier
   * @param {number} limit     - Max requests per window
   * @param {number} windowMs  - Window size in milliseconds
   * @returns {{ allowed: boolean, remaining: number, resetTime: number }}
   */
  check(key, limit, windowMs) {
    const now = Date.now();
    const windowStart = now - windowMs;

    // Get existing timestamps and filter out expired ones
    const timestamps = (this.store.get(key) || []).filter(
      (ts) => ts > windowStart
    );

    const resetTime = timestamps.length > 0 ? timestamps[0] + windowMs : now + windowMs;

    if (timestamps.length < limit) {
      timestamps.push(now);
      this.store.set(key, timestamps);
      return {
        allowed: true,
        remaining: limit - timestamps.length,
        resetTime,
      };
    }

    // Limit exceeded
    return {
      allowed: false,
      remaining: 0,
      resetTime,
    };
  }

  reset(key) {
    this.store.delete(key);
  }

  cleanup() {
    this.store.clear();
  }
}

module.exports = SlidingWindow;
```

## Strategy 3: Token Bucket — `src/strategies/tokenBucket.js`

```javascript
/**
 * Token Bucket Strategy
 * Tokens are added at a fixed rate up to a maximum capacity.
 * Each request consumes one token. Allows controlled bursting.
 */
class TokenBucket {
  constructor() {
    // store: { key -> { tokens, lastRefill } }
    this.store = new Map();
  }

  /**
   * Check if a request is allowed.
   * @param {string} key          - Unique identifier
   * @param {number} capacity     - Max tokens (burst limit)
   * @param {number} refillRate   - Tokens added per second
   * @returns {{ allowed: boolean, remaining: number, resetTime: number }}
   */
  check(key, capacity, refillRate) {
    const now = Date.now();
    let bucket = this.store.get(key);

    if (!bucket) {
      // New bucket starts full
      bucket = { tokens: capacity, lastRefill: now };
      this.store.set(key, bucket);
    }

    // Refill tokens based on elapsed time
    const elapsed = (now - bucket.lastRefill) / 1000; // seconds
    const tokensToAdd = elapsed * refillRate;

    bucket.tokens = Math.min(capacity, bucket.tokens + tokensToAdd);
    bucket.lastRefill = now;

    const timeUntilNextToken = bucket.tokens < 1
      ? Math.ceil((1 - bucket.tokens) / refillRate * 1000)
      : 0;

    if (bucket.tokens >= 1) {
      bucket.tokens -= 1;
      return {
        allowed: true,
        remaining: Math.floor(bucket.tokens),
        resetTime: now + timeUntilNextToken,
      };
    }

    // No tokens available
    return {
      allowed: false,
      remaining: 0,
      resetTime: now + timeUntilNextToken,
    };
  }

  reset(key) {
    this.store.delete(key);
  }

  cleanup() {
    this.store.clear();
  }
}

module.exports = TokenBucket;
```

## Core Rate Limiter — `src/rateLimiter.js`

```javascript
const FixedWindow = require('./strategies/fixedWindow');
const SlidingWindow = require('./strategies/slidingWindow');
const TokenBucket = require('./strategies/tokenBucket');

const STRATEGIES = {
  fixedWindow: 'fixedWindow',
  slidingWindow: 'slidingWindow',
  tokenBucket: 'tokenBucket',
};

class RateLimiter {
  /**
   * @param {object} options
   * @param {string}   options.strategy     - 'fixedWindow' | 'slidingWindow' | 'tokenBucket'
   * @param {number}   options.limit        - Max requests (or token capacity)
   * @param {number}   options.windowMs     - Window in ms (fixedWindow / slidingWindow)
   * @param {number}   options.refillRate   - Tokens/sec (tokenBucket only)
   * @param {function} options.keyGenerator - (req) => string  [optional]
   * @param {function} options.onLimitReached - (req, res) => void  [optional]
   */
  constructor(options = {}) {
    this.strategy = options.strategy || STRATEGIES.slidingWindow;
    this.limit = options.limit || 100;
    this.windowMs = options.windowMs || 60 * 1000; // 1 minute default
    this.refillRate = options.refillRate || 10;
    this.keyGenerator = options.keyGenerator || this._defaultKeyGenerator;
    this.onLimitReached = options.onLimitReached || this._defaultOnLimitReached;

    this.handler = this._createHandler();

    // Periodically clean up stale entries to prevent memory leaks
    this._cleanupInterval = setInterval(
      () => this.handler.cleanup(),
      this.windowMs * 2
    );

    // Allow the interval to be garbage collected if the process exits
    if (this._cleanupInterval.unref) {
      this._cleanupInterval.unref();
    }
  }

  _createHandler() {
    switch (this.strategy) {
      case STRATEGIES.fixedWindow:
        return new FixedWindow();
      case STRATEGIES.slidingWindow:
        return new SlidingWindow();
      case STRATEGIES.tokenBucket:
        return new TokenBucket();
      default:
        throw new Error(`Unknown strategy: "${this.strategy}"`);
    }
  }

  _defaultKeyGenerator(req) {
    // Use X-Forwarded-For for clients behind a proxy, fallback to remoteAddress
    return (
      req.headers['x-forwarded-for']?.split(',')[0].trim() ||
      req.socket?.remoteAddress ||
      'unknown'
    );
  }

  _defaultOnLimitReached(req, res) {
    res.status(429).json({
      error: 'Too Many Requests',
      message: 'You have exceeded the rate limit. Please try again later.',
    });
  }

  /**
   * Returns an Express middleware function.
   */
  middleware() {
    return (req, res, next) => {
      const key = this.keyGenerator(req);
      let result;

      if (this.strategy === STRATEGIES.tokenBucket) {
        result = this.handler.check(key, this.limit, this.refillRate);
      } else {
        result = this.handler.check(key, this.limit, this.windowMs);
      }

      // Set standard rate limit headers
      res.setHeader('X-RateLimit-Limit', this.limit);
      res.setHeader('X-RateLimit-Remaining', result.remaining);
      res.setHeader('X-RateLimit-Reset', Math.ceil(result.resetTime / 1000)); // Unix timestamp

      if (!result.allowed) {
        res.setHeader('Retry-After', Math.ceil((result.resetTime - Date.now()) / 1000));
        return this.onLimitReached(req, res);
      }

      next();
    };
  }

  /**
   * Manually reset the limit for a specific key.
   */
  reset(key) {
    this.handler.reset(key);
  }

  /**
   * Stop the cleanup interval.
   */
  destroy() {
    clearInterval(this._cleanupInterval);
  }
}

module.exports = { RateLimiter, STRATEGIES };
```

## Express Middleware Helper — `src/middleware.js`

```javascript
const { RateLimiter } = require('./rateLimiter');

/**
 * Convenience factory for common rate limiter configurations.
 */

const createApiLimiter = () =>
  new RateLimiter({
    strategy: 'slidingWindow',
    limit: 100,
    windowMs: 60 * 1000, // 100 req/min
  }).middleware();

const createAuthLimiter = () =>
  new RateLimiter({
    strategy: 'fixedWindow',
    limit: 5,
    windowMs: 15 * 60 * 1000, // 5 attempts per 15 min
    onLimitReached: (req, res) => {
      res.status(429).json({
        error: 'Too Many Requests',
        message: 'Too many login attempts. Please try again in 15 minutes.',
      });
    },
  }).middleware();

const createBurstLimiter = () =>
  new RateLimiter({
    strategy: 'tokenBucket',
    limit: 20,       // burst capacity
    refillRate: 5,   // 5 tokens/sec steady state
  }).middleware();

module.exports = { createApiLimiter, createAuthLimiter, createBurstLimiter };
```

## Demo Server — `server.js`

```javascript
const express = require('express');
const { RateLimiter } = require('./src/rateLimiter');
const { createApiLimiter, createAuthLimiter, createBurstLimiter } = require('./src/middleware');

const app = express();
app.use(express.json());

// ── Global limiter (all routes) ──────────────────────────────────────────────
const globalLimiter = new RateLimiter({
  strategy: 'slidingWindow',
  limit: 200,
  windowMs: 60 * 1000,
});
app.use(globalLimiter.middleware());

// ── Public API ───────────────────────────────────────────────────────────────
app.get('/api/data', createApiLimiter(), (req, res) => {
  res.json({ message: 'Here is your data!', timestamp: Date.now() });
});

// ── Auth route (strict limiter) ───────────────────────────────────────────────
app.post('/auth/login', createAuthLimiter(), (req, res) => {
  res.json({ message: 'Login successful (demo)' });
});

// ── Burst-tolerant route ─────────────────────────────────────────────────────
app.get('/api/stream', createBurstLimiter(), (req, res) => {
  res.json({ message: 'Streaming data...', timestamp: Date.now() });
});

// ── Per-user limiter (custom key) ─────────────────────────────────────────────
const userLimiter = new RateLimiter({
  strategy: 'tokenBucket',
  limit: 10,
  refillRate: 2,
  keyGenerator: (req) => req.headers['x-user-id'] || 'anonymous',
}).middleware();

app.get('/api/user/resource', userLimiter, (req, res) => {
  res.json({ message: 'User resource accessed' });
});

// ── Health check (no rate limit) ─────────────────────────────────────────────
app.get('/health', (req, res) => res.json({ status: 'ok' }));

// ── Error handler ─────────────────────────────────────────────────────────────
app.use((err, req, res, next) => {
  console.error(err.stack);
  res.status(500).json({ error: 'Internal Server Error' });
});

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => {
  console.log(`Server running on http://localhost:${PORT}`);
  console.log('\nEndpoints:');
  console.log('  GET  /api/data        — Sliding window (100 req/min)');
  console.log('  POST /auth/login      — Fixed window   (5 req/15min)');
  console.log('  GET  /api/stream      — Token bucket   (burst: 20, rate: 5/s)');
  console.log('  GET  /api/user/resource — Token bucket per user-id');
  console.log('  GET  /health          — No rate limit');
});

module.exports = app;
```

## Tests — `tests/rateLimiter.test.js`

```javascript
const request = require('supertest');
const app = require('../server');
const { RateLimiter } = require('../src/rateLimiter');
const FixedWindow = require('../src/strategies/fixedWindow');
const SlidingWindow = require('../src/strategies/slidingWindow');
const TokenBucket = require('../src/strategies/tokenBucket');

// ── Unit Tests ────────────────────────────────────────────────────────────────

describe('FixedWindow', () => {
  let fw;
  beforeEach(() => (fw = new FixedWindow()));

  test('allows requests within limit', () => {
    const r1 = fw.check('ip1', 3, 1000);
    const r2 = fw.check('ip1', 3, 1000);
    const r3 = fw.check('ip1', 3, 1000);
    expect(r1.allowed).toBe(true);
    expect(r2.allowed).toBe(true);
    expect(r3.allowed).toBe(true);
    expect(r3.remaining).toBe(0);
  });

  test('blocks requests exceeding limit', () => {
    fw.check('ip1', 2, 1000);
    fw.check('ip1', 2, 1000);
    const r = fw.check('ip1', 2, 1000);
    expect(r.allowed).toBe(false);
    expect(r.remaining).toBe(0);
  });

  test('resets after window expires', async () => {
    fw.check('ip1', 1, 50);
    fw.check('ip1', 1, 50); // blocked
    await new Promise((r) => setTimeout(r, 60));
    const r = fw.check('ip1', 1, 50);
    expect(r.allowed).toBe(true);
  });

  test('tracks different keys independently', () => {
    fw.check('ip1', 1, 1000);
    const r = fw.check('ip2', 1, 1000);
    expect(r.allowed).toBe(true);
  });
});

describe('SlidingWindow', () => {
  let sw;
  beforeEach(() => (sw = new SlidingWindow()));

  test('allows requests within limit', () => {
    expect(sw.check('ip1', 3, 1000).allowed).toBe(true);
    expect(sw.check('ip1', 3, 1000).allowed).toBe(true);
    expect(sw.check('ip1', 3, 1000).allowed).toBe(true);
  });

  test('blocks requests exceeding limit', () => {
    sw.check('ip1', 2, 1000);
    sw.check('ip1', 2, 1000);
    expect(sw.check('ip1', 2, 1000).allowed).toBe(false);
  });

  test('slides window correctly', async () => {
    sw.check('ip1', 2, 100);
    sw.check('ip1', 2, 100);
    expect(sw.check('ip1', 2, 100).allowed).toBe(false);
    await new Promise((r) => setTimeout(r, 110));
    expect(sw.check('ip1', 2, 100).allowed).toBe(true);
  });
});

describe('TokenBucket', () => {
  let tb;
  beforeEach(() => (tb = new TokenBucket()));

  test('starts with full bucket', () => {
    const r = tb.check('ip1', 5, 1);
    expect(r.allowed).toBe(true);
    expect(r.remaining).toBe(4);
  });

  test('blocks when bucket is empty', () => {
    for (let i = 0; i < 5; i++) tb.check('ip1', 5, 1);
    expect(tb.check('ip1', 5, 1).allowed).toBe(false);
  });

  test('refills tokens over time', async () => {
    tb.check('ip1', 1, 10); // drain
    expect(tb.check('ip1', 1, 10).allowed).toBe(false);
    await new Promise((r) => setTimeout(r, 150)); // wait for refill
    expect(tb.check('ip1', 1, 10).allowed).toBe(true);
  });
});

// ── Integration Tests ─────────────────────────────────────────────────────────

describe('RateLimiter middleware', () => {
  test('sets rate limit headers', async () => {
    const res = await request(app).get('/api/data');
    expect(res.headers['x-ratelimit-limit']).toBeDefined();
    expect(res.headers['x-ratelimit-remaining']).toBeDefined();
    expect(res.headers['x-ratelimit-reset']).toBeDefined();
  });

  test('returns 429 when limit exceeded', async () => {
    const limiter = new RateLimiter({ strategy: 'fixedWindow', limit: 2, windowMs: 5000 });
    const express = require('express');
    const testApp = express();
    testApp.use(limiter.middleware());
    testApp.get('/test', (req, res) => res.json({ ok: true }));

    await request(testApp).get('/test');
    await request(testApp).get('/test');
    const res = await request(testApp).get('/test');

    expect(res.status).toBe(429);
    expect(res.body.error).toBe('Too Many Requests');
    expect(res.headers['retry-after']).toBeDefined();
    limiter.destroy();
  });

  test('health check is not blocked', async () => {
    const res = await request(app).get('/health');
    expect(res.status).toBe(200);
  });
});
```

## Strategy Comparison

| Strategy | Accuracy | Memory | Burst Handling | Best For |
|---|---|---|---|---|
| **Fixed Window** | Low | Low | Poor (boundary bursts) | Simple counters |
| **Sliding Window** | High | Medium | Good | API rate limiting |
| **Token Bucket** | High | Low | Excellent | Streaming / bursty traffic |
