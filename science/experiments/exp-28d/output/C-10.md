## 1. Memory Store

```javascript
// src/store/memoryStore.js

class MemoryStore {
  constructor() {
    this.store = new Map();
    this.cleanupInterval = setInterval(() => this.cleanup(), 60_000);
    this.cleanupInterval.unref(); // Don't block process exit
  }

  async get(key) {
    const entry = this.store.get(key);
    if (!entry) return null;
    if (Date.now() > entry.expiresAt) {
      this.store.delete(key);
      return null;
    }
    return entry.value;
  }

  async set(key, value, ttlMs) {
    this.store.set(key, {
      value,
      expiresAt: Date.now() + ttlMs,
    });
  }

  async increment(key, ttlMs) {
    const entry = this.store.get(key);
    const now = Date.now();

    if (!entry || now > entry.expiresAt) {
      this.store.set(key, { value: 1, expiresAt: now + ttlMs });
      return 1;
    }

    entry.value += 1;
    return entry.value;
  }

  async delete(key) {
    this.store.delete(key);
  }

  cleanup() {
    const now = Date.now();
    for (const [key, entry] of this.store.entries()) {
      if (now > entry.expiresAt) {
        this.store.delete(key);
      }
    }
  }

  destroy() {
    clearInterval(this.cleanupInterval);
    this.store.clear();
  }
}

module.exports = MemoryStore;
```

```javascript
class RedisStore {
  /**
   * @param {import('ioredis').Redis} client - ioredis client instance
   */
  constructor(client) {
    if (!client) throw new Error('Redis client is required');
    this.client = client;
  }

  async get(key) {
    const value = await this.client.get(key);
    return value ? JSON.parse(value) : null;
  }

  async set(key, value, ttlMs) {
    await this.client.set(key, JSON.stringify(value), 'PX', ttlMs);
  }

  async increment(key, ttlMs) {
    // Atomic increment using Lua script
    const script = `
      local current = redis.call('GET', KEYS[1])
      if current == false then
        redis.call('SET', KEYS[1], 1, 'PX', ARGV[1])
        return 1
      else
        return redis.call('INCR', KEYS[1])
      end
    `;
    return await this.client.eval(script, 1, key, ttlMs);
  }

  async delete(key) {
    await this.client.del(key);
  }
}

module.exports = RedisStore;
```

## 3. Fixed Window Strategy

```javascript
// src/strategies/fixedWindow.js

class FixedWindowStrategy {
  /**
   * @param {object} options
   * @param {number} options.windowMs    - Window size in milliseconds
   * @param {number} options.maxRequests - Max requests per window
   * @param {object} options.store       - Store instance
   */
  constructor({ windowMs, maxRequests, store }) {
    this.windowMs = windowMs;
    this.maxRequests = maxRequests;
    this.store = store;
  }

  _getWindowKey(key) {
    const windowId = Math.floor(Date.now() / this.windowMs);
    return `fw:${key}:${windowId}`;
  }

  async consume(key) {
    const windowKey = this._getWindowKey(key);
    const count = await this.store.increment(windowKey, this.windowMs);

    const remaining = Math.max(0, this.maxRequests - count);
    const windowStart = Math.floor(Date.now() / this.windowMs) * this.windowMs;
    const resetTime = new Date(windowStart + this.windowMs);

    return {
      allowed: count <= this.maxRequests,
      count,
      remaining,
      resetTime,
      limit: this.maxRequests,
    };
  }

  async reset(key) {
    const windowKey = this._getWindowKey(key);
    await this.store.delete(windowKey);
  }
}

module.exports = FixedWindowStrategy;
```

## 4. Sliding Window Strategy

```javascript
// src/strategies/slidingWindow.js

class SlidingWindowStrategy {
  /**
   * Uses a sliding window log to track requests with timestamps.
   * More accurate than fixed window but uses more memory.
   *
   * @param {object} options
   * @param {number} options.windowMs    - Window size in milliseconds
   * @param {number} options.maxRequests - Max requests per window
   * @param {object} options.store       - Store instance
   */
  constructor({ windowMs, maxRequests, store }) {
    this.windowMs = windowMs;
    this.maxRequests = maxRequests;
    this.store = store;
  }

  async consume(key) {
    const storeKey = `sw:${key}`;
    const now = Date.now();
    const windowStart = now - this.windowMs;

    // Get existing timestamps
    let timestamps = (await this.store.get(storeKey)) || [];

    // Filter out timestamps outside the window
    timestamps = timestamps.filter((ts) => ts > windowStart);

    // Add current request
    timestamps.push(now);

    // Save back with TTL equal to window size
    await this.store.set(storeKey, timestamps, this.windowMs);

    const count = timestamps.length;
    const remaining = Math.max(0, this.maxRequests - count);
    const oldestTimestamp = timestamps[0] || now;
    const resetTime = new Date(oldestTimestamp + this.windowMs);

    return {
      allowed: count <= this.maxRequests,
      count,
      remaining,
      resetTime,
      limit: this.maxRequests,
    };
  }

  async reset(key) {
    await this.store.delete(`sw:${key}`);
  }
}

module.exports = SlidingWindowStrategy;
```

## 5. Token Bucket Strategy

```javascript
// src/strategies/tokenBucket.js

class TokenBucketStrategy {
  /**
   * Tokens are added at a fixed rate. Each request consumes one token.
   * Allows bursting up to the bucket capacity.
   *
   * @param {object} options
   * @param {number} options.capacity       - Max tokens in the bucket
   * @param {number} options.refillRate     - Tokens added per second
   * @param {object} options.store          - Store instance
   */
  constructor({ capacity, refillRate, store }) {
    this.capacity = capacity;
    this.refillRate = refillRate; // tokens per second
    this.store = store;
  }

  async consume(key) {
    const storeKey = `tb:${key}`;
    const now = Date.now();

    let bucket = await this.store.get(storeKey);

    if (!bucket) {
      bucket = { tokens: this.capacity, lastRefill: now };
    }

    // Calculate tokens to add based on elapsed time
    const elapsed = (now - bucket.lastRefill) / 1000; // seconds
    const tokensToAdd = elapsed * this.refillRate;
    bucket.tokens = Math.min(this.capacity, bucket.tokens + tokensToAdd);
    bucket.lastRefill = now;

    const allowed = bucket.tokens >= 1;

    if (allowed) {
      bucket.tokens -= 1;
    }

    // TTL: time for a full refill from empty
    const ttlMs = (this.capacity / this.refillRate) * 1000 * 2;
    await this.store.set(storeKey, bucket, ttlMs);

    const resetTime = new Date(
      now + ((1 - bucket.tokens) / this.refillRate) * 1000
    );

    return {
      allowed,
      tokens: Math.floor(bucket.tokens),
      remaining: Math.floor(bucket.tokens),
      resetTime,
      limit: this.capacity,
    };
  }

  async reset(key) {
    await this.store.delete(`tb:${key}`);
  }
}

module.exports = TokenBucketStrategy;
```

## 6. Core Rate Limiter

```javascript
// src/rateLimiter.js

const MemoryStore = require('./store/memoryStore');
const FixedWindowStrategy = require('./strategies/fixedWindow');
const SlidingWindowStrategy = require('./strategies/slidingWindow');
const TokenBucketStrategy = require('./strategies/tokenBucket');

const STRATEGIES = {
  fixedWindow: FixedWindowStrategy,
  slidingWindow: SlidingWindowStrategy,
  tokenBucket: TokenBucketStrategy,
};

class RateLimiter {
  /**
   * @param {object}   options
   * @param {string}   [options.strategy='fixedWindow'] - 'fixedWindow' | 'slidingWindow' | 'tokenBucket'
   * @param {number}   [options.windowMs=60000]         - Window in ms (fixedWindow / slidingWindow)
   * @param {number}   [options.maxRequests=100]        - Max requests per window
   * @param {number}   [options.capacity=10]            - Bucket capacity (tokenBucket)
   * @param {number}   [options.refillRate=1]           - Tokens/sec (tokenBucket)
   * @param {object}   [options.store]                  - Custom store (default: MemoryStore)
   * @param {function} [options.keyGenerator]           - Custom key function (req) => string
   * @param {function} [options.onLimitReached]         - Called when limit is hit
   * @param {boolean}  [options.skipFailedRequests]     - Don't count failed responses
   * @param {boolean}  [options.skipSuccessfulRequests] - Don't count successful responses
   */
  constructor(options = {}) {
    const {
      strategy = 'fixedWindow',
      windowMs = 60_000,
      maxRequests = 100,
      capacity = 10,
      refillRate = 1,
      store = new MemoryStore(),
      keyGenerator = (req) => req.ip || req.connection?.remoteAddress || 'global',
      onLimitReached = null,
      skipFailedRequests = false,
      skipSuccessfulRequests = false,
    } = options;

    if (!STRATEGIES[strategy]) {
      throw new Error(`Unknown strategy: "${strategy}". Choose from: ${Object.keys(STRATEGIES).join(', ')}`);
    }

    this.store = store;
    this.keyGenerator = keyGenerator;
    this.onLimitReached = onLimitReached;
    this.skipFailedRequests = skipFailedRequests;
    this.skipSuccessfulRequests = skipSuccessfulRequests;

    const StrategyClass = STRATEGIES[strategy];
    this.strategy = new StrategyClass({
      windowMs,
      maxRequests,
      capacity,
      refillRate,
      store,
    });
  }

  /**
   * Check rate limit for a given key (use outside of HTTP context)
   * @param {string} key
   * @returns {Promise<object>} result
   */
  async check(key) {
    return this.strategy.consume(key);
  }

  /**
   * Reset rate limit for a given key
   * @param {string} key
   */
  async reset(key) {
    return this.strategy.reset(key);
  }
}

module.exports = RateLimiter;
```

## 7. Express Middleware

```javascript
// src/middleware.js

const RateLimiter = require('./rateLimiter');

/**
 * Creates an Express middleware from a RateLimiter instance or options.
 * @param {RateLimiter|object} limiterOrOptions
 */
function createMiddleware(limiterOrOptions = {}) {
  const limiter =
    limiterOrOptions instanceof RateLimiter
      ? limiterOrOptions
      : new RateLimiter(limiterOrOptions);

  return async function rateLimitMiddleware(req, res, next) {
    const key = limiter.keyGenerator(req);

    let result;
    try {
      result = await limiter.strategy.consume(key);
    } catch (err) {
      console.error('[RateLimiter] Store error:', err);
      return next(); // Fail open — don't block on store errors
    }

    // Set standard rate limit headers
    res.setHeader('X-RateLimit-Limit', result.limit);
    res.setHeader('X-RateLimit-Remaining', result.remaining);
    res.setHeader('X-RateLimit-Reset', Math.ceil(result.resetTime.getTime() / 1000));
    res.setHeader('X-RateLimit-Policy', `${result.limit};w=${Math.ceil((result.resetTime - Date.now()) / 1000)}`);

    if (!result.allowed) {
      res.setHeader('Retry-After', Math.ceil((result.resetTime - Date.now()) / 1000));

      if (limiter.onLimitReached) {
        limiter.onLimitReached(req, res, result);
      }

      return res.status(429).json({
        error: 'Too Many Requests',
        message: 'Rate limit exceeded. Please try again later.',
        retryAfter: Math.ceil((result.resetTime - Date.now()) / 1000),
      });
    }

    // Handle skip options by intercepting response
    if (limiter.skipFailedRequests || limiter.skipSuccessfulRequests) {
      const originalEnd = res.end.bind(res);
      res.end = async function (...args) {
        const shouldSkip =
          (limiter.skipFailedRequests && res.statusCode >= 400) ||
          (limiter.skipSuccessfulRequests && res.statusCode < 400);

        if (shouldSkip) {
          await limiter.reset(key);
        }

        return originalEnd(...args);
      };
    }

    next();
  };
}

module.exports = { createMiddleware, RateLimiter };
```

## 8. Express Server Example

```javascript
// server.js

const express = require('express');
const { createMiddleware } = require('./src/middleware');

const app = express();
app.use(express.json());

// ── Global limiter: 100 req/min per IP ──────────────────────────────────────
const globalLimiter = createMiddleware({
  strategy: 'fixedWindow',
  windowMs: 60_000,
  maxRequests: 100,
});

// ── Strict limiter for auth routes: 5 req/15min ─────────────────────────────
const authLimiter = createMiddleware({
  strategy: 'slidingWindow',
  windowMs: 15 * 60_000,
  maxRequests: 5,
  keyGenerator: (req) => `auth:${req.ip}`,
  onLimitReached: (req, res, result) => {
    console.warn(`Auth rate limit hit for IP: ${req.ip}`);
  },
});

// ── API limiter using token bucket (allows bursting) ────────────────────────
const apiLimiter = createMiddleware({
  strategy: 'tokenBucket',
  capacity: 20,    // burst up to 20
  refillRate: 5,   // refill 5 tokens/sec
  keyGenerator: (req) => `api:${req.headers['x-api-key'] || req.ip}`,
});

app.use(globalLimiter);

app.post('/auth/login', authLimiter, (req, res) => {
  res.json({ message: 'Login successful' });
});

app.post('/auth/register', authLimiter, (req, res) => {
  res.json({ message: 'Registration successful' });
});

app.get('/api/data', apiLimiter, (req, res) => {
  res.json({ data: 'Here is your data', timestamp: new Date() });
});

app.get('/health', (req, res) => {
  res.json({ status: 'ok' });
});

// Error handler
app.use((err, req, res, next) => {
  console.error(err);
  res.status(500).json({ error: 'Internal Server Error' });
});

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => console.log(`Server running on port ${PORT}`));

module.exports = app;
```

## Strategy Comparison

| Strategy | Accuracy | Memory | Burst Handling | Best For |
|---|---|---|---|---|
| **Fixed Window** | Medium | Low | Poor (edge spikes) | Simple APIs |
| **Sliding Window** | High | Medium | Good | Auth, sensitive routes |
| **Token Bucket** | High | Low | Excellent | APIs needing burst tolerance |

## Key Features

- **3 strategies** — Fixed Window, Sliding Window, Token Bucket
- **2 stores** — In-memory (default) and Redis (distributed)
- **Standard headers** — `X-RateLimit-*` and `Retry-After`
- **Flexible key generation** — by IP, API key, user ID, or any custom logic
- **Fail open** — store errors don't block requests
- **Skip options** — optionally ignore failed/successful responses
- **Callbacks** — hook into limit-reached events
