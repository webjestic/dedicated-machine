## 1. Memory Store

```javascript
// src/store/memoryStore.js

class MemoryStore {
  constructor() {
    this.store = new Map();
    this.cleanupInterval = setInterval(() => this.cleanup(), 60_000);
    this.cleanupInterval.unref(); // Don't block process exit
  }

  async get(key) {
    const entry = this.store.get(key);
    if (!entry) return null;
    if (Date.now() > entry.expiresAt) {
      this.store.delete(key);
      return null;
    }
    return entry.value;
  }

  async set(key, value, ttlMs) {
    this.store.set(key, {
      value,
      expiresAt: Date.now() + ttlMs,
    });
  }

  async increment(key, ttlMs) {
    const entry = this.store.get(key);
    const now = Date.now();

    if (!entry || now > entry.expiresAt) {
      this.store.set(key, { value: 1, expiresAt: now + ttlMs });
      return 1;
    }

    entry.value += 1;
    return entry.value;
  }

  async delete(key) {
    this.store.delete(key);
  }

  cleanup() {
    const now = Date.now();
    for (const [key, entry] of this.store.entries()) {
      if (now > entry.expiresAt) {
        this.store.delete(key);
      }
    }
  }

  destroy() {
    clearInterval(this.cleanupInterval);
    this.store.clear();
  }
}

module.exports = MemoryStore;
```

// src/store/redisStore.js

class RedisStore {
  /**
   * @param {import('ioredis').Redis} client - ioredis client instance
   */
  constructor(client) {
    if (!client) throw new Error('Redis client is required');
    this.client = client;
  }

  async get(key) {
    const value = await this.client.get(key);
    return value ? JSON.parse(value) : null;
  }

  async set(key, value, ttlMs) {
    await this.client.set(key, JSON.stringify(value), 'PX', ttlMs);
  }

  async increment(key, ttlMs) {
    // Atomic increment using Lua script
    const script = `
      local current = redis.call('GET', KEYS[1])
      if current == false then
        redis.call('SET', KEYS[1], 1, 'PX', ARGV[1])
        return 1
      else
        return redis.call('INCR', KEYS[1])
      end
    `;
    return await this.client.eval(script, 1, key, ttlMs);
  }

  async delete(key) {
    await this.client.del(key);
  }
}

module.exports = RedisStore;

## 3. Fixed Window Strategy

```javascript
// src/strategies/fixedWindow.js

class FixedWindowStrategy {
  /**
   * @param {object} options
   * @param {number} options.windowMs    - Window size in milliseconds
   * @param {number} options.maxRequests - Max requests per window
   */
  constructor({ windowMs, maxRequests }) {
    this.windowMs = windowMs;
    this.maxRequests = maxRequests;
  }

  async check(store, key) {
    const count = await store.increment(key, this.windowMs);
    const remaining = Math.max(0, this.maxRequests - count);
    const allowed = count <= this.maxRequests;

    return {
      allowed,
      count,
      remaining,
      resetAt: new Date(
        Math.ceil(Date.now() / this.windowMs) * this.windowMs
      ),
      limit: this.maxRequests,
    };
  }
}

module.exports = FixedWindowStrategy;
```

## 4. Sliding Window Strategy

```javascript
// src/strategies/slidingWindow.js

class SlidingWindowStrategy {
  /**
   * Uses a sliding log of timestamps for precise rate limiting.
   * @param {object} options
   * @param {number} options.windowMs    - Window size in milliseconds
   * @param {number} options.maxRequests - Max requests per window
   */
  constructor({ windowMs, maxRequests }) {
    this.windowMs = windowMs;
    this.maxRequests = maxRequests;
  }

  async check(store, key) {
    const now = Date.now();
    const windowStart = now - this.windowMs;

    // Get existing timestamps
    let timestamps = (await store.get(key)) || [];

    // Filter out expired timestamps
    timestamps = timestamps.filter((ts) => ts > windowStart);

    const count = timestamps.length;
    const allowed = count < this.maxRequests;

    if (allowed) {
      timestamps.push(now);
      await store.set(key, timestamps, this.windowMs);
    }

    const oldest = timestamps[0] || now;
    const resetAt = new Date(oldest + this.windowMs);

    return {
      allowed,
      count: allowed ? count + 1 : count,
      remaining: Math.max(0, this.maxRequests - timestamps.length),
      resetAt,
      limit: this.maxRequests,
    };
  }
}

module.exports = SlidingWindowStrategy;
```

## 5. Token Bucket Strategy

```javascript
// src/strategies/tokenBucket.js

class TokenBucketStrategy {
  /**
   * @param {object} options
   * @param {number} options.capacity      - Max tokens in bucket
   * @param {number} options.refillRate    - Tokens added per second
   * @param {number} options.tokensPerRequest - Tokens consumed per request (default: 1)
   */
  constructor({ capacity, refillRate, tokensPerRequest = 1 }) {
    this.capacity = capacity;
    this.refillRate = refillRate;
    this.tokensPerRequest = tokensPerRequest;
  }

  async check(store, key) {
    const now = Date.now();
    const bucket = (await store.get(key)) || {
      tokens: this.capacity,
      lastRefill: now,
    };

    // Calculate tokens to add since last refill
    const elapsed = (now - bucket.lastRefill) / 1000; // seconds
    const tokensToAdd = elapsed * this.refillRate;
    bucket.tokens = Math.min(this.capacity, bucket.tokens + tokensToAdd);
    bucket.lastRefill = now;

    const allowed = bucket.tokens >= this.tokensPerRequest;

    if (allowed) {
      bucket.tokens -= this.tokensPerRequest;
    }

    // TTL: time until bucket would be empty if no refill
    const ttlMs = (this.capacity / this.refillRate) * 1000 * 2;
    await store.set(key, bucket, ttlMs);

    const timeToNextToken =
      bucket.tokens < this.tokensPerRequest
        ? ((this.tokensPerRequest - bucket.tokens) / this.refillRate) * 1000
        : 0;

    return {
      allowed,
      tokens: Math.floor(bucket.tokens),
      remaining: Math.floor(bucket.tokens / this.tokensPerRequest),
      resetAt: new Date(now + timeToNextToken),
      limit: this.capacity,
    };
  }
}

module.exports = TokenBucketStrategy;
```

## 6. Core Rate Limiter

```javascript
// src/rateLimiter.js

const MemoryStore = require('./store/memoryStore');
const FixedWindowStrategy = require('./strategies/fixedWindow');
const SlidingWindowStrategy = require('./strategies/slidingWindow');
const TokenBucketStrategy = require('./strategies/tokenBucket');

const STRATEGIES = {
  fixedWindow: FixedWindowStrategy,
  slidingWindow: SlidingWindowStrategy,
  tokenBucket: TokenBucketStrategy,
};

class RateLimiter {
  /**
   * @param {object}   options
   * @param {string}   [options.strategy='fixedWindow'] - Rate limiting strategy
   * @param {object}   [options.store]                  - Storage backend
   * @param {function} [options.keyGenerator]           - Custom key generator
   * @param {number}   [options.windowMs=60000]         - Window in ms
   * @param {number}   [options.maxRequests=100]        - Max requests per window
   * @param {number}   [options.capacity]               - Token bucket capacity
   * @param {number}   [options.refillRate]             - Token bucket refill rate
   * @param {function} [options.onLimitReached]         - Callback when limit hit
   */
  constructor(options = {}) {
    const {
      strategy = 'fixedWindow',
      store,
      keyGenerator,
      onLimitReached,
      ...strategyOptions
    } = options;

    if (!STRATEGIES[strategy]) {
      throw new Error(
        `Unknown strategy: "${strategy}". Choose from: ${Object.keys(STRATEGIES).join(', ')}`
      );
    }

    this.store = store || new MemoryStore();
    this.strategy = new STRATEGIES[strategy](strategyOptions);
    this.keyGenerator = keyGenerator || this._defaultKeyGenerator;
    this.onLimitReached = onLimitReached || null;
  }

  _defaultKeyGenerator(identifier) {
    return `rl:${identifier}`;
  }

  /**
   * Check if a request is allowed.
   * @param {string} identifier - Unique identifier (e.g., IP, user ID)
   * @returns {Promise<RateLimitResult>}
   */
  async check(identifier) {
    const key = this.keyGenerator(identifier);
    const result = await this.strategy.check(this.store, key);

    if (!result.allowed && this.onLimitReached) {
      await this.onLimitReached(identifier, result);
    }

    return result;
  }

  /**
   * Reset the rate limit for an identifier.
   * @param {string} identifier
   */
  async reset(identifier) {
    const key = this.keyGenerator(identifier);
    await this.store.delete(key);
  }

  destroy() {
    if (this.store.destroy) {
      this.store.destroy();
    }
  }
}

module.exports = RateLimiter;
```

## 7. Express Middleware

```javascript
// src/middleware.js

const RateLimiter = require('./rateLimiter');

/**
 * Creates an Express middleware for rate limiting.
 * @param {object} options - RateLimiter options + middleware-specific options
 * @param {function} [options.keyGenerator] - Extract key from req (default: IP)
 * @param {function} [options.handler]      - Custom response handler
 * @param {boolean}  [options.skipFailed]   - Don't count failed requests
 * @param {function} [options.skip]         - Skip rate limiting for certain requests
 */
function createMiddleware(options = {}) {
  const {
    handler,
    skip,
    skipFailed = false,
    keyGenerator,
    ...limiterOptions
  } = options;

  // Default key generator uses IP address
  const resolveKey = keyGenerator || ((req) => req.ip || req.connection.remoteAddress);

  const limiter = new RateLimiter({
    ...limiterOptions,
    keyGenerator: (id) => `rl:${id}`,
  });

  const defaultHandler = (req, res, result) => {
    res.set({
      'X-RateLimit-Limit': result.limit,
      'X-RateLimit-Remaining': 0,
      'X-RateLimit-Reset': Math.ceil(result.resetAt.getTime() / 1000),
      'Retry-After': Math.ceil((result.resetAt - Date.now()) / 1000),
    });

    res.status(429).json({
      error: 'Too Many Requests',
      message: 'Rate limit exceeded. Please try again later.',
      retryAfter: Math.ceil((result.resetAt - Date.now()) / 1000),
    });
  };

  const responseHandler = handler || defaultHandler;

  return async (req, res, next) => {
    try {
      // Allow skipping for certain requests (e.g., health checks)
      if (skip && (await skip(req))) {
        return next();
      }

      const identifier = resolveKey(req);
      const result = await limiter.check(identifier);

      // Set rate limit headers on all responses
      res.set({
        'X-RateLimit-Limit': result.limit,
        'X-RateLimit-Remaining': result.remaining,
        'X-RateLimit-Reset': Math.ceil(result.resetAt.getTime() / 1000),
      });

      if (!result.allowed) {
        return responseHandler(req, res, result);
      }

      // Optionally don't count failed requests
      if (skipFailed) {
        res.on('finish', async () => {
          if (res.statusCode >= 400) {
            await limiter.reset(identifier);
          }
        });
      }

      next();
    } catch (err) {
      // Fail open - don't block requests if rate limiter errors
      console.error('[RateLimiter] Error:', err.message);
      next();
    }
  };
}

module.exports = { createMiddleware };
```

## 8. Express Server Example

```javascript
// server.js

const express = require('express');
const { createMiddleware } = require('./src/middleware');

const app = express();
app.use(express.json());

// ── Global limiter: 100 req/min per IP ──────────────────────────────────────
const globalLimiter = createMiddleware({
  strategy: 'fixedWindow',
  windowMs: 60_000,
  maxRequests: 100,
  skip: (req) => req.path === '/health',
});

// ── Strict limiter for auth routes: 10 req/15min ────────────────────────────
const authLimiter = createMiddleware({
  strategy: 'slidingWindow',
  windowMs: 15 * 60_000,
  maxRequests: 10,
  handler: (req, res, result) => {
    res.status(429).json({
      error: 'Too many login attempts',
      retryAfter: Math.ceil((result.resetAt - Date.now()) / 1000),
    });
  },
});

// ── Token bucket for API: burst-friendly ────────────────────────────────────
const apiLimiter = createMiddleware({
  strategy: 'tokenBucket',
  capacity: 20,
  refillRate: 5, // 5 tokens/sec
  keyGenerator: (req) => req.headers['x-api-key'] || req.ip,
});

// ── Routes ───────────────────────────────────────────────────────────────────
app.get('/health', (req, res) => res.json({ status: 'ok' }));

app.use('/api', globalLimiter, apiLimiter);
app.get('/api/data', (req, res) => res.json({ data: 'Here is your data!' }));

app.use('/auth', authLimiter);
app.post('/auth/login', (req, res) => res.json({ token: 'mock-jwt-token' }));

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => console.log(`Server running on port ${PORT}`));
```

## Strategy Comparison

| Strategy | Best For | Pros | Cons |
|---|---|---|---|
| **Fixed Window** | Simple APIs | Fast, low memory | Burst at window edges |
| **Sliding Window** | Precise limiting | No edge bursts | Higher memory usage |
| **Token Bucket** | Bursty traffic | Allows bursts, smooth | More complex state |
