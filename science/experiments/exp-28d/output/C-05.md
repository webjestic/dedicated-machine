## 1. Memory Store

```javascript
// src/stores/memoryStore.js

class MemoryStore {
  constructor() {
    this.store = new Map();
    this.cleanupInterval = setInterval(() => this.cleanup(), 60_000);
    this.cleanupInterval.unref(); // Don't block process exit
  }

  async get(key) {
    const entry = this.store.get(key);
    if (!entry) return null;
    if (Date.now() > entry.expiresAt) {
      this.store.delete(key);
      return null;
    }
    return entry.value;
  }

  async set(key, value, ttlMs) {
    this.store.set(key, {
      value,
      expiresAt: Date.now() + ttlMs,
    });
  }

  async increment(key, ttlMs) {
    const entry = this.store.get(key);
    const now = Date.now();

    if (!entry || now > entry.expiresAt) {
      this.store.set(key, { value: 1, expiresAt: now + ttlMs });
      return 1;
    }

    entry.value += 1;
    return entry.value;
  }

  async delete(key) {
    this.store.delete(key);
  }

  cleanup() {
    const now = Date.now();
    for (const [key, entry] of this.store.entries()) {
      if (now > entry.expiresAt) {
        this.store.delete(key);
      }
    }
  }

  destroy() {
    clearInterval(this.cleanupInterval);
    this.store.clear();
  }
}

module.exports = MemoryStore;
```

```javascript
class RedisStore {
  /**
   * @param {import('ioredis').Redis} client - ioredis client instance
   */
  constructor(client) {
    if (!client) throw new Error('Redis client is required');
    this.client = client;
  }

  async get(key) {
    const value = await this.client.get(key);
    return value ? JSON.parse(value) : null;
  }

  async set(key, value, ttlMs) {
    await this.client.set(key, JSON.stringify(value), 'PX', ttlMs);
  }

  async increment(key, ttlMs) {
    // Atomic increment using Lua script
    const script = `
      local current = redis.call('incr', KEYS[1])
      if current == 1 then
        redis.call('pexpire', KEYS[1], ARGV[1])
      end
      return current
    `;
    return await this.client.eval(script, 1, key, ttlMs);
  }

  async delete(key) {
    await this.client.del(key);
  }
}

module.exports = RedisStore;
```

## 3. Fixed Window Strategy

```javascript
// src/strategies/fixedWindow.js

class FixedWindowStrategy {
  /**
   * @param {object} options
   * @param {number} options.windowMs     - Window size in milliseconds
   * @param {number} options.maxRequests  - Max requests per window
   * @param {object} options.store        - Storage backend
   */
  constructor({ windowMs, maxRequests, store }) {
    this.windowMs = windowMs;
    this.maxRequests = maxRequests;
    this.store = store;
  }

  _getWindowKey(key) {
    const windowId = Math.floor(Date.now() / this.windowMs);
    return `fw:${key}:${windowId}`;
  }

  async consume(key) {
    const windowKey = this._getWindowKey(key);
    const count = await this.store.increment(windowKey, this.windowMs);

    const remaining = Math.max(0, this.maxRequests - count);
    const resetAt = (Math.floor(Date.now() / this.windowMs) + 1) * this.windowMs;

    return {
      allowed: count <= this.maxRequests,
      count,
      remaining,
      resetAt,
      retryAfter: count > this.maxRequests ? Math.ceil((resetAt - Date.now()) / 1000) : 0,
    };
  }

  async reset(key) {
    const windowKey = this._getWindowKey(key);
    await this.store.delete(windowKey);
  }
}

module.exports = FixedWindowStrategy;
```

## 4. Sliding Window Strategy

```javascript
// src/strategies/slidingWindow.js

class SlidingWindowStrategy {
  /**
   * Uses a sliding log of timestamps for precise rate limiting.
   * More accurate than fixed window but uses more memory.
   *
   * @param {object} options
   * @param {number} options.windowMs     - Window size in milliseconds
   * @param {number} options.maxRequests  - Max requests per window
   * @param {object} options.store        - Storage backend
   */
  constructor({ windowMs, maxRequests, store }) {
    this.windowMs = windowMs;
    this.maxRequests = maxRequests;
    this.store = store;
  }

  async consume(key) {
    const storeKey = `sw:${key}`;
    const now = Date.now();
    const windowStart = now - this.windowMs;

    // Get existing timestamps
    let timestamps = (await this.store.get(storeKey)) || [];

    // Remove expired timestamps outside the window
    timestamps = timestamps.filter((ts) => ts > windowStart);

    const count = timestamps.length;
    const allowed = count < this.maxRequests;

    if (allowed) {
      timestamps.push(now);
      await this.store.set(storeKey, timestamps, this.windowMs);
    }

    const oldest = timestamps[0] || now;
    const resetAt = oldest + this.windowMs;

    return {
      allowed,
      count: allowed ? count + 1 : count,
      remaining: Math.max(0, this.maxRequests - timestamps.length),
      resetAt,
      retryAfter: !allowed ? Math.ceil((resetAt - now) / 1000) : 0,
    };
  }

  async reset(key) {
    await this.store.delete(`sw:${key}`);
  }
}

module.exports = SlidingWindowStrategy;
```

## 5. Token Bucket Strategy

```javascript
// src/strategies/tokenBucket.js

class TokenBucketStrategy {
  /**
   * Allows bursting up to `capacity` requests, refilling at `refillRate` tokens/sec.
   *
   * @param {object} options
   * @param {number} options.capacity    - Max tokens (burst size)
   * @param {number} options.refillRate  - Tokens added per second
   * @param {object} options.store       - Storage backend
   */
  constructor({ capacity, refillRate, store }) {
    this.capacity = capacity;
    this.refillRate = refillRate; // tokens per second
    this.store = store;
  }

  async consume(key, tokensRequested = 1) {
    const storeKey = `tb:${key}`;
    const now = Date.now();

    let bucket = await this.store.get(storeKey);

    if (!bucket) {
      bucket = { tokens: this.capacity, lastRefill: now };
    }

    // Refill tokens based on elapsed time
    const elapsed = (now - bucket.lastRefill) / 1000; // seconds
    const tokensToAdd = elapsed * this.refillRate;
    bucket.tokens = Math.min(this.capacity, bucket.tokens + tokensToAdd);
    bucket.lastRefill = now;

    const allowed = bucket.tokens >= tokensRequested;

    if (allowed) {
      bucket.tokens -= tokensRequested;
    }

    // TTL: time to fully refill the bucket
    const ttlMs = Math.ceil((this.capacity / this.refillRate) * 1000) + 1000;
    await this.store.set(storeKey, bucket, ttlMs);

    const timeToRefill = allowed
      ? 0
      : Math.ceil((tokensRequested - bucket.tokens) / this.refillRate);

    return {
      allowed,
      tokens: Math.floor(bucket.tokens),
      remaining: Math.floor(bucket.tokens),
      resetAt: now + timeToRefill * 1000,
      retryAfter: allowed ? 0 : timeToRefill,
    };
  }

  async reset(key) {
    await this.store.delete(`tb:${key}`);
  }
}

module.exports = TokenBucketStrategy;
```

## 6. Core Rate Limiter

```javascript
// src/rateLimiter.js

const MemoryStore = require('./stores/memoryStore');
const FixedWindowStrategy = require('./strategies/fixedWindow');
const SlidingWindowStrategy = require('./strategies/slidingWindow');
const TokenBucketStrategy = require('./strategies/tokenBucket');

const STRATEGIES = {
  fixedWindow: FixedWindowStrategy,
  slidingWindow: SlidingWindowStrategy,
  tokenBucket: TokenBucketStrategy,
};

class RateLimiter {
  /**
   * @param {object}   options
   * @param {string}   [options.strategy='fixedWindow']  - 'fixedWindow' | 'slidingWindow' | 'tokenBucket'
   * @param {number}   [options.windowMs=60000]          - Window duration in ms
   * @param {number}   [options.maxRequests=100]         - Max requests per window
   * @param {number}   [options.capacity=10]             - Token bucket capacity
   * @param {number}   [options.refillRate=1]            - Token bucket refill rate (tokens/sec)
   * @param {object}   [options.store]                   - Custom store (defaults to MemoryStore)
   * @param {Function} [options.keyGenerator]            - Custom key generator fn(req) => string
   * @param {Function} [options.onLimitReached]          - Callback when limit is hit
   * @param {boolean}  [options.skipFailedRequests]      - Don't count failed responses
   * @param {boolean}  [options.skipSuccessfulRequests]  - Don't count successful responses
   */
  constructor(options = {}) {
    const {
      strategy = 'fixedWindow',
      windowMs = 60_000,
      maxRequests = 100,
      capacity = 10,
      refillRate = 1,
      store = new MemoryStore(),
      keyGenerator = (req) => req.ip || req.connection?.remoteAddress || 'global',
      onLimitReached = null,
      skipFailedRequests = false,
      skipSuccessfulRequests = false,
    } = options;

    if (!STRATEGIES[strategy]) {
      throw new Error(`Unknown strategy: "${strategy}". Choose from: ${Object.keys(STRATEGIES).join(', ')}`);
    }

    this.keyGenerator = keyGenerator;
    this.onLimitReached = onLimitReached;
    this.skipFailedRequests = skipFailedRequests;
    this.skipSuccessfulRequests = skipSuccessfulRequests;

    const StrategyClass = STRATEGIES[strategy];

    this.strategy =
      strategy === 'tokenBucket'
        ? new StrategyClass({ capacity, refillRate, store })
        : new StrategyClass({ windowMs, maxRequests, store });
  }

  async check(key) {
    return this.strategy.consume(key);
  }

  async reset(key) {
    return this.strategy.reset(key);
  }
}

module.exports = RateLimiter;
```

## 7. Express Middleware

```javascript
// src/middleware.js

const RateLimiter = require('./rateLimiter');

/**
 * Creates an Express rate-limiting middleware.
 * @param {object} options - RateLimiter options + middleware-specific options
 * @param {string} [options.message]      - Custom error message
 * @param {number} [options.statusCode]   - HTTP status code (default: 429)
 * @param {boolean} [options.headers]     - Send rate limit headers (default: true)
 */
function createRateLimiter(options = {}) {
  const {
    message = 'Too many requests, please try again later.',
    statusCode = 429,
    headers = true,
    ...limiterOptions
  } = options;

  const limiter = new RateLimiter(limiterOptions);

  return async function rateLimitMiddleware(req, res, next) {
    try {
      const key = limiter.keyGenerator(req);
      const result = await limiter.check(key);

      if (headers) {
        res.setHeader('X-RateLimit-Limit', limiterOptions.maxRequests || limiterOptions.capacity || 100);
        res.setHeader('X-RateLimit-Remaining', result.remaining);
        res.setHeader('X-RateLimit-Reset', Math.ceil(result.resetAt / 1000));

        if (!result.allowed) {
          res.setHeader('Retry-After', result.retryAfter);
        }
      }

      if (!result.allowed) {
        if (typeof limiter.onLimitReached === 'function') {
          limiter.onLimitReached(req, res, result);
        }

        return res.status(statusCode).json({
          error: message,
          retryAfter: result.retryAfter,
        });
      }

      // Support skipping based on response status
      if (limiter.skipFailedRequests || limiter.skipSuccessfulRequests) {
        const originalEnd = res.end.bind(res);
        res.end = async (...args) => {
          const isSuccess = res.statusCode < 400;
          if (
            (limiter.skipSuccessfulRequests && isSuccess) ||
            (limiter.skipFailedRequests && !isSuccess)
          ) {
            await limiter.reset(key);
          }
          originalEnd(...args);
        };
      }

      next();
    } catch (err) {
      // Fail open — don't block requests if rate limiter errors
      console.error('[RateLimiter] Error:', err.message);
      next();
    }
  };
}

module.exports = { createRateLimiter, RateLimiter };
```

## 8. Express Server Example

```javascript
// server.js

const express = require('express');
const { createRateLimiter } = require('./src/middleware');

const app = express();
app.use(express.json());

// ── Global limiter (fixed window) ──────────────────────────────────────────
const globalLimiter = createRateLimiter({
  strategy: 'fixedWindow',
  windowMs: 60_000,   // 1 minute
  maxRequests: 100,
  message: 'Global rate limit exceeded.',
});

// ── Strict limiter for auth routes (sliding window) ────────────────────────
const authLimiter = createRateLimiter({
  strategy: 'slidingWindow',
  windowMs: 15 * 60_000, // 15 minutes
  maxRequests: 10,
  message: 'Too many login attempts. Please try again later.',
  keyGenerator: (req) => `auth:${req.ip}`,
});

// ── API limiter with token bucket (allows bursting) ────────────────────────
const apiLimiter = createRateLimiter({
  strategy: 'tokenBucket',
  capacity: 20,      // burst up to 20 requests
  refillRate: 5,     // refill 5 tokens/sec
  keyGenerator: (req) => `api:${req.headers['x-api-key'] || req.ip}`,
  onLimitReached: (req, res, result) => {
    console.warn(`[RateLimit] Key exhausted. Retry after ${result.retryAfter}s`);
  },
});

app.use(globalLimiter);

app.post('/auth/login', authLimiter, (req, res) => {
  res.json({ message: 'Login successful' });
});

app.get('/api/data', apiLimiter, (req, res) => {
  res.json({ data: 'Here is your data', timestamp: Date.now() });
});

app.get('/health', (req, res) => {
  res.json({ status: 'ok' });
});

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => console.log(`Server running on port ${PORT}`));

module.exports = app;
```

## 9. Tests

```javascript
// tests/rateLimiter.test.js

const MemoryStore = require('../src/stores/memoryStore');
const FixedWindowStrategy = require('../src/strategies/fixedWindow');
const SlidingWindowStrategy = require('../src/strategies/slidingWindow');
const TokenBucketStrategy = require('../src/strategies/tokenBucket');

// ── Helpers ────────────────────────────────────────────────────────────────

function makeStore() {
  return new MemoryStore();
}

async function exhaust(strategy, key, times) {
  const results = [];
  for (let i = 0; i < times; i++) {
    results.push(await strategy.consume(key));
  }
  return results;
}

// ── Fixed Window ───────────────────────────────────────────────────────────

async function testFixedWindow() {
  console.log('\n── Fixed Window ──────────────────────────────────────────');
  const store = makeStore();
  const strategy = new FixedWindowStrategy({ windowMs: 5000, maxRequests: 3, store });

  const results = await exhaust(strategy, 'user1', 5);

  results.forEach((r, i) => {
    const status = r.allowed ? '✅ allowed' : '❌ blocked';
    console.log(`  Request ${i + 1}: ${status} | count=${r.count} remaining=${r.remaining}`);
  });

  console.assert(results[0].allowed === true,  'Request 1 should be allowed');
  console.assert(results[2].allowed === true,  'Request 3 should be allowed');
  console.assert(results[3].allowed === false, 'Request 4 should be blocked');
  console.assert(results[4].allowed === false, 'Request 5 should be blocked');
  console.log('  ✅ Fixed Window tests passed');

  store.destroy();
}

// ── Sliding Window ─────────────────────────────────────────────────────────

async function testSlidingWindow() {
  console.log('\n── Sliding Window ────────────────────────────────────────');
  const store = makeStore();
  const strategy = new SlidingWindowStrategy({ windowMs: 5000, maxRequests: 3, store });

  const results = await exhaust(strategy, 'user2', 5);

  results.forEach((r, i) => {
    const status = r.allowed ? '✅ allowed' : '❌ blocked';
    console.log(`  Request ${i + 1}: ${status} | count=${r.count} remaining=${r.remaining}`);
  });

  console.assert(results[0].allowed === true,  'Request 1 should be allowed');
  console.assert(results[2].allowed === true,  'Request 3 should be allowed');
  console.assert(results[3].allowed === false, 'Request 4 should be blocked');
  console.log('  ✅ Sliding Window tests passed');

  store.destroy();
}

// ── Token Bucket ───────────────────────────────────────────────────────────

async function testTokenBucket() {
  console.log('\n── Token Bucket ──────────────────────────────────────────');
  const store = makeStore();
  const strategy = new TokenBucketStrategy({ capacity: 3, refillRate: 1, store });

  // Burst 3 requests immediately
  const burst = await exhaust(strategy, 'user3', 4);
  burst.forEach((r, i) => {
    const status = r.allowed ? '✅ allowed' : '❌ blocked';
    console.log(`  Burst ${i + 1}: ${status} | tokens=${r.tokens}`);
  });

  console.assert(burst[0].allowed === true,  'Burst 1 should be allowed');
  console.assert(burst[2].allowed === true,  'Burst 3 should be allowed');
  console.assert(burst[3].allowed === false, 'Burst 4 should be blocked (bucket empty)');

  // Wait 2 seconds for refill
  console.log('  ⏳ Waiting 2s for token refill...');
  await new Promise((r) => setTimeout(r, 2000));

  const afterRefill = await strategy.consume('user3');
  console.log(`  After refill: ${afterRefill.allowed ? '✅ allowed' : '❌ blocked'} | tokens=${afterRefill.tokens}`);
  console.assert(afterRefill.allowed === true, 'Should be allowed after refill');

  console.log('  ✅ Token Bucket tests passed');
  store.destroy();
}

// ── Reset ──────────────────────────────────────────────────────────────────

async function testReset() {
  console.log('\n── Reset ─────────────────────────────────────────────────');
  const store = makeStore();
  const strategy = new FixedWindowStrategy({ windowMs: 60_000, maxRequests: 2, store });

  await exhaust(strategy, 'user4', 2);
  const blocked = await strategy.consume('user4');
  console.assert(blocked.allowed === false, 'Should be blocked before reset');
  console.log('  Blocked before reset ✅');

  await strategy.reset('user4');
  const afterReset = await strategy.consume('user4');
  console.assert(afterReset.allowed === true, 'Should be allowed after reset');
  console.log('  Allowed after reset ✅');

  store.destroy();
}

  }
})();
```

## Strategy Comparison

| Strategy | Accuracy | Memory | Burst Handling | Best For |
|---|---|---|---|---|
| **Fixed Window** | ⭐⭐ | Low | ❌ Boundary spikes | Simple APIs |
| **Sliding Window** | ⭐⭐⭐ | Medium | ✅ Smooth | Auth, sensitive routes |
| **Token Bucket** | ⭐⭐⭐ | Low | ✅ Controlled bursts | APIs with variable load |
