## 1. Memory Store

```javascript
// src/store/memoryStore.js

class MemoryStore {
  constructor() {
    this.store = new Map();
    this.cleanupInterval = setInterval(() => this.cleanup(), 60_000);
    // Prevent keeping the process alive
    this.cleanupInterval.unref?.();
  }

  async get(key) {
    const entry = this.store.get(key);
    if (!entry) return null;
    if (Date.now() > entry.expiresAt) {
      this.store.delete(key);
      return null;
    }
    return entry.value;
  }

  async set(key, value, ttlMs) {
    this.store.set(key, {
      value,
      expiresAt: Date.now() + ttlMs,
    });
  }

  async increment(key, ttlMs) {
    const entry = this.store.get(key);
    const now = Date.now();

    if (!entry || now > entry.expiresAt) {
      this.store.set(key, { value: 1, expiresAt: now + ttlMs });
      return 1;
    }

    entry.value += 1;
    return entry.value;
  }

  async delete(key) {
    this.store.delete(key);
  }

  cleanup() {
    const now = Date.now();
    for (const [key, entry] of this.store.entries()) {
      if (now > entry.expiresAt) {
        this.store.delete(key);
      }
    }
  }

  destroy() {
    clearInterval(this.cleanupInterval);
    this.store.clear();
  }
}

module.exports = MemoryStore;
```

```javascript
class RedisStore {
  /**
   * @param {import('ioredis').Redis} client - ioredis client instance
   */
  constructor(client) {
    if (!client) throw new Error('Redis client is required');
    this.client = client;
  }

  async get(key) {
    const value = await this.client.get(key);
    return value ? JSON.parse(value) : null;
  }

  async set(key, value, ttlMs) {
    await this.client.set(key, JSON.stringify(value), 'PX', ttlMs);
  }

  async increment(key, ttlMs) {
    // Atomic increment using a Lua script
    const script = `
      local current = redis.call('GET', KEYS[1])
      if current == false then
        redis.call('SET', KEYS[1], 1, 'PX', ARGV[1])
        return 1
      else
        return redis.call('INCR', KEYS[1])
      end
    `;
    return this.client.eval(script, 1, key, ttlMs);
  }

  async delete(key) {
    await this.client.del(key);
  }
}

module.exports = RedisStore;
```

## 3. Strategies

### Fixed Window

```javascript
// src/strategies/fixedWindow.js

class FixedWindowStrategy {
  /**
   * @param {object} options
   * @param {number} options.windowMs    - Window size in milliseconds
   * @param {number} options.maxRequests - Max requests per window
   * @param {object} options.store       - Store instance
   */
  constructor({ windowMs, maxRequests, store }) {
    this.windowMs = windowMs;
    this.maxRequests = maxRequests;
    this.store = store;
  }

  async check(key) {
    const count = await this.store.increment(key, this.windowMs);
    const remaining = Math.max(0, this.maxRequests - count);
    const allowed = count <= this.maxRequests;

    return {
      allowed,
      count,
      remaining,
      resetAfterMs: this.windowMs,
      retryAfterMs: allowed ? 0 : this.windowMs,
    };
  }

  async reset(key) {
    await this.store.delete(key);
  }
}

module.exports = FixedWindowStrategy;
```

### Sliding Window

```javascript
// src/strategies/slidingWindow.js

class SlidingWindowStrategy {
  /**
   * Uses a log of timestamps to track requests within a rolling window.
   * More accurate than fixed window but uses more memory.
   *
   * @param {object} options
   * @param {number} options.windowMs    - Window size in milliseconds
   * @param {number} options.maxRequests - Max requests per window
   * @param {object} options.store       - Store instance
   */
  constructor({ windowMs, maxRequests, store }) {
    this.windowMs = windowMs;
    this.maxRequests = maxRequests;
    this.store = store;
  }

  async check(key) {
    const now = Date.now();
    const windowStart = now - this.windowMs;

    // Retrieve existing timestamps
    let timestamps = (await this.store.get(key)) || [];

    // Remove timestamps outside the current window
    timestamps = timestamps.filter((ts) => ts > windowStart);

    // Add current request timestamp
    timestamps.push(now);

    // Persist updated timestamps with TTL
    await this.store.set(key, timestamps, this.windowMs);

    const count = timestamps.length;
    const allowed = count <= this.maxRequests;
    const remaining = Math.max(0, this.maxRequests - count);

    // Time until the oldest request in the window expires
    const oldestTs = timestamps[0];
    const retryAfterMs = allowed ? 0 : oldestTs + this.windowMs - now;

    return {
      allowed,
      count,
      remaining,
      resetAfterMs: oldestTs ? oldestTs + this.windowMs - now : this.windowMs,
      retryAfterMs,
    };
  }

  async reset(key) {
    await this.store.delete(key);
  }
}

module.exports = SlidingWindowStrategy;
```

### Token Bucket

```javascript
// src/strategies/tokenBucket.js

class TokenBucketStrategy {
  /**
   * Tokens refill at a constant rate. Allows bursting up to `capacity`.
   *
   * @param {object} options
   * @param {number} options.capacity      - Max tokens (burst limit)
   * @param {number} options.refillRate    - Tokens added per second
   * @param {number} options.tokensPerRequest - Tokens consumed per request (default: 1)
   * @param {object} options.store         - Store instance
   */
  constructor({ capacity, refillRate, tokensPerRequest = 1, store }) {
    this.capacity = capacity;
    this.refillRate = refillRate;         // tokens/second
    this.tokensPerRequest = tokensPerRequest;
    this.store = store;
    this.ttlMs = Math.ceil((capacity / refillRate) * 1000) * 2; // generous TTL
  }

  async check(key) {
    const now = Date.now();
    const bucket = (await this.store.get(key)) || {
      tokens: this.capacity,
      lastRefill: now,
    };

    // Refill tokens based on elapsed time
    const elapsed = (now - bucket.lastRefill) / 1000; // seconds
    bucket.tokens = Math.min(
      this.capacity,
      bucket.tokens + elapsed * this.refillRate
    );
    bucket.lastRefill = now;

    const allowed = bucket.tokens >= this.tokensPerRequest;

    if (allowed) {
      bucket.tokens -= this.tokensPerRequest;
    }

    await this.store.set(key, bucket, this.ttlMs);

    const remaining = Math.floor(bucket.tokens);
    const retryAfterMs = allowed
      ? 0
      : Math.ceil(((this.tokensPerRequest - bucket.tokens) / this.refillRate) * 1000);

    return {
      allowed,
      tokens: bucket.tokens,
      remaining,
      capacity: this.capacity,
      retryAfterMs,
    };
  }

  async reset(key) {
    await this.store.delete(key);
  }
}

module.exports = TokenBucketStrategy;
```

## 4. Core Rate Limiter

```javascript
// src/rateLimiter.js

const MemoryStore = require('./store/memoryStore');
const FixedWindowStrategy = require('./strategies/fixedWindow');
const SlidingWindowStrategy = require('./strategies/slidingWindow');
const TokenBucketStrategy = require('./strategies/tokenBucket');

const STRATEGIES = {
  fixedWindow: FixedWindowStrategy,
  slidingWindow: SlidingWindowStrategy,
  tokenBucket: TokenBucketStrategy,
};

class RateLimiter {
  /**
   * @param {object}   options
   * @param {string}   [options.strategy='fixedWindow'] - 'fixedWindow' | 'slidingWindow' | 'tokenBucket'
   * @param {number}   [options.windowMs=60000]         - Window in ms (fixed/sliding)
   * @param {number}   [options.maxRequests=100]        - Max requests per window
   * @param {number}   [options.capacity=10]            - Bucket capacity (tokenBucket)
   * @param {number}   [options.refillRate=1]           - Tokens/sec (tokenBucket)
   * @param {function} [options.keyGenerator]           - (req) => string
   * @param {object}   [options.store]                  - Custom store
   * @param {function} [options.onLimitReached]         - (req, res, info) => void
   */
  constructor(options = {}) {
    const {
      strategy = 'fixedWindow',
      windowMs = 60_000,
      maxRequests = 100,
      capacity = 10,
      refillRate = 1,
      tokensPerRequest = 1,
      keyGenerator = (req) => req.ip || req.connection?.remoteAddress || 'global',
      store = new MemoryStore(),
      onLimitReached = null,
    } = options;

    if (!STRATEGIES[strategy]) {
      throw new Error(`Unknown strategy: "${strategy}". Choose from: ${Object.keys(STRATEGIES).join(', ')}`);
    }

    this.keyGenerator = keyGenerator;
    this.store = store;
    this.onLimitReached = onLimitReached;

    const strategyOptions =
      strategy === 'tokenBucket'
        ? { capacity, refillRate, tokensPerRequest, store }
        : { windowMs, maxRequests, store };

    this.strategy = new STRATEGIES[strategy](strategyOptions);
  }

  /**
   * Programmatically check a key without going through HTTP middleware.
   * @param {string} key
   * @returns {Promise<object>} result
   */
  async check(key) {
    return this.strategy.check(key);
  }

  /**
   * Reset the limit for a given key.
   * @param {string} key
   */
  async reset(key) {
    return this.strategy.reset(key);
  }

  /**
   * Returns an Express/Connect-compatible middleware function.
   */
  middleware() {
    return async (req, res, next) => {
      const key = this.keyGenerator(req);

      try {
        const result = await this.strategy.check(key);

        // Set standard rate limit headers
        res.setHeader('X-RateLimit-Limit',     result.capacity ?? result.remaining + (result.count ?? 0));
        res.setHeader('X-RateLimit-Remaining', result.remaining ?? 0);

        if (result.retryAfterMs > 0) {
          res.setHeader('Retry-After', Math.ceil(result.retryAfterMs / 1000));
        }

        if (!result.allowed) {
          if (this.onLimitReached) {
            return this.onLimitReached(req, res, result);
          }

          return res.status(429).json({
            error: 'Too Many Requests',
            message: 'Rate limit exceeded. Please try again later.',
            retryAfter: Math.ceil(result.retryAfterMs / 1000),
          });
        }

        next();
      } catch (err) {
        // Fail open — don't block requests if the store is unavailable
        console.error('[RateLimiter] Store error:', err.message);
        next();
      }
    };
  }
}

module.exports = RateLimiter;
```

```javascript
// src/middleware.js

const RateLimiter = require('./rateLimiter');

/**
 * Quick-create middleware with sensible defaults.
 *
 * @example
 * app.use(createRateLimiter({ maxRequests: 50, windowMs: 30_000 }));
 */
function createRateLimiter(options = {}) {
  const limiter = new RateLimiter(options);
  return limiter.middleware();
}

module.exports = { createRateLimiter, RateLimiter };
```

## 6. Express Server Demo

```javascript
// server.js

const express = require('express');
const { createRateLimiter, RateLimiter } = require('./src/middleware');

const app = express();
app.use(express.json());

// ── Global limiter: 100 req / 15 min ──────────────────────────────────────────
app.use(
  createRateLimiter({
    strategy: 'fixedWindow',
    windowMs: 15 * 60 * 1000,
    maxRequests: 100,
  })
);

// ── Strict API limiter (sliding window): 20 req / min ─────────────────────────
const apiLimiter = createRateLimiter({
  strategy: 'slidingWindow',
  windowMs: 60_000,
  maxRequests: 20,
  keyGenerator: (req) => `api:${req.ip}`,
});

// ── Login limiter (token bucket): burst of 5, refill 1/sec ───────────────────
const loginLimiter = createRateLimiter({
  strategy: 'tokenBucket',
  capacity: 5,
  refillRate: 1,
  keyGenerator: (req) => `login:${req.ip}`,
  onLimitReached: (req, res, info) => {
    console.warn(`[RateLimit] Login blocked for IP: ${req.ip}`);
    res.status(429).json({
      error: 'Too many login attempts',
      retryAfter: Math.ceil(info.retryAfterMs / 1000),
    });
  },
});

// ── Per-user limiter ──────────────────────────────────────────────────────────
const userLimiter = new RateLimiter({
  strategy: 'slidingWindow',
  windowMs: 60_000,
  maxRequests: 30,
  keyGenerator: (req) => `user:${req.headers['x-user-id'] || req.ip}`,
});

// ── Routes ────────────────────────────────────────────────────────────────────
app.get('/', (req, res) => {
  res.json({ message: 'Hello! You are within the rate limit.' });
});

app.get('/api/data', apiLimiter, (req, res) => {
  res.json({ data: 'Here is your data.' });
});

app.post('/auth/login', loginLimiter, (req, res) => {
  res.json({ message: 'Login successful.' });
});

app.get('/user/profile', userLimiter.middleware(), (req, res) => {
  res.json({ profile: 'User profile data.' });
});

// Programmatic usage example
app.get('/check', async (req, res) => {
  const result = await userLimiter.check(`user:${req.ip}`);
  res.json({ rateLimitInfo: result });
});

app.listen(3000, () => console.log('Server running on http://localhost:3000'));
```

## Summary

| Feature | Details |
|---|---|
| **Strategies** | Fixed Window, Sliding Window, Token Bucket |
| **Stores** | In-Memory (built-in), Redis (plug-in) |
| **Headers** | `X-RateLimit-Limit`, `X-RateLimit-Remaining`, `Retry-After` |
| **Fail-open** | Store errors don't block requests |
| **Key generation** | Customizable per-route (IP, user ID, API key, etc.) |
| **Programmatic API** | `limiter.check(key)` / `limiter.reset(key)` |
| **Custom handlers** | `onLimitReached` callback |
