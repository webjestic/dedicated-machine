## 2. Strategies

### Fixed Window

```javascript
// src/strategies/fixedWindow.js

class FixedWindow {
  /**
   * Allows N requests per fixed time window.
   * Simple but can allow 2x traffic at window boundaries.
   *
   * @param {number} limit     - Max requests per window
   * @param {number} windowMs  - Window size in milliseconds
   */
  constructor(limit, windowMs) {
    if (limit <= 0) throw new Error("Limit must be a positive number");
    if (windowMs <= 0) throw new Error("Window must be a positive number");

    this.limit = limit;
    this.windowMs = windowMs;
    this.store = new Map(); // { key -> { count, resetAt } }
  }

  /**
   * Check if a request is allowed for the given key.
   * @param {string} key - Unique identifier (e.g., IP address)
   * @returns {{ allowed: boolean, remaining: number, resetAt: number }}
   */
  check(key) {
    const now = Date.now();
    const record = this.store.get(key);

    // No record or window has expired — start a fresh window
    if (!record || now >= record.resetAt) {
      const resetAt = now + this.windowMs;
      this.store.set(key, { count: 1, resetAt });
      return { allowed: true, remaining: this.limit - 1, resetAt };
    }

    // Within the current window
    if (record.count < this.limit) {
      record.count++;
      return {
        allowed: true,
        remaining: this.limit - record.count,
        resetAt: record.resetAt,
      };
    }

    // Limit exceeded
    return { allowed: false, remaining: 0, resetAt: record.resetAt };
  }

  reset(key) {
    this.store.delete(key);
  }

  cleanup() {
    const now = Date.now();
    for (const [key, record] of this.store) {
      if (now >= record.resetAt) this.store.delete(key);
    }
  }
}

module.exports = FixedWindow;
```

### Sliding Window

```javascript
// src/strategies/slidingWindow.js

class SlidingWindow {
  /**
   * Allows N requests per rolling time window.
   * Smoother than fixed window — no boundary burst problem.
   *
   * @param {number} limit     - Max requests per window
   * @param {number} windowMs  - Window size in milliseconds
   */
  constructor(limit, windowMs) {
    if (limit <= 0) throw new Error("Limit must be a positive number");
    if (windowMs <= 0) throw new Error("Window must be a positive number");

    this.limit = limit;
    this.windowMs = windowMs;
    this.store = new Map(); // { key -> [timestamps] }
  }

  /**
   * Check if a request is allowed for the given key.
   * @param {string} key
   * @returns {{ allowed: boolean, remaining: number, resetAt: number }}
   */
  check(key) {
    const now = Date.now();
    const windowStart = now - this.windowMs;

    // Get existing timestamps and filter out expired ones
    const timestamps = (this.store.get(key) || []).filter(
      (ts) => ts > windowStart
    );

    const resetAt = (timestamps[0] || now) + this.windowMs;

    if (timestamps.length < this.limit) {
      timestamps.push(now);
      this.store.set(key, timestamps);
      return {
        allowed: true,
        remaining: this.limit - timestamps.length,
        resetAt,
      };
    }

    return { allowed: false, remaining: 0, resetAt };
  }

  reset(key) {
    this.store.delete(key);
  }

  cleanup() {
    const now = Date.now();
    for (const [key, timestamps] of this.store) {
      const valid = timestamps.filter((ts) => ts > now - this.windowMs);
      if (valid.length === 0) {
        this.store.delete(key);
      } else {
        this.store.set(key, valid);
      }
    }
  }
}

module.exports = SlidingWindow;
```

### Token Bucket

```javascript
// src/strategies/tokenBucket.js

class TokenBucket {
  /**
   * Tokens refill at a steady rate. Allows controlled bursting.
   *
   * @param {number} capacity      - Max tokens (burst size)
   * @param {number} refillRate    - Tokens added per second
   */
  constructor(capacity, refillRate) {
    if (capacity <= 0) throw new Error("Capacity must be a positive number");
    if (refillRate <= 0) throw new Error("Refill rate must be a positive number");

    this.capacity = capacity;
    this.refillRate = refillRate; // tokens per second
    this.store = new Map(); // { key -> { tokens, lastRefill } }
  }

  _refill(record) {
    const now = Date.now();
    const elapsed = (now - record.lastRefill) / 1000; // seconds
    const tokensToAdd = elapsed * this.refillRate;

    record.tokens = Math.min(this.capacity, record.tokens + tokensToAdd);
    record.lastRefill = now;
  }

  /**
   * Check if a request is allowed for the given key.
   * @param {string} key
   * @param {number} [cost=1] - Token cost of this request
   * @returns {{ allowed: boolean, remaining: number, resetAt: number }}
   */
  check(key, cost = 1) {
    const now = Date.now();

    if (!this.store.has(key)) {
      this.store.set(key, { tokens: this.capacity, lastRefill: now });
    }

    const record = this.store.get(key);
    this._refill(record);

    // Time until the bucket has enough tokens for the next request
    const resetAt =
      record.tokens < cost
        ? now + Math.ceil(((cost - record.tokens) / this.refillRate) * 1000)
        : now;

    if (record.tokens >= cost) {
      record.tokens -= cost;
      return {
        allowed: true,
        remaining: Math.floor(record.tokens),
        resetAt,
      };
    }

    return { allowed: false, remaining: 0, resetAt };
  }

  reset(key) {
    this.store.delete(key);
  }

  cleanup() {
    // Remove buckets that are full (idle clients)
    for (const [key, record] of this.store) {
      this._refill(record);
      if (record.tokens >= this.capacity) this.store.delete(key);
    }
  }
}

module.exports = TokenBucket;
```

## 3. Core Rate Limiter

```javascript
// src/rateLimiter.js

const FixedWindow = require("./strategies/fixedWindow");
const SlidingWindow = require("./strategies/slidingWindow");
const TokenBucket = require("./strategies/tokenBucket");

const STRATEGIES = {
  fixedWindow: FixedWindow,
  slidingWindow: SlidingWindow,
  tokenBucket: TokenBucket,
};

class RateLimiter {
  /**
   * @param {object} options
   * @param {'fixedWindow'|'slidingWindow'|'tokenBucket'} options.strategy
   * @param {number}   options.limit        - Max requests (or capacity for token bucket)
   * @param {number}   options.windowMs     - Window in ms (fixedWindow / slidingWindow)
   * @param {number}   options.refillRate   - Tokens/sec (tokenBucket only)
   * @param {number}   [options.cleanupIntervalMs=60000] - How often to purge stale entries
   * @param {Function} [options.keyGenerator] - Custom key function (default: returns 'global')
   */
  constructor(options = {}) {
    const {
      strategy = "slidingWindow",
      limit = 60,
      windowMs = 60_000,
      refillRate = 1,
      cleanupIntervalMs = 60_000,
      keyGenerator = () => "global",
    } = options;

    if (!STRATEGIES[strategy]) {
      throw new Error(
        `Unknown strategy "${strategy}". Choose from: ${Object.keys(STRATEGIES).join(", ")}`
      );
    }

    this.keyGenerator = keyGenerator;

    // Instantiate the chosen strategy
    if (strategy === "tokenBucket") {
      this.strategy = new STRATEGIES[strategy](limit, refillRate);
    } else {
      this.strategy = new STRATEGIES[strategy](limit, windowMs);
    }

    // Periodically clean up stale entries to prevent memory leaks
    this._cleanupTimer = setInterval(
      () => this.strategy.cleanup(),
      cleanupIntervalMs
    ).unref(); // unref so it doesn't keep the process alive
  }

  /**
   * Check whether a request should be allowed.
   * @param {string} [key] - Override the key (skips keyGenerator)
   * @returns {{ allowed: boolean, remaining: number, resetAt: number }}
   */
  check(key) {
    const resolvedKey = key ?? this.keyGenerator();
    return this.strategy.check(resolvedKey);
  }

  /**
   * Reset the limit for a specific key.
   * @param {string} key
   */
  reset(key) {
    this.strategy.reset(key);
  }

  /**
   * Stop the cleanup timer (important for tests / graceful shutdown).
   */
  destroy() {
    clearInterval(this._cleanupTimer);
  }
}

module.exports = RateLimiter;
```

## 4. Express Middleware

```javascript
// src/middleware.js

const RateLimiter = require("./rateLimiter");

/**
 * Creates an Express rate-limiting middleware.
 *
 * @param {object} options - Same as RateLimiter options, plus:
 * @param {Function} [options.keyGenerator]  - (req) => string
 * @param {Function} [options.onLimitReached] - (req, res, next, info) => void
 * @param {string}   [options.message]       - Response message when limited
 * @param {number}   [options.statusCode]    - HTTP status (default: 429)
 * @param {boolean}  [options.headers]       - Send rate-limit headers (default: true)
 */
function createRateLimiter(options = {}) {
  const {
    keyGenerator = (req) => req.ip,
    onLimitReached = null,
    message = "Too many requests. Please try again later.",
    statusCode = 429,
    headers = true,
    ...limiterOptions
  } = options;

  const limiter = new RateLimiter({
    ...limiterOptions,
    keyGenerator: () => "unused", // key resolved per-request below
  });

  function middleware(req, res, next) {
    const key = keyGenerator(req);
    const { allowed, remaining, resetAt } = limiter.check(key);

    if (headers) {
      res.setHeader("X-RateLimit-Limit", limiterOptions.limit ?? 60);
      res.setHeader("X-RateLimit-Remaining", remaining);
      res.setHeader("X-RateLimit-Reset", Math.ceil(resetAt / 1000)); // Unix seconds
      if (!allowed) {
        res.setHeader(
          "Retry-After",
          Math.ceil((resetAt - Date.now()) / 1000)
        );
      }
    }

    if (allowed) return next();

    if (onLimitReached) return onLimitReached(req, res, next, { remaining, resetAt });

    res.status(statusCode).json({ error: message, retryAfter: Math.ceil((resetAt - Date.now()) / 1000) });
  }

  // Expose the underlying limiter for manual resets, etc.
  middleware.limiter = limiter;

  return middleware;
}

module.exports = createRateLimiter;
```

## 5. Express Server (Demo)

```javascript
// server.js

const express = require("express");
const createRateLimiter = require("./src/middleware");

const app = express();
app.use(express.json());

// ── Global limiter (sliding window, 10 req / 10s) ──────────────────────────
const globalLimiter = createRateLimiter({
  strategy: "slidingWindow",
  limit: 10,
  windowMs: 10_000,
  keyGenerator: (req) => req.ip,
});

// ── Strict limiter for auth routes (fixed window, 5 req / 15s) ────────────
const authLimiter = createRateLimiter({
  strategy: "fixedWindow",
  limit: 5,
  windowMs: 15_000,
  keyGenerator: (req) => `auth:${req.ip}`,
  message: "Too many login attempts. Please wait before trying again.",
});

// ── Token bucket for API (capacity 20, refill 2 tokens/sec) ───────────────
const apiLimiter = createRateLimiter({
  strategy: "tokenBucket",
  limit: 20,        // capacity
  refillRate: 2,    // tokens per second
  keyGenerator: (req) => `api:${req.headers["x-api-key"] || req.ip}`,
});

// ── Routes ─────────────────────────────────────────────────────────────────
app.get("/", globalLimiter, (req, res) => {
  res.json({ message: "Hello! You are within the rate limit." });
});

app.post("/auth/login", authLimiter, (req, res) => {
  res.json({ message: "Login endpoint" });
});

app.get("/api/data", apiLimiter, (req, res) => {
  res.json({ data: "Here is your data", timestamp: Date.now() });
});

// ── Admin: manually reset a key ────────────────────────────────────────────
app.delete("/admin/reset/:key", (req, res) => {
  apiLimiter.limiter.reset(req.params.key);
  res.json({ message: `Rate limit reset for key: ${req.params.key}` });
});

// ── Error handler ──────────────────────────────────────────────────────────
app.use((err, req, res, _next) => {
  console.error(err);
  res.status(500).json({ error: "Internal server error" });
});

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => console.log(`Server running on http://localhost:${PORT}`));

module.exports = app;
```

## Strategy Comparison

| Strategy | Pros | Cons | Best For |
|---|---|---|---|
| **Fixed Window** | Simple, low memory | Boundary burst (2× traffic at edges) | Simple use cases |
| **Sliding Window** | Smooth, accurate | Higher memory (stores timestamps) | APIs, general use |
| **Token Bucket** | Allows controlled bursts | Slightly complex | Streaming, bursty clients |
