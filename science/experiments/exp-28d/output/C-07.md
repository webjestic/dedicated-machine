## 1. Memory Store

```javascript
// src/store/memoryStore.js
class MemoryStore {
  constructor() {
    this.store = new Map();
    this.cleanupInterval = setInterval(() => this.cleanup(), 60_000);
    this.cleanupInterval.unref(); // Don't block process exit
  }

  async get(key) {
    const entry = this.store.get(key);
    if (!entry) return null;
    if (Date.now() > entry.expiresAt) {
      this.store.delete(key);
      return null;
    }
    return entry.value;
  }

  async set(key, value, ttlMs) {
    this.store.set(key, {
      value,
      expiresAt: Date.now() + ttlMs,
    });
  }

  async increment(key, ttlMs) {
    const entry = this.store.get(key);
    const now = Date.now();

    if (!entry || now > entry.expiresAt) {
      this.store.set(key, { value: 1, expiresAt: now + ttlMs });
      return 1;
    }

    entry.value += 1;
    return entry.value;
  }

  async delete(key) {
    this.store.delete(key);
  }

  cleanup() {
    const now = Date.now();
    for (const [key, entry] of this.store.entries()) {
      if (now > entry.expiresAt) {
        this.store.delete(key);
      }
    }
  }

  destroy() {
    clearInterval(this.cleanupInterval);
    this.store.clear();
  }
}

module.exports = MemoryStore;
```

// src/store/redisStore.js
class RedisStore {
  /**
   * @param {import('ioredis').Redis} client - ioredis client instance
   */
  constructor(client) {
    if (!client) throw new Error('Redis client is required');
    this.client = client;
  }

  async get(key) {
    const value = await this.client.get(key);
    return value ? JSON.parse(value) : null;
  }

  async set(key, value, ttlMs) {
    await this.client.set(key, JSON.stringify(value), 'PX', ttlMs);
  }

  async increment(key, ttlMs) {
    // Atomic increment using a Lua script
    const script = `
      local current = redis.call('GET', KEYS[1])
      if current == false then
        redis.call('SET', KEYS[1], 1, 'PX', ARGV[1])
        return 1
      else
        return redis.call('INCR', KEYS[1])
      end
    `;
    return this.client.eval(script, 1, key, ttlMs);
  }

  async delete(key) {
    await this.client.del(key);
  }
}

module.exports = RedisStore;

## 3. Fixed Window Strategy

```javascript
// src/strategies/fixedWindow.js

/**
 * Fixed Window Rate Limiting
 * Simple counter per time window. Resets at window boundaries.
 * Weakness: allows burst at window edges (2x limit possible).
 */
class FixedWindow {
  /**
   * @param {object} options
   * @param {number} options.windowMs   - Window size in milliseconds
   * @param {number} options.maxRequests - Max requests per window
   * @param {object} options.store      - Storage backend
   */
  constructor({ windowMs, maxRequests, store }) {
    this.windowMs = windowMs;
    this.maxRequests = maxRequests;
    this.store = store;
  }

  async check(key) {
    const count = await this.store.increment(key, this.windowMs);
    const remaining = Math.max(0, this.maxRequests - count);
    const allowed = count <= this.maxRequests;

    return {
      allowed,
      count,
      remaining,
      resetAfterMs: this.windowMs,
      strategy: 'fixed-window',
    };
  }

  async reset(key) {
    await this.store.delete(key);
  }
}

module.exports = FixedWindow;
```

## 4. Sliding Window Strategy

```javascript
// src/strategies/slidingWindow.js

/**
 * Sliding Window Rate Limiting
 * Tracks individual request timestamps for precise rate limiting.
 * More accurate than fixed window but uses more memory.
 */
class SlidingWindow {
  /**
   * @param {object} options
   * @param {number} options.windowMs    - Window size in milliseconds
   * @param {number} options.maxRequests - Max requests per window
   * @param {object} options.store       - Storage backend
   */
  constructor({ windowMs, maxRequests, store }) {
    this.windowMs = windowMs;
    this.maxRequests = maxRequests;
    this.store = store;
  }

  async check(key) {
    const now = Date.now();
    const windowStart = now - this.windowMs;

    // Get existing timestamps
    let timestamps = (await this.store.get(key)) || [];

    // Remove expired timestamps outside the window
    timestamps = timestamps.filter((ts) => ts > windowStart);

    const count = timestamps.length;
    const allowed = count < this.maxRequests;

    if (allowed) {
      timestamps.push(now);
      await this.store.set(key, timestamps, this.windowMs);
    }

    const remaining = Math.max(0, this.maxRequests - timestamps.length);

    // Time until oldest request expires
    const resetAfterMs =
      timestamps.length > 0
        ? Math.max(0, timestamps[0] + this.windowMs - now)
        : this.windowMs;

    return {
      allowed,
      count: timestamps.length,
      remaining,
      resetAfterMs,
      strategy: 'sliding-window',
    };
  }

  async reset(key) {
    await this.store.delete(key);
  }
}

module.exports = SlidingWindow;
```

## 5. Token Bucket Strategy

```javascript
// src/strategies/tokenBucket.js

/**
 * Token Bucket Rate Limiting
 * Tokens refill at a constant rate. Allows controlled bursting.
 * Best for APIs that want to allow short bursts while maintaining avg rate.
 */
class TokenBucket {
  /**
   * @param {object} options
   * @param {number} options.capacity      - Max tokens (burst size)
   * @param {number} options.refillRate    - Tokens added per second
   * @param {number} options.refillIntervalMs - How often to refill (ms)
   * @param {object} options.store         - Storage backend
   */
  constructor({ capacity, refillRate, refillIntervalMs = 1000, store }) {
    this.capacity = capacity;
    this.refillRate = refillRate;
    this.refillIntervalMs = refillIntervalMs;
    this.store = store;
  }

  async check(key, tokensRequested = 1) {
    const now = Date.now();
    let bucket = await this.store.get(key);

    if (!bucket) {
      bucket = {
        tokens: this.capacity,
        lastRefill: now,
      };
    }

    // Calculate tokens to add based on elapsed time
    const elapsed = now - bucket.lastRefill;
    const tokensToAdd =
      (elapsed / this.refillIntervalMs) * this.refillRate;

    bucket.tokens = Math.min(this.capacity, bucket.tokens + tokensToAdd);
    bucket.lastRefill = now;

    const allowed = bucket.tokens >= tokensRequested;

    if (allowed) {
      bucket.tokens -= tokensRequested;
    }

    // TTL: time to refill from 0 to capacity
    const ttlMs =
      (this.capacity / this.refillRate) * this.refillIntervalMs * 2;
    await this.store.set(key, bucket, ttlMs);

    const timeToNextToken =
      bucket.tokens < 1
        ? Math.ceil(
            ((1 - bucket.tokens) / this.refillRate) * this.refillIntervalMs
          )
        : 0;

    return {
      allowed,
      tokens: Math.floor(bucket.tokens),
      remaining: Math.floor(bucket.tokens),
      resetAfterMs: timeToNextToken,
      strategy: 'token-bucket',
    };
  }

  async reset(key) {
    await this.store.delete(key);
  }
}

module.exports = TokenBucket;
```

## 6. Core Rate Limiter

```javascript
// src/rateLimiter.js
const MemoryStore = require('./store/memoryStore');
const FixedWindow = require('./strategies/fixedWindow');
const SlidingWindow = require('./strategies/slidingWindow');
const TokenBucket = require('./strategies/tokenBucket');

const STRATEGIES = {
  'fixed-window': FixedWindow,
  'sliding-window': SlidingWindow,
  'token-bucket': TokenBucket,
};

class RateLimiter {
  /**
   * @param {object} options
   * @param {string} [options.strategy='sliding-window'] - Rate limiting strategy
   * @param {number} [options.windowMs=60000]            - Window in ms (fixed/sliding)
   * @param {number} [options.maxRequests=100]           - Max requests per window
   * @param {number} [options.capacity=10]               - Token bucket capacity
   * @param {number} [options.refillRate=1]              - Token bucket refill rate/sec
   * @param {object} [options.store]                     - Custom store (default: MemoryStore)
   * @param {function} [options.keyGenerator]            - Custom key function
   * @param {function} [options.onLimitReached]          - Callback when limit hit
   */
  constructor(options = {}) {
    const {
      strategy = 'sliding-window',
      windowMs = 60_000,
      maxRequests = 100,
      capacity = 10,
      refillRate = 1,
      refillIntervalMs = 1000,
      store = new MemoryStore(),
      keyGenerator = null,
      onLimitReached = null,
    } = options;

    if (!STRATEGIES[strategy]) {
      throw new Error(
        `Unknown strategy: "${strategy}". Choose from: ${Object.keys(STRATEGIES).join(', ')}`
      );
    }

    this.store = store;
    this.keyGenerator = keyGenerator;
    this.onLimitReached = onLimitReached;

    const StrategyClass = STRATEGIES[strategy];
    this.strategy = new StrategyClass({
      windowMs,
      maxRequests,
      capacity,
      refillRate,
      refillIntervalMs,
      store,
    });
  }

  /**
   * Check if a request should be allowed
   * @param {string} identifier - Unique key (IP, user ID, API key, etc.)
   * @returns {Promise<RateLimitResult>}
   */
  async check(identifier) {
    if (!identifier) throw new Error('Identifier is required');

    const key = this.keyGenerator
      ? this.keyGenerator(identifier)
      : `rl:${identifier}`;

    const result = await this.strategy.check(key);

    if (!result.allowed && this.onLimitReached) {
      await this.onLimitReached(identifier, result);
    }

    return result;
  }

  /**
   * Reset limits for an identifier
   * @param {string} identifier
   */
  async reset(identifier) {
    const key = this.keyGenerator
      ? this.keyGenerator(identifier)
      : `rl:${identifier}`;
    await this.strategy.reset(key);
  }
}

module.exports = RateLimiter;
```

## 7. Express Middleware

```javascript
// src/middleware.js
const RateLimiter = require('./rateLimiter');

/**
 * Creates an Express rate limiting middleware
 * @param {object} options - RateLimiter options plus middleware-specific options
 * @param {function} [options.keyGenerator] - Extract key from req (default: IP)
 * @param {function} [options.skip]         - Skip rate limiting for certain requests
 * @param {object}  [options.headers]       - Custom header names
 */
function createRateLimitMiddleware(options = {}) {
  const {
    skip = null,
    headers = {},
    message = 'Too many requests, please try again later.',
    statusCode = 429,
    ...limiterOptions
  } = options;

  const headerNames = {
    limit: 'X-RateLimit-Limit',
    remaining: 'X-RateLimit-Remaining',
    reset: 'X-RateLimit-Reset',
    retryAfter: 'Retry-After',
    ...headers,
  };

  // Default key: client IP address
  if (!limiterOptions.keyGenerator) {
    limiterOptions.keyGenerator = (identifier) => `rl:${identifier}`;
  }

  const limiter = new RateLimiter(limiterOptions);

  return async function rateLimitMiddleware(req, res, next) {
    try {
      // Allow skipping for certain routes/conditions
      if (skip && (await skip(req))) {
        return next();
      }

      // Extract identifier from request
      const identifier =
        req.ip ||
        req.headers['x-forwarded-for']?.split(',')[0].trim() ||
        req.connection.remoteAddress ||
        'unknown';

      const result = await limiter.check(identifier);

      // Set rate limit headers
      const limit =
        limiterOptions.maxRequests || limiterOptions.capacity || 100;
      res.setHeader(headerNames.limit, limit);
      res.setHeader(headerNames.remaining, result.remaining);
      res.setHeader(
        headerNames.reset,
        Math.ceil(Date.now() / 1000 + result.resetAfterMs / 1000)
      );

      if (!result.allowed) {
        res.setHeader(
          headerNames.retryAfter,
          Math.ceil(result.resetAfterMs / 1000)
        );

        return res.status(statusCode).json({
          error: message,
          retryAfter: Math.ceil(result.resetAfterMs / 1000),
        });
      }

      next();
    } catch (err) {
      // Fail open: if rate limiter errors, allow the request
      console.error('[RateLimiter] Error:', err.message);
      next();
    }
  };
}

module.exports = { createRateLimitMiddleware, RateLimiter };
```

## 8. Express Server Example

```javascript
// server.js
const express = require('express');
const { createRateLimitMiddleware } = require('./src/middleware');

const app = express();
app.use(express.json());

// ── Global limiter: 100 req/min per IP ──────────────────────────────────────
const globalLimiter = createRateLimitMiddleware({
  strategy: 'sliding-window',
  windowMs: 60_000,
  maxRequests: 100,
});

// ── Strict limiter for auth routes: 5 req/15min ─────────────────────────────
const authLimiter = createRateLimitMiddleware({
  strategy: 'fixed-window',
  windowMs: 15 * 60_000,
  maxRequests: 5,
  message: 'Too many login attempts. Please try again in 15 minutes.',
});

// ── Token bucket for API: burst of 20, refill 2/sec ─────────────────────────
const apiLimiter = createRateLimitMiddleware({
  strategy: 'token-bucket',
  capacity: 20,
  refillRate: 2,
  refillIntervalMs: 1000,
});

// Apply global limiter to all routes
app.use(globalLimiter);

app.post('/auth/login', authLimiter, (req, res) => {
  res.json({ message: 'Login successful' });
});

app.get('/api/data', apiLimiter, (req, res) => {
  res.json({ data: 'Here is your data', timestamp: Date.now() });
});

app.get('/health', (req, res) => {
  res.json({ status: 'ok' });
});

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => {
  console.log(`Server running on http://localhost:${PORT}`);
});

module.exports = app;
```

## Strategy Comparison

| Strategy | Accuracy | Memory | Burst Handling | Best For |
|---|---|---|---|---|
| **Fixed Window** | Low | Low | Poor (edge bursts) | Simple counters |
| **Sliding Window** | High | Medium | Good | API rate limits |
| **Token Bucket** | High | Low | Excellent | Bursty traffic |

## Key Design Decisions

- **Fail open** — if the store errors, requests are allowed through (availability over security)
- **Pluggable stores** — swap `MemoryStore` for `RedisStore` for distributed deployments
- **Atomic operations** — Redis Lua scripts prevent race conditions
- **TTL-based cleanup** — memory store auto-expires entries to prevent leaks
- **Strategy pattern** — easily extend with new algorithms (e.g., leaky bucket)
