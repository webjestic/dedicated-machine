# Production Rate Limiter in Node.js

Let me design this fully before writing a line of code.

---

## Phase 1: Requirements Design

### What algorithm should I use?

| Algorithm | Pros | Cons | Best For |
|---|---|---|---|
| Fixed Window | Simple, fast | Burst at window edges (2x spike) | Internal tools |
| Sliding Window Log | Exact | High memory (stores every timestamp) | Low-traffic, exact billing |
| Sliding Window Counter | Good approximation, low memory | ~0.003% error at boundary | Most APIs |
| Token Bucket | Smooth bursting allowed | State harder to reason about | Streaming, SDKs |
| Leaky Bucket | Perfectly smooth output | Drops bursts entirely | Queue-based systems |

**Decision: Sliding Window Counter** — best balance of accuracy, memory efficiency, and production operability for a general-purpose API rate limiter.

### What are the key design decisions?

```
1. Identifier strategy     → by IP, API key, user ID, or composite
2. Storage backend         → in-process Map vs Redis (distributed)
3. Atomicity               → race conditions under concurrent requests
4. Headers                 → must tell clients what happened and when to retry
5. Failure mode            → if Redis is down, do we fail open or closed?
6. Multiple limits         → per-second AND per-day on same key?
7. Bypass/whitelist        → some callers must never be limited
8. Observability           → we need to know when this fires in production
```

## Phase 2: Architecture

```
Incoming Request
      │
      ▼
┌─────────────────────────────────────────────────────┐
│                  Rate Limiter Middleware              │
│                                                      │
│  1. Extract identifier (IP / API key / user ID)      │
│  2. Check whitelist → pass through immediately       │
│  3. For each rule that applies to this identifier:   │
│     a. Compute sliding window count (Redis ATOMIC)   │
│     b. If any rule exceeded → 429, set Retry-After   │
│  4. Attach rate limit headers to all responses       │
│  5. On storage failure → fail open (log + alert)     │
└─────────────────────────────────────────────────────┘
      │                          │
   Allowed                    Blocked
      │                          │
      ▼                          ▼
 next()                   429 + headers
```

### Sliding Window Counter — how it works

```
Window size: 60 seconds
Current time: 12:00:45

Previous window (12:00:00 - 12:00:00): count = 80 requests
Current window (12:00:00 - 12:00:45): count = 30 requests

Weight of previous window:
  = (60 - 45) / 60 = 0.25

Estimated count = (80 × 0.25) + 30 = 50 requests

Limit: 100 → 50 < 100 → ALLOW
```

This smooths the boundary spike that kills fixed-window limiters.

### `src/store/memoryStore.js`

```javascript
/**
 * In-process store for development and testing.
 * Implements the same interface as redisStore so the algorithm
 * layer never knows which backing store it is talking to.
 *
 * NOT safe for multi-process or multi-instance deployments.
 * Counts are lost on restart.
 */

class MemoryStore {
  constructor() {
    // Map<key, { count: number, expiresAt: number }>
    this._windows = new Map();

    // Prevent unbounded memory growth in long-running dev servers.
    // In production this is handled by Redis TTL automatically.
    this._cleanupInterval = setInterval(
      () => this._evictExpired(),
      30_000 // every 30 seconds
    );

    // Allow Node process to exit even if this interval is still running
    this._cleanupInterval.unref();
  }

  /**
   * Atomically increment a counter and set its TTL on first write.
   *
   * @param {string} key        - namespaced window key
   * @param {number} ttlSeconds - how long this window lives
   * @returns {{ count: number, ttlMs: number }}
   */
  async increment(key, ttlSeconds) {
    const now = Date.now();
    const existing = this._windows.get(key);

    if (!existing || existing.expiresAt <= now) {
      // First request in this window
      const expiresAt = now + ttlSeconds * 1000;
      this._windows.set(key, { count: 1, expiresAt });
      return { count: 1, ttlMs: ttlSeconds * 1000 };
    }

    existing.count += 1;
    return {
      count: existing.count,
      ttlMs: existing.expiresAt - now,
    };
  }

  /**
   * Read current count without incrementing.
   * Used to build response headers without side effects.
   *
   * @param {string} key
   * @returns {{ count: number, ttlMs: number } | null}
   */
  async get(key) {
    const now = Date.now();
    const entry = this._windows.get(key);

    if (!entry || entry.expiresAt <= now) {
      return null;
    }

    return {
      count: entry.count,
      ttlMs: entry.expiresAt - now,
    };
  }

  /**
   * Wipe all state. Used in tests between cases.
   */
  async flush() {
    this._windows.clear();
  }

  /**
   * Remove expired entries to prevent memory growth.
   * Called automatically on an interval; also callable in tests.
   */
  _evictExpired() {
    const now = Date.now();
    for (const [key, entry] of this._windows) {
      if (entry.expiresAt <= now) {
        this._windows.delete(key);
      }
    }
  }

  /**
   * Graceful shutdown — stop the cleanup interval.
   */
  async close() {
    clearInterval(this._cleanupInterval);
  }
}

module.exports = { MemoryStore };
```

### `src/store/redisStore.js`

```javascript
/**
 * Redis-backed store for production use.
 *
 * Uses a Lua script to make increment + TTL-set atomic.
 * Without atomicity, two concurrent requests on a new key can
 * both see count=0, both set TTL, and the window resets early.
 *
 * Requires: ioredis  (npm install ioredis)
 */

const Redis = require('ioredis');

/**
 * Lua script executed atomically on the Redis server.
 *
 * KEYS[1]  = the window key
 * ARGV[1]  = TTL in seconds (only applied on first write)
 *
 * Returns: { new_count, remaining_ttl_ms }
 *
 * Why Lua and not a transaction (MULTI/EXEC)?
 * MULTI/EXEC is optimistic — it retries on conflict, which adds
 * latency under contention. Lua runs as a single atomic command.
 */
const INCREMENT_SCRIPT = `
  local key    = KEYS[1]
  local ttl    = tonumber(ARGV[1])
  local count  = redis.call('INCR', key)

  if count == 1 then
    -- First increment in this window: set expiry
    redis.call('EXPIRE', key, ttl)
  end

  local pttl = redis.call('PTTL', key)
  return { count, pttl }
`;

class RedisStore {
  /**
   * @param {object} options
   * @param {string} [options.host]
   * @param {number} [options.port]
   * @param {string} [options.password]
   * @param {number} [options.db]
   * @param {object} [options.redisClient]  - inject a pre-built client (testing)
   */
  constructor(options = {}) {
    if (options.redisClient) {
      this._client = options.redisClient;
    } else {
      this._client = new Redis({
        host: options.host ?? 'localhost',
        port: options.port ?? 6379,
        password: options.password,
        db: options.db ?? 0,

        // Fail fast rather than queue requests during an outage.
        // The middleware catches this and fails open.
        enableOfflineQueue: false,

        // Reconnect with exponential backoff, cap at 10 seconds
        retryStrategy: (times) => Math.min(times * 200, 10_000),

        lazyConnect: true,
      });
    }

    // Surface connection errors so the middleware can catch them
    this._client.on('error', (err) => {
      // ioredis emits these; without a listener Node would crash
      // The actual error handling happens in the middleware layer
      console.error('[RedisStore] connection error:', err.message);
    });

    this._scriptSha = null; // populated on first use via SCRIPT LOAD
  }

  /**
   * Connect to Redis. Call this during app startup so the first
   * request does not pay the connection cost.
   */
  async connect() {
    await this._client.connect();
    // Pre-load the Lua script so we call EVALSHA (hash lookup)
    // instead of EVAL (full script transfer) on every request
    this._scriptSha = await this._client.script('LOAD', INCREMENT_SCRIPT);
  }

  /**
   * Atomically increment a counter and set TTL on first write.
   *
   * @param {string} key
   * @param {number} ttlSeconds
   * @returns {{ count: number, ttlMs: number }}
   * @throws if Redis is unreachable
   */
  async increment(key, ttlSeconds) {
    let result;

    try {
      result = await this._client.evalsha(
        this._scriptSha,
        1,           // number of keys
        key,         // KEYS[1]
        ttlSeconds   // ARGV[1]
      );
    } catch (err) {
      // Script was flushed (Redis restart, SCRIPT FLUSH).
      // Reload and retry once.
      if (err.message.includes('NOSCRIPT')) {
        this._scriptSha = await this._client.script('LOAD', INCREMENT_SCRIPT);
        result = await this._client.evalsha(
          this._scriptSha,
          1,
          key,
          ttlSeconds
        );
      } else {
        throw err; // propagate to middleware for fail-open handling
      }
    }

    const [count, pttl] = result;
    return { count, ttlMs: pttl };
  }

  /**
   * Read current count without incrementing.
   *
   * @param {string} key
   * @returns {{ count: number, ttlMs: number } | null}
   */
  async get(key) {
    const [count, pttl] = await Promise.all([
      this._client.get(key),
      this._client.pttl(key),
    ]);

    if (count === null) return null;

    return {
      count: parseInt(count, 10),
      ttlMs: pttl > 0 ? pttl : 0,
    };
  }

  /**
   * Graceful shutdown.
   */
  async close() {
    await this._client.quit();
  }
}

module.exports = { RedisStore };
```

### `src/algorithms/slidingWindow.js`

```javascript
/**
 * Sliding Window Counter algorithm.
 *
 * Approximates a true sliding window by weighting the previous
 * window's count by how much of it overlaps the current window.
 *
 * Error rate: < 0.003% in practice (proven by Cloudflare's analysis).
 *
 * This module is pure logic — no I/O, no Express, fully unit-testable.
 */

/**
 * @typedef {object} WindowResult
 * @property {boolean} allowed         - whether this request should pass
 * @property {number}  count           - estimated requests in sliding window
 * @property {number}  limit           - the configured limit
 * @property {number}  remaining       - requests left before limit
 * @property {number}  retryAfterMs    - ms until a blocked request could retry
 * @property {number}  resetMs         - ms until current window resets
 * @property {string}  currentKey      - Redis/memory key for current window
 * @property {string}  previousKey     - Redis/memory key for previous window
 * @property {number}  windowSizeMs    - window duration in milliseconds
 */

/**
 * Compute whether a request is within the rate limit.
 *
 * @param {object} params
 * @param {string} params.identifier      - e.g. "ip:192.168.1.1"
 * @param {number} params.limit           - max requests per window
 * @param {number} params.windowSizeMs    - window duration in milliseconds
 * @param {number} params.currentCount    - count in current window (after increment)
 * @param {number} params.previousCount   - count in previous window
 * @param {number} params.now             - current timestamp (ms) — injected for testability
 * @returns {WindowResult}
 */
function evaluate({
  identifier,
  limit,
  windowSizeMs,
  currentCount,
  previousCount,
  now,
}) {
  // How far into the current window are we? (0.0 → 1.0)
  const windowStart = Math.floor(now / windowSizeMs) * windowSizeMs;
  const elapsedMs   = now - windowStart;
  const elapsedFraction = elapsedMs / windowSizeMs;

  // Weight the previous window by how much of it still "overlaps"
  // the sliding window ending at `now`
  const previousWeight = 1 - elapsedFraction;
  const estimatedCount = previousCount * previousWeight + currentCount;

  const allowed   = estimatedCount <= limit;
  const remaining = Math.max(0, limit - Math.ceil(estimatedCount));

  // If blocked: tell the client how long until the oldest requests
  // fall out of the sliding window. Conservatively: rest of current window.
  const resetMs      = windowSizeMs - elapsedMs;
  const retryAfterMs = allowed ? 0 : resetMs;

  return {
    allowed,
    count: Math.ceil(estimatedCount),
    limit,
    remaining,
    retryAfterMs,
    resetMs,
    // Expose keys so the middleware can pass them to the store
    currentKey:   buildKey(identifier, windowSizeMs, now, 'current'),
    previousKey:  buildKey(identifier, windowSizeMs, now, 'previous'),
    windowSizeMs,
  };
}

/**
 * Build a namespaced, time-bucketed store key.
 *
 * Format: rl:{identifier}:{windowSizeMs}:{bucketIndex}
 *
 * Example: rl:ip:127.0.0.1:60000:28512480
 *   → IP 127.0.0.1, 60-second window, bucket starting at
 *     28512480 * 60000ms = Unix time of window start
 *
 * @param {string} identifier
 * @param {number} windowSizeMs
 * @param {number} now
 * @param {'current'|'previous'} which
 * @returns {string}
 */
function buildKey(identifier, windowSizeMs, now, which) {
  const bucketIndex = Math.floor(now / windowSizeMs);
  const index = which === 'current' ? bucketIndex : bucketIndex - 1;
  return `rl:${identifier}:${windowSizeMs}:${index}`;
}

module.exports = { evaluate, buildKey };
```

### `src/utils/identifiers.js`

```javascript
/**
 * Strategies for extracting a rate-limit identifier from a request.
 *
 * The identifier determines who shares a rate limit bucket.
 * Choosing the wrong one is a common production mistake:
 *
 *   - IP only      → broken behind a shared NAT or corporate proxy
 *   - API key only → one compromised key can't affect others, but
 *                    unauthenticated endpoints have no protection
 *   - Composite    → most robust for mixed auth/unauth APIs
 */

/**
 * Extract the real client IP, respecting common proxy headers.
 *
 * SECURITY NOTE: X-Forwarded-For can be spoofed by clients unless
 * your infrastructure strips/overwrites it. Only trust it if you
 * control the proxy layer (load balancer, CDN, API gateway).
 *
 * @param {import('express').Request} req
 * @returns {string}
 */
function extractIP(req) {
  const forwarded = req.headers['x-forwarded-for'];
  if (forwarded) {
    // X-Forwarded-For: client, proxy1, proxy2
    // The leftmost is the originating client (if your proxy appends)
    return forwarded.split(',')[0].trim();
  }
  return req.socket?.remoteAddress ?? 'unknown';
}

/**
 * Extract an API key from common header locations.
 *
 * @param {import('express').Request} req
 * @returns {string | null}
 */
function extractAPIKey(req) {
  // Authorization: Bearer <token>
  const authHeader = req.headers['authorization'];
  if (authHeader?.startsWith('Bearer ')) {
    return authHeader.slice(7).trim() || null;
  }

  // X-API-Key: <key>
  const apiKeyHeader = req.headers['x-api-key'];
  if (apiKeyHeader) return apiKeyHeader.trim();

  // ?api_key=<key> query param (less preferred — shows up in logs)
  if (req.query?.api_key) return String(req.query.api_key).trim();

  return null;
}

/**
 * Composite identifier: prefer API key, fall back to IP.
 * This is the recommended default for most APIs.
 *
 * @param {import('express').Request} req
 * @returns {string}
 */
function extractComposite(req) {
  const apiKey = extractAPIKey(req);
  if (apiKey) return `key:${apiKey}`;
  return `ip:${extractIP(req)}`;
}

module.exports = { extractIP, extractAPIKey, extractComposite };
```

### `src/config/rules.js`

```javascript
/**
 * Rate limit rule definitions.
 *
 * A "rule" describes a limit that applies to a set of requests.
 * Multiple rules can apply to the same request — all must pass.
 *
 * Rules are evaluated in order. The first rule to block a request
 * wins (its Retry-After is used in the response).
 *
 * Design note: keeping rules in config (not hardcoded in middleware)
 * means you can load them from a database, feature flags, or env vars
 * without touching application logic.
 */

/**
 * @typedef {object} RateLimitRule
 * @property {string}   name          - human-readable name for logs/metrics
 * @property {number}   limit         - max requests per window
 * @property {number}   windowMs      - window size in milliseconds
 * @property {string[]} [methods]     - HTTP methods this rule applies to (omit = all)
 * @property {RegExp}   [pathPattern] - URL paths this rule applies to (omit = all)
 */

/** @type {RateLimitRule[]} */
const rules = [
  {
    // Tight per-second limit to blunt burst attacks
    name: 'burst-protection',
    limit: 20,
    windowMs: 1_000, // 1 second
  },
  {
    // Standard per-minute limit for general API usage
    name: 'per-minute',
    limit: 100,
    windowMs: 60_000, // 1 minute
  },
  {
    // Stricter limit on expensive write operations
    name: 'writes-per-minute',
    limit: 20,
    windowMs: 60_000,
    methods: ['POST', 'PUT', 'PATCH', 'DELETE'],
  },
  {
    // Very tight limit on authentication endpoints to slow credential stuffing
    name: 'auth-endpoints',
    limit: 5,
    windowMs: 60_000,
    pathPattern: /^\/auth\//,
  },
];

/**
 * Identifiers that are never rate limited.
 * Use sparingly — internal health checks, trusted monitoring agents.
 *
 * SECURITY: never put customer API keys here. Use higher limits instead.
 */
const whitelist = new Set([
  'ip:127.0.0.1',
  'ip:::1', // IPv6 localhost
]);

/**
 * Filter rules to those that apply to a given request.
 *
 * @param {import('express').Request} req
 * @returns {RateLimitRule[]}
 */
function getRulesForRequest(req) {
  return rules.filter((rule) => {
    if (rule.methods && !rule.methods.includes(req.method)) return false;
    if (rule.pathPattern && !rule.pathPattern.test(req.path)) return false;
    return true;
  });
}

module.exports = { rules, whitelist, getRulesForRequest };
```

### `src/middleware/rateLimiter.js`

```javascript
/**
 * Rate limiter middleware factory.
 *
 * Usage:
 *   const limiter = createRateLimiter({ store, getIdentifier });
 *   app.use(limiter);
 *
 * This is the integration layer — it wires together:
 *   - identifier extraction
 *   - rule selection
 *   - the sliding window algorithm
 *   - the backing store
 *   - response headers
 *   - failure handling
 *   - observability hooks
 */

const { evaluate } = require('../algorithms/slidingWindow');
const { extractComposite } = require('../utils/identifiers');
const { getRulesForRequest, whitelist } = require('../config/rules');

/**
 * @typedef {object} RateLimiterOptions
 * @property {object}   store                   - MemoryStore or RedisStore instance
 * @property {Function} [getIdentifier]         - (req) => string
 * @property {Function} [onLimitReached]        - (req, result) => void  (metrics hook)
 * @property {Function} [onError]               - (err, req) => void     (error hook)
 * @property {boolean}  [failOpen=true]         - pass requests through if store is down
 * @property {boolean}  [dryRun=false]          - evaluate but never block (shadow mode)
 */

/**
 * Create the rate limiter middleware.
 *
 * @param {RateLimiterOptions} options
 * @returns {import('express').RequestHandler}
 */
function createRateLimiter(options = {}) {
  const {
    store,
    getIdentifier  = extractComposite,
    onLimitReached = defaultOnLimitReached,
    onError        = defaultOnError,
    failOpen       = true,
    dryRun         = false,
  } = options;

  if (!store) {
    throw new Error('[RateLimiter] options.store is required');
  }

  return async function rateLimiterMiddleware(req, res, next) {
    const identifier = getIdentifier(req);

    // ── Whitelist check ──────────────────────────────────────────────
    if (whitelist.has(identifier)) {
      setHeader(res, 'X-RateLimit-Bypassed', 'whitelisted');
      return next();
    }

    const applicableRules = getRulesForRequest(req);

    // No rules match this request (shouldn't happen with a catch-all rule,
    // but be explicit rather than silently passing everything)
    if (applicableRules.length === 0) {
      return next();
    }

    // ── Evaluate all applicable rules ────────────────────────────────
    let blocked       = null; // the first rule that blocks this request
    let mostRestrictive = null; // for headers on allowed requests

    try {
      for (const rule of applicableRules) {
        const result = await evaluateRule({
          store,
          identifier,
          rule,
        });

        // Track the most restrictive result for headers (lowest remaining)
        if (
          !mostRestrictive ||
          result.remaining < mostRestrictive.remaining
        ) {
          mostRestrictive = { rule, result };
        }

        if (!result.allowed) {
          blocked = { rule, result };
          break; // first blocking rule wins
        }
      }
    } catch (err) {
      onError(err, req);

      if (failOpen) {
        // Store is down — let the request through rather than
        // taking down the entire API. Log and alert separately.
        setHeader(res, 'X-RateLimit-Store-Error', 'true');
        return next();
      } else {
        // Fail closed — safer for security-sensitive endpoints
        return res.status(503).json({
          error: 'Service temporarily unavailable',
          message: 'Rate limiter storage is unreachable',
        });
      }
    }

    // ── Attach rate limit headers ────────────────────────────────────
    if (mostRestrictive) {
      attachHeaders(res, mostRestrictive.rule, mostRestrictive.result);
    }

    // ── Block or allow ───────────────────────────────────────────────
    if (blocked && !dryRun) {
      onLimitReached(req, blocked);

      attachHeaders(res, blocked.rule, blocked.result); // override with blocking rule

      const retryAfterSeconds = Math.ceil(blocked.result.retryAfterMs / 1000);

      res.setHeader('Retry-After', retryAfterSeconds);

      return res.status(429).json({
        error: 'Too Many Requests',
        message: `Rate limit exceeded: ${blocked.rule.name}`,
        retryAfter: retryAfterSeconds,
        limit: blocked.result.limit,
        windowMs: blocked.rule.windowMs,
      });
    }

    if (blocked && dryRun) {
      // Shadow mode: would have blocked, but let through
      setHeader(res, 'X-RateLimit-DryRun-WouldBlock', blocked.rule.name);
    }

    return next();
  };
}

// ── Internal helpers ───────────────────────────────────────────────────────

/**
 * Run the sliding window algorithm for one rule against the store.
 *
 * @param {{ store, identifier, rule }} params
 * @returns {Promise<WindowResult>}
 */
async function evaluateRule({ store, identifier, rule }) {
  const now          = Date.now();
  const windowSizeMs = rule.windowMs;

  // Build keys for current and previous windows
  const bucketIndex  = Math.floor(now / windowSizeMs);
  const currentKey   = `rl:${identifier}:${windowSizeMs}:${bucketIndex}`;
  const previousKey  = `rl:${identifier}:${windowSizeMs}:${bucketIndex - 1}`;

  // Increment current window, read previous window (non-incrementing)
  const [currentData, previousData] = await Promise.all([
    store.increment(currentKey, Math.ceil(windowSizeMs / 1000) * 2), // TTL = 2 windows
    store.get(previousKey),
  ]);

  const currentCount  = currentData.count;
  const previousCount = previousData?.count ?? 0;

  return evaluate({
    identifier,
    limit: rule.limit,
    windowSizeMs,
    currentCount,
    previousCount,
    now,
  });
}

/**
 * Set standard rate limit response headers.
 * Following the IETF draft: https://datatracker.ietf.org/doc/html/draft-ietf-httpapi-ratelimit-headers
 *
 * @param {import('express').Response} res
 * @param {object} rule
 * @param {object} result
 */
function attachHeaders(res, rule, result) {
  setHeader(res, 'X-RateLimit-Limit',     result.limit);
  setHeader(res, 'X-RateLimit-Remaining', result.remaining);
  setHeader(res, 'X-RateLimit-Reset',     Math.ceil(result.resetMs / 1000));
  setHeader(res, 'X-RateLimit-Policy',    `${result.limit};w=${rule.windowMs / 1000}`);
}

/**
 * Safe header setter — avoids throwing on already-sent responses.
 */
function setHeader(res, name, value) {
  if (!res.headersSent) {
    res.setHeader(name, String(value));
  }
}

/**
 * Default observability hooks — replace with your metrics/logging system.
 */
function defaultOnLimitReached(req, { rule, result }) {
  console.warn(
    `[RateLimiter] BLOCKED | rule=${rule.name} | ` +
    `identifier=${result.identifier ?? 'unknown'} | ` +
    `count=${result.count} | limit=${result.limit} | ` +
    `path=${req.path} | method=${req.method}`
  );
}

function defaultOnError(err, req) {
  console.error(
    `[RateLimiter] STORE ERROR | path=${req.path} | ` +
    `error=${err.message}`
  );
}
```
