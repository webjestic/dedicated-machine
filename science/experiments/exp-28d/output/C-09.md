## 1. Memory Store

```javascript
// src/store/memoryStore.js
class MemoryStore {
  constructor() {
    this.store = new Map();
    this.cleanupInterval = setInterval(() => this.cleanup(), 60_000);
    this.cleanupInterval.unref(); // Don't block process exit
  }

  async get(key) {
    return this.store.get(key) ?? null;
  }

  async set(key, value, ttlMs) {
    this.store.set(key, {
      value,
      expiresAt: Date.now() + ttlMs,
    });
  }

  async increment(key, ttlMs) {
    const entry = this.store.get(key);
    const now = Date.now();

    if (!entry || entry.expiresAt <= now) {
      this.store.set(key, { value: 1, expiresAt: now + ttlMs });
      return 1;
    }

    entry.value += 1;
    return entry.value;
  }

  async delete(key) {
    this.store.delete(key);
  }

  cleanup() {
    const now = Date.now();
    for (const [key, entry] of this.store.entries()) {
      if (entry.expiresAt <= now) {
        this.store.delete(key);
      }
    }
  }

  destroy() {
    clearInterval(this.cleanupInterval);
    this.store.clear();
  }
}

module.exports = MemoryStore;
```

// src/store/redisStore.js
class RedisStore {
  /**
   * @param {import('ioredis').Redis} client - ioredis client instance
   */
  constructor(client) {
    if (!client) throw new Error('Redis client is required');
    this.client = client;
  }

  async get(key) {
    const value = await this.client.get(key);
    return value ? JSON.parse(value) : null;
  }

  async set(key, value, ttlMs) {
    await this.client.set(key, JSON.stringify(value), 'PX', ttlMs);
  }

  async increment(key, ttlMs) {
    // Atomic increment using Lua script
    const script = `
      local current = redis.call('GET', KEYS[1])
      if current == false then
        redis.call('SET', KEYS[1], 1, 'PX', ARGV[1])
        return 1
      else
        return redis.call('INCR', KEYS[1])
      end
    `;
    return this.client.eval(script, 1, key, ttlMs);
  }

  async delete(key) {
    await this.client.del(key);
  }
}

module.exports = RedisStore;

## 3. Fixed Window Strategy

```javascript
// src/strategies/fixedWindow.js

/**
 * Fixed Window Rate Limiting
 * Simple counter per time window. Resets at the end of each window.
 * Weakness: allows burst at window boundaries.
 */
class FixedWindowStrategy {
  /**
   * @param {object} options
   * @param {number} options.windowMs   - Window size in milliseconds
   * @param {number} options.maxRequests - Max requests per window
   */
  constructor({ windowMs, maxRequests }) {
    if (!windowMs || windowMs <= 0) throw new Error('windowMs must be positive');
    if (!maxRequests || maxRequests <= 0) throw new Error('maxRequests must be positive');

    this.windowMs = windowMs;
    this.maxRequests = maxRequests;
  }

  async check(store, key) {
    const count = await store.increment(key, this.windowMs);
    const remaining = Math.max(0, this.maxRequests - count);
    const allowed = count <= this.maxRequests;

    return {
      allowed,
      count,
      remaining,
      resetAfterMs: this.windowMs,
      retryAfterMs: allowed ? 0 : this.windowMs,
    };
  }
}

module.exports = FixedWindowStrategy;
```

## 4. Sliding Window Strategy

```javascript
// src/strategies/slidingWindow.js

/**
 * Sliding Window Log Rate Limiting
 * Tracks individual request timestamps. More accurate than fixed window.
 * Uses more memory but eliminates boundary bursts.
 */
class SlidingWindowStrategy {
  /**
   * @param {object} options
   * @param {number} options.windowMs    - Window size in milliseconds
   * @param {number} options.maxRequests - Max requests per window
   */
  constructor({ windowMs, maxRequests }) {
    if (!windowMs || windowMs <= 0) throw new Error('windowMs must be positive');
    if (!maxRequests || maxRequests <= 0) throw new Error('maxRequests must be positive');

    this.windowMs = windowMs;
    this.maxRequests = maxRequests;
  }

  async check(store, key) {
    const now = Date.now();
    const windowStart = now - this.windowMs;

    // Get existing timestamps
    let timestamps = (await store.get(key))?.timestamps ?? [];

    // Remove expired timestamps outside the window
    timestamps = timestamps.filter((ts) => ts > windowStart);

    const count = timestamps.length;
    const allowed = count < this.maxRequests;

    if (allowed) {
      timestamps.push(now);
    }

    // Store updated timestamps with TTL
    await store.set(key, { timestamps }, this.windowMs);

    const remaining = Math.max(0, this.maxRequests - timestamps.length);

    // Time until oldest request expires
    const retryAfterMs =
      !allowed && timestamps.length > 0
        ? timestamps[0] - windowStart
        : 0;

    return {
      allowed,
      count: timestamps.length,
      remaining,
      resetAfterMs: this.windowMs,
      retryAfterMs,
    };
  }
}

module.exports = SlidingWindowStrategy;
```

## 5. Token Bucket Strategy

```javascript
// src/strategies/tokenBucket.js

/**
 * Token Bucket Rate Limiting
 * Tokens refill at a steady rate. Allows controlled bursting.
 * Best for APIs that want to allow short bursts while enforcing average rate.
 */
class TokenBucketStrategy {
  /**
   * @param {object} options
   * @param {number} options.capacity      - Max tokens (burst size)
   * @param {number} options.refillRate    - Tokens added per second
   * @param {number} options.tokensPerRequest - Tokens consumed per request (default: 1)
   */
  constructor({ capacity, refillRate, tokensPerRequest = 1 }) {
    if (!capacity || capacity <= 0) throw new Error('capacity must be positive');
    if (!refillRate || refillRate <= 0) throw new Error('refillRate must be positive');

    this.capacity = capacity;
    this.refillRate = refillRate;         // tokens per second
    this.tokensPerRequest = tokensPerRequest;
    this.ttlMs = Math.ceil((capacity / refillRate) * 1000) * 2; // generous TTL
  }

  async check(store, key) {
    const now = Date.now();
    const existing = await store.get(key);

    let tokens;
    let lastRefill;

    if (!existing) {
      // New bucket — start full
      tokens = this.capacity;
      lastRefill = now;
    } else {
      lastRefill = existing.lastRefill;
      const elapsedSeconds = (now - lastRefill) / 1000;
      const refilled = elapsedSeconds * this.refillRate;

      tokens = Math.min(this.capacity, existing.tokens + refilled);
      lastRefill = now;
    }

    const allowed = tokens >= this.tokensPerRequest;

    if (allowed) {
      tokens -= this.tokensPerRequest;
    }

    await store.set(key, { tokens, lastRefill }, this.ttlMs);

    const remaining = Math.floor(tokens);

    // How long until enough tokens are available
    const tokensNeeded = this.tokensPerRequest - tokens;
    const retryAfterMs = allowed
      ? 0
      : Math.ceil((tokensNeeded / this.refillRate) * 1000);

    return {
      allowed,
      tokens: Math.floor(tokens),
      remaining,
      resetAfterMs: Math.ceil((this.capacity / this.refillRate) * 1000),
      retryAfterMs,
    };
  }
}

module.exports = TokenBucketStrategy;
```

## 6. Core Rate Limiter

```javascript
// src/rateLimiter.js
const MemoryStore = require('./store/memoryStore');
const FixedWindowStrategy = require('./strategies/fixedWindow');
const SlidingWindowStrategy = require('./strategies/slidingWindow');
const TokenBucketStrategy = require('./strategies/tokenBucket');

const STRATEGIES = {
  fixedWindow: FixedWindowStrategy,
  slidingWindow: SlidingWindowStrategy,
  tokenBucket: TokenBucketStrategy,
};

class RateLimiter {
  /**
   * @param {object} options
   * @param {string}  options.strategy      - 'fixedWindow' | 'slidingWindow' | 'tokenBucket'
   * @param {object}  options.store         - Storage backend (MemoryStore | RedisStore)
   * @param {string}  [options.keyPrefix]   - Prefix for all keys
   * @param {object}  options.strategyOptions - Options passed to the strategy
   */
  constructor({ strategy = 'fixedWindow', store, keyPrefix = 'rl', ...strategyOptions }) {
    const StrategyClass = STRATEGIES[strategy];
    if (!StrategyClass) {
      throw new Error(`Unknown strategy: "${strategy}". Valid: ${Object.keys(STRATEGIES).join(', ')}`);
    }

    this.store = store ?? new MemoryStore();
    this.strategy = new StrategyClass(strategyOptions);
    this.keyPrefix = keyPrefix;
  }

  /**
   * Check and consume a rate limit slot for the given identifier.
   * @param {string} identifier - e.g., IP address, user ID, API key
   * @returns {Promise<RateLimitResult>}
   */
  async consume(identifier) {
    if (!identifier) throw new Error('identifier is required');

    const key = `${this.keyPrefix}:${identifier}`;

    try {
      const result = await this.strategy.check(this.store, key);
      return {
        ...result,
        identifier,
        key,
      };
    } catch (err) {
      // Fail open — allow request if store is unavailable
      console.error('[RateLimiter] Store error, failing open:', err.message);
      return {
        allowed: true,
        remaining: -1,
        retryAfterMs: 0,
        error: err.message,
      };
    }
  }

  /**
   * Reset the rate limit for a given identifier.
   * @param {string} identifier
   */
  async reset(identifier) {
    const key = `${this.keyPrefix}:${identifier}`;
    await this.store.delete(key);
  }

  /**
   * Destroy the limiter and clean up resources.
   */
  destroy() {
    if (typeof this.store.destroy === 'function') {
      this.store.destroy();
    }
  }
}

module.exports = { RateLimiter, MemoryStore, STRATEGIES };
```

## 7. Express Middleware

```javascript
// src/middleware.js
const { RateLimiter } = require('./rateLimiter');

/**
 * Creates an Express rate-limiting middleware.
 *
 * @param {object} options
 * @param {RateLimiter|object} options.limiter       - RateLimiter instance or config to create one
 * @param {function}           [options.keyGenerator] - (req) => string
 * @param {function}           [options.onLimited]    - (req, res, result) => void
 * @param {boolean}            [options.setHeaders]   - Add rate limit headers (default: true)
 * @param {string}             [options.message]      - Error message when limited
 */
function createMiddleware({
  limiter,
  keyGenerator = (req) => req.ip,
  onLimited,
  setHeaders = true,
  message = 'Too many requests, please try again later.',
} = {}) {
  // Allow passing raw config instead of a RateLimiter instance
  const rateLimiter = limiter instanceof RateLimiter ? limiter : new RateLimiter(limiter);

  return async function rateLimitMiddleware(req, res, next) {
    try {
      const identifier = keyGenerator(req);
      const result = await rateLimiter.consume(identifier);

      if (setHeaders) {
        applyHeaders(res, result);
      }

      if (!result.allowed) {
        if (typeof onLimited === 'function') {
          return onLimited(req, res, result);
        }

        return res.status(429).json({
          error: 'Too Many Requests',
          message,
          retryAfterMs: result.retryAfterMs,
        });
      }

      next();
    } catch (err) {
      // Fail open on unexpected errors
      console.error('[RateLimitMiddleware] Unexpected error:', err);
      next();
    }
  };
}

function applyHeaders(res, result) {
  if (result.remaining !== undefined) {
    res.setHeader('X-RateLimit-Remaining', result.remaining);
  }
  if (result.resetAfterMs !== undefined) {
    res.setHeader('X-RateLimit-Reset', Math.ceil(result.resetAfterMs / 1000));
  }
  if (!result.allowed && result.retryAfterMs) {
    res.setHeader('Retry-After', Math.ceil(result.retryAfterMs / 1000));
  }
}

module.exports = { createMiddleware };
```

## 8. Express Server (Demo)

```javascript
// server.js
const express = require('express');
const { RateLimiter } = require('./src/rateLimiter');
const { createMiddleware } = require('./src/middleware');

const app = express();
app.use(express.json());

// ── Strategy 1: Fixed Window ──────────────────────────────────────────────────
const fixedLimiter = new RateLimiter({
  strategy: 'fixedWindow',
  windowMs: 60_000,   // 1 minute
  maxRequests: 10,
  keyPrefix: 'fw',
});

// ── Strategy 2: Sliding Window ────────────────────────────────────────────────
const slidingLimiter = new RateLimiter({
  strategy: 'slidingWindow',
  windowMs: 60_000,
  maxRequests: 10,
  keyPrefix: 'sw',
});

// ── Strategy 3: Token Bucket ──────────────────────────────────────────────────
const tokenLimiter = new RateLimiter({
  strategy: 'tokenBucket',
  capacity: 10,        // max burst of 10
  refillRate: 2,       // 2 tokens/second
  keyPrefix: 'tb',
});

// ── Routes ────────────────────────────────────────────────────────────────────
app.get(
  '/fixed',
  createMiddleware({ limiter: fixedLimiter }),
  (req, res) => res.json({ message: 'Fixed window — request allowed' }),
);

app.get(
  '/sliding',
  createMiddleware({ limiter: slidingLimiter }),
  (req, res) => res.json({ message: 'Sliding window — request allowed' }),
);

app.get(
  '/token',
  createMiddleware({ limiter: tokenLimiter }),
  (req, res) => res.json({ message: 'Token bucket — request allowed' }),
);

// Per-user limiting (keyed by user ID from header)
const userLimiter = new RateLimiter({
  strategy: 'slidingWindow',
  windowMs: 60_000,
  maxRequests: 5,
  keyPrefix: 'user',
});

app.get(
  '/user-limited',
  createMiddleware({
    limiter: userLimiter,
    keyGenerator: (req) => req.headers['x-user-id'] ?? req.ip,
    onLimited: (req, res, result) => {
      res.status(429).json({
        error: 'Rate limit exceeded',
        retryAfterSeconds: Math.ceil(result.retryAfterMs / 1000),
      });
    },
  }),
  (req, res) => res.json({ message: 'User-scoped — request allowed' }),
);

// Manual consume (programmatic use)
app.get('/manual', async (req, res) => {
  const result = await fixedLimiter.consume(req.ip);

  if (!result.allowed) {
    return res.status(429).json({ error: 'Rate limited', result });
  }

  res.json({ message: 'Allowed', result });
});

app.listen(3000, () => console.log('Server running on http://localhost:3000'));
```

// tests/rateLimiter.test.js
const { RateLimiter } = require('../src/rateLimiter');
const MemoryStore = require('../src/store/memoryStore');

// ── Helpers ───────────────────────────────────────────────────────────────────
function sleep(ms) {
  return new Promise((resolve) => setTimeout(resolve, ms));
}

async function runRequests(limiter, id, count) {
  const results = [];
  for (let i = 0; i < count; i++) {
    results.push(await limiter.consume(id));
  }
  return results;
}

// ── Fixed Window ──────────────────────────────────────────────────────────────
describe('Fixed Window Strategy', () => {
  let limiter;

  beforeEach(() => {
    limiter = new RateLimiter({
      strategy: 'fixedWindow',
      windowMs: 500,
      maxRequests: 3,
    });
  });

  afterEach(() => limiter.destroy());

  test('allows requests within limit', async () => {
    const results = await runRequests(limiter, 'user1', 3);
    expect(results.every((r) => r.allowed)).toBe(true);
  });

  test('blocks requests over limit', async () => {
    const results = await runRequests(limiter, 'user1', 5);
    const allowed = results.filter((r) => r.allowed).length;
    const blocked = results.filter((r) => !r.allowed).length;
    expect(allowed).toBe(3);
    expect(blocked).toBe(2);
  });

  test('resets after window expires', async () => {
    await runRequests(limiter, 'user1', 3);
    await sleep(600);
    const result = await limiter.consume('user1');
    expect(result.allowed).toBe(true);
  });

  test('isolates different identifiers', async () => {
    await runRequests(limiter, 'user1', 3);
    const result = await limiter.consume('user2');
    expect(result.allowed).toBe(true);
  });

  test('reset() clears the limit', async () => {
    await runRequests(limiter, 'user1', 3);
    await limiter.reset('user1');
    const result = await limiter.consume('user1');
    expect(result.allowed).toBe(true);

## Strategy Comparison

| Strategy | Accuracy | Memory | Burst Handling | Best For |
|---|---|---|---|---|
| **Fixed Window** | Low | Low | Poor (boundary bursts) | Simple APIs |
| **Sliding Window** | High | Medium | Good | General purpose |
| **Token Bucket** | High | Low | Excellent (controlled) | APIs with burst allowance |

## Key Design Decisions

- **Fail Open** — if the store is unavailable, requests are allowed through to avoid cascading failures
- **Pluggable Store** — swap `MemoryStore` for `RedisStore` for distributed/multi-instance deployments
- **Atomic Operations** — Redis store uses Lua scripts to prevent race conditions
- **TTL Cleanup** — Memory store auto-purges expired keys every 60 seconds
- **Standard Headers** — Sets `X-RateLimit-Remaining`, `X-RateLimit-Reset`, and `Retry-After`
