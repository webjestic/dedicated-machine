## 1. Memory Store

```javascript
// src/stores/memoryStore.js
class MemoryStore {
  constructor() {
    this.store = new Map();
    this.cleanupInterval = setInterval(() => this.cleanup(), 60_000);
    // Prevent the interval from blocking process exit
    if (this.cleanupInterval.unref) {
      this.cleanupInterval.unref();
    }
  }

  async get(key) {
    return this.store.get(key) ?? null;
  }

  async set(key, value, ttlMs) {
    this.store.set(key, {
      value,
      expiresAt: Date.now() + ttlMs,
    });
  }

  async increment(key, ttlMs) {
    const entry = this.store.get(key);
    const now = Date.now();

    if (!entry || entry.expiresAt <= now) {
      this.store.set(key, { value: 1, expiresAt: now + ttlMs });
      return 1;
    }

    entry.value += 1;
    return entry.value;
  }

  async delete(key) {
    this.store.delete(key);
  }

  cleanup() {
    const now = Date.now();
    for (const [key, entry] of this.store.entries()) {
      if (entry.expiresAt <= now) {
        this.store.delete(key);
      }
    }
  }

  destroy() {
    clearInterval(this.cleanupInterval);
    this.store.clear();
  }
}

module.exports = MemoryStore;
```

// src/stores/redisStore.js
class RedisStore {
  /**
   * @param {import('ioredis').Redis} client - ioredis client instance
   */
  constructor(client) {
    if (!client) throw new Error('Redis client is required');
    this.client = client;
  }

  async get(key) {
    const value = await this.client.get(key);
    return value ? JSON.parse(value) : null;
  }

  async set(key, value, ttlMs) {
    await this.client.set(key, JSON.stringify(value), 'PX', ttlMs);
  }

  async increment(key, ttlMs) {
    // Atomic increment using a Lua script
    const script = `
      local current = redis.call('GET', KEYS[1])
      if current == false then
        redis.call('SET', KEYS[1], 1, 'PX', ARGV[1])
        return 1
      else
        return redis.call('INCR', KEYS[1])
      end
    `;
    return await this.client.eval(script, 1, key, ttlMs);
  }

  async delete(key) {
    await this.client.del(key);
  }
}

module.exports = RedisStore;

## 3. Fixed Window Strategy

```javascript
// src/strategies/fixedWindow.js

/**
 * Fixed Window Rate Limiting
 * Simple counter that resets at fixed time intervals.
 * Con: Allows burst traffic at window boundaries.
 */
class FixedWindowStrategy {
  /**
   * @param {object} options
   * @param {number} options.windowMs   - Window size in milliseconds
   * @param {number} options.maxRequests - Max requests per window
   * @param {object} options.store      - Storage backend
   */
  constructor({ windowMs, maxRequests, store }) {
    this.windowMs = windowMs;
    this.maxRequests = maxRequests;
    this.store = store;
  }

  async check(key) {
    const count = await this.store.increment(key, this.windowMs);
    const remaining = Math.max(0, this.maxRequests - count);
    const allowed = count <= this.maxRequests;

    return {
      allowed,
      count,
      remaining,
      resetAfterMs: this.windowMs,
      retryAfterMs: allowed ? null : this.windowMs,
    };
  }

  async reset(key) {
    await this.store.delete(key);
  }
}

module.exports = FixedWindowStrategy;
```

## 4. Sliding Window Strategy

```javascript
// src/strategies/slidingWindow.js

/**
 * Sliding Window Log Rate Limiting
 * Tracks individual request timestamps for precise rate limiting.
 * More memory-intensive but eliminates boundary bursts.
 */
class SlidingWindowStrategy {
  constructor({ windowMs, maxRequests, store }) {
    this.windowMs = windowMs;
    this.maxRequests = maxRequests;
    this.store = store;
  }

  async check(key) {
    const now = Date.now();
    const windowStart = now - this.windowMs;

    // Retrieve existing timestamps
    const data = await this.store.get(key);
    let timestamps = data?.timestamps ?? [];

    // Remove expired timestamps outside the window
    timestamps = timestamps.filter((ts) => ts > windowStart);

    // Add current request timestamp
    timestamps.push(now);

    // Persist updated timestamps
    await this.store.set(key, { timestamps }, this.windowMs);

    const count = timestamps.length;
    const allowed = count <= this.maxRequests;
    const remaining = Math.max(0, this.maxRequests - count);

    // Time until the oldest request falls outside the window
    const retryAfterMs =
      !allowed && timestamps.length > 0
        ? timestamps[0] + this.windowMs - now
        : null;

    return {
      allowed,
      count,
      remaining,
      resetAfterMs: timestamps[0] ? timestamps[0] + this.windowMs - now : this.windowMs,
      retryAfterMs,
    };
  }

  async reset(key) {
    await this.store.delete(key);
  }
}

module.exports = SlidingWindowStrategy;
```

## 5. Token Bucket Strategy

```javascript
// src/strategies/tokenBucket.js

/**
 * Token Bucket Rate Limiting
 * Allows controlled bursting while maintaining an average rate.
 * Tokens refill at a steady rate up to a maximum capacity.
 */
class TokenBucketStrategy {
  /**
   * @param {object} options
   * @param {number} options.capacity      - Max tokens (burst limit)
   * @param {number} options.refillRate    - Tokens added per second
   * @param {number} options.refillIntervalMs - How often to refill (ms)
   * @param {object} options.store
   */
  constructor({ capacity, refillRate, refillIntervalMs = 1000, store }) {
    this.capacity = capacity;
    this.refillRate = refillRate;
    this.refillIntervalMs = refillIntervalMs;
    this.store = store;
  }

  async check(key) {
    const now = Date.now();
    const data = await this.store.get(key);

    let tokens;
    let lastRefill;

    if (!data) {
      // New bucket starts full
      tokens = this.capacity;
      lastRefill = now;
    } else {
      tokens = data.tokens;
      lastRefill = data.lastRefill;

      // Calculate tokens to add since last refill
      const elapsed = now - lastRefill;
      const intervals = Math.floor(elapsed / this.refillIntervalMs);
      const tokensToAdd = intervals * this.refillRate;

      if (tokensToAdd > 0) {
        tokens = Math.min(this.capacity, tokens + tokensToAdd);
        lastRefill = lastRefill + intervals * this.refillIntervalMs;
      }
    }

    const allowed = tokens >= 1;

    if (allowed) {
      tokens -= 1;
    }

    // TTL: time to refill the entire bucket from empty
    const ttlMs = (this.capacity / this.refillRate) * this.refillIntervalMs * 2;
    await this.store.set(key, { tokens, lastRefill }, ttlMs);

    const msPerToken = this.refillIntervalMs / this.refillRate;
    const retryAfterMs = allowed ? null : Math.ceil(msPerToken);

    return {
      allowed,
      count: this.capacity - tokens,
      remaining: Math.floor(tokens),
      resetAfterMs: msPerToken,
      retryAfterMs,
    };
  }

  async reset(key) {
    await this.store.delete(key);
  }
}

module.exports = TokenBucketStrategy;
```

## 6. Core Rate Limiter

```javascript
// src/rateLimiter.js
const MemoryStore = require('./stores/memoryStore');
const FixedWindowStrategy = require('./strategies/fixedWindow');
const SlidingWindowStrategy = require('./strategies/slidingWindow');
const TokenBucketStrategy = require('./strategies/tokenBucket');

const STRATEGIES = {
  fixedWindow: FixedWindowStrategy,
  slidingWindow: SlidingWindowStrategy,
  tokenBucket: TokenBucketStrategy,
};

class RateLimiter {
  /**
   * @param {object}   options
   * @param {string}   [options.strategy='fixedWindow']  - Rate limiting strategy
   * @param {number}   [options.windowMs=60000]          - Window in ms (fixed/sliding)
   * @param {number}   [options.maxRequests=100]         - Max requests per window
   * @param {number}   [options.capacity=10]             - Token bucket capacity
   * @param {number}   [options.refillRate=1]            - Token bucket refill rate/s
   * @param {object}   [options.store]                   - Custom store (default: MemoryStore)
   * @param {function} [options.keyGenerator]            - Custom key function (req) => string
   * @param {function} [options.onLimitReached]          - Hook called when limit is hit
   * @param {boolean}  [options.skipFailedRequests]      - Don't count failed responses
   * @param {number[]} [options.skipSuccessfulRequests]  - Don't count these status codes
   */
  constructor(options = {}) {
    const {
      strategy = 'fixedWindow',
      windowMs = 60_000,
      maxRequests = 100,
      capacity = 10,
      refillRate = 1,
      refillIntervalMs = 1000,
      store,
      keyGenerator,
      onLimitReached,
      skipFailedRequests = false,
    } = options;

    if (!STRATEGIES[strategy]) {
      throw new Error(
        `Unknown strategy: "${strategy}". Choose from: ${Object.keys(STRATEGIES).join(', ')}`
      );
    }

    this.store = store ?? new MemoryStore();
    this.keyGenerator = keyGenerator ?? this._defaultKeyGenerator;
    this.onLimitReached = onLimitReached ?? null;
    this.skipFailedRequests = skipFailedRequests;

    const StrategyClass = STRATEGIES[strategy];

    // Pass relevant options to each strategy
    this.strategy = new StrategyClass({
      windowMs,
      maxRequests,
      capacity,
      refillRate,
      refillIntervalMs,
      store: this.store,
    });
  }

  _defaultKeyGenerator(req) {
    // Support proxies by checking forwarded headers
    const forwarded = req.headers?.['x-forwarded-for'];
    const ip = forwarded
      ? forwarded.split(',')[0].trim()
      : req.socket?.remoteAddress ?? 'unknown';
    return `rl:${ip}`;
  }

  async consume(key) {
    return this.strategy.check(key);
  }

  async reset(key) {
    return this.strategy.reset(key);
  }
}

module.exports = RateLimiter;
```

## 7. Express Middleware

```javascript
// src/middleware.js
const RateLimiter = require('./rateLimiter');

/**
 * Creates an Express-compatible rate limiting middleware.
 * @param {object} options - RateLimiter options + middleware-specific options
 * @returns {function} Express middleware
 */
function createRateLimitMiddleware(options = {}) {
  const {
    statusCode = 429,
    message = 'Too many requests, please try again later.',
    headers = true,
    handler,
    ...limiterOptions
  } = options;

  const limiter = new RateLimiter(limiterOptions);

  return async function rateLimitMiddleware(req, res, next) {
    try {
      const key = limiter.keyGenerator(req);
      const result = await limiter.consume(key);

      if (headers) {
        res.setHeader('X-RateLimit-Limit',
          limiterOptions.maxRequests ?? limiterOptions.capacity ?? 100);
        res.setHeader('X-RateLimit-Remaining', result.remaining);
        res.setHeader('X-RateLimit-Reset',
          Math.ceil((Date.now() + result.resetAfterMs) / 1000));

        if (!result.allowed && result.retryAfterMs) {
          res.setHeader('Retry-After', Math.ceil(result.retryAfterMs / 1000));
        }
      }

      if (!result.allowed) {
        if (limiter.onLimitReached) {
          limiter.onLimitReached(req, res, result);
        }

        if (handler) {
          return handler(req, res, next, result);
        }

        return res.status(statusCode).json({
          error: message,
          retryAfter: result.retryAfterMs
            ? Math.ceil(result.retryAfterMs / 1000)
            : null,
        });
      }

      // Optionally undo count for failed responses
      if (limiter.skipFailedRequests) {
        res.on('finish', async () => {
          if (res.statusCode >= 400) {
            await limiter.reset(key);
          }
        });
      }

      next();
    } catch (err) {
      // Fail open — don't block requests if the limiter errors
      console.error('[RateLimiter] Error:', err);
      next();
    }
  };
}

module.exports = { createRateLimitMiddleware, RateLimiter };
```

## 8. Example Express Server

```javascript
// server.js
const express = require('express');
const { createRateLimitMiddleware } = require('./src/middleware');

const app = express();
app.use(express.json());

// ── Global limiter (fixed window) ────────────────────────────────────────────
const globalLimiter = createRateLimitMiddleware({
  strategy: 'fixedWindow',
  windowMs: 60_000,   // 1 minute
  maxRequests: 100,
  message: 'Global rate limit exceeded.',
});

// ── Auth limiter (sliding window — stricter) ──────────────────────────────────
const authLimiter = createRateLimitMiddleware({
  strategy: 'slidingWindow',
  windowMs: 15 * 60_000,  // 15 minutes
  maxRequests: 10,
  message: 'Too many login attempts.',
  keyGenerator: (req) => `auth:${req.ip}`,
});

// ── API limiter (token bucket — allows bursting) ──────────────────────────────
const apiLimiter = createRateLimitMiddleware({
  strategy: 'tokenBucket',
  capacity: 20,       // burst up to 20
  refillRate: 5,      // refill 5 tokens/second
  keyGenerator: (req) => `api:${req.headers['x-api-key'] ?? req.ip}`,
});

// Apply global limiter to all routes
app.use(globalLimiter);

app.post('/auth/login', authLimiter, (req, res) => {
  res.json({ message: 'Login successful' });
});

app.get('/api/data', apiLimiter, (req, res) => {
  res.json({ data: 'Here is your data', timestamp: Date.now() });
});

app.get('/health', (req, res) => {
  res.json({ status: 'ok' });
});

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => console.log(`Server running on port ${PORT}`));

module.exports = app;
```

## Strategy Comparison

| Strategy | Accuracy | Memory | Burst Handling | Best For |
|---|---|---|---|---|
| **Fixed Window** | Low | Low | Poor (boundary spikes) | Simple global limits |
| **Sliding Window** | High | Medium | Good | API endpoints |
| **Token Bucket** | High | Low | Excellent | Bursty traffic |

## Key Design Decisions

- **Fail open** — if the store errors, requests are allowed through (availability > strict limiting)
- **Atomic increments** — Redis store uses a Lua script to prevent race conditions
- **Pluggable stores** — swap `MemoryStore` for `RedisStore` in distributed environments
- **Standard headers** — sets `X-RateLimit-*` and `Retry-After` headers per RFC 6585
- **Proxy-aware** — reads `X-Forwarded-For` for real client IPs behind load balancers
