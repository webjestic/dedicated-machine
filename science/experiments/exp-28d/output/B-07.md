# Production Rate Limiter in Node.js

Let me work through this properly — requirements first, then implementation, then operational gaps.

---

## Phase 1: Requirements Design

### Functional Requirements
- Limit requests per identifier (IP, user ID, API key) within a sliding time window
- Return standard rate limit headers on every response
- Allow configurable limits per route or client tier
- Reject requests that exceed limits with proper HTTP 429 responses
- Support multiple algorithms (token bucket for burst tolerance, sliding window log for precision)

### Non-Functional Requirements
- Sub-millisecond overhead on the hot path
- Survive Redis failures without taking down the application (fail-open vs fail-closed is a decision, not a default)
- Consistent behavior across multiple application instances
- No memory leaks under sustained load

### Explicit Design Decisions Made Here
| Decision | Choice | Reason |
|---|---|---|
| Algorithm | Sliding window counter | Balance of accuracy vs memory vs atomicity |
| Storage | Redis with in-memory fallback | Distributed consistency, degradation path |
| Failure mode | Fail-open with alerting | Availability over strict limiting |
| Identifier | API key → User ID → IP (priority) | Avoid punishing shared IPs (NAT, offices) |
| Headers | `RateLimit-*` (IETF draft-6) + `Retry-After` | Modern standard |

### `scripts/lua/slidingWindow.lua`
```lua
-- Atomic sliding window counter in Redis
-- This runs as a single atomic operation — no race conditions across instances
-- KEYS[1] = rate limit key
-- ARGV[1] = current timestamp (milliseconds)
-- ARGV[2] = window size (milliseconds)
-- ARGV[3] = max requests allowed
-- ARGV[4] = TTL for the key (seconds)

local key = KEYS[1]
local now = tonumber(ARGV[1])
local window = tonumber(ARGV[2])
local limit = tonumber(ARGV[3])
local ttl = tonumber(ARGV[4])

-- Remove entries outside the current window
local windowStart = now - window
redis.call('ZREMRANGEBYSCORE', key, '-inf', windowStart)

-- Count requests in the current window
local count = redis.call('ZCARD', key)

if count < limit then
  -- Add current request with timestamp as both score and member
  -- Appending a random suffix handles multiple requests at identical millisecond timestamps
  redis.call('ZADD', key, now, now .. '-' .. math.random(1, 1000000))
  redis.call('EXPIRE', key, ttl)
  return {1, count + 1, limit}  -- {allowed, current_count, limit}
else
  -- Get oldest entry to calculate retry-after
  local oldest = redis.call('ZRANGE', key, 0, 0, 'WITHSCORES')
  local retryAfter = 0
  if #oldest > 0 then
    retryAfter = math.ceil((tonumber(oldest[2]) + window - now) / 1000)
  end
  return {0, count, limit, retryAfter}  -- {denied, current_count, limit, retry_after_seconds}
end
```

```javascript
'use strict';

const { createClient } = require('redis');
const fs = require('fs');
const path = require('path');

class RedisStore {
  constructor(config = {}) {
    this.config = {
      url: config.url || process.env.REDIS_URL || 'redis://localhost:6379',
      connectTimeout: config.connectTimeout || 5000,
      commandTimeout: config.commandTimeout || 500, // Critical: don't let Redis slow your API
      retryAttempts: config.retryAttempts || 3,
      keyPrefix: config.keyPrefix || 'rl:',
      ...config,
    };

    this.client = null;
    this.isConnected = false;
    this.luaScript = null;
    this.scriptSha = null;

    // Metrics — wire these to your observability stack
    this.metrics = {
      hits: 0,
      misses: 0,
      errors: 0,
      timeouts: 0,
    };
  }

  async connect() {
    this.client = createClient({
      url: this.config.url,
      socket: {
        connectTimeout: this.config.connectTimeout,
        commandTimeout: this.config.commandTimeout,
        reconnectStrategy: (retries) => {
          if (retries > this.config.retryAttempts) {
            // Stop retrying — caller handles fallback
            return new Error('Redis max reconnection attempts reached');
          }
          // Exponential backoff: 100ms, 200ms, 400ms
          return Math.min(retries * 100, 3000);
        },
      },
    });

    this.client.on('error', (err) => {
      this.metrics.errors++;
      // Do NOT throw here — event emitter errors crash the process if unhandled
      console.error('[RedisStore] Redis client error:', err.message);
      this.isConnected = false;
    });

    this.client.on('connect', () => {
      this.isConnected = true;
      console.info('[RedisStore] Redis connected');
    });

    this.client.on('reconnecting', () => {
      console.warn('[RedisStore] Redis reconnecting...');
    });

    await this.client.connect();
    await this._loadLuaScript();
  }

  async _loadLuaScript() {
    const scriptPath = path.join(__dirname, '../../scripts/lua/slidingWindow.lua');
    this.luaScript = fs.readFileSync(scriptPath, 'utf8');

    // Load script into Redis and store SHA for EVALSHA (more efficient than EVAL every time)
    this.scriptSha = await this.client.scriptLoad(this.luaScript);
    console.info('[RedisStore] Lua script loaded, SHA:', this.scriptSha);
  }

  async increment(key, windowMs, limit) {
    if (!this.isConnected || !this.client) {
      throw new Error('Redis not connected');
    }

    const prefixedKey = this.config.keyPrefix + key;
    const now = Date.now();
    const ttlSeconds = Math.ceil(windowMs / 1000) + 1; // +1 buffer

    try {
      // EVALSHA is faster than EVAL — uses cached compiled script
      const result = await this.client.evalSha(this.scriptSha, {
        keys: [prefixedKey],
        arguments: [
          String(now),
          String(windowMs),
          String(limit),
          String(ttlSeconds),
        ],
      });

      return this._parseResult(result);
    } catch (err) {
      // NOSCRIPT means Redis was restarted and lost the cached script — reload and retry once
      if (err.message && err.message.includes('NOSCRIPT')) {
        console.warn('[RedisStore] Script cache miss, reloading...');
        await this._loadLuaScript();

        const result = await this.client.evalSha(this.scriptSha, {
          keys: [prefixedKey],
          arguments: [
            String(now),
            String(windowMs),
            String(limit),
            String(ttlSeconds),
          ],
        });

        return this._parseResult(result);
      }

      this.metrics.errors++;
      throw err;
    }
  }

  _parseResult(result) {
    const allowed = result[0] === 1;
    const current = Number(result[1]);
    const limit = Number(result[2]);
    const retryAfter = result[3] ? Number(result[3]) : 0;

    return { allowed, current, limit, retryAfter };
  }

  async disconnect() {
    if (this.client) {
      await this.client.quit();
      this.isConnected = false;
    }
  }

  getMetrics() {
    return { ...this.metrics };
  }

  isHealthy() {
    return this.isConnected;
  }
}

module.exports = RedisStore;
```

### `src/store/memoryStore.js`
```javascript
'use strict';

// In-memory fallback store — used when Redis is unavailable
// IMPORTANT: This does NOT share state across processes/instances
// It is a degraded mode, not an equivalent replacement
// Under high load with many workers, limits will be N times more permissive
// where N = number of worker processes

class MemoryStore {
  constructor(config = {}) {
    // Map of key -> { requests: [timestamps], lastCleanup: timestamp }
    this.store = new Map();

    // Prevent unbounded memory growth under attack or high cardinality keys
    this.maxKeys = config.maxKeys || 100000;

    // Periodic cleanup of expired windows
    this.cleanupIntervalMs = config.cleanupIntervalMs || 60000;
    this._cleanupTimer = null;

    this.metrics = {
      hits: 0,
      evictions: 0,
    };
  }

  start() {
    this._cleanupTimer = setInterval(() => {
      this._cleanup();
    }, this.cleanupIntervalMs);

    // Don't prevent process exit
    if (this._cleanupTimer.unref) {
      this._cleanupTimer.unref();
    }
  }

  stop() {
    if (this._cleanupTimer) {
      clearInterval(this._cleanupTimer);
      this._cleanupTimer = null;
    }
  }

  increment(key, windowMs, limit) {
    const now = Date.now();
    const windowStart = now - windowMs;

    if (!this.store.has(key)) {
      // Enforce max key limit — evict oldest entry if at capacity
      if (this.store.size >= this.maxKeys) {
        const oldestKey = this.store.keys().next().value;
        this.store.delete(oldestKey);
        this.metrics.evictions++;
      }
      this.store.set(key, { requests: [] });
    }

    const entry = this.store.get(key);

    // Slide the window — remove timestamps outside the window
    entry.requests = entry.requests.filter((ts) => ts > windowStart);

    const current = entry.requests.length;

    if (current < limit) {
      entry.requests.push(now);
      this.metrics.hits++;
      return { allowed: true, current: current + 1, limit, retryAfter: 0 };
    }

    // Calculate when the oldest request will fall out of the window
    const oldestInWindow = entry.requests[0];
    const retryAfter = Math.ceil((oldestInWindow + windowMs - now) / 1000);

    return {
      allowed: false,
      current,
      limit,
      retryAfter: Math.max(retryAfter, 1),
    };
  }

  _cleanup() {
    const now = Date.now();
    let cleaned = 0;

    for (const [key, entry] of this.store.entries()) {
      // If the entry has no recent requests, remove it entirely
      if (entry.requests.length === 0) {
        this.store.delete(key);
        cleaned++;
      }
    }

    if (cleaned > 0) {
      console.debug(`[MemoryStore] Cleaned ${cleaned} expired entries`);
    }
  }

  isHealthy() {
    return true; // Always available — but degraded
  }

  getMetrics() {
    return {
      ...this.metrics,
      keyCount: this.store.size,
    };
  }
}

module.exports = MemoryStore;
```

```javascript
const TIERS = {
  // Unauthenticated requests — strictest limits

    limit: 1000,

  internal: {
    windowMs: 60 * 1000,
    limit: 10000,
    burstLimit: 500,
  },
};

// Per-route overrides — more specific rules take precedence
const ROUTE_OVERRIDES = {
  // Auth endpoints need stricter limits — primary target for credential stuffing
  'POST /auth/login': {
    anonymous: { windowMs: 15 * 60 * 1000, limit: 5 },
    standard: { windowMs: 15 * 60 * 1000, limit: 10 },
  },

  'POST /auth/register': {
    anonymous: { windowMs: 60 * 60 * 1000, limit: 3 },

  'POST /auth/reset-password': {

  // Expensive endpoints
```

### `src/utils/identifierExtractor.js`
```javascript
'use strict';

// Extract the best available identifier for rate limiting
// Priority: API Key > Authenticated User ID > IP Address
// Each has different trust levels and abuse surface areas

function extractIdentifier(req) {
  // 1. API Key — most reliable, tied to a specific client
  const apiKey = req.headers['x-api-key'];
  if (apiKey && isValidApiKeyFormat(apiKey)) {
    return { id: `apikey:${apiKey}`, tier: req.apiKeyTier || 'standard' };
  }

  // 2. Authenticated user from JWT/session middleware
  // Assumes upstream auth middleware sets req.user
  if (req.user && req.user.id) {
    return {
      id: `user:${req.user.id}`,
      tier: req.user.tier || 'standard',
    };
  }

  // 3. IP address — least reliable, but always available
  // WARNING: X-Forwarded-For can be spoofed — only trust it if you control
  // the proxy layer. If behind a load balancer you own, this is fine.
  // If accepting arbitrary client headers, use req.socket.remoteAddress only.
  const ip = extractIP(req);
  return { id: `ip:${ip}`, tier: 'anonymous' };
}

function extractIP(req) {
  // Trust X-Forwarded-For only if explicitly configured to do so
  // In production behind a known proxy, use the leftmost non-trusted IP
  const forwardedFor = req.headers['x-forwarded-for'];
  if (forwardedFor && process.env.TRUST_PROXY === 'true') {
    // Take the first IP — leftmost is the original client
    const ips = forwardedFor.split(',').map((ip) => ip.trim());
    return ips[0];
  }

  return req.socket?.remoteAddress || req.connection?.remoteAddress || 'unknown';
}

function isValidApiKeyFormat(key) {
  // Basic sanity check — prevents using garbage values as cache keys
  // Adjust regex to match your actual API key format
  return typeof key === 'string' && /^[a-zA-Z0-9_-]{20,64}$/.test(key);
}

function buildRateLimitKey(identifier, routeKey, windowMs) {
  // Include window size in key so different window configs don't collide
  return `${identifier}:${routeKey}:${windowMs}`;
}

module.exports = { extractIdentifier, extractIP, buildRateLimitKey };
```

### `src/utils/headerBuilder.js`

```javascript
'use strict';

// IETF draft-ietf-httpapi-ratelimit-headers-06
// https://datatracker.ietf.org/doc/draft-ietf-httpapi-ratelimit-headers/
// Using the modern RateLimit-* headers, not the non-standard X-RateLimit-* variants
// Include both for maximum client compatibility during transition

function buildRateLimitHeaders(result, windowMs) {
  const resetTimestamp = Math.ceil((Date.now() + windowMs) / 1000);
  const remaining = Math.max(result.limit - result.current, 0);

  const headers = {
    // Modern IETF headers
    'RateLimit-Limit': String(result.limit),
    'RateLimit-Remaining': String(remaining),
    'RateLimit-Reset': String(resetTimestamp),

    // Legacy X- headers — many clients still expect these
    'X-RateLimit-Limit': String(result.limit),
    'X-RateLimit-Remaining': String(remaining),
    'X-RateLimit-Reset': String(resetTimestamp),
  };

  if (!result.allowed) {
    // RFC 6585 — required on 429 responses
    headers['Retry-After'] = String(result.retryAfter || 1);
  }

  return headers;
}

function applyHeaders(res, headers) {
  for (const [key, value] of Object.entries(headers)) {
    res.setHeader(key, value);
  }
}

module.exports = { buildRateLimitHeaders, applyHeaders };
```

### `src/middleware/rateLimiter.js`
```javascript
'use strict';

const RedisStore = require('../store/redisStore');
const MemoryStore = require('../store/memoryStore');
const { getTierConfig } = require('../config/tiers');
const { extractIdentifier, buildRateLimitKey } = require('../utils/identifierExtractor');
const { buildRateLimitHeaders, applyHeaders } = require('../utils/headerBuilder');

class RateLimiter {
  constructor(config = {}) {
    this.config = {
      failOpen: config.failOpen !== false, // Default: fail open (allow requests on error)
      skipSuccessfulRequests: config.skipSuccessfulRequests || false,
      skipFailedRequests: config.skipFailedRequests || false,
      keyGenerator: config.keyGenerator || null, // Custom key override
      onRateLimited: config.onRateLimited || null, // Hook for logging/alerting
      onError: config.onError || null,             // Hook for error observability
      ...config,
    };

    // Primary store — Redis for distributed consistency
    this.primaryStore = config.redisStore || null;

    // Fallback store — in-memory, single-process only
    this.fallbackStore = new MemoryStore({
      maxKeys: config.maxMemoryKeys || 100000,
    });
    this.fallbackStore.start();

    // Track whether we're in degraded mode
    this.usingFallback = false;

    // Metrics for observability
    this.metrics = {
      allowed: 0,
      blocked: 0,
      errors: 0,
      fallbackActivations: 0,
    };
  }

  // Returns Express middleware function
  middleware() {
    return async (req, res, next) => {
      try {
        await this._processRequest(req, res, next);
      } catch (err) {
        // This should never be reached — _processRequest handles its own errors
        // But we catch here as a last resort to prevent unhandled rejections
        console.error('[RateLimiter] Unhandled error in middleware:', err);
        this.metrics.errors++;

        if (this.config.failOpen) {
          next();
        } else {
          res.status(503).json({ error: 'Rate limiter unavailable' });
        }
      }
    };
  }

  async _processRequest(req, res, next) {
    // Extract identifier and tier
    const { id, tier } = this.config.keyGenerator
      ? this.config.keyGenerator(req)
      : extractIdentifier(req);

    // Build route key for per-route overrides
    const routeKey = `${req.method} ${req.route?.path || req.path}`;

    // Get config for this tier + route combination
    const tierConfig = getTierConfig(tier, routeKey);
    const { windowMs, limit } = tierConfig;

    // Build the actual Redis/memory key
    const rateLimitKey = buildRateLimitKey(id, routeKey, windowMs);

    // Attempt rate limit check
    let result;
    let storeUsed = 'primary';

    try {
      if (this.primaryStore && this.primaryStore.isHealthy()) {
        result = await this._withTimeout(
          this.primaryStore.increment(rateLimitKey, windowMs, limit),
          this.config.redisTimeoutMs || 500
        );
        this.usingFallback = false;
      } else {
        throw new Error('Primary store unavailable');
      }
    } catch (err) {
      // Primary store failed — fall back to memory store
      storeUsed = 'fallback';

      if (!this.usingFallback) {
        this.usingFallback = true;
        this.metrics.fallbackActivations++;
        console.warn('[RateLimiter] Falling back to memory store:', err.message);

        // This is where you'd fire an alert to PagerDuty/OpsGenie
        if (this.config.onError) {
          this.config.onError(err, { tier, routeKey, id });
        }
      }

      if (this.config.failOpen) {
        // Allow the request through without counting it
        // Preferred for most APIs — Redis outage shouldn't cause user-facing errors
        // But document this decision: attackers can exploit Redis outages
        result = { allowed: true, current: 0, limit, retryAfter: 0 };
      } else {
        // Use memory store — imprecise across instances but better than nothing
        result = this.fallbackStore.increment(rateLimitKey, windowMs, limit);
      }
    }

    // Build and apply rate limit headers — send on EVERY response, not just 429s
    const headers = buildRateLimitHeaders(result, windowMs);
    applyHeaders(res, headers);

    // Attach metadata to request for downstream logging
    req.rateLimitInfo = {
      tier,
      key: rateLimitKey,
      result,
      storeUsed,
    };

    if (result.allowed) {
      this.metrics.allowed++;
      next();
    } else {
      this.metrics.blocked++;

      if (this.config.onRateLimited) {
        this.config.onRateLimited(req, res, result);
      }

      // Log blocked requests — these are signals, not noise
      console.warn('[RateLimiter] Request blocked', {
        id,
        tier,
        routeKey,
        current: result.current,
        limit: result.limit,
        retryAfter: result.retryAfter,
        ip: req.socket?.remoteAddress,
      });

      res.status(429).json({
        error: 'Too Many Requests',
        message: `Rate limit exceeded. Try again in ${result.retryAfter} seconds.`,
        retryAfter: result.retryAfter,
      });
    }
  }

  // Prevent slow Redis from blocking your entire request pipeline
  _withTimeout(promise, timeoutMs) {
    return new Promise((resolve, reject) => {
      const timer = setTimeout(() => {
        reject(new Error(`Rate limiter store timeout after ${timeoutMs}ms`));
      }, timeoutMs);

      promise
        .then((result) => {
          clearTimeout(timer);
          resolve(result);
        })
        .catch((err) => {
          clearTimeout(timer);
          reject(err);
        });
    });
  }

  getMetrics() {
    return {
      ...this.metrics,
      usingFallback: this.usingFallback,
      primaryStoreHealthy: this.primaryStore?.isHealthy() || false,
      fallbackStoreMetrics: this.fallbackStore.getMetrics(),
    };
  }

  async shutdown() {
    this.fallbackStore.stop();
    if (this.primaryStore) {
      await this.primaryStore.disconnect();
    }
  }
}

module.exports = RateLimiter;
```

### `src/algorithms/tokenBucket.js`
```javascript
'use strict';

// Token Bucket — better for burst-tolerant APIs
// Tokens refill at a steady rate; clients can save up burst capacity
// Use this when you want to allow short bursts but control sustained rate
// Trade-off vs sliding window: slightly more complex, allows bursts

class TokenBucket {
  constructor(config = {}) {
    this.capacity = config.capacity || 100;       // Max tokens (burst ceiling)
    this.refillRate = config.refillRate || 10;    // Tokens added per second
    this.store = config.store;                    // Must be a store with get/set
  }

  async consume(key, tokensRequested = 1) {
    const now = Date.now();
    const bucket = await this._getBucket(key, now);

    // Refill tokens based on elapsed time
    const elapsed = (now - bucket.lastRefill) / 1000; // seconds
    const tokensToAdd = elapsed * this.refillRate;
    bucket.tokens = Math.min(this.capacity, bucket.tokens + tokensToAdd);
    bucket.lastRefill = now;

    if (bucket.tokens >= tokensRequested) {
      bucket.tokens -= tokensRequested;
      await this._saveBucket(key, bucket);
      return {
        allowed: true,
        remaining: Math.floor(bucket.tokens),
        limit: this.capacity,
        retryAfter: 0,
      };
    }

    // Calculate when enough tokens will be available
    const tokensNeeded = tokensRequested - bucket.tokens;
    const retryAfter = Math.ceil(tokensNeeded / this.refillRate);

    await this._saveBucket(key, bucket);
    return {
      allowed: false,
      remaining: 0,
      limit: this.capacity,
      retryAfter,
    };
  }

  async _getBucket(key, now) {
    // Implementation depends on store interface
    // Left as interface definition — wire to Redis hash or memory map
    const existing = await this.store.get(key);
    if (!existing) {
      return { tokens: this.capacity, lastRefill: now };
    }
    return existing;
  }

  async _saveBucket(key, bucket) {
    const ttl = Math.ceil(this.capacity / this.refillRate) + 60;
    await this.store.set(key, bucket, ttl);
  }
}

module.exports = TokenBucket;
```

'use strict';

const MemoryStore = require('../../src/store/memoryStore');

describe('MemoryStore - Sliding Window Counter', () => {
  let store;

  beforeEach(() => {
    store = new MemoryStore({ maxKeys: 1000 });
    store.start();
  });

  afterEach(() => {
    store.stop();
  });

  test('allows requests under the limit', () => {

    expect(result.current).toBe(1);
    expect(result.limit).toBe(10);
  });

  test('blocks requests at the limit', () => {

      store.increment('test-key', 60000, 10);
    }
    const result = store.increment('test-key', 60000, 10);
    expect(result.allowed).toBe(false);
    expect(result.current).toBe(10);
  });

  test('resets count after window expires', async () => {
    const shortWindow = 100; // 100ms window
    for (let i = 0; i < 5; i++) {
      store.increment('test-key', shortWindow, 5);
    }

    // Window expires
    await new Promise((resolve) => setTimeout(resolve, 150));

    const result = store.increment('test-key', shortWindow, 5);
    expect(result.allowed).toBe(true);
    expect(result.current).toBe(1);
  });
