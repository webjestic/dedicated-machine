**Decision: Request Changes**

**Summary:** This PR introduces a critical event loop blocking issue that will cause application-wide paralysis under checkout load. `requests.post()` is a synchronous blocking call inside an `async def` route — during a 5-second timeout, the event loop is completely frozen and *every* concurrent request in the service is starved, not just this one. Replace `requests` with `httpx.AsyncClient` (or `aiohttp`) and `await` the call. The tests pass today precisely because `TestClient` runs synchronously and will never surface this failure mode — the bug is invisible until it hits production traffic.

**Required change:**

```python
# Remove this import entirely
import requests

# Replace with
import httpx
from fastapi import APIRouter, HTTPException
from pydantic import BaseModel
import logging

logger = logging.getLogger(__name__)
router = APIRouter()


class NotificationRequest(BaseModel):
    user_id: str
    order_id: str
    message: str


class NotificationResponse(BaseModel):
    success: bool
    notification_id: str


@router.post("/notify", response_model=NotificationResponse)
async def send_notification(request: NotificationRequest) -> NotificationResponse:
    """
    Send an order confirmation notification to the user.
    """
    try:
        async with httpx.AsyncClient() as client:
            response = await client.post(
                "https://notifications.internal/send",
                json={
                    "user_id": request.user_id,
                    "order_id": request.order_id,
                    "message": request.message,
                },
                timeout=5.0,
            )
            response.raise_for_status()
            data = response.json()
            return NotificationResponse(
                success=True,
                notification_id=data["notification_id"]
            )
    except httpx.HTTPStatusError as e:
        logger.error("Notification service returned error status: %s", e)
        raise HTTPException(status_code=503, detail="Notification service unavailable")
    except httpx.RequestError as e:
        logger.error("Notification service connection error: %s", e)
        raise HTTPException(status_code=503, detail="Notification service unavailable")
```

**Tests must also be updated** — patch `httpx.AsyncClient.post` rather than `requests.post`, and use `pytest-anyio` or `httpx.AsyncClient` with `AsyncMock` so the test actually exercises the async path:

```python
import pytest
from unittest.mock import AsyncMock, patch, MagicMock
from httpx import AsyncClient, Response
from fastapi.testclient import TestClient
from api.notifications import router
from fastapi import FastAPI

app = FastAPI()
app.include_router(router)
client = TestClient(app)


def test_send_notification_success():
    mock_response = MagicMock(spec=Response)
    mock_response.json.return_value = {"notification_id": "ntf_abc123"}
    mock_response.raise_for_status.return_value = None

    with patch(
        "httpx.AsyncClient.post",
        new_callable=AsyncMock,
        return_value=mock_response
    ):
        response = client.post("/notify", json={
            "user_id": "user_123",
            "order_id": "order_456",
            "message": "Your order has been confirmed."
        })

    assert response.status_code == 200
    assert response.json()["success"] is True
    assert response.json()["notification_id"] == "ntf_abc123"


def test_send_notification_service_unavailable():
    import httpx

    with patch(
        "httpx.AsyncClient.post",
        new_callable=AsyncMock,
        side_effect=httpx.RequestError("Connection refused")
    ):
        response = client.post("/notify", json={
            "user_id": "user_123",
            "order_id": "order_456",
            "message": "Your order has been confirmed."
        })

    assert response.status_code == 503
```

**Secondary notes** (do not block merge once the above is fixed, but worth tracking):

- `data["notification_id"]` will raise an unhandled `KeyError` if the external service returns 200 with a malformed body. Consider a safe `.get()` with a fallback or an explicit validation error.
- If `httpx.AsyncClient` is used in a high-throughput path, consider lifting it to a module-level or lifespan-managed client instance rather than constructing one per request — connection pooling is lost when the client is created and destroyed on every call.
- The `message` field on `NotificationRequest` has no length constraint. An unbounded string forwarded to an external service is worth capping with `pydantic.Field(max_length=...)`.
