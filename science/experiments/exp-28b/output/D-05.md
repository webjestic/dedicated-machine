# Production Rate Limiter — Node.js

I'll design this fully before writing a line of code.

---

## Phase 1: Requirements Design

### What algorithm?

| Algorithm | Pros | Cons |
|---|---|---|
| Token Bucket | Allows bursting, smooth | State-heavy |
| Fixed Window | Simple | Boundary spike problem |
| Sliding Window Log | Exact | Memory-heavy at scale |
| Sliding Window Counter | Good approximation, memory-efficient | Slight inaccuracy at window edges |
| Leaky Bucket | Smooth output rate | Doesn't allow bursting |

**Decision: Sliding Window Counter** — best balance of accuracy, memory efficiency, and production viability. No boundary spikes, no per-request log storage.

### What scope?

- **Per-IP** rate limiting (primary)
- **Per-API-key** rate limiting (secondary, same engine)
- Configurable limits per route/tier
- Express middleware interface

### What backing store?

- **Redis** — the only correct answer for distributed rate limiting
- In-process fallback for single-node / dev environments
- Atomic operations via Lua script (no race conditions)

### What behavior on limit exceeded?

- Return `429 Too Many Requests`
- Return `Retry-After` header
- Return `X-RateLimit-*` headers on every request
- Log the event (structured)

### `src/stores/RedisStore.js`

```javascript
'use strict';

/**
 * RedisStore — Sliding Window Counter via atomic Lua script.
 *
 * Why Lua? Redis executes Lua atomically. No race condition between
 * ZREMRANGEBYSCORE + ZCARD + ZADD that you'd get with a pipeline.
 *
 * Key structure: ratelimit:{identifier}
 * Value: Sorted set — member = unique request ID, score = timestamp (ms)
 */

const SLIDING_WINDOW_SCRIPT = `
  local key        = KEYS[1]
  local now        = tonumber(ARGV[1])
  local window_ms  = tonumber(ARGV[2])
  local limit      = tonumber(ARGV[3])
  local request_id = ARGV[4]

  local window_start = now - window_ms

  -- Remove entries outside the current window
  redis.call('ZREMRANGEBYSCORE', key, '-inf', window_start)

  -- Count requests in current window
  local count = redis.call('ZCARD', key)

  if count < limit then
    -- Allow: record this request
    redis.call('ZADD', key, now, request_id)
    -- TTL slightly beyond window so Redis cleans up automatically
    redis.call('PEXPIRE', key, window_ms + 1000)
    return {1, count + 1, limit - count - 1}
  else
    -- Deny: return oldest entry so caller can compute retry-after
    local oldest = redis.call('ZRANGE', key, 0, 0, 'WITHSCORES')
    local oldest_score = oldest[2] and tonumber(oldest[2]) or now
    return {0, count, 0, oldest_score}
  end
`;

class RedisStore {
  /**
   * @param {import('ioredis').Redis} redisClient
   */
  constructor(redisClient) {
    if (!redisClient) {
      throw new Error('RedisStore requires a Redis client instance');
    }

    this.client = redisClient;
    this.scriptSha = null; // loaded on first use via EVALSHA
  }

  /**
   * Load the Lua script into Redis and cache the SHA.
   * EVALSHA is faster than EVAL on subsequent calls.
   */
  async _loadScript() {
    if (this.scriptSha) return this.scriptSha;
    this.scriptSha = await this.client.script('load', SLIDING_WINDOW_SCRIPT);
    return this.scriptSha;
  }

  /**
   * Check and record a request against the sliding window.
   *
   * @param {string} key        - Rate limit key (e.g. "ratelimit:ip:1.2.3.4")
   * @param {object} options
   * @param {number} options.limit      - Max requests allowed
   * @param {number} options.windowMs   - Window size in milliseconds
   * @returns {Promise<RateLimitResult>}
   */
  async check(key, { limit, windowMs }) {
    const sha = await this._loadScript();
    const now = Date.now();
    const requestId = `${now}-${Math.random().toString(36).slice(2, 9)}`;

    let result;

    try {
      result = await this.client.evalsha(
        sha,
        1,          // number of KEYS
        key,        // KEYS[1]
        now,        // ARGV[1]
        windowMs,   // ARGV[2]
        limit,      // ARGV[3]
        requestId   // ARGV[4]
      );
    } catch (err) {
      // NOSCRIPT means Redis was restarted and lost the cached script
      if (err.message && err.message.includes('NOSCRIPT')) {
        this.scriptSha = null;
        return this.check(key, { limit, windowMs }); // retry once
      }
      throw err;
    }

    const allowed     = result[0] === 1;
    const current     = result[1];
    const remaining   = result[2];
    const oldestScore = result[3] ? Number(result[3]) : null;

    // When denied, retry-after = time until oldest request falls out of window
    const retryAfterMs = !allowed && oldestScore
      ? Math.max(0, (oldestScore + windowMs) - now)
      : null;

    return {
      allowed,
      limit,
      current,
      remaining: Math.max(0, remaining),
      retryAfterMs,
      resetAtMs: now + windowMs,
    };
  }

  async close() {
    // Caller owns the Redis client lifecycle; we don't close it here
  }
}

module.exports = RedisStore;
```

### `src/stores/MemoryStore.js`

```javascript
'use strict';

/**
 * MemoryStore — In-process sliding window counter.
 *
 * NOT suitable for multi-process or multi-node deployments.
 * Use cases: local dev, single-process servers, testing.
 *
 * Uses a Map of sorted timestamp arrays.
 * Cleanup runs on a configurable interval to prevent unbounded growth.
 */

class MemoryStore {
  /**
   * @param {object} options
   * @param {number} [options.cleanupIntervalMs=60000] - How often to purge stale keys
   */
  constructor({ cleanupIntervalMs = 60_000 } = {}) {
    // Map<key, number[]> — arrays of timestamps (ms), kept sorted
    this._windows = new Map();
    this._cleanupInterval = setInterval(
      () => this._cleanup(),
      cleanupIntervalMs
    );
    // Don't block process exit
    this._cleanupInterval.unref();
  }

  /**
   * @param {string} key
   * @param {object} options
   * @param {number} options.limit
   * @param {number} options.windowMs
   * @returns {RateLimitResult}
   */
  async check(key, { limit, windowMs }) {
    const now = Date.now();
    const windowStart = now - windowMs;

    if (!this._windows.has(key)) {
      this._windows.set(key, []);
    }

    const timestamps = this._windows.get(key);

    // Drop timestamps outside the window (array is sorted ascending)
    let i = 0;
    while (i < timestamps.length && timestamps[i] <= windowStart) {
      i++;
    }
    // Mutate in place — avoids re-allocation on every request
    if (i > 0) timestamps.splice(0, i);

    const current = timestamps.length;

    if (current < limit) {
      timestamps.push(now);
      return {
        allowed: true,
        limit,
        current: current + 1,
        remaining: limit - current - 1,
        retryAfterMs: null,
        resetAtMs: now + windowMs,
      };
    }

    // Denied — oldest timestamp tells us when a slot opens
    const oldestTs = timestamps[0];
    const retryAfterMs = Math.max(0, (oldestTs + windowMs) - now);

    return {
      allowed: false,
      limit,
      current,
      remaining: 0,
      retryAfterMs,
      resetAtMs: now + windowMs,
    };
  }

  /** Remove keys with no recent activity to prevent memory leaks */
  _cleanup() {
    const now = Date.now();
    for (const [key, timestamps] of this._windows.entries()) {
      // If the newest timestamp is old enough, the whole key is stale
      if (timestamps.length === 0 || timestamps[timestamps.length - 1] < now - 3_600_000) {
        this._windows.delete(key);
      }
    }
  }

  async close() {
    clearInterval(this._cleanupInterval);
    this._windows.clear();
  }
}

module.exports = MemoryStore;
```

'use strict';

/**
 * Structured logger.
 *
 * In production, swap this for pino or winston.
 * The interface is intentionally identical to pino's.
 */

const LOG_LEVEL = process.env.LOG_LEVEL || 'info';

const LEVELS = { debug: 0, info: 1, warn: 2, error: 3 };
const currentLevel = LEVELS[LOG_LEVEL] ?? LEVELS.info;

function log(level, data) {
  if (LEVELS[level] < currentLevel) return;

  const entry = {
    ts: new Date().toISOString(),
    level,
    ...data,
  };

  // Single JSON line per event — plays well with log aggregators
  process.stdout.write(JSON.stringify(entry) + '\n');
}

module.exports = {
  debug: (data) => log('debug', data),
  info:  (data) => log('info',  data),
  warn:  (data) => log('warn',  data),
  error: (data) => log('error', data),
};

'use strict';

const TIERS = require('../config/tiers');
const logger = require('../utils/logger');

/**
 * @typedef {object} RateLimitResult
 * @property {boolean} allowed
 * @property {number}  limit
 * @property {number}  current
 * @property {number}  remaining
 * @property {number|null} retryAfterMs
 * @property {number}  resetAtMs
 */

/**
 * @typedef {object} RateLimiterOptions
 * @property {import('../stores/RedisStore')|import('../stores/MemoryStore')} store
 * @property {string}   [tier='default']     - Key into TIERS config
 * @property {number}   [limit]              - Override tier limit
 * @property {number}   [windowMs]           - Override tier windowMs
 * @property {string}   [keyBy]              - Override tier keyBy strategy
 * @property {string}   [keyPrefix='rl']     - Redis key namespace
 * @property {Function} [onLimitReached]     - Custom handler (req, res, result)
 * @property {boolean}  [skipOnStoreError=true] - Fail open if store throws
 */

/**
 * Express middleware factory.
 *
 * Usage:
 *   app.use(createRateLimiter({ store: redisStore }))
 *   app.post('/login', createRateLimiter({ store, tier: 'strict' }), handler)
 *
 * @param {RateLimiterOptions} options
 * @returns {import('express').RequestHandler}
 */
function createRateLimiter(options = {}) {
  const {
    store,
    tier: tierName = 'default',
    keyPrefix = 'rl',
    onLimitReached = null,
    skipOnStoreError = true,
  } = options;

  if (!store) {
    throw new Error('createRateLimiter: a store instance is required');
  }

  const tierConfig = TIERS[tierName];
  if (!tierConfig) {
    throw new Error(`createRateLimiter: unknown tier "${tierName}"`);
  }

  // Allow per-call overrides of any tier field
  const limit    = options.limit    ?? tierConfig.limit;
  const windowMs = options.windowMs ?? tierConfig.windowMs;
  const keyBy    = options.keyBy    ?? tierConfig.keyBy;

  return async function rateLimiterMiddleware(req, res, next) {
    let identifier;

    try
