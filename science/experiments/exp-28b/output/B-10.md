# Production Rate Limiter in Node.js

Let me build this properly — working code first, then an honest operational assessment.

## Architecture Decision

I'll implement **sliding window log** with a **token bucket** fallback, backed by Redis for distributed state. Single-process in-memory is a toy; production means multiple instances.

### `src/config.js`

```javascript
'use strict';

/**
 * Configuration with explicit defaults and validation.
 * Every value that affects safety has a comment explaining why it exists.
 */

const DEFAULT_CONFIG = {
  // How many requests are allowed in the window
  limit: 100,

  // Window size in milliseconds
  windowMs: 60_000,

  // Key prefix — prevents collisions if you share a Redis instance
  keyPrefix: 'rl:',

  // What to key on. Receives (req) and returns a string.
  // Default: IP address. Override for user-scoped limiting.
  keyGenerator: (req) => {
    // X-Forwarded-For requires trust in your proxy chain.
    // If you're not behind a trusted proxy, this can be spoofed.
    // See OPERATIONAL GAPS section.
    return (
      req.ip ||
      req.headers['x-forwarded-for']?.split(',')[0].trim() ||
      req.socket?.remoteAddress ||
      'unknown'
    );
  },

  // Called when the limit is exceeded. Override to customize response.
  onLimitReached: (req, res, options) => {
    res.status(429).json({
      error: 'Too Many Requests',
      retryAfter: Math.ceil(options.retryAfter / 1000),
      limit: options.limit,
      windowMs: options.windowMs,
    });
  },

  // If the backend (Redis) is unavailable, should we fail open or closed?
  // FAIL OPEN  = allow requests through (availability over security)
  // FAIL CLOSED = block requests (security over availability)
  // This is a business decision. Default: fail open with logging.
  failOpen: true,

  // Skip rate limiting for these conditions (e.g., internal health checks)
  skip: () => false,

  // Headers to send back to clients
  headers: {
    limit: 'X-RateLimit-Limit',
    remaining: 'X-RateLimit-Remaining',
    reset: 'X-RateLimit-Reset',
    retryAfter: 'Retry-After',
  },

  // Algorithm: 'sliding-window-log' | 'token-bucket' | 'fixed-window'
  algorithm: 'sliding-window-log',
};

function validateConfig(config) {
  const errors = [];

  if (typeof config.limit !== 'number' || config.limit <= 0) {
    errors.push('limit must be a positive number');
  }

  if (typeof config.windowMs !== 'number' || config.windowMs <= 0) {
    errors.push('windowMs must be a positive number');
  }

  if (!['sliding-window-log', 'token-bucket', 'fixed-window'].includes(config.algorithm)) {
    errors.push(`algorithm must be one of: sliding-window-log, token-bucket, fixed-window`);
  }

  if (typeof config.failOpen !== 'boolean') {
    errors.push('failOpen must be a boolean — this is a safety decision, be explicit');
  }

  if (errors.length > 0) {
    throw new Error(`Rate limiter configuration errors:\n${errors.map(e => `  - ${e}`).join('\n')}`);
  }

  return config;
}

function createConfig(overrides = {}) {
  const merged = { ...DEFAULT_CONFIG, ...overrides };
  return validateConfig(merged);
}

module.exports = { createConfig, DEFAULT_CONFIG };
```

### `src/backends/redisBackend.js`

```javascript
'use strict';

const fs = require('fs');
const path = require('path');

/**
 * Redis backend using Lua scripts for atomicity.
 *
 * WHY LUA SCRIPTS:
 * Any multi-step operation (check count → increment → set expiry) is a
 * race condition without atomicity. Redis executes Lua scripts atomically —
 * no other commands run between steps. This is non-negotiable for correctness.
 *
 * WHY NOT TRANSACTIONS (MULTI/EXEC):
 * WATCH + MULTI/EXEC requires retry logic on conflict and doesn't work
 * cleanly with Redis Cluster across different key slots.
 */

class RedisBackend {
  constructor(redisClient, options = {}) {
    if (!redisClient) {
      throw new Error('RedisBackend requires a redis client instance');
    }

    this.client = redisClient;
    this.logger = options.logger || console;

    // Load Lua scripts at construction time, not at request time
    this._scripts = this._loadScripts();

    // Track connection state for fail-open/fail-closed logic
    this._isConnected = true;
    this._setupConnectionHandlers();
  }

  _loadScripts() {
    const scriptDir = path.join(__dirname, '../../scripts/lua');

    return {
      slidingWindow: fs.readFileSync(
        path.join(scriptDir, 'slidingWindow.lua'),
        'utf8'
      ),
      tokenBucket: fs.readFileSync(
        path.join(scriptDir, 'tokenBucket.lua'),
        'utf8'
      ),
      fixedWindow: fs.readFileSync(
        path.join(scriptDir, 'fixedWindow.lua'),
        'utf8'
      ),
    };
  }

  _setupConnectionHandlers() {
    // ioredis emits these events — use them to track health
    this.client.on('connect', () => {
      this._isConnected = true;
      this.logger.info('[RateLimiter] Redis connection established');
    });

    this.client.on('error', (err) => {
      this._isConnected = false;
      // Log but don't throw — let fail-open/closed logic handle it
      this.logger.error('[RateLimiter] Redis error:', err.message);
    });

    this.client.on('close', () => {
      this._isConnected = false;
      this.logger.warn('[RateLimiter] Redis connection closed');
    });
  }

  isHealthy() {
    return this._isConnected;
  }

  async slidingWindowCheck(key, limit, windowMs, now) {
    const windowStart = now - windowMs;

    try {
      // Returns [count, oldestTimestamp]
      const result = await this.client.eval(
        this._scripts.slidingWindow,
        1,           // number of keys
        key,         // KEYS[1]
        windowStart, // ARGV[1]: remove entries older than this
        now,         // ARGV[2]: score for new entry
        limit,       // ARGV[3]: limit to check against
        windowMs     // ARGV[4]: TTL for the key
      );

      const count = result[0];
      const allowed = result[1] === 1;

      return {
        allowed,
        count,
        remaining: Math.max(0, limit - count),
        resetMs: windowMs, // sliding window resets relative to oldest entry
      };
    } catch (err) {
      this.logger.error('[RateLimiter] slidingWindowCheck error:', err.message);
      throw err; // Let caller handle fail-open/closed
    }
  }

  async tokenBucketCheck(key, capacity, refillRate, refillIntervalMs, now) {
    try {
      const result = await this.client.eval(
        this._scripts.tokenBucket,
        1,
        key,
        capacity,
        refillRate,
        refillIntervalMs,
        now
      );

      const tokens = result[0];
      const allowed = result[1] === 1;
      const nextRefillMs = result[2];

      return {
        allowed,
        remaining: Math.max(0, tokens),
        resetMs: nextRefillMs,
      };
    } catch (err) {
      this.logger.error('[RateLimiter] tokenBucketCheck error:', err.message);
      throw err;
    }
  }

  async fixedWindowCheck(key, limit, windowMs, now) {
    const windowId = Math.floor(now / windowMs);
    const windowKey = `${key}:${windowId}`;
    const windowExpiry = Math.ceil(windowMs / 1000) + 1; // seconds, +1 buffer

    try {
      const result = await this.client.eval(
        this._scripts.fixedWindow,
        1,
        windowKey,
        limit,
        windowExpiry
      );

      const count = result[0];
      const allowed = result[1] === 1;
      const resetMs = (windowId + 1) * windowMs - now;

      return {
        allowed,
        count,
        remaining: Math.max(0, limit - count),
        resetMs,
      };
    } catch (err) {
      this.logger.error('[RateLimiter] fixedWindowCheck error:', err.message);
      throw err;
    }
  }

  async ping() {
    try {
      await this.client.ping();
      return true;
    } catch {
      return false;
    }
  }
}

module.exports = { RedisBackend };
```

### `src/backends/memoryBackend.js`

```javascript
'use strict';

/**
 * In-memory backend for single-process use or local development.
 *
 * CRITICAL WARNING: This backend does NOT work correctly when you have
 * multiple Node.js processes or instances. Each process has its own
 * memory — limits are per-process, not per-user across your fleet.
 *
 * Use cases:
 *   - Local development
 *   - Single-process applications (not behind a load balancer)
 *   - Testing without Redis
 *
 * NOT suitable for:
 *   - Any horizontally scaled deployment
 *   - PM2 cluster mode
 *   - Kubernetes with multiple pods
 *   - Any load-balanced setup
 */

class MemoryBackend {
  constructor(options = {}) {
    this.logger = options.logger || console;

    // Map of key -> { count, windowStart, tokens, lastRefill }
    this._store = new Map();

    // Periodic cleanup to prevent unbounded memory growth
    // This is the kind of thing that bites you at 3am
    this._cleanupInterval = setInterval(
      () => this._cleanup(),
      options.cleanupIntervalMs || 60_000
    );

    // Allow the process to exit even if this interval is running
    if (this._cleanupInterval.unref) {
      this._cleanupInterval.unref();
    }

    this.logger.warn(
      '[RateLimiter] Using in-memory backend. ' +
      'This does not work across multiple processes or instances.'
    );
  }

  isHealthy() {
    return true; // Memory is always available (until it isn't — see OOM risks)
  }

  slidingWindowCheck(key, limit, windowMs, now) {
    const windowStart = now - windowMs;

    if (!this._store.has(key)) {
      this._store.set(key, { type: 'sliding', timestamps: [] });
    }

    const record = this._store.get(key);

    // Remove expired timestamps
    record.timestamps = record.timestamps.filter(ts => ts > windowStart);

    const count = record.timestamps.length;

    if (count >= limit) {
      return {
        allowed: false,
        count,
        remaining: 0,
        resetMs: record.timestamps[0] + windowMs - now,
      };
    }

    record.timestamps.push(now);

    return {
      allowed: true,
      count: count + 1,
      remaining: limit - count - 1,
      resetMs: windowMs,
    };
  }

  tokenBucketCheck(key, capacity, refillRate, refillIntervalMs, now) {
    if (!this._store.has(key)) {
      this._store.set(key, {
        type: 'token',
        tokens: capacity,
        lastRefill: now,
      });
    }

    const record = this._store.get(key);

    // Calculate tokens to add based on elapsed time
    const elapsed = now - record.lastRefill;
    const intervalsElapsed = Math.floor(elapsed / refillIntervalMs);
    const tokensToAdd = intervalsElapsed * refillRate;

    if (tokensToAdd > 0) {
      record.tokens = Math.min(capacity, record.tokens + tokensToAdd);
      record.lastRefill = record.lastRefill + (intervalsElapsed * refillIntervalMs);
    }

    if (record.tokens < 1) {
      const nextRefillMs = refillIntervalMs - (now - record.lastRefill);
      return {
        allowed: false,
        remaining: 0,
        resetMs: nextRefillMs,
      };
    }

    record.tokens -= 1;

    return {
      allowed: true,
      remaining: record.tokens,
      resetMs: refillIntervalMs - (now - record.lastRefill),
    };
  }

  fixedWindowCheck(key, limit, windowMs, now) {
    const windowId = Math.floor(now / windowMs);
    const windowKey = `${key}:${windowId}`;
    const resetMs = (windowId + 1) * windowMs - now;

    if (!this._store.has(windowKey)) {
      this._store.set(windowKey, { type: 'fixed', count: 0, windowId });
    }

    const record = this._store.get(windowKey);

    if (record.count >= limit) {
      return {
        allowed: false,
        count: record.count,
        remaining: 0,
        resetMs,
      };
