## Storage Backends

### Memory Store

```javascript
// src/stores/memoryStore.js
'use strict';

/**
 * In-process memory store.
 * Fast, zero dependencies — but does NOT share state across processes.
 * Use this for single-process deployments or testing.
 *
 * Includes a periodic sweep to evict expired keys so the Map
 * doesn't grow without bound in long-running processes.
 */
class MemoryStore {
  /**
   * @param {object}  opts
   * @param {number}  [opts.sweepIntervalMs=60_000]  How often to purge stale keys.
   */
  constructor(opts = {}) {
    this._store = new Map();
    this._sweepIntervalMs = opts.sweepIntervalMs ?? 60_000;
    this._sweepTimer = setInterval(() => this._sweep(), this._sweepIntervalMs);

    // Don't keep the process alive just for housekeeping.
    if (this._sweepTimer.unref) this._sweepTimer.unref();
  }

  // ─── Fixed / Sliding Window ───────────────────────────────────────────────

  /**
   * Atomically increment a counter and set its TTL on first touch.
   * Returns { count, ttlMs } where ttlMs is the remaining window time.
   */
  async increment(key, windowMs) {
    const now = Date.now();
    const entry = this._store.get(key);

    if (!entry || now >= entry.expiresAt) {
      // First request in this window.
      const expiresAt = now + windowMs;
      this._store.set(key, { count: 1, expiresAt });
      return { count: 1, ttlMs: windowMs };
    }

    entry.count += 1;
    return { count: entry.count, ttlMs: entry.expiresAt - now };
  }

  // ─── Sliding Window Log ───────────────────────────────────────────────────

  /**
   * Append a timestamp to a sorted list and prune entries older than windowMs.
   * Returns the number of requests in the current window.
   */
  async appendAndCount(key, windowMs) {
    const now = Date.now();
    const cutoff = now - windowMs;
    const entry = this._store.get(key) ?? { timestamps: [], expiresAt: 0 };

    // Prune expired timestamps (list is kept in insertion order ≈ sorted).
    entry.timestamps = entry.timestamps.filter((ts) => ts > cutoff);
    entry.timestamps.push(now);
    entry.expiresAt = now + windowMs;

    this._store.set(key, entry);
    return entry.timestamps.length;
  }

  // ─── Token Bucket ─────────────────────────────────────────────────────────

  /**
   * Fetch the current bucket state.
   * Returns null when no bucket exists yet (first request).
   */
  async getBucket(key) {
    return this._store.get(key) ?? null;
  }

  /** Persist bucket state. */
  async setBucket(key, bucket) {
    this._store.set(key, bucket);
  }

  // ─── Generic ──────────────────────────────────────────────────────────────

  async delete(key) {
    this._store.delete(key);
  }

  async reset() {
    this._store.clear();
  }

  /** Release the sweep timer (call in tests / graceful shutdown). */
  destroy() {
    clearInterval(this._sweepTimer);
  }

  // ─── Private ──────────────────────────────────────────────────────────────

  _sweep() {
    const now = Date.now();
    for (const [key, entry] of this._store) {
      if (now >= entry.expiresAt) this._store.delete(key);
    }
  }
}

module.exports = MemoryStore;
```

### Redis Store

```javascript
// src/stores/redisStore.js
'use strict';

const { StoreError } = require('../errors');

/**
 * Redis-backed store — safe for multi-process / multi-server deployments.
 *
 * Requires an ioredis (or node-redis v4) client passed in by the caller.
 * All mutating operations use Lua scripts so they are atomic even under
 * high concurrency without MULTI/EXEC overhead.
 */
class RedisStore {
  /**
   * @param {object} client  An ioredis or node-redis v4 client instance.
   * @param {object} [opts]
   * @param {string} [opts.keyPrefix='rl:']  Namespace prefix for all keys.
   */
  constructor(client, opts = {}) {
    if (!client) throw new Error('RedisStore requires a Redis client');
    this._client = client;
    this._prefix = opts.keyPrefix ?? 'rl:';

    // Pre-register Lua scripts as named commands (ioredis defineCommand).
    // Falls back to plain eval for node-redis compatibility.
    this._defineScripts();
  }

  // ─── Fixed Window ─────────────────────────────────────────────────────────

  /**
   * Atomically increment counter; set TTL only on creation.
   * Returns { count, ttlMs }.
   */
  async increment(key, windowMs) {
    const k = this._k(key);
    try {
      const [[, count], [, ttlPx]] = await this._client
        .multi()
        .incr(k)
        .pttl(k)
        .exec();

      // First request — the key was just created, set its expiry.
      if (count === 1) {
        await this._client.pexpire(k, windowMs);
        return { count: 1, ttlMs: windowMs };
      }

      return { count, ttlMs: ttlPx > 0 ? ttlPx : windowMs };
    } catch (err) {
      throw new StoreError('Redis increment failed', err);
    }
  }

  // ─── Sliding Window Log ───────────────────────────────────────────────────

  /**
   * Uses a Redis Sorted Set where score = timestamp.
   * Atomically removes old members, adds current timestamp, returns count.
   */
  async appendAndCount(key, windowMs) {
    const k = this._k(key);
    const now = Date.now();
    const cutoff = now - windowMs;

    // Lua keeps the whole operation atomic.
    const script = `
      redis.call('ZREMRANGEBYSCORE', KEYS[1], '-inf', ARGV[1])
      redis.call('ZADD', KEYS[1], ARGV[2], ARGV[2])
      redis.call('PEXPIRE', KEYS[1], ARGV[3])
      return redis.call('ZCARD', KEYS[1])
    `;

    try {
      const count = await this._eval(script, 1, k, cutoff, now, windowMs);
      return Number(count);
    } catch (err) {
      throw new StoreError('Redis appendAndCount failed', err);
    }
  }

  // ─── Token Bucket ─────────────────────────────────────────────────────────

  async getBucket(key) {
    try {
      const raw = await this._client.hgetall(this._k(key));
      if (!raw || Object.keys(raw).length === 0) return null;
      return {
        tokens: parseFloat(raw.tokens),
        lastRefill: parseInt(raw.lastRefill, 10),
      };
    } catch (err) {
      throw new StoreError('Redis getBucket failed', err);
    }
  }

  async setBucket(key, bucket, ttlMs) {
    const k = this._k(key);
    try {
      await this._client
        .multi()
        .hset(k, 'tokens', String(bucket.tokens), 'lastRefill', String(bucket.lastRefill))
        .pexpire(k, ttlMs)
        .exec();
    } catch (err) {
      throw new StoreError('Redis setBucket failed', err);
    }
  }

  // ─── Generic ──────────────────────────────────────────────────────────────

  async delete(key) {
    await this._client.del(this._k(key));
  }

  async reset(pattern = '*') {
    const keys = await this._client.keys(this._k(pattern));
    if (keys.length) await this._client.del(...keys);
  }

  // ─── Private ──────────────────────────────────────────────────────────────

  _k(key) {
    return `${this._prefix}${key}`;
  }

  _defineScripts() {
    // ioredis exposes defineCommand; node-redis uses client.eval directly.
    // We normalise via _eval() so callers don't need to branch.
  }

  async _eval(script, numKeys, ...args) {
    if (typeof this._client.eval === 'function') {
      return this._client.eval(script, numKeys, ...args);
    }
    // node-redis v4
    return this._client.eval(script, { keys: args.slice(0, numKeys), arguments: args.slice(numKeys).map(String) });
  }
}

module.exports = RedisStore;
```

## Algorithms

### 1. Fixed Window

```javascript
// src/algorithms/fixedWindow.js
'use strict';

/**
 * Fixed Window Counter
 *
 * Simplest algorithm. Divides time into discrete windows of `windowMs`.
 * Counts requests per window per key. Resets hard at the window boundary.
 *
 * Pros:  Very low memory, O(1) per request.
 * Cons:  Susceptible to burst at window edges (2× limit possible).
 */
class FixedWindow {
  /**
   * @param {object} store     A MemoryStore or RedisStore instance.
   * @param {object} opts
   * @param {number} opts.windowMs   Window duration in milliseconds.
   * @param {number} opts.maxRequests  Max allowed requests per window.
   */
  constructor(store, opts) {
    this._store = store;
    this._windowMs = opts.windowMs;
    this._max = opts.maxRequests;
  }

  /**
   * @param {string} key  Unique identifier (e.g. IP address, user ID).
   * @returns {Promise<RateLimitResult>}
   */
  async check(key) {
    const { count, ttlMs } = await this._store.increment(key, this._windowMs);
    const remaining = Math.max(0, this._max - count);
    const allowed = count <= this._max;

    return {
      allowed,
      count,
      remaining,
      resetMs: ttlMs,
      retryAfterMs: allowed ? 0 : ttlMs,
      limit: this._max,
    };
  }
}

module.exports = FixedWindow;
```

### 2. Sliding Window

```javascript
// src/algorithms/slidingWindow.js
'use strict';

/**
 * Sliding Window Log
 *
 * Maintains a timestamped log of every request within the last `windowMs`.
 * Provides accurate rate limiting without the edge-burst problem of fixed windows.
 *
 * Pros:  Accurate, no edge bursting.
 * Cons:  Higher memory — O(requests_in_window) per key.
 *        Use fixed window for very high-traffic keys.
 */
class SlidingWindow {
  constructor(store, opts) {
    this._store = store;
    this._windowMs = opts.windowMs;
    this._max = opts.maxRequests;
  }

  async check(key) {
    const count = await this._store.appendAndCount(key, this._windowMs);
    const allowed = count <= this._max;
    const remaining = Math.max(0, this._max - count);

    return {
      allowed,
      count,
      remaining,
      resetMs: this._windowMs,
      retryAfterMs: allowed ? 0 : this._windowMs,
      limit: this._max,
    };
  }
}

module.exports = SlidingWindow;
```

### 3. Token Bucket

```javascript
// src/algorithms/tokenBucket.js
'use strict';

/**
 * Token Bucket
 *
 * A bucket holds up to `capacity` tokens. Tokens refill at `refillRate`
 * tokens per second. Each request consumes `tokensPerRequest` tokens.
 *
 * Pros:  Handles bursts gracefully up to bucket capacity.
 *        Smooth average throughput.
 * Cons:  Slightly more complex state (two fields per key).
 */
class TokenBucket {
  /**
   * @param {object} store
   * @param {object} opts
   * @param {number} opts.capacity         Max tokens the bucket can hold.
   * @param {number} opts.refillRate       Tokens added per second.
   * @param {number} [opts.tokensPerRequest=1]  Tokens consumed per request.
   */
  constructor(store, opts) {
    this._store = store;
    this._capacity = opts.capacity;
    this._refillRate = opts.refillRate;          // tokens / second
    this._tokensPerRequest = opts.tokensPerRequest ?? 1;
    // TTL: keep bucket alive for 2× the time needed to fully refill.
    this._ttlMs = Math.ceil((this._capacity / this._refillRate) * 2 * 1000);
  }

  async check(key) {
    const now = Date.now();
    let bucket = await this._store.getBucket(key);

    if (!bucket) {
      // Brand-new bucket — start full.
      bucket = { tokens: this._capacity, lastRefill: now };
    } else {
      // Refill tokens based on elapsed time.
      const elapsedSec = (now - bucket.lastRefill) / 1000;
      const refilled = elapsedSec * this._refillRate;
      bucket.tokens = Math.min(this._capacity, bucket.tokens + refilled);
      bucket.lastRefill = now;
    }

    const
