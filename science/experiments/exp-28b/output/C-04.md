## 1. Memory Store

```javascript
// src/stores/memoryStore.js

class MemoryStore {
  constructor(cleanupInterval = 60000) {
    this.store = new Map();

    // Periodically clean up expired entries
    this.cleanupTimer = setInterval(() => {
      this.cleanup();
    }, cleanupInterval);

    // Allow process to exit even if timer is running
    if (this.cleanupTimer.unref) {
      this.cleanupTimer.unref();
    }
  }

  async get(key) {
    const entry = this.store.get(key);
    if (!entry) return null;

    if (Date.now() > entry.expiresAt) {
      this.store.delete(key);
      return null;
    }

    return entry.value;
  }

  async set(key, value, ttlMs) {
    this.store.set(key, {
      value,
      expiresAt: Date.now() + ttlMs,
    });
  }

  async increment(key, ttlMs) {
    const entry = this.store.get(key);
    const now = Date.now();

    if (!entry || now > entry.expiresAt) {
      this.store.set(key, { value: 1, expiresAt: now + ttlMs });
      return 1;
    }

    entry.value += 1;
    return entry.value;
  }

  async delete(key) {
    this.store.delete(key);
  }

  async reset() {
    this.store.clear();
  }

  cleanup() {
    const now = Date.now();
    for (const [key, entry] of this.store.entries()) {
      if (now > entry.expiresAt) {
        this.store.delete(key);
      }
    }
  }

  destroy() {
    clearInterval(this.cleanupTimer);
    this.store.clear();
  }
}

module.exports = MemoryStore;
```

// src/stores/redisStore.js

class RedisStore {
  constructor(redisClient) {
    if (!redisClient) {
      throw new Error("Redis client is required");
    }
    this.client = redisClient;
  }

  async get(key) {
    const value = await this.client.get(key);
    return value ? JSON.parse(value) : null;
  }

  async set(key, value, ttlMs) {
    await this.client.set(key, JSON.stringify(value), {
      PX: ttlMs, // Millisecond expiry
    });
  }

  async increment(key, ttlMs) {
    // Atomic increment using Lua script
    const script = `
      local current = redis.call('GET', KEYS[1])
      if current == false then
        redis.call('SET', KEYS[1], 1, 'PX', ARGV[1])
        return 1
      else
        return redis.call('INCR', KEYS[1])
      end
    `;

    const result = await this.client.eval(script, {
      keys: [key],
      arguments: [String(ttlMs)],
    });

    return result;
  }

  async delete(key) {
    await this.client.del(key);
  }

  async reset() {
    await this.client.flushDb();
  }
}

module.exports = RedisStore;

## 3. Fixed Window Strategy

```javascript
// src/strategies/fixedWindow.js

class FixedWindowStrategy {
  /**
   * Fixed Window Rate Limiting
   * Simple counter per time window. Resets at the end of each window.
   *
   * Pros: Simple, low memory
   * Cons: Burst traffic at window boundaries
   *
   * @param {object} store - Storage backend
   * @param {number} windowMs - Window size in milliseconds
   * @param {number} maxRequests - Max requests per window
   */
  constructor(store, windowMs, maxRequests) {
    this.store = store;
    this.windowMs = windowMs;
    this.maxRequests = maxRequests;
  }

  async check(key) {
    const windowKey = this._getWindowKey(key);
    const count = await this.store.increment(windowKey, this.windowMs);

    const remaining = Math.max(0, this.maxRequests - count);
    const windowStart = this._getWindowStart();
    const resetTime = new Date(windowStart + this.windowMs);

    return {
      allowed: count <= this.maxRequests,
      limit: this.maxRequests,
      remaining,
      resetTime,
      retryAfter: count > this.maxRequests
        ? Math.ceil((resetTime - Date.now()) / 1000)
        : 0,
    };
  }

  async reset(key) {
    const windowKey = this._getWindowKey(key);
    await this.store.delete(windowKey);
  }

  _getWindowStart() {
    return Math.floor(Date.now() / this.windowMs) * this.windowMs;
  }

  _getWindowKey(key) {
    return `fw:${key}:${this._getWindowStart()}`;
  }
}

module.exports = FixedWindowStrategy;
```

## 4. Sliding Window Strategy

```javascript
// src/strategies/slidingWindow.js

class SlidingWindowStrategy {
  /**
   * Sliding Window Rate Limiting
   * Tracks individual request timestamps within the window.
   * More accurate than fixed window — no boundary burst issues.
   *
   * Pros: Smooth rate limiting
   * Cons: Higher memory usage (stores timestamps)
   *
   * @param {object} store - Storage backend
   * @param {number} windowMs - Window size in milliseconds
   * @param {number} maxRequests - Max requests per window
   */
  constructor(store, windowMs, maxRequests) {
    this.store = store;
    this.windowMs = windowMs;
    this.maxRequests = maxRequests;
  }

  async check(key) {
    const now = Date.now();
    const windowStart = now - this.windowMs;
    const storeKey = `sw:${key}`;

    // Get existing timestamps
    let timestamps = (await this.store.get(storeKey)) || [];

    // Remove expired timestamps outside the window
    timestamps = timestamps.filter((ts) => ts > windowStart);

    const count = timestamps.length;
    const allowed = count < this.maxRequests;

    if (allowed) {
      timestamps.push(now);
      await this.store.set(storeKey, timestamps, this.windowMs);
    }

    const remaining = Math.max(0, this.maxRequests - timestamps.length);

    // Retry after the oldest request in the window expires
    const oldestTimestamp = timestamps[0];
    const retryAfter =
      !allowed && oldestTimestamp
        ? Math.ceil((oldestTimestamp + this.windowMs - now) / 1000)
        : 0;

    return {
      allowed,
      limit: this.maxRequests,
      remaining,
      resetTime: new Date(now + this.windowMs),
      retryAfter,
    };
  }

  async reset(key) {
    await this.store.delete(`sw:${key}`);
  }
}

module.exports = SlidingWindowStrategy;
```

## 5. Token Bucket Strategy

```javascript
// src/strategies/tokenBucket.js

class TokenBucketStrategy {
  /**
   * Token Bucket Rate Limiting
   * Tokens are added at a fixed rate. Each request consumes one token.
   * Allows controlled bursting up to bucket capacity.
   *
   * Pros: Handles bursts gracefully, smooth traffic
   * Cons: Slightly more complex state
   *
   * @param {object} store - Storage backend
   * @param {number} capacity - Max tokens in the bucket
   * @param {number} refillRate - Tokens added per second
   * @param {number} tokensPerRequest - Tokens consumed per request
   */
  constructor(store, capacity, refillRate, tokensPerRequest = 1) {
    this.store = store;
    this.capacity = capacity;
    this.refillRate = refillRate; // tokens per second
    this.tokensPerRequest = tokensPerRequest;
  }

  async check(key) {
    const now = Date.now();
    const storeKey = `tb:${key}`;

    let bucket = await this.store.get(storeKey);

    if (!bucket) {
      // New bucket starts full
      bucket = {
        tokens: this.capacity,
        lastRefill: now,
      };
    }

    // Calculate tokens to add since last refill
    const elapsed = (now - bucket.lastRefill) / 1000; // seconds
    const tokensToAdd = elapsed * this.refillRate;
    bucket.tokens = Math.min(this.capacity, bucket.tokens + tokensToAdd);
    bucket.lastRefill = now;

    const allowed = bucket.tokens >= this.tokensPerRequest;

    if (allowed) {
      bucket.tokens -= this.tokensPerRequest;
    }

    // TTL: time until bucket would be empty even if no requests come
    const ttlMs = (this.capacity / this.refillRate) * 1000 * 2;
    await this.store.set(storeKey, bucket, ttlMs);

    const remaining = Math.floor(bucket.tokens);
    const retryAfter = !allowed
      ? Math.ceil(
          (this.tokensPerRequest - bucket.tokens) / this.refillRate
        )
      : 0;

    return {
      allowed,
      limit: this.capacity,
      remaining,
      resetTime: new Date(now + (this.capacity / this.refillRate) * 1000),
      retryAfter,
    };
  }

  async reset(key) {
    await this.store.delete(`tb:${key}`);
  }
}

module.exports = TokenBucketStrategy;
```

## 6. Core Rate Limiter

```javascript
// src/rateLimiter.js

const MemoryStore = require("./stores/memoryStore");
const FixedWindowStrategy = require("./strategies/fixedWindow");
const SlidingWindowStrategy = require("./strategies/slidingWindow");
const TokenBucketStrategy = require("./strategies/tokenBucket");

const STRATEGIES = {
  fixedWindow: FixedWindowStrategy,
  slidingWindow: SlidingWindowStrategy,
  tokenBucket: TokenBucketStrategy,
};

class RateLimiter {
  /**
   * @param {object} options
   * @param {string}   options.strategy      - 'fixedWindow' | 'slidingWindow' | 'tokenBucket'
   * @param {number}   options.windowMs      - Window duration in ms (fixedWindow/slidingWindow)
   * @param {number}   options.maxRequests   - Max requests per window
   * @param {number}   options.capacity      - Bucket capacity (tokenBucket)
   * @param {number}   options.refillRate    - Tokens/sec (tokenBucket)
   * @param {object}   options.store         - Custom store (optional)
   * @param {function} options.keyGenerator  - Custom key function (req) => string
   * @param {function} options.onLimitReached - Callback when limit is hit
   * @param {boolean}  options.skipFailedRequests - Don't count failed responses
   * @param {boolean}  options.trustProxy    - Trust X-Forwarded-For header
   */
  constructor(options = {}) {
    const {
      strategy = "fixedWindow",
      windowMs = 60 * 1000,
      maxRequests = 100,
      capacity = maxRequests,
      refillRate = maxRequests / (windowMs / 1000),
      store = new MemoryStore(),
      keyGenerator = null,
      onLimitReached = null,
      skipFailedRequests = false,
      trustProxy = false,
    } = options;

    if (!STRATEGIES[strategy]) {
      throw new Error(
        `Unknown strategy: "${strategy}". Choose from: ${Object.keys(STRATEGIES).join(", ")}`
      );
    }

    this.store = store;
    this.keyGenerator = keyGenerator || this._defaultKeyGenerator(trustProxy);
    this.onLimitReached = onLimitReached;
    this.skipFailedRequests = skipFailedRequests;

    // Instantiate strategy
    if (strategy === "tokenBucket") {
      this.strategy = new STRATEGIES[strategy](store, capacity, refillRate);
    } else {
      this.strategy = new STRATEGIES[strategy](store, windowMs, maxRequests);
    }
  }

  /**
   * Check rate limit for a given identifier
   * @param {string} identifier - Unique key (IP, user ID, etc.)
   * @returns {Promise<RateLimitResult>}
   */
  async check(identifier) {
    return this.strategy.check(identifier);
  }

  /**
   * Reset rate limit for a given identifier
   * @param {string} identifier
   */
  async reset(identifier) {
    return this.strategy.reset(identifier);
  }

  /**
   * Express/Connect middleware
   */
  middleware() {
    return async (req, res, next) => {
      const key = this.keyGenerator(req);
      const result = await this.check(key);

      // Set standard rate limit headers
      res.setHeader("X-RateLimit-Limit", result.limit);
      res.setHeader("X-RateLimit-Remaining", result.remaining);
      res.setHeader(
        "X-RateLimit-Reset",
        Math.ceil(result.resetTime.getTime() / 1000)
      );

      if (!result.allowed) {
        res.setHeader("Retry-After", result.retryAfter);

        if (this.onLimitReached) {
          this.onLimitReached(req, res, key, result);
        }

        return res.status(429).json({
          error: "Too Many Requests",
          message: `Rate limit exceeded. Try again in ${result.retryAfter} seconds.`,
          retryAfter: result.retryAfter,
        });
      }

      // Optionally skip counting failed requests
      if (this.skipFailedRequests) {
        const originalEnd = res.end.bind(res);
        res.end = (...args) => {
          if (res.statusCode >= 400) {
            this.reset(key).catch(console.error);
          }
          return originalEnd(...args);
        };
      }

      next();
    };
  }

  _defaultKeyGenerator(trustProxy) {
    return (req) => {
      if (trustProxy) {
        const forwarded = req.headers["x-forwarded-for"];
        if (forwarded) {
          return forwarded.split(",")[0].trim();
        }
      }
      return (
