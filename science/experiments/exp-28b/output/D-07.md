# Production Rate Limiter in Node.js

Let me work through this properly — requirements first, then implementation, then operational gaps.

---

## Phase 1: Requirements Design

### Functional Requirements
- Limit requests per identifier (IP, user ID, API key) within a time window
- Support multiple rate limit tiers (different limits for different consumers)
- Return standard rate limit headers on every response
- Reject requests that exceed limits with proper HTTP 429 responses
- Support sliding window (more accurate than fixed window)

### Non-Functional Requirements
- Sub-millisecond overhead on the hot path
- Survive Redis failures without taking down the API (fail-open or fail-closed, configurable)
- Atomic operations — no race conditions under concurrent load
- Horizontally scalable across multiple Node.js instances

### Explicit Design Decisions
| Decision | Choice | Reason |
|---|---|---|
| Algorithm | Sliding window log via Redis | Accurate, no boundary burst problem |
| Fallback | In-memory fixed window | Redis outage shouldn't kill the API |
| Atomicity | Lua script in Redis | Atomic read-modify-write, no race condition |
| Identity | Pluggable key resolver | IP, user ID, API key — caller decides |
| Failure mode | Configurable fail-open/closed | Different APIs have different risk tolerance |

### `src/config.js`

```javascript
'use strict';

/**
 * Configuration schema with explicit defaults and validation.
 *
 * Every value has a reason for its default. Nothing is arbitrary.
 */

const DEFAULT_CONFIG = {
  // Time window in milliseconds
  // 60000ms = 1 minute is the most common API rate limit window
  windowMs: 60_000,

  // Maximum requests allowed within the window per unique key
  maxRequests: 100,

  // What to do when Redis is unreachable
  // 'open'  — allow the request through (availability over security)
  // 'closed' — reject the request (security over availability)
  // Choose based on your threat model. Public APIs often use 'open'.
  failMode: 'open',

  // How long to keep a key in Redis after last request (seconds)
  // Should be >= windowMs / 1000 to avoid premature expiry
  // We use 2x as a safety margin
  keyTtlSeconds: 120,

  // HTTP status code for rate limited responses
  // 429 is the correct code per RFC 6585
  statusCode: 429,

  // Message returned to the client
  message: 'Too many requests. Please retry after the window resets.',

  // Whether to skip rate limiting entirely (useful for testing)
  skip: false,

  // Function to extract the rate limit key from a request
  // Default: IP address. Override for user-based limiting.
  keyGenerator: (req) => {
    // Trust X-Forwarded-For only if you're behind a known proxy
    // Raw socket IP is safer for untrusted environments
    return (
      req.ip ||
      req.headers['x-forwarded-for']?.split(',')[0].trim() ||
      req.socket.remoteAddress ||
      'unknown'
    );
  },

  // Called when a request is rate limited — use for metrics/logging
  onLimitReached: (req, key, limitInfo) => {
    // Default: no-op. Wire this to your metrics system.
  },

  // Redis connection options (ioredis format)
  redis: {
    host: process.env.REDIS_HOST || '127.0.0.1',
    port: parseInt(process.env.REDIS_PORT || '6379', 10),
    password: process.env.REDIS_PASSWORD || undefined,
    db: parseInt(process.env.REDIS_DB || '0', 10),
    // Connection timeout — fail fast rather than hang
    connectTimeout: 2000,
    // Command timeout — rate limiting must be fast
    commandTimeout: 500,
    // Retry strategy: back off, but don't retry forever
    // Returns null to stop retrying after 3 attempts
    maxRetriesPerRequest: 3,
    lazyConnect: true,
  },
};

/**
 * Merges user config with defaults and validates the result.
 * Throws early with a clear message rather than failing at runtime.
 */
function buildConfig(userConfig = {}) {
  const config = {
    ...DEFAULT_CONFIG,
    ...userConfig,
    redis: {
      ...DEFAULT_CONFIG.redis,
      ...(userConfig.redis || {}),
    },
  };

  // Validate required numeric fields
  if (typeof config.windowMs !== 'number' || config.windowMs <= 0) {
    throw new Error(`[RateLimiter] windowMs must be a positive number, got: ${config.windowMs}`);
  }

  if (typeof config.maxRequests !== 'number' || config.maxRequests <= 0) {
    throw new Error(`[RateLimiter] maxRequests must be a positive number, got: ${config.maxRequests}`);
  }

  if (!['open', 'closed'].includes(config.failMode)) {
    throw new Error(`[RateLimiter] failMode must be 'open' or 'closed', got: ${config.failMode}`);
  }

  if (typeof config.keyGenerator !== 'function') {
    throw new Error('[RateLimiter] keyGenerator must be a function');
  }

  // Ensure TTL is sufficient to cover the window
  const minTtl = Math.ceil(config.windowMs / 1000);
  if (config.keyTtlSeconds < minTtl) {
    console.warn(
      `[RateLimiter] keyTtlSeconds (${config.keyTtlSeconds}) is less than windowMs in seconds ` +
      `(${minTtl}). Keys may expire before the window ends. Setting to ${minTtl * 2}.`
    );
    config.keyTtlSeconds = minTtl * 2;
  }

  return config;
}

module.exports = { buildConfig, DEFAULT_CONFIG };

### `src/store/redisStore.js`

```javascript
'use strict';

const Redis = require('ioredis');

/**
 * Manages the Redis connection with explicit health tracking.
 *
 * The critical design decision here: we track connection health
 * ourselves rather than relying on ioredis error events alone.
 * This gives the rate limiter a synchronous, cheap way to check
 * "is Redis available right now?" on every request without
 * adding async overhead to the hot path.
 */
class RedisStore {
  constructor(redisConfig) {
    this._config = redisConfig;
    this._client = null;
    this._isHealthy = false;
    this._connectionAttempts = 0;
  }

  async connect() {
    this._client = new Redis(this._config);

    // Track health state on connection events
    this._client.on('connect', () => {
      this._isHealthy = true;
      this._connectionAttempts = 0;
      console.info('[RateLimiter:Redis] Connected');
    });

    this._client.on('ready', () => {
      this._isHealthy = true;
      console.info('[RateLimiter:Redis] Ready');
    });

    this._client.on('error', (err) => {
      // Log but don't throw — the rate limiter handles degraded state
      console.error('[RateLimiter:Redis] Error:', err.message);
    });

    this._client.on('close', () => {
      this._isHealthy = false;
      console.warn('[RateLimiter:Redis] Connection closed');
    });

    this._client.on('reconnecting', (delay) => {
      this._isHealthy = false;
      this._connectionAttempts++;
      console.warn(`[RateLimiter:Redis] Reconnecting in ${delay}ms (attempt ${this._connectionAttempts})`);
    });

    // Explicitly connect and surface failure at startup
    // rather than silently falling back on first request
    try {
      await this._client.connect();
    } catch (err) {
      console.error('[RateLimiter:Redis] Initial connection failed:', err.message);
      // Don't throw — allow the app to start in degraded mode
      // The failMode config determines what happens on each request
    }

    return this;
  }

  get client() {
    return this._client;
  }

  get isHealthy() {
    return this._isHealthy;
  }

  async disconnect() {
    if (this._client) {
      await this._client.quit();
      this._isHealthy = false;
    }
  }
}

module.exports = RedisStore;
```

### `src/algorithms/slidingWindowRedis.js`

```javascript
'use strict';

/**
 * Sliding Window Log algorithm implemented via Redis + Lua.
 *
 * WHY SLIDING WINDOW OVER FIXED WINDOW:
 * Fixed window has a well-known boundary burst problem.
 * A client can make maxRequests at 00:59 and maxRequests at 01:01
 * — double the allowed rate in a 2-second span.
 *
 * Sliding window tracks every request timestamp in a sorted set.
 * The window is always [now - windowMs, now], so the burst
 * problem doesn't exist.
 *
 * COST: O(N) memory per key where N = requests in the window.
 * For most APIs this is fine. For very high-volume keys (millions
 * of requests/window), consider sliding window counter instead.
 *
 * WHY LUA:
 * Redis is single-threaded, and Lua scripts execute atomically.
 * Without Lua, a check-then-increment would be a race condition:
 *   Thread A reads count=99  (limit=100)
 *   Thread B reads count=99  (limit=100)
 *   Thread A increments → 100 (allowed)
 *   Thread B increments → 101 (should be blocked, but isn't)
 *
 * The Lua script makes read + write a single atomic operation.
 */

// The Lua script that runs atomically in Redis.
// KEYS[1] = the rate limit key (e.g., "rl:127.0.0.1")
// ARGV[1] = current timestamp in milliseconds
// ARGV[2] = window size in milliseconds
// ARGV[3] = max requests allowed
// ARGV[4] = key TTL in seconds
//
// Returns: [current_count, is_allowed (1=yes, 0=no), oldest_timestamp_in_window]
const SLIDING_WINDOW_SCRIPT = `
local key        = KEYS[1]
local now        = tonumber(ARGV[1])
local window_ms  = tonumber(ARGV[2])
local max_req    = tonumber(ARGV[3])
local ttl_sec    = tonumber(ARGV[4])
local window_start = now - window_ms

-- Remove all entries outside the current window
-- ZREMRANGEBYSCORE removes members with scores between -inf and window_start
redis.call('ZREMRANGEBYSCORE', key, '-inf', window_start)

-- Count remaining entries (requests within the window)
local current_count = redis.call('ZCARD', key)

-- Determine if this request is allowed
local is_allowed = 0
if current_count < max_req then
  is_allowed = 1
  -- Add current request with timestamp as score
  -- Use timestamp + a random suffix as member to handle
  -- multiple requests at the exact same millisecond
  local member = now .. '-' .. math.random(1, 1000000)
  redis.call('ZADD', key, now, member)
  current_count = current_count + 1
end

-- Reset TTL on every access to keep active keys alive
-- but allow inactive keys to expire naturally
redis.call('EXPIRE', key, ttl_sec)

-- Get the oldest entry to calculate reset time
local oldest = redis.call('ZRANGE', key, 0, 0, 'WITHSCORES')
local oldest_ts = now
if #oldest > 0 then
  oldest_ts = tonumber(oldest[2])
end

return {current_count, is_allowed, oldest_ts}
`;

class SlidingWindowRedis {
  constructor(redisClient, config) {
    this._client = redisClient;
    this._config = config;
    // Pre-load the script into Redis for slightly faster execution
    // SHA is stable for the same script content
    this._scriptSha = null;
    this._loadScript();
  }

  async _loadScript() {
    try {
      this._scriptSha = await this._client.script('LOAD', SLIDING_WINDOW_SCRIPT);
    } catch (err) {
      // Non-fatal: we'll fall back to EVAL on each call
      console.warn('[RateLimiter:Redis] Could not pre-load Lua script:', err.message);
    }
  }

  /**
   * Check and increment the rate limit for a given key.
   *
   * @param {string} key - The rate limit identifier
   * @returns {Promise<{allowed: boolean, current: number, limit: number, resetMs: number, remainingMs: number}>}
   */
  async check(key) {
    const now = Date.now();
    const redisKey = `rl:${key}`;

    const args = [
      redisKey,
      now.toString(),
      this._config.windowMs.toString(),
      this._config.maxRequests.toString(),
      this._config.keyTtlSeconds.toString(),
    ];

    let result;

    // Try EVALSHA first (faster — uses cached script)
    // Fall back to EVAL if script was flushed from Redis
    if (this._scriptSha) {
      try {
        result = await this._client.evalsha(this._scriptSha, 1, ...args);

          // Script was flushed (e.g., Redis restart) — reload and retry
          await this._loadScript();
          result = await
