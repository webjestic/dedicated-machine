# Production Rate Limiter in Node.js

Let me build this properly — working code first, then an honest operational assessment.

## Architecture Decision

I'll implement **sliding window log** (most accurate) with a **token bucket** fallback pattern, backed by Redis for distributed state. I'll also include an in-memory fallback for single-instance deployments.

### `src/stores/redisStore.js`

```javascript
const Redis = require('ioredis');
const { StoreConnectionError } = require('../errors');
const config = require('../config');

/**
 * Redis store with:
 * - Lua scripts for atomic operations (prevents race conditions)
 * - Connection health tracking
 * - Explicit error surfacing (no silent swallowing)
 */
class RedisStore {
  constructor(redisConfig = config.redis) {
    this.client = new Redis({
      host: redisConfig.host,
      port: redisConfig.port,
      password: redisConfig.password,
      tls: redisConfig.tls ? {} : undefined,
      connectTimeout: redisConfig.connectTimeout,
      commandTimeout: redisConfig.commandTimeout,
      maxRetriesPerRequest: redisConfig.maxRetriesPerRequest,
      // ioredis will emit 'error' events — we must handle them or Node crashes
      lazyConnect: true,
    });

    this.isHealthy = false;
    this._registerEventHandlers();

    // Lua script: sliding window counter (atomic increment + expire)
    // Returns [current_count, ttl_ms]
    this.slidingWindowScript = `
      local key = KEYS[1]
      local window = tonumber(ARGV[1])
      local now = tonumber(ARGV[2])
      local limit = tonumber(ARGV[3])

      -- Remove entries outside the current window
      redis.call('ZREMRANGEBYSCORE', key, 0, now - window)

      -- Count current entries
      local count = redis.call('ZCARD', key)

      if count < limit then
        -- Add current request with score = timestamp
        redis.call('ZADD', key, now, now .. '-' .. math.random(1000000))
        -- Set expiry slightly beyond window to allow cleanup
        redis.call('PEXPIRE', key, window + 1000)
        return {count + 1, 0}
      else
        -- Return oldest entry TTL so client knows when to retry
        local oldest = redis.call('ZRANGE', key, 0, 0, 'WITHSCORES')
        local retryAfterMs = 0
        if oldest[2] then
          retryAfterMs = window - (now - tonumber(oldest[2]))
        end
        return {count, retryAfterMs}
      end
    `;

    // Lua script: token bucket
    // Returns [tokens_remaining, allowed (0|1)]
    this.tokenBucketScript = `
      local key = KEYS[1]
      local capacity = tonumber(ARGV[1])
      local refillRate = tonumber(ARGV[2])  -- tokens per ms
      local now = tonumber(ARGV[3])
      local requested = tonumber(ARGV[4])

      local data = redis.call('HMGET', key, 'tokens', 'last_refill')
      local tokens = tonumber(data[1]) or capacity
      local lastRefill = tonumber(data[2]) or now

      -- Calculate tokens to add based on elapsed time
      local elapsed = now - lastRefill
      local newTokens = math.min(capacity, tokens + (elapsed * refillRate))

      if newTokens >= requested then
        newTokens = newTokens - requested
        redis.call('HMSET', key, 'tokens', newTokens, 'last_refill', now)
        redis.call('PEXPIRE', key, math.ceil(capacity / refillRate) + 1000)
        return {math.floor(newTokens), 1}
      else
        redis.call('HMSET', key, 'tokens', newTokens, 'last_refill', now)
        redis.call('PEXPIRE', key, math.ceil(capacity / refillRate) + 1000)
        return {math.floor(newTokens), 0}
      end
    `;
  }

  async connect() {
    try {
      await this.client.connect();
      this.isHealthy = true;
    } catch (err) {
      throw new StoreConnectionError('Failed to connect to Redis', {
        host: config.redis.host,
        port: config.redis.port,
        originalError: err.message,
      });
    }
  }

  _registerEventHandlers() {
    this.client.on('connect', () => {
      this.isHealthy = true;
      // In production: emit to your metrics/alerting system
      console.info('[RedisStore] Connected');
    });

    this.client.on('ready', () => {
      this.isHealthy = true;
    });

    this.client.on('error', (err) => {
      // CRITICAL: Must handle this event or Node.js will crash on unhandled error
      this.isHealthy = false;
      // In production: increment error counter, alert if sustained
      console.error('[RedisStore] Error:', err.message);
    });

    this.client.on('close', () => {
      this.isHealthy = false;
      console.warn('[RedisStore] Connection closed');
    });

    this.client.on('reconnecting', (delay) => {
      console.warn(`[RedisStore] Reconnecting in ${delay}ms`);
    });
  }

  /**
   * Execute sliding window log check atomically via Lua.
   * Lua scripts in Redis are atomic — no race conditions between
   * the read-check-write operations.
   */
  async slidingWindowCheck({ key, windowMs, maxRequests, now }) {
    const result = await this.client.eval(
      this.slidingWindowScript,
      1,           // number of keys
      key,         // KEYS[1]
      windowMs,    // ARGV[1]
      now,         // ARGV[2]
      maxRequests  // ARGV[3]
    );

    return {
      count: result[0],
      retryAfterMs: result[1],
      allowed: result[0] <= maxRequests,
    };
  }

  /**
   * Execute token bucket check atomically via Lua.
   */
  async tokenBucketCheck({ key, capacity, refillRatePerMs, now, requested = 1 }) {
    const result = await this.client.eval(
      this.tokenBucketScript,
      1,
      key,
      capacity,
      refillRatePerMs,
      now,
      requested
    );

    return {
      tokensRemaining: result[0],
      allowed: result[1] === 1,
    };
  }

  async ping() {
    try {
      const result = await this.client.ping();
      return result === 'PONG';
    } catch {
      return false;
    }
  }

  async disconnect() {
    await this.client.quit();
  }
}

module.exports = RedisStore;

### `src/stores/memoryStore.js`

```javascript
/**
 * In-memory store for single-instance deployments or local development.
 *
 * OPERATIONAL WARNING: This store does NOT share state across processes.
 * If you run multiple instances (load balanced, PM2 cluster, k8s pods),
 * each instance enforces limits independently, meaning your effective
 * limit is maxRequests * instanceCount. Do not use in production
 * multi-instance deployments without understanding this.
 */
class MemoryStore {
  constructor() {
    this.windows = new Map();
    this.buckets = new Map();
    this.isHealthy = true;

    // Periodic cleanup to prevent unbounded memory growth.
    // In production, monitor memory usage of this store.
    this._cleanupInterval = setInterval(() => this._cleanup(), 60_000);

    // Allow Node.js to exit even if this interval is running
    if (this._cleanupInterval.unref) {
      this._cleanupInterval.unref();
    }
  }

  async slidingWindowCheck({ key, windowMs, maxRequests, now }) {
    if (!this.windows.has(key)) {
      this.windows.set(key, []);
    }

    const timestamps = this.windows.get(key);
    const windowStart = now - windowMs;

    // Remove expired entries
    while (timestamps.length > 0 && timestamps[0] <= windowStart) {
      timestamps.shift();
    }

    const count = timestamps.length;

    if (count < maxRequests) {
      timestamps.push(now);
      return { count: count + 1, retryAfterMs: 0, allowed: true };
    }

    const retryAfterMs = timestamps[0] + windowMs - now;
    return { count, retryAfterMs: Math.max(0, retryAfterMs), allowed: false };
  }

  async tokenBucketCheck({ key, capacity, refillRatePerMs, now, requested = 1 }) {
    if (!this.buckets.has(key)) {
      this.buckets.set(key, { tokens: capacity, lastRefill: now });
    }

    const bucket = this.buckets.get(key);
    const elapsed = now - bucket.lastRefill;
    bucket.tokens = Math.min(capacity, bucket.tokens + elapsed * refillRatePerMs);
    bucket.lastRefill = now;

    if (bucket.tokens >= requested) {
      bucket.tokens -= requested;
      return { tokensRemaining: Math.floor(bucket.tokens), allowed: true };
    }

    return { tokensRemaining: Math.floor(bucket.tokens), allowed: false };
  }

  _cleanup() {
    const now = Date.now();

    // Clean up sliding window entries older than any reasonable window
    for (const [key, timestamps] of this.windows.entries()) {
      if (timestamps.length === 0) {
        this.windows.delete(key);
      }
    }

    // Clean up stale token buckets (no activity for 10 minutes)
    for (const [key, bucket] of this.buckets.entries()) {
      if (now - bucket.lastRefill > 600_000) {
        this.buckets.delete(key);
      }
    }
  }

  async ping() {
    return true;
  }

  async disconnect() {
    clearInterval(this._cleanupInterval);
  }
}

module.exports = MemoryStore;
```

### `src/algorithms/slidingWindowCounter.js`

```javascript
/**
 * Sliding Window Counter
 *
 * A memory-efficient approximation of sliding window log.
 * Uses two fixed windows (current + previous) and weights the previous
 * window's count by how far into the current window we are.
 *
 * Trade-off: ~0-10% error rate at window boundaries vs O(1) space per key.
 * For most APIs, this is the right default — accurate enough, cheap to run.
 *
 * Formula:
 *   estimated_count = prev_count * (1 - elapsed/window) + curr_count
 */
class SlidingWindowCounter {
  constructor(store) {
    this.store = store;
  }

  async check({ key, windowMs, maxRequests, now = Date.now() }) {
    const windowIndex = Math.floor(now / windowMs);
    const currentKey = `${key}:${windowIndex}`;
    const previousKey = `${key}:${windowIndex - 1}`;

    // In a real implementation, this would use
