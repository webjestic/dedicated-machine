### `src/storage/memoryStore.js`

In-memory store with TTL cleanup — suitable for single-process or testing.

```javascript
'use strict';

/**
 * In-memory store with automatic TTL-based expiration.
 * Not suitable for multi-instance deployments — use RedisStore instead.
 */
class MemoryStore {
  constructor({ cleanupIntervalMs = 60_000 } = {}) {
    // Map<key, { value, expiresAt }>
    this._store = new Map();
    this._cleanupInterval = setInterval(
      () => this._cleanup(),
      cleanupIntervalMs
    ).unref(); // Don't block process exit
  }

  /**
   * Atomically increment a counter and set TTL if it doesn't exist.
   * Returns the new value and whether this was the first increment.
   */
  async increment(key, windowSizeMs) {
    const now = Date.now();
    const existing = this._store.get(key);

    if (existing && existing.expiresAt > now) {
      existing.value += 1;
      return { value: existing.value, ttl: existing.expiresAt - now };
    }

    // Key doesn't exist or has expired — start a new window
    const expiresAt = now + windowSizeMs;
    this._store.set(key, { value: 1, expiresAt });
    return { value: 1, ttl: windowSizeMs };
  }

  async get(key) {
    const now = Date.now();
    const entry = this._store.get(key);
    if (!entry || entry.expiresAt <= now) return null;
    return { value: entry.value, ttl: entry.expiresAt - now };
  }

  async set(key, value, ttlMs) {
    this._store.set(key, {
      value,
      expiresAt: Date.now() + ttlMs,
    });
  }

  async delete(key) {
    this._store.delete(key);
  }

  async reset(key) {
    this._store.delete(key);
  }

  _cleanup() {
    const now = Date.now();
    for (const [key, entry] of this._store) {
      if (entry.expiresAt <= now) {
        this._store.delete(key);
      }
    }
  }

  destroy() {
    clearInterval(this._cleanupInterval);
    this._store.clear();
  }
}

module.exports = MemoryStore;
```

### `src/storage/redisStore.js`

Redis-backed store using atomic Lua scripts — safe for distributed deployments.

```javascript
'use strict';

const { StorageError } = require('../errors');

/**
 * Redis-backed store using atomic Lua scripts.
 * Compatible with ioredis and node-redis v4+.
 *
 * All critical operations are atomic via Lua to prevent race conditions
 * in distributed environments.
 */
class RedisStore {
  constructor(redisClient, { keyPrefix = 'rl:' } = {}) {
    if (!redisClient) throw new Error('RedisStore requires a Redis client');
    this._client = redisClient;
    this._keyPrefix = keyPrefix;

    // Lua script: atomically increment and set expiry on first call
    // Returns [newValue, ttlMs]
    this._incrementScript = `
      local key = KEYS[1]
      local windowMs = tonumber(ARGV[1])
      local now = tonumber(ARGV[2])

      local current = redis.call('GET', key)

      if current == false then
        redis.call('SET', key, 1, 'PX', windowMs)
        return {1, windowMs}
      else
        local newVal = redis.call('INCR', key)
        local pttl = redis.call('PTTL', key)
        -- Safeguard: if TTL was lost somehow, reset it
        if pttl < 0 then
          redis.call('PEXPIRE', key, windowMs)
          pttl = windowMs
        end
        return {newVal, pttl}
      end
    `;

    // Lua script for sliding window log — add timestamp, trim old entries, count
    this._slidingWindowScript = `
      local key = KEYS[1]
      local now = tonumber(ARGV[1])
      local windowMs = tonumber(ARGV[2])
      local maxRequests = tonumber(ARGV[3])
      local windowStart = now - windowMs

      -- Remove entries outside the window
      redis.call('ZREMRANGEBYSCORE', key, '-inf', windowStart)

      -- Count current entries
      local count = redis.call('ZCARD', key)

      if count < maxRequests then
        -- Add current request with score = timestamp
        redis.call('ZADD', key, now, now .. '-' .. math.random(1000000))
        redis.call('PEXPIRE', key, windowMs)
        return {1, count + 1}  -- allowed, new count
      else
        local oldest = redis.call('ZRANGE', key, 0, 0, 'WITHSCORES')
        local resetAt = 0
        if oldest[2] then
          resetAt = tonumber(oldest[2]) + windowMs
        end
        return {0, count, resetAt}  -- blocked, count, resetAt
      end
    `;
  }

  _prefixKey(key) {
    return `${this._keyPrefix}${key}`;
  }

  async increment(key, windowSizeMs) {
    try {
      const prefixedKey = this._prefixKey(key);
      const result = await this._client.eval(
        this._incrementScript,
        1,
        prefixedKey,
        windowSizeMs.toString(),
        Date.now().toString()
      );
      return { value: result[0], ttl: result[1] };
    } catch (err) {
      throw new StorageError(`Redis increment failed for key: ${key}`, err);
    }
  }

  async slidingWindowCheck(key, windowSizeMs, maxRequests) {
    try {
      const prefixedKey = this._prefixKey(key);
      const result = await this._client.eval(
        this._slidingWindowScript,
        1,
        prefixedKey,
        Date.now().toString(),
        windowSizeMs.toString(),
        maxRequests.toString()
      );
      return {
        allowed: result[0] === 1,
        count: result[1],
        resetAt: result[2] || null,
      };
    } catch (err) {
      throw new StorageError(`Redis sliding window check failed for key: ${key}`, err);
    }
  }

  async get(key) {
    try {
      const val = await this._client.get(this._prefixKey(key));
      if (val === null) return null;
      const ttl = await this._client.pttl(this._prefixKey(key));
      return { value: JSON.parse(val), ttl };
    } catch (err) {
      throw new StorageError(`Redis get failed for key: ${key}`, err);
    }
  }

  async set(key, value, ttlMs) {
    try {
      await this._client.set(
        this._prefixKey(key),
        JSON.stringify(value),
        'PX',
        ttlMs
      );
    } catch (err) {
      throw new StorageError(`Redis set failed for key: ${key}`, err);
    }
  }

  async delete(key) {
    try {
      await this._client.del(this._prefixKey(key));
    } catch (err) {
      throw new StorageError(`Redis delete failed for key: ${key}`, err);
    }
  }

  async reset(key) {
    return this.delete(key);
  }
}

module.exports = RedisStore;
```

### `src/algorithms/fixedWindow.js`

```javascript
'use strict';

/**
 * Fixed Window Counter
 *
 * Simple and memory-efficient. Divides time into fixed windows and
 * counts requests per window.
 *
 * Weakness: Burst vulnerability at window boundaries — a client can
 * send 2x the limit by hitting the end of one window and start of the next.
 * Use SlidingWindow if this matters for your use case.
 */
class FixedWindowAlgorithm {
  constructor(store, { windowSizeMs, maxRequests }) {
    if (!windowSizeMs || windowSizeMs <= 0)
      throw new Error('windowSizeMs must be a positive number');
    if (!maxRequests || maxRequests <= 0)
      throw new Error('maxRequests must be a positive number');

    this._store = store;
    this._windowSizeMs = windowSizeMs;
    this._maxRequests = maxRequests;
  }

  /**
   * @param {string} key - Unique identifier (e.g., IP, user ID)
   * @returns {Promise<RateLimitResult>}
   */
  async check(key) {
    const { value, ttl } = await this._store.increment(key, this._windowSizeMs);

    const remaining = Math.max(0, this._maxRequests - value);
    const resetAt = new Date(Date.now() + ttl);
    const allowed = value <= this._maxRequests;

    return {
      allowed,
      limit: this._maxRequests,
      remaining,
      resetAt,
      retryAfter: allowed ? null : Math.ceil(ttl / 1000),
    };
  }
}

module.exports = FixedWindowAlgorithm;
```

### `src/algorithms/slidingWindow.js`

```javascript
'use strict';

/**
 * Sliding Window Log
 *
 * Tracks exact timestamps of each request in a sorted set.
 * Provides precise rate limiting without boundary burst issues.
 *
 * Trade-off: Higher memory usage (O(requests) per key vs O(1) for fixed window).
 * Best for: APIs where accuracy matters more than memory efficiency.
 *
 * Note: Full accuracy requires Redis (sorted sets). MemoryStore falls back
 * to an approximation using two fixed windows.
 */
class SlidingWindowAlgorithm {
  constructor(store, { windowSizeMs, maxRequests }) {
    if (!windowSizeMs || windowSizeMs <= 0)
      throw new Error('windowSizeMs must be a positive number');
    if (!maxRequests || maxRequests <= 0)
      throw new Error('maxRequests must be a positive number');

    this._store = store;
    this._windowSizeMs = windowSizeMs;
    this._maxRequests = maxRequests;
    this._hasNativeSlidingWindow = typeof store.slidingWindowCheck === 'function';
  }

  async check(key) {
    // Use native Redis sorted set implementation if available
    if (this._hasNativeSlidingWindow) {
      return this._checkRedis(key);
    }
    // Fall back to sliding window approximation for MemoryStore
    return this._checkApproximation(key);
  }

  async _checkRedis(key) {
    const { allowed, count, resetAt } = await this._store.slidingWindowCheck(
      key,
      this._windowSizeMs,
      this._maxRequests
    );

    const remaining = Math.max(0, this._maxRequests - count);
    const resetDate = resetAt ? new Date(resetAt) : new Date(Date.now() + this._windowSizeMs);

    return {
      allowed,
      limit: this._maxRequests,
      remaining: allowed ? remaining : 0,
      resetAt: resetDate,
      retryAfter: allowed ? null : Math.ceil((resetDate - Date.now()) / 1000),
    };
  }

  /**
   * Approximation: uses two fixed windows to estimate sliding count.
   * Formula: count = (previousWindow * overlap%) + currentWindow
   * This is the same approach used by Cloudflare and Nginx.
   */
  async _checkApproximation(key) {
    const now = Date.now();
    const currentWindowStart = Math.floor(now / this._windowSizeMs) * this._windowSizeMs;
    const previousWindowKey = `${key}:${currentWindowStart - this._windowSizeMs}`;
    const currentWindowKey = `${key}:${currentWindowStart}`;

    const [prevEntry, currEntry] = await Promise.all([
      this._store.get(previousWindowKey),
      this._store.get(currentWindowKey),
    ]);

    const previousCount = prevEntry?.value || 0;
    const currentCount = currEntry?.value || 0;

    // How far into the current window are we? (0.0 to 1.0)
    const elapsedInWindow = now - currentWindowStart;
    const overlapFraction = 1 - elapsedInWindow / this._windowSizeMs;

    const estimatedCount = Math.floor(previousCount * overlapFraction) + currentCount;
    const allowed = estimatedCount < this._maxRequests;

    if (allowed) {
      await this._store.increment(currentWindowKey, this._windowSizeMs * 2);
    }

    const remaining = Math.max(0, this._maxRequests - estimatedCount - (allowed ? 1 : 0));
    const resetAt = new Date(currentWindowStart + this._windowSizeMs);

    return {
      allowed,
      limit: this._maxRequests,
      remaining,
      resetAt,
      retryAfter: allowed ? null : Math.ceil((resetAt - now) / 1000),
    };
  }
}

module.exports = SlidingWindowAlgorithm;
```
