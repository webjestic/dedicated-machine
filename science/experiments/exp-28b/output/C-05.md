## 2. Memory Store — `src/stores/memoryStore.js`

```javascript
/**
 * In-memory store for rate limiter data.
 * Suitable for single-instance applications.
 */
class MemoryStore {
  constructor() {
    this.store = new Map();
    this._startCleanupTimer();
  }

  /**
   * Get the current record for a key.
   * @param {string} key
   * @returns {object|null}
   */
  async get(key) {
    const record = this.store.get(key);
    if (!record) return null;

    // Return null if the record has expired
    if (Date.now() > record.expiresAt) {
      this.store.delete(key);
      return null;
    }

    return record;
  }

  /**
   * Set a record for a key with a TTL.
   * @param {string} key
   * @param {object} value
   * @param {number} ttlMs - Time to live in milliseconds
   */
  async set(key, value, ttlMs) {
    this.store.set(key, {
      ...value,
      expiresAt: Date.now() + ttlMs,
    });
  }

  /**
   * Increment a counter for a key.
   * @param {string} key
   * @param {number} ttlMs
   * @returns {number} New count
   */
  async increment(key, ttlMs) {
    const record = await this.get(key);

    if (!record) {
      await this.set(key, { count: 1 }, ttlMs);
      return 1;
    }

    record.count += 1;
    this.store.set(key, record);
    return record.count;
  }

  /**
   * Delete a key from the store.
   * @param {string} key
   */
  async delete(key) {
    this.store.delete(key);
  }

  /**
   * Periodically clean up expired keys to prevent memory leaks.
   */
  _startCleanupTimer() {
    this._cleanupInterval = setInterval(() => {
      const now = Date.now();
      for (const [key, record] of this.store.entries()) {
        if (now > record.expiresAt) {
          this.store.delete(key);
        }
      }
    }, 60_000); // Run every 60 seconds

    // Allow the process to exit even if this timer is active
    if (this._cleanupInterval.unref) {
      this._cleanupInterval.unref();
    }
  }

  destroy() {
    clearInterval(this._cleanupInterval);
    this.store.clear();
  }
}

module.exports = MemoryStore;
```

/**
 * Redis-backed store for rate limiter data.
 * Suitable for distributed / multi-instance applications.
 */
class RedisStore {
  /**
   * @param {import('ioredis').Redis} redisClient - An ioredis client instance
   */
  constructor(redisClient) {
    if (!redisClient) {
      throw new Error("RedisStore requires a Redis client instance.");
    }
    this.client = redisClient;
  }

  /**
   * Get the current record for a key.
   * @param {string} key
   * @returns {object|null}
   */
  async get(key) {
    const data = await this.client.get(key);
    return data ? JSON.parse(data) : null;
  }

  /**
   * Set a record for a key with a TTL.
   * @param {string} key
   * @param {object} value
   * @param {number} ttlMs
   */
  async set(key, value, ttlMs) {
    await this.client.set(key, JSON.stringify(value), "PX", ttlMs);
  }

  /**
   * Atomically increment a counter using a Lua script.
   * Ensures thread-safety in distributed environments.
   * @param {string} key
   * @param {number} ttlMs
   * @returns {number} New count
   */
  async increment(key, ttlMs) {
    const luaScript = `
      local current = redis.call('GET', KEYS[1])
      if current == false then
        redis.call('SET', KEYS[1], 1, 'PX', ARGV[1])
        return 1
      else
        return redis.call('INCR', KEYS[1])
      end
    `;
    return await this.client.eval(luaScript, 1, key, ttlMs);
  }

  /**
   * Delete a key from the store.
   * @param {string} key
   */
  async delete(key) {
    await this.client.del(key);
  }
}

module.exports = RedisStore;

## 4. Fixed Window Strategy — `src/strategies/fixedWindow.js`

```javascript
/**
 * Fixed Window Rate Limiting Strategy
 *
 * Counts requests in fixed time windows (e.g., 0-60s, 60-120s).
 * Simple and memory-efficient, but susceptible to burst traffic
 * at window boundaries.
 */
class FixedWindowStrategy {
  /**
   * @param {object} options
   * @param {number} options.windowMs - Window size in milliseconds
   * @param {number} options.maxRequests - Max requests per window
   * @param {object} options.store - Storage backend
   */
  constructor({ windowMs, maxRequests, store }) {
    this.windowMs = windowMs;
    this.maxRequests = maxRequests;
    this.store = store;
  }

  /**
   * Check if a request is allowed for the given key.
   * @param {string} key - Unique identifier (e.g., IP address)
   * @returns {object} Result with allowed status and metadata
   */
  async check(key) {
    const windowKey = `fw:${key}:${Math.floor(Date.now() / this.windowMs)}`;
    const count = await this.store.increment(windowKey, this.windowMs);

    const remaining = Math.max(0, this.maxRequests - count);
    const allowed = count <= this.maxRequests;

    return {
      allowed,
      count,
      remaining,
      resetTime: new Date(
        Math.ceil(Date.now() / this.windowMs) * this.windowMs
      ),
      strategy: "fixed-window",
    };
  }
}

module.exports = FixedWindowStrategy;
```

## 5. Sliding Window Strategy — `src/strategies/slidingWindow.js`

```javascript
/**
 * Sliding Window Rate Limiting Strategy
 *
 * Tracks individual request timestamps within a rolling time window.
 * More accurate than fixed window — eliminates boundary burst issues.
 * Higher memory usage due to storing individual timestamps.
 */
class SlidingWindowStrategy {
  /**
   * @param {object} options
   * @param {number} options.windowMs - Window size in milliseconds
   * @param {number} options.maxRequests - Max requests per window
   * @param {object} options.store - Storage backend
   */
  constructor({ windowMs, maxRequests, store }) {
    this.windowMs = windowMs;
    this.maxRequests = maxRequests;
    this.store = store;
  }

  /**
   * Check if a request is allowed for the given key.
   * @param {string} key
   * @returns {object} Result with allowed status and metadata
   */
  async check(key) {
    const storeKey = `sw:${key}`;
    const now = Date.now();
    const windowStart = now - this.windowMs;

    // Retrieve existing timestamps
    const record = await this.store.get(storeKey);
    let timestamps = record ? record.timestamps : [];

    // Remove timestamps outside the current window
    timestamps = timestamps.filter((ts) => ts > windowStart);

    const count = timestamps.length;
    const allowed = count < this.maxRequests;

    if (allowed) {
      timestamps.push(now);
      await this.store.set(storeKey, { timestamps }, this.windowMs);
    }

    const remaining = Math.max(0, this.maxRequests - timestamps.length);
    const oldestTimestamp = timestamps[0];
    const resetTime = oldestTimestamp
      ? new Date(oldestTimestamp + this.windowMs)
      : new Date(now + this.windowMs);

    return {
      allowed,
      count: timestamps.length,
      remaining,
      resetTime,
      strategy: "sliding-window",
    };
  }
}

module.exports = SlidingWindowStrategy;
```

## 6. Token Bucket Strategy — `src/strategies/tokenBucket.js`

```javascript
/**
 * Token Bucket Rate Limiting Strategy
 *
 * Tokens accumulate at a steady refill rate up to a maximum capacity.
 * Each request consumes one token. Allows controlled bursting.
 * Ideal for APIs that want to allow short bursts while enforcing
 * an average rate over time.
 */
class TokenBucketStrategy {
  /**
   * @param {object} options
   * @param {number} options.capacity - Max tokens (burst size)
   * @param {number} options.refillRate - Tokens added per second
   * @param {object} options.store - Storage backend
   */
  constructor({ capacity, refillRate, store }) {
    this.capacity = capacity;
    this.refillRate = refillRate; // tokens per second
    this.store = store;
  }

  /**
   * Check if a request is allowed for the given key.
   * @param {string} key
   * @returns {object} Result with allowed status and metadata
   */
  async check(key) {
    const storeKey = `tb:${key}`;
    const now = Date.now();

    const record = await this.store.get(storeKey);

    let tokens;
    let lastRefill;

    if (!record) {
      // New client: start with a full bucket
      tokens = this.capacity;
      lastRefill = now;
    } else {
      // Calculate tokens added since last request
      const elapsedSeconds = (now - record.lastRefill) / 1000;
      const tokensToAdd = elapsedSeconds * this.refillRate;
      tokens = Math.min(this.capacity, record.tokens + tokensToAdd);
      lastRefill = now;
    }

    const allowed = tokens >= 1;

    if (allowed) {
      tokens -= 1;
    }

    // TTL: time until bucket would be full from current state
    const ttlMs = ((this.capacity - tokens) / this.refillRate) * 1000 + 5000;
    await this.store.set(storeKey, { tokens, lastRefill }, ttlMs);

    const secondsUntilRefill = allowed ? 0 : (1 - tokens) / this.refillRate;

    return {
      allowed,
      tokens: Math.floor(tokens),
      remaining: Math.floor(tokens),
      resetTime: new Date(now + secondsUntilRefill * 1000),
      strategy: "token-bucket",
    };
  }
}

module.exports = TokenBucketStrategy;
```

## 7. Core Rate Limiter — `src/rateLimiter.js`

```javascript
const MemoryStore = require("./stores/memoryStore");
const FixedWindowStrategy = require("./strategies/fixedWindow");
const SlidingWindowStrategy = require("./strategies/slidingWindow");
const TokenBucketStrategy = require("./strategies/tokenBucket");

const STRATEGIES = {
  "fixed-window": FixedWindowStrategy,
  "sliding-window": SlidingWindowStrategy,
  "token-bucket": TokenBucketStrategy,
};

class RateLimiter {
  /**
   * @param {object} options
   * @param {string}  [options.strategy='fixed-window']  - Rate limiting strategy
   * @param {number}  [options.windowMs=60000]           - Window size in ms (fixed/sliding)
   * @param {number}  [options.maxRequests=100]          - Max requests per window
   * @param {number}  [options.capacity=10]              - Token bucket capacity
   * @param {number}  [options.refillRate=1]             - Token bucket refill rate (tokens/sec)
   * @param {object}  [options.store]                    - Custom store (defaults to MemoryStore)
   * @param {Function}[options.keyGenerator]             - Custom key generator function
   * @param {Function}[options.onLimitReached]           - Callback when limit is hit
   * @param {boolean} [options.skipFailedRequests=false] - Don't count failed responses
   * @param {boolean} [options.skipSuccessfulRequests=false] - Don't count successful responses
   */
  constructor(options = {}) {
    const {
      strategy = "fixed-window",
      windowMs = 60_000,
      maxRequests = 100,
      capacity = 10,
      refillRate = 1,
      store = new MemoryStore(),
      keyGenerator = (req) => req.ip || req.connection?.remoteAddress || "unknown",
      onLimitReached = null,
      skipFailedRequests = false,
      skipSuccessfulRequests = false,
    } = options;

    if (!STRATEGIES[strategy]) {
      throw new Error(
        `Unknown strategy: "${strategy}". Valid options: ${Object.keys(STRATEGIES).join(", ")}`
      );
    }

    this.keyGenerator = keyGenerator;
    this.onLimitReached = onLimitReached;
    this.skipFailedRequests = skipFailedRequests;
    this.skipSuccessfulRequests = skipSuccessfulRequests;

    // Instantiate the chosen strategy
    const strategyOptions = { windowMs, maxRequests, capacity, refillRate, store };
    this.strategy = new STRATEGIES[strategy](strategyOptions);
  }

  /**
   * Check rate limit for a given key without side effects.
   * Useful for manual checks outside of HTTP middleware.
   * @param {string} key
   * @returns {Promise<object>}
