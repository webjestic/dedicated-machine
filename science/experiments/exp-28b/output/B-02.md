'use strict';

const { createClient } = require('redis');

/**
 * Redis store with connection lifecycle management.
 *
 * Design decisions:
 * - Separate store from limiter logic so you can swap backends
 * - Explicit connect/disconnect — never rely on implicit connection state
 * - Circuit breaker flag so Redis failure doesn't take down your app
 */
class RedisStore {
  constructor(options = {}) {
    this.options = {
      url: options.url || process.env.REDIS_URL || 'redis://localhost:6379',
      socket: {
        connectTimeout: options.connectTimeout || 5000,
        lazyConnect: false,
      },
      // Retry with backoff — don't hammer a recovering Redis
      retry_strategy: options.retryStrategy || defaultRetryStrategy,
      ...options.redisOptions,
    };

    this.client = null;
    this.isConnected = false;
    this.isShuttingDown = false;
  }

  async connect() {
    if (this.client && this.isConnected) return;

    this.client = createClient(this.options);

    this.client.on('connect', () => {
      this.isConnected = true;
    });

    this.client.on('error', (err) => {
      // Log but don't throw — handled at call sites
      console.error('[RedisStore] Connection error:', err.message);
      this.isConnected = false;
    });

    this.client.on('reconnecting', () => {
      console.warn('[RedisStore] Reconnecting...');
    });

    this.client.on('end', () => {
      if (!this.isShuttingDown) {
        console.warn('[RedisStore] Connection closed unexpectedly');
      }
      this.isConnected = false;
    });

    await this.client.connect();
    this.isConnected = true;
  }

  async disconnect() {
    this.isShuttingDown = true;
    if (this.client) {
      await this.client.quit();
      this.client = null;
      this.isConnected = false;
    }
  }

  getClient() {
    if (!this.client || !this.isConnected) {
      throw new RedisUnavailableError('Redis client is not connected');
    }
    return this.client;
  }

  async ping() {
    try {
      const result = await this.client.ping();
      return result === 'PONG';
    } catch {
      return false;
    }
  }
}

function defaultRetryStrategy(retries) {
  if (retries > 10) return new Error('Redis retry limit exceeded');
  return Math.min(retries * 100, 3000); // cap at 3s
}

class RedisUnavailableError extends Error {
  constructor(message) {
    super(message);
    this.name = 'RedisUnavailableError';
  }
}

module.exports = { RedisStore, RedisUnavailableError };
```

### `src/limiters/sliding-window-redis.js`
```javascript
'use strict';

/**
 * Sliding Window Log Rate Limiter — Redis backend
 *
 * Algorithm:
 *   For each request, store a timestamped entry in a sorted set.
 *   Count entries within [now - windowMs, now].
 *   If count >= limit, reject.
 *
 * Why sorted sets:
 *   ZRANGEBYSCORE gives O(log N + M) range queries.
 *   ZREMRANGEBYSCORE prunes expired entries atomically with the check.
 *
 * Why a Lua script:
 *   The check-then-increment is a race condition without atomicity.
 *   Two concurrent requests at the limit would both pass the check
 *   before either increments. Lua scripts run atomically in Redis.
 *
 * Tradeoff vs sliding window counter:
 *   This is more accurate but uses more memory (one entry per request
 *   vs two counters). For high-volume keys, consider sliding window
 *   counter instead. See token-bucket-redis.js for an alternative.
 */

const crypto = require('crypto');

// Atomic Lua script — this is the critical section
// Returns: [current_count, requests_remaining, oldest_request_timestamp]
const SLIDING_WINDOW_SCRIPT = `
local key = KEYS[1]
local now = tonumber(ARGV[1])
local window_ms = tonumber(ARGV[2])
local limit = tonumber(ARGV[3])
local request_id = ARGV[4]
local window_start = now - window_ms

-- Remove entries outside the current window
redis.call('ZREMRANGEBYSCORE', key, '-inf', window_start)

-- Count current requests in window
local current_count = redis.call('ZCARD', key)

if current_count < limit then
  -- Add this request with timestamp as score
  redis.call('ZADD', key, now, request_id)
  -- Set TTL so keys self-clean (window_ms + buffer in seconds)
  redis.call('PEXPIRE', key, window_ms + 1000)
  return {current_count + 1, limit - current_count - 1, 0, 1}
else
  -- Get oldest entry to tell client when their window resets
  local oldest = redis.call('ZRANGE', key, 0, 0, 'WITHSCORES')
  local reset_at = 0
  if #oldest > 0 then
    reset_at = tonumber(oldest[2]) + window_ms
  end
  return {current_count, 0, reset_at, 0}
end
`;

class SlidingWindowRedisLimiter {
  /**
   * @param {object} store - RedisStore instance
   * @param {object} options
   * @param {number} options.limit - Max requests per window
   * @param {number} options.windowMs - Window size in milliseconds
   * @param {string} [options.keyPrefix] - Redis key namespace
   * @param {function} [options.keyGenerator] - (req) => string
   * @param {function} [options.onLimitReached] - Hook for logging/alerting
   * @param {boolean} [options.failOpen] - Allow requests if Redis is down
   */
  constructor(store, options = {}) {
    this.validateOptions(options);

    this.store = store;
    this.limit = options.limit;
    this.windowMs = options.windowMs;
    this.keyPrefix = options.keyPrefix || 'rl';
    this.keyGenerator = options.keyGenerator || defaultKeyGenerator;
    this.onLimitReached = options.onLimitReached || null;
    this.failOpen = options.failOpen !== undefined ? options.failOpen : true;

    // Load script SHA for EVALSHA (more efficient than EVAL on every call)
    this._scriptSha = null;
  }

  validateOptions(options) {
    if (!options.limit || typeof options.limit !== 'number' || options.limit < 1) {
      throw new TypeError('options.limit must be a positive number');
    }
    if (!options.windowMs || typeof options.windowMs !== 'number' || options.windowMs < 100) {
      throw new TypeError('options.windowMs must be a number >= 100ms');
    }
  }

  async _loadScript() {
    if (this._scriptSha) return this._scriptSha;

    const client = this.store.getClient();
    this._scriptSha = await client.scriptLoad(SLIDING_WINDOW_SCRIPT);
    return this._scriptSha;
  }

  /**
   * Core check method — call this directly if not using Express middleware
   *
   * @param {string} identifier - The key to rate limit on (IP, user ID, API key)
   * @returns {RateLimitResult}
   */
  async check(identifier) {
    const key = `${this.keyPrefix}:${identifier}`;
    const now = Date.now();
    const requestId = `${now}-${crypto.randomBytes(4).toString('hex')}`;

    try {
      const sha = await this._loadScript();
      const client = this.store.getClient();

      const result = await client.evalSha(sha, {
        keys: [key],
        arguments: [
          now.toString(),
          this.windowMs.toString(),
          this.limit.toString(),
          requestId,
        ],
      });

      const [count, remaining, resetAt, allowed] = result;
      const isAllowed = allowed === 1;

      if (!isAllowed && this.onLimitReached) {
        // Don't await — this is a side effect, don't let it slow the response
        Promise.resolve(
          this.onLimitReached({ identifier, key, count, limit: this.limit })
        ).catch((err) => console.error('[RateLimiter] onLimitReached hook error:', err));
      }

      return {
        allowed: isAllowed,
        limit: this.limit,
        remaining: Math.max(0, remaining),
        count,
        resetAt: resetAt ? new Date(resetAt) : new Date(now + this.windowMs),
        retryAfter: isAllowed ? null : Math.ceil((resetAt - now) / 1000),
      };
    } catch (err) {
      return this._handleStoreError(err, identifier);
    }
  }

  _handleStoreError(err, identifier) {
    console.error('[RateLimiter] Store error:', {
      error: err.message,
      identifier,
      failOpen: this.failOpen,
    });

    if (this.failOpen) {
      // Allow the request through — availability over strict limiting
      return {
        allowed: true,
        limit: this.limit,
        remaining: -1, // Sentinel: indicates degraded mode
        count: -1,
        resetAt: new Date(Date.now() + this.windowMs),
        retryAfter: null,
        degraded: true, // Caller can log/alert on this
      };
    }

    // Fail closed — reject when Redis is unavailable
    // Use this for security-critical endpoints
    return {
      allowed: false,
      limit: this.limit,
      remaining: 0,
      count: -1,
      resetAt: new Date(Date.now() + this.windowMs),
      retryAfter: 1,
      degraded: true,
    };
  }

  async reset(identifier) {
    const key = `${this.keyPrefix}:${identifier}`;
    try {
      const client = this.store.getClient();
      await client.del(key);
    } catch (err) {
      console.error('[RateLimiter] Reset error:', err.message);
    }
  }

  async getStatus(identifier) {
    const key = `${this.keyPrefix}:${identifier}`;
    const now = Date.now();
    const windowStart = now - this.windowMs;

    try {
      const client = this.store.getClient();
      await client.zRemRangeByScore(key, '-inf', windowStart);
      const count = await client.zCard(key);

      return {
        identifier,
        count,
        limit: this.limit,
        remaining: Math.max(0, this.limit - count),
        windowMs: this.windowMs,
      };
    } catch (err) {
      throw new Error(`Failed to get rate limit status: ${err.message}`);
    }
  }
}

function defaultKeyGenerator(req) {
  // X-Forwarded-For requires trust — see operational notes
  const forwarded = req.headers['x-forwarded-for'];
  const ip = forwarded ? forwarded.split(',')[0].trim() : req.socket.remoteAddress;
  return ip || 'unknown';
}

module.exports = { SlidingWindowRedisLimiter };

### `src/limiters/sliding-window-memory.js`
```javascript
'use strict';

/**
 * In-memory sliding window limiter.
 *
 * WHEN TO USE THIS:
 *   - Single-instance deployments only
 *   - Development and testing
 *   - As a first-pass filter before Redis (defense in depth)
 *
 * WHEN NOT TO USE THIS:
 *   - Any multi-instance deployment — limits are per-process, not global
 *   - If your process restarts frequently (state is lost)
 *
 * Memory management:
 *   Without cleanup, the timestamps Map grows unbounded.
 *   The cleanup interval prunes expired windows.
 *   Tune cleanupIntervalMs based on your key cardinality.
 */
class SlidingWindowMemoryLimiter {
  constructor(options = {}) {
    if (!options.limit || !options.windowMs) {
      throw new TypeError('options.limit and options.windowMs are required');
    }

    this.limit = options.limit;
    this.windowMs = options.windowMs;
    this.keyPrefix = options.keyPrefix || 'rl';
    this.keyGenerator = options.keyGenerator || defaultKeyGenerator;

    // Map<key, number[]> — array of timestamps
    this._store = new Map();

    // Prevent unbounded memory growth
    this._cleanupInterval = setInterval(
      () => this._cleanup(),
      options.cleanupIntervalMs || 60_000
    );

    // Don't keep process alive just for cleanup
    if (this._cleanupInterval.unref) {
      this._cleanupInterval.unref();
    }
  }

    const key = `${this.keyPrefix}:${identifier}`;
    const now = Date.now();

      this._store.set(key, []);
