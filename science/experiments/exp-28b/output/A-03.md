```javascript
// src/stores/memoryStore.js

/**
 * In-memory store with TTL-based expiry.
 * Fine for single-process / testing. Not suitable for multi-instance deployments.
 */
class MemoryStore {
  constructor() {
    this.store = new Map();

    // Periodic cleanup to prevent unbounded memory growth
    this._cleanupInterval = setInterval(() => this._cleanup(), 60_000);
    this._cleanupInterval.unref(); // Don't block process exit
  }

  async get(key) {
    const entry = this.store.get(key);
    if (!entry) return null;
    if (Date.now() > entry.expiresAt) {
      this.store.delete(key);
      return null;
    }
    return entry.value;
  }

  async set(key, value, ttlMs) {
    this.store.set(key, {
      value,
      expiresAt: Date.now() + ttlMs,
    });
  }

  async increment(key, ttlMs) {
    const entry = this.store.get(key);
    const now = Date.now();

    if (!entry || now > entry.expiresAt) {
      this.store.set(key, { value: 1, expiresAt: now + ttlMs });
      return { count: 1, ttlMs };
    }

    entry.value += 1;
    return { count: entry.value, ttlMs: entry.expiresAt - now };
  }

  async delete(key) {
    this.store.delete(key);
  }

  // Sorted set operations (for sliding window)
  async zAdd(key, score, member, ttlMs) {
    let zset = this.store.get(key);
    const now = Date.now();

    if (!zset || now > zset.expiresAt) {
      zset = { value: new Map(), expiresAt: now + ttlMs };
      this.store.set(key, zset);
    }

    zset.value.set(member, score);
    zset.expiresAt = now + ttlMs; // Refresh TTL
  }

  async zRemRangeByScore(key, min, max) {
    const zset = this.store.get(key);
    if (!zset) return 0;

    let removed = 0;
    for (const [member, score] of zset.value.entries()) {
      if (score >= min && score <= max) {
        zset.value.delete(member);
        removed++;
      }
    }
    return removed;
  }

  async zCard(key) {
    const zset = this.store.get(key);
    if (!zset || Date.now() > zset.expiresAt) return 0;
    return zset.value.size;
  }

  _cleanup() {
    const now = Date.now();
    for (const [key, entry] of this.store.entries()) {
      if (now > entry.expiresAt) {
        this.store.delete(key);
      }
    }
  }

  destroy() {
    clearInterval(this._cleanupInterval);
    this.store.clear();
  }
}

module.exports = MemoryStore;
```

```javascript
// src/stores/redisStore.js
const { createClient } = require('redis');
const { StoreConnectionError } = require('../errors');

/**
 * Redis-backed store. Supports clustering and multi-instance deployments.
 * Uses Lua scripts for atomic operations — critical for correctness under load.
 */
class RedisStore {
  constructor(options = {}) {
    this.client = options.client ?? createClient(options.redisOptions ?? {});
    this.connected = false;
    this._connectPromise = null;

    // Lua script: atomic increment with TTL set only on first call
    this.incrementScript = `
      local current = redis.call('INCR', KEYS[1])
      if current == 1 then
        redis.call('PEXPIRE', KEYS[1], ARGV[1])
      end
      local ttl = redis.call('PTTL', KEYS[1])
      return {current, ttl}
    `;

    // Lua script: atomic sliding window add + trim + count
    this.slidingWindowScript = `
      local key     = KEYS[1]
      local now     = tonumber(ARGV[1])
      local window  = tonumber(ARGV[2])
      local member  = ARGV[3]
      local ttlMs   = ARGV[4]

      -- Remove entries outside the window
      redis.call('ZREMRANGEBYSCORE', key, '-inf', now - window)

      -- Add current request
      redis.call('ZADD', key, now, member)

      -- Refresh expiry
      redis.call('PEXPIRE', key, ttlMs)

      -- Return count
      return redis.call('ZCARD', key)
    `;
  }

  async connect() {
    if (this.connected) return;
    if (this._connectPromise) return this._connectPromise;

    this._connectPromise = this.client.connect()
      .then(() => { this.connected = true; })
      .catch((err) => {
        this._connectPromise = null;
        throw new StoreConnectionError(`Redis connection failed: ${err.message}`);
      });

    return this._connectPromise;
  }

  async increment(key, ttlMs) {
    await this.connect();
    const result = await this.client.eval(this.incrementScript, {
      keys: [key],
      arguments: [String(ttlMs)],
    });
    return { count: result[0], ttlMs: result[1] };
  }

  async get(key) {
    await this.connect();
    const val = await this.client.get(key);
    return val ? JSON.parse(val) : null;
  }

  async set(key, value, ttlMs) {
    await this.connect();
    await this.client.set(key, JSON.stringify(value), { PX: ttlMs });
  }

  async delete(key) {
    await this.connect();
    await this.client.del(key);
  }

  async slidingWindowIncrement(key, nowMs, windowMs, member, ttlMs) {
    await this.connect();
    const count = await this.client.eval(this.slidingWindowScript, {
      keys: [key],
      arguments: [String(nowMs), String(windowMs), member, String(ttlMs)],
    });
    return Number(count);
  }

  async disconnect() {
    if (this.connected) {
      await this.client.disconnect();
      this.connected = false;
    }
  }
}

module.exports = RedisStore;
```

## Algorithms

```javascript
// src/algorithms/fixedWindow.js

/**
 * Fixed Window Counter
 *
 * Simple and memory-efficient. Weakness: burst traffic at window boundaries
 * can allow up to 2x the limit (end of one window + start of next).
 *
 * Best for: coarse-grained limits where boundary bursts are acceptable.
 */
class FixedWindowAlgorithm {
  constructor(store) {
    this.store = store;
  }

  /**
   * @param {string} key       - Rate limit key (e.g., "rl:user:123")
   * @param {number} limit     - Max requests per window
   * @param {number} windowMs  - Window size in milliseconds
   * @returns {{ allowed: boolean, count: number, remaining: number, resetAt: Date, retryAfter: number }}
   */
  async check(key, limit, windowMs) {
    const windowKey = `${key}:${Math.floor(Date.now() / windowMs)}`;
    const { count, ttlMs } = await this.store.increment(windowKey, windowMs);

    const remaining = Math.max(0, limit - count);
    const resetAt = new Date(Date.now() + ttlMs);

    return {
      allowed: count <= limit,
      count,
      remaining,
      resetAt,
      retryAfter: count > limit ? Math.ceil(ttlMs / 1000) : 0,
    };
  }
}

module.exports = FixedWindowAlgorithm;
```

```javascript
// src/algorithms/slidingWindow.js
const { randomUUID } = require('crypto');

/**
 * Sliding Window Log
 *
 * Tracks exact request timestamps in a sorted set. Eliminates the boundary
 * burst problem of fixed windows. More memory-intensive (O(requests) per key).
 *
 * Best for: strict rate limiting where accuracy matters most.
 */
class SlidingWindowAlgorithm {
  constructor(store) {
    this.store = store;
  }

  async check(key, limit, windowMs) {
    const now = Date.now();
    const member = `${now}-${randomUUID()}`;

    let count;

    // Redis store has an optimized atomic Lua script path
    if (typeof this.store.slidingWindowIncrement === 'function') {
      count = await this.store.slidingWindowIncrement(
        key,
        now,
        windowMs,
        member,
        windowMs * 2, // TTL buffer
      );
    } else {
      // Memory store path
      await this.store.zRemRangeByScore(key, 0, now - windowMs);
      await this.store.zAdd(key, now, member, windowMs * 2);
      count = await this.store.zCard(key);
    }

    const remaining = Math.max(0, limit - count);
    const resetAt = new Date(now + windowMs);

    return {
      allowed: count <= limit,
      count,
      remaining,
      resetAt,
      retryAfter: count > limit ? Math.ceil(windowMs / 1000) : 0,
    };
  }
}

module.exports = SlidingWindowAlgorithm;
```

```javascript
// src/algorithms/tokenBucket.js

/**
 * Token Bucket
 *
 * Tokens refill at a steady rate up to a maximum capacity (burst allowance).
 * Naturally handles bursts while enforcing a sustained rate limit.
 *
 * State: { tokens: number, lastRefill: number }
 *
 * Best for: APIs where you want to allow short bursts but control sustained throughput.
 */
class TokenBucketAlgorithm {
  constructor(store) {
    this.store = store;
  }

  /**
   * @param {string} key            - Rate limit key
   * @param {number} capacity       - Max tokens (burst size)
   * @param {number} refillRate     - Tokens added per second
   * @param {number} tokensPerReq   - Tokens consumed per request (default 1)
   */
  async check(key, capacity, refillRate, tokensPerReq = 1) {
    const now = Date.now();
    const ttlMs = Math.ceil((capacity / refillRate) * 1000) * 2; // Generous TTL

    let bucket = await this.store.get(key);

    if (!bucket) {
      // First request: full bucket
      bucket = { tokens: capacity, lastRefill: now };
    } else {
      // Refill tokens based on elapsed time
      const elapsedSeconds = (now - bucket.lastRefill) / 1000;
      const tokensToAdd = elapsedSeconds * refillRate;
      bucket.tokens = Math.min(capacity, bucket.tokens + tokensToAdd);
      bucket.lastRefill = now;
    }

    const allowed = bucket.tokens >= tokensPerReq;

    if (allowed) {
      bucket.tokens -= tokensPerReq;
    }

    await this.store.set(key, bucket, ttlMs);

    const remaining = Math.floor(bucket.tokens);
    const tokensNeeded = tokensPerReq - bucket.tokens;
    const retryAfterSeconds = allowed ? 0 : tokensNeeded / refillRate;

    return {
      allowed,
      tokens: bucket.tokens,
      remaining,
      resetAt: new Date(now + (retryAfterSeconds * 1000)),
      retryAfter: allowed ? 0 : Math.ceil(retryAfterSeconds),
    };
  }
}

module.exports = TokenBucketAlgorithm;
```

// src/RateLimiter.js
const FixedWindowAlgorithm = require('./algorithms/fixedWindow');
const SlidingWindowAlgorithm = require('./algorithms/slidingWindow');
const TokenBucketAlgorithm = require('./algorithms/tokenBucket');
const MemoryStore = require('./stores/memoryStore');
const { RateLimitError } = require('./errors');

const ALGORITHMS = {
  FIXED_WINDOW: 'fixed_window',
  SLIDING_WINDOW: 'sliding_window',
  TOKEN_BUCKET: 'token_bucket',
};

class RateLimiter {
  /**
   * @param {object} options
   * @param {string}  options.algorithm      - One of ALGORITHMS.*
   * @param {object}  options.store          - Storage backend (MemoryStore | RedisStore)
   * @param {number}  options.limit          - Max requests (fixed/sliding) or capacity (token bucket)
   * @param {number}  options.windowMs       - Window size in ms (fixed/sliding window)
   * @param {number}  options.refillRate     - Tokens/second (token bucket)
   * @param {number}  options.tokensPerReq   - Tokens per request (token bucket, default 1)
   * @param {string}  options.keyPrefix      - Prefix for all keys (default 'rl')
   * @param {boolean} options.throwOnLimit   - Throw RateLimitError instead of returning result
   * @param {boolean} options.failOpen       - Allow requests if store errors (default true)
   */
  constructor(options = {}) {
    this.algorithm = options.
