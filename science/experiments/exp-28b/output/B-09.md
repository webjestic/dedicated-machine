# Production Rate Limiter in Node.js

Let me build this properly — working code is the easy part. I'll also tell you exactly what you need to run this safely in production.

## Architecture Decision

I'm implementing a **sliding window log** algorithm backed by Redis. Here's why:

| Algorithm | Pros | Cons |
|---|---|---|
| Fixed window | Simple, fast | Burst at window edges (2x traffic) |
| Token bucket | Smooth bursting | Complex distributed sync |
| Sliding window counter | Fast, low memory | ~10% error at window edges |
| **Sliding window log** | Exact, no edge bursts | More memory per user |

For most production APIs, sliding window log is the right default. I'll make the algorithm swappable.

### `src/config.js`

```javascript
'use strict';

/**
 * All limits, timeouts, and behaviors are explicit and documented.
 * No magic numbers buried in logic.
 */
const defaults = {
  // Window duration in milliseconds
  windowMs: 60_000,

  // Maximum requests allowed per window per key
  max: 100,

  // How to derive the rate limit key from a request.
  // WARNING: IP-based keying is naive — see operational notes.
  keyGenerator: (req) => {
    const forwarded = req.headers['x-forwarded-for'];
    const ip = forwarded
      ? forwarded.split(',')[0].trim()   // leftmost is client IP
      : req.socket?.remoteAddress ?? 'unknown';
    return `rl:${ip}`;
  },

  // When the store is unavailable, should we block or allow traffic?
  // 'open'  — fail open  (allow requests, log the error) — safer for availability
  // 'closed'— fail closed (block requests)               — safer for abuse prevention
  // Choose based on your threat model. Default: fail open.
  failMode: 'open',

  // Skip rate limiting for certain requests (e.g., health checks)
  skip: () => false,

  // Called when a request is rate limited — use for metrics/alerting
  onLimitReached: (req, key, info) => {},

  // Standard headers to return
  standardHeaders: true,   // RateLimit-* (RFC 6585 draft)
  legacyHeaders: false,    // X-RateLimit-* (older clients)

  // Redis key TTL multiplier — keys live longer than the window
  // to handle clock skew and delayed cleanup
  ttlMultiplier: 2,
};

function buildConfig(options = {}) {
  const config = { ...defaults, ...options };

  if (config.max <= 0) throw new Error('config.max must be > 0');
  if (config.windowMs <= 0) throw new Error('config.windowMs must be > 0');
  if (!['open', 'closed'].includes(config.failMode)) {
    throw new Error('config.failMode must be "open" or "closed"');
  }

  return config;
}

module.exports = { buildConfig, defaults };

### `src/store/redisStore.js`

```javascript
'use strict';

const { StoreConnectionError } = require('../errors');

/**
 * Redis store using a Lua script for atomic sliding window log.
 *
 * WHY LUA: The check-then-increment must be atomic. Without it,
 * two concurrent requests can both read "99 requests" and both
 * be allowed, blowing past the limit. Redis executes Lua scripts
 * atomically — no MULTI/EXEC needed.
 *
 * WHY NOT MULTI/EXEC: WATCH + MULTI/EXEC requires retry logic on
 * contention. Under high load this degrades badly. Lua is simpler
 * and faster.
 */

// Script returns: [current_count, ttl_ms_remaining]
// Arguments:
//   KEYS[1]  — the rate limit key
//   ARGV[1]  — current timestamp in ms
//   ARGV[2]  — window size in ms
//   ARGV[3]  — max allowed requests
//   ARGV[4]  — TTL for the key in ms
const SLIDING_WINDOW_LOG_SCRIPT = `
local key        = KEYS[1]
local now        = tonumber(ARGV[1])
local window     = tonumber(ARGV[2])
local max        = tonumber(ARGV[3])
local ttl        = tonumber(ARGV[4])
local window_start = now - window

-- Remove timestamps outside the current window
redis.call('ZREMRANGEBYSCORE', key, '-inf', window_start)

-- Count requests in the current window
local count = redis.call('ZCARD', key)

if count < max then
  -- Add current request with timestamp as both score and member
  -- Member must be unique: timestamp + random suffix handles
  -- multiple requests at the exact same millisecond
  local member = now .. ':' .. redis.call('INCR', key .. ':seq')
  redis.call('ZADD', key, now, member)
  redis.call('PEXPIRE', key, ttl)
  -- Also expire the sequence counter
  redis.call('PEXPIRE', key .. ':seq', ttl)
  count = count + 1
end

-- Return count and time until oldest entry expires
local oldest = redis.call('ZRANGE', key, 0, 0, 'WITHSCORES')
local reset_after = window
if #oldest > 0 then
  reset_after = window - (now - tonumber(oldest[2]))
end

return {count, reset_after, max}
`;

class RedisStore {
  /**
   * @param {import('ioredis').Redis | import('ioredis').Cluster} client
   * @param {object} options
   */
  constructor(client, options = {}) {
    if (!client) throw new Error('RedisStore requires a Redis client');

    this.client = client;
    this.scriptSha = null;
    this.isReady = false;

    // Bind so we can use as event handlers
    this._onReady = this._onReady.bind(this);
    this._onError = this._onError.bind(this);

    this.client.on('ready', this._onReady);
    this.client.on('error', this._onError);

    // If client is already connected
    if (client.status === 'ready') {
      this._loadScript();
    }
  }

  async _onReady() {
    this.isReady = true;
    await this._loadScript();
  }

  _onError(err) {
    this.isReady = false;
    // Don't throw — the rate limiter's failMode handles this
  }

  /**
   * Pre-load the Lua script using SCRIPT LOAD.
   * Subsequent calls use EVALSHA — faster, less bandwidth.
   * If Redis restarts and flushes scripts, we fall back to EVAL.
   */
  async _loadScript() {
    try {
      this.scriptSha = await this.client.script('LOAD', SLIDING_WINDOW_LOG_SCRIPT);
    } catch (err) {
      this.scriptSha = null;
    }
  }

  /**
   * Increment and check the rate limit for a key.
   *
   * @returns {{ count: number, resetAfterMs: number, limit: number, allowed: boolean }}
   */
  async increment(key, windowMs, max, ttlMs) {
    const now = Date.now();
    const args = [key, now, windowMs, max, ttlMs];

    let result;

    try {
      if (this.scriptSha) {
        try {
          result = await this.client.evalsha(this.scriptSha, 1, ...args);
        } catch (err) {
          // NOSCRIPT: Redis flushed its script cache (restart, SCRIPT FLUSH)
          if (err.message && err.message.includes('NOSCRIPT')) {
            await this._loadScript();
            result = await this.client.eval(SLIDING_WINDOW_LOG_SCRIPT, 1, ...args);
          } else {
            throw err;
          }
        }
      } else {
        result = await this.client.eval(SLIDING_WINDOW_LOG_SCRIPT, 1, ...args);
      }
    } catch (err) {
      throw new StoreConnectionError('Redis operation failed', err);
    }

    const [count, resetAfterMs, limit] = result.map(Number);

    return {
      count,
      resetAfterMs: Math.max(0, resetAfterMs),
      limit,
      allowed: count <= max,
    };
  }

  async close() {
    this.client.off('ready', this._onReady);
    this.client.off('error', this._onError);
  }
}

module.exports = { RedisStore };
```

### `src/store/memoryStore.js`

```javascript
'use strict';

/**
 * In-memory store for local development and testing.
 *
 * PRODUCTION WARNING: This store is NOT suitable for production because:
 * 1. State is not shared across multiple Node.js processes or instances
 * 2. State is lost on process restart
 * 3. Memory grows unbounded if cleanup doesn't run (mitigated here with interval)
 *
 * Use RedisStore in production.
 */
class MemoryStore {
  constructor() {
    this.windows = new Map(); // key -> sorted array of timestamps

    // Periodic cleanup of expired windows to prevent memory leak
    this._cleanupInterval = setInterval(() => this._cleanup(), 60_000);
    this._cleanupInterval.unref(); // Don't block process exit
  }

  async increment(key, windowMs, max, ttlMs) {
    const now = Date.now();
    const windowStart = now - windowMs;

    if (!this.windows.has(key)) {
      this.windows.set(key, []);
    }

    const timestamps = this.windows.get(key);

    // Remove expired timestamps
    let i = 0;
    while (i < timestamps.length && timestamps[i] <= windowStart) i++;
    timestamps.splice(0, i);

    const count = timestamps.length;
    const allowed = count < max;

    if (allowed) {
      timestamps.push(now);
    }

    const resetAfterMs = timestamps.length > 0
      ? windowMs - (now - timestamps[0])
      : windowMs;

    return {
      count: allowed ? count + 1 : count,
      resetAfterMs: Math.max(0, resetAfterMs),
      limit: max,
      allowed,
    };
  }

  _cleanup() {
    const now = Date.now();
    for (const [key, timestamps] of this.windows.entries()) {
      if (timestamps.length === 0 || now - timestamps[timestamps.length - 1] > 120_000) {
        this.windows.delete(key);
      }
    }
  }

  close() {
    clearInterval(this._cleanupInterval);
    this.windows.clear();
  }
}

module.exports = { MemoryStore };
```

### `src/middleware/express.js`

```javascript
'use strict';

const { buildConfig } = require('../config');
const { StoreConnectionError } = require('../errors');

/**
 * Express middleware factory.
 *
 * @param {object} store   - RedisStore or MemoryStore instance
 * @param {object} options - See config.js for all options
 */
function createRateLimiter(store, options = {}) {
  if (!store) throw new Error('createRateLimiter requires a store');

  const config = buildConfig(options);

  return async function rateLimiterMiddleware(req, res, next) {
    // Allow certain requests to bypass limiting entirely
    if (config.skip(req)) {
      return next();
    }

    const key = config.keyGenerator(req);
    const ttlMs = config.windowMs * config.ttlMultiplier;

    let result;

    try {
      result = await store.increment(key, config.windowMs, config.max, ttlMs);
    } catch (err) {
      // Store is unavailable — apply failMode policy
      if (err instanceof StoreConnectionError || err.name === 'StoreConnectionError') {
        // Always log — this is an infrastructure problem
        console.error('[rate-limiter] Store error:', {
          key,
          error: err.message,
          cause: err.cause?.message,
        });

        if (config.failMode === 'open') {
          // Allow the request through, but signal degraded state
          res.setHeader('X-RateLimit-Store-Error', '1');
          return next();
        } else {
          // Fail closed — return 503 (not 429, this is our fault not theirs)
          return res.status(503).json({
            error: 'Service temporarily unavailable',
            retryAfter: 5,
          });
        }
      }

      // Unexpected error — don't swallow it
      return next(err);
    }

    req.rateLimit = {
      key,
      limit: result.limit,
      current: result.count,
      remaining: Math.max(0, result.limit - result.count),
      resetAfterMs: result.resetAfterMs,
      allowed: result.allowed,
