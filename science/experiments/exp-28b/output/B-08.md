# Production Rate Limiter in Node.js

Let me build this properly — working code first, then an honest accounting of what separates "it runs" from "it runs safely in production."

---

## Architecture Decisions (made explicit)

| Decision | Choice | Reason |
|---|---|---|
| Algorithm | Sliding window log | Accurate, no boundary burst problem of fixed window |
| Storage | Redis | Shared state across multiple Node processes/instances |
| Fallback | Fail-open with logging | Availability over perfect rate limiting during Redis outage |
| Key strategy | Pluggable | IP, user ID, API key, or composite |

'use strict';

/**
 * Thrown when a request exceeds the configured rate limit.
 * Carries the metadata callers need to build a correct 429 response.
 */
class RateLimitExceededError extends Error {
  /**
   * @param {object} params
   * @param {string} params.key        - The rate limit key that was exceeded
   * @param {number} params.limit      - Maximum requests allowed in the window
   * @param {number} params.windowMs   - Window duration in milliseconds
   * @param {number} params.retryAfter - Milliseconds until the client may retry
   */
  constructor({ key, limit, windowMs, retryAfter }) {
    super(`Rate limit exceeded for key: ${key}`);
    this.name = 'RateLimitExceededError';
    this.key = key;
    this.limit = limit;
    this.windowMs = windowMs;
    this.retryAfter = retryAfter;
    // Preserve stack trace in V8
    if (Error.captureStackTrace) {
      Error.captureStackTrace(this, RateLimitExceededError);
    }
  }
}

/**
 * Thrown when the backing store is unavailable.
 * Consumers decide whether to fail-open or fail-closed.
 */
class StoreUnavailableError extends Error {
  constructor(message, cause) {
    super(message);
    this.name = 'StoreUnavailableError';
    this.cause = cause;
  }
}

module.exports = { RateLimitExceededError, StoreUnavailableError };

### `src/RedisStore.js`

```javascript
'use strict';

const { StoreUnavailableError } = require('./errors');

/**
 * Sliding window log implemented with a Redis sorted set.
 *
 * Key layout:  rl:{namespace}:{key}
 * Members:     UUID or timestamp-based unique ID
 * Scores:      Unix timestamp in milliseconds (used for range queries)
 *
 * Why a sorted set instead of a simple counter?
 *   A counter gives you a fixed window. A sorted set gives you a true
 *   sliding window — no burst at window boundaries.
 *
 * Atomicity:
 *   All read-modify-write operations run inside a single Lua script.
 *   This eliminates the TOCTOU race condition you get if you do
 *   ZCOUNT → ZADD as two separate round trips.
 */

// Lua script executed atomically on the Redis server.
// Arguments:
//   KEYS[1]  - the sorted set key
//   ARGV[1]  - current timestamp in ms (string)
//   ARGV[2]  - window size in ms (string)
//   ARGV[3]  - max requests allowed in the window (string)
//   ARGV[4]  - TTL for the key in seconds (string) — slightly > window to allow cleanup
//
// Returns: { allowed (0|1), current_count, oldest_timestamp_in_window }
const SLIDING_WINDOW_SCRIPT = `
local key          = KEYS[1]
local now          = tonumber(ARGV[1])
local window_ms    = tonumber(ARGV[2])
local limit        = tonumber(ARGV[3])
local ttl_seconds  = tonumber(ARGV[4])
local window_start = now - window_ms

-- Remove entries that have fallen outside the current window.
-- This keeps the sorted set from growing unboundedly.
redis.call('ZREMRANGEBYSCORE', key, '-inf', window_start)

-- Count how many requests remain inside the window.
local count = redis.call('ZCARD', key)

if count < limit then
  -- Add this request. Score = timestamp, member = timestamp+random suffix
  -- to handle multiple requests arriving in the same millisecond.
  local member = tostring(now) .. ':' .. tostring(math.random(1, 1000000))
  redis.call('ZADD', key, now, member)
  -- Refresh TTL so idle keys expire automatically.
  redis.call('EXPIRE', key, ttl_seconds)
  return {1, count + 1, window_start}
else
  -- Limit exceeded. Return the score of the oldest entry so the caller
  -- can compute an accurate Retry-After value.
  local oldest = redis.call('ZRANGE', key, 0, 0, 'WITHSCORES')
  local oldest_score = oldest[2] and tonumber(oldest[2]) or now
  return {0, count, oldest_score}
end
`;

class RedisStore {
  /**
   * @param {object} params
   * @param {object} params.client      - ioredis (or node-redis v4) client instance.
   *                                      Caller owns the connection lifecycle.
   * @param {string} [params.namespace] - Key prefix to avoid collisions with other apps.
   * @param {number} [params.commandTimeoutMs] - Max ms to wait for a Redis command.
   */
  constructor({ client, namespace = 'rl', commandTimeoutMs = 200 }) {
    if (!client) {
      throw new Error('RedisStore requires a Redis client instance');
    }
    this.client = client;
    this.namespace = namespace;
    this.commandTimeoutMs = commandTimeoutMs;

    // Pre-load the Lua script so Redis can cache it by SHA.
    // evalsha is faster than eval on hot paths.
    this._scriptSha = null;
    this._loadScript();
  }

  async _loadScript() {
    try {
      this._scriptSha = await this.client.script('LOAD', SLIDING_WINDOW_SCRIPT);
    } catch (err) {
      // Non-fatal at construction time — we fall back to EVAL if SHA is missing.
      console.warn('[RedisStore] Failed to pre-load Lua script:', err.message);
    }
  }

  /**
   * Check and increment the request count for a key.
   *
   * @param {string} key       - Logical rate limit key (e.g. user ID, IP)
   * @param {number} windowMs  - Sliding window duration in milliseconds
   * @param {number} limit     - Maximum requests allowed within the window
   * @returns {Promise<{allowed: boolean, count: number, retryAfterMs: number}>}
   * @throws {StoreUnavailableError}
   */
  async increment(key, windowMs, limit) {
    const redisKey = `${this.namespace}:${key}`;
    const now = Date.now();
    // TTL slightly longer than the window so Redis cleans up stale keys.
    const ttlSeconds = Math.ceil((windowMs / 1000) * 1.5);

    const args = [
      String(now),
      String(windowMs),
      String(limit),
      String(ttlSeconds),
    ];

    let result;
    try {
      result = await this._execWithTimeout(
        this._runScript(redisKey, args),
        this.commandTimeoutMs
      );
    } catch (err) {
      throw new StoreUnavailableError(
        `Redis command failed for key "${redisKey}": ${err.message}`,
        err
      );
    }

    const [allowed, count, oldestScore] = result;
    const retryAfterMs = allowed
      ? 0
      : Math.max(0, oldestScore + windowMs - now);

    return {
      allowed: allowed === 1,
      count: Number(count),
      retryAfterMs: Math.ceil(retryAfterMs),
    };
  }

  /**
   * Run the Lua script, preferring EVALSHA (faster) with EVAL fallback.
   */
  async _runScript(key, args) {
    if (this._scriptSha) {
      try {
        return await this.client.evalsha(this._scriptSha, 1, key, ...args);
      } catch (err) {
        // NOSCRIPT means Redis lost the script (e.g. after a restart).
        // Fall through to EVAL and reload the SHA.
        if (!err.message.includes('NOSCRIPT')) throw err;
        this._scriptSha = null;
        this._loadScript(); // async reload for next time
      }
    }
    return this.client.eval(SLIDING_WINDOW_SCRIPT, 1, key, ...args);
  }

  /**
   * Wraps a promise with a hard timeout so a slow Redis never blocks
   * the request pipeline indefinitely.
   */
  _execWithTimeout(promise, timeoutMs) {
    return new Promise((resolve, reject) => {
      const timer = setTimeout(() => {
        reject(new Error(`Redis command timed out after ${timeoutMs}ms`));
      }, timeoutMs);

      promise
        .then((result) => { clearTimeout(timer); resolve(result); })
        .catch((err)   => { clearTimeout(timer); reject(err); });
    });
  }

  /**
   * Retrieve the current request count without incrementing.
   * Useful for monitoring endpoints.
   */
  async getCount(key, windowMs) {
    const redisKey = `${this.namespace}:${key}`;
    const windowStart = Date.now() - windowMs;
    try {
      await this.client.zremrangebyscore(redisKey, '-inf', windowStart);
      return await this.client.zcard(redisKey);
    } catch (err) {
      throw new StoreUnavailableError(`Redis read failed: ${err.message}`, err);
    }
  }

  /**
   * Manually reset the counter for a key.
   * Useful for admin tooling or test teardown.
   */
  async reset(key) {
    const redisKey = `${this.namespace}:${key}`;
    try {
      await this.client.del(redisKey);
    } catch (err) {
      throw new StoreUnavailableError(`Redis delete failed: ${err.message}`, err);
    }
  }
}

module.exports = RedisStore;
```

### `src/MemoryStore.js`

```javascript
'use strict';

/**
 * In-process sliding window store.
 *
 * Use cases:
 *   1. Unit tests — no Redis dependency needed.
 *   2. Single-instance deployments where shared state is not required.
 *   3. Fail-open fallback when Redis is unavailable.
 *
 * NOT suitable for multi-instance deployments as a primary store —
 * each instance maintains independent counters.
 */
class MemoryStore {
  constructor() {
    // Map<key, Array<timestamp>>
    this._windows = new Map();

    // Periodic cleanup prevents unbounded memory growth from stale keys.
    // In production, tune this interval based on your key cardinality.
    this._cleanupInterval = setInterval(() => this._cleanup(), 60_000);

    // Don't prevent the process from exiting cleanly.
    if (this._cleanupInterval.unref) {
      this._cleanupInterval.unref();
    }
  }

  /**
   * @param {string} key
   * @param {number} windowMs
   * @param {number} limit
   * @returns {Promise<{allowed: boolean, count: number, retryAfterMs: number}>}
   */
  async increment(key, windowMs, limit) {
    const now = Date.now();
    const windowStart = now - windowMs;

    if (!this._windows.has(key)) {
      this._windows.set(key, []);
    }

    const timestamps = this._windows.get(key);

    // Evict expired entries — O(n) but n is bounded by `limit`.
    const validStart = timestamps.findIndex((ts) => ts > windowStart);
    const valid = validStart === -1 ? [] : timestamps.slice(validStart);

    if (valid.length < limit) {
      valid.push(now);
      this._windows.set(key, valid);
      return { allowed: true, count: valid.length, retryAfterMs: 0 };
    }

    // Oldest entry in the window determines when the client can retry.
    const retryAfterMs = Math.max(0, valid[0] + windowMs - now);
    return { allowed: false, count: valid.length, retryAfterMs };
  }

  async getCount(key, windowMs) {
    const windowStart = Date.now() - windowMs;
    const timestamps = this._windows.get(key) || [];
    return timestamps.filter((ts) => ts > windowStart).length;
  }

  async reset(key) {
    this._windows.delete(key);
  }

  /** Remove all keys with no recent activity to prevent memory leaks. */
  _cleanup() {
    const now = Date.now();
    for (const [key, timestamps] of this._windows.entries()) {
      // If the most recent timestamp is older than 1 hour, evict the key.
      if (!timestamps.length || now - timestamps[timestamps.length - 1] > 3_600_000) {
        this._windows.delete(key);
      }
    }
  }

  destroy() {
    clearInterval(this._cleanupInterval);
    this._windows.clear();
  }
}

module.exports = MemoryStore;
```
