### `src/storage/memoryStore.js`

```javascript
/**
 * In-memory store with TTL-based expiration.
 *
 * Good for: single-process apps, testing, development.
 * Not good for: multi-instance deployments (no shared state).
 *
 * Uses a Map with lazy expiration + periodic cleanup to avoid
 * unbounded memory growth under high cardinality keys.
 */
class MemoryStore {
  constructor({ cleanupIntervalMs = 60_000 } = {}) {
    this.store = new Map();

    // Periodic cleanup prevents memory leaks when you have
    // high-cardinality keys (e.g., per-IP limiting with many IPs)
    this.cleanupTimer = setInterval(
      () => this._cleanup(),
      cleanupIntervalMs
    );

    // Don't block process exit
    if (this.cleanupTimer.unref) {
      this.cleanupTimer.unref();
    }
  }

  /**
   * Atomically increment a counter and set TTL on first write.
   * Returns { count, ttlMs } where ttlMs is time until expiry.
   */
  async increment(key, windowMs) {
    const now = Date.now();
    const entry = this.store.get(key);

    if (!entry || now >= entry.expiresAt) {
      // New window
      const expiresAt = now + windowMs;
      this.store.set(key, { count: 1, expiresAt });
      return { count: 1, ttlMs: windowMs };
    }

    entry.count += 1;
    return { count: entry.count, ttlMs: entry.expiresAt - now };
  }

  async get(key) {
    const now = Date.now();
    const entry = this.store.get(key);

    if (!entry || now >= entry.expiresAt) {
      return null;
    }

    return { count: entry.count, ttlMs: entry.expiresAt - now };
  }

  async set(key, value, ttlMs) {
    this.store.set(key, {
      ...value,
      expiresAt: Date.now() + ttlMs,
    });
  }

  async delete(key) {
    this.store.delete(key);
  }

  async reset(key) {
    this.store.delete(key);
  }

  _cleanup() {
    const now = Date.now();
    for (const [key, entry] of this.store.entries()) {
      if (now >= entry.expiresAt) {
        this.store.delete(key);
      }
    }
  }

  destroy() {
    clearInterval(this.cleanupTimer);
    this.store.clear();
  }
}

module.exports = MemoryStore;
```

### `src/storage/redisStore.js`

```javascript
const { StorageError } = require('../errors');

/**
 * Redis-backed store for distributed rate limiting.
 *
 * Uses Lua scripts for atomic operations — critical for correctness
 * under concurrent requests. Without atomicity, you get race conditions
 * where two requests both read count=4, both increment to 5, and you
 * allow 2x your intended limit.
 *
 * Requires: `npm install ioredis`
 * Pass in your own ioredis client for connection pooling control.
 */
class RedisStore {
  constructor(redisClient) {
    if (!redisClient) {
      throw new Error('RedisStore requires a redis client instance');
    }

    this.client = redisClient;
    this._defineScripts();
  }

  _defineScripts() {
    /**
     * Atomic increment with window initialization.
     *
     * KEYS[1] = rate limit key
     * ARGV[1] = window duration in milliseconds
     *
     * Returns: [count, pttl]
     * - count: current request count in this window
     * - pttl: remaining TTL in milliseconds
     */
    this.incrementScript = `
      local key = KEYS[1]
      local windowMs = tonumber(ARGV[1])

      local count = redis.call('INCR', key)

      if count == 1 then
        redis.call('PEXPIRE', key, windowMs)
      end

      local pttl = redis.call('PTTL', key)
      return {count, pttl}
    `;

    /**
     * Token bucket: atomic get-and-update.
     *
     * KEYS[1] = bucket key
     * ARGV[1] = capacity (max tokens)
     * ARGV[2] = refill rate (tokens per ms)
     * ARGV[3] = current timestamp (ms)
     * ARGV[4] = tokens to consume
     *
     * Returns: [allowed (0|1), remaining_tokens, retry_after_ms]
     */
    this.tokenBucketScript = `
      local key = KEYS[1]
      local capacity = tonumber(ARGV[1])
      local refillRate = tonumber(ARGV[2])
      local now = tonumber(ARGV[3])
      local requested = tonumber(ARGV[4])

      local bucket = redis.call('HMGET', key, 'tokens', 'lastRefill')
      local tokens = tonumber(bucket[1])
      local lastRefill = tonumber(bucket[2])

      if tokens == nil then
        tokens = capacity
        lastRefill = now
      else
        local elapsed = now - lastRefill
        local refilled = elapsed * refillRate
        tokens = math.min(capacity, tokens + refilled)
        lastRefill = now
      end

      local allowed = 0
      local retryAfter = 0

      if tokens >= requested then
        tokens = tokens - requested
        allowed = 1
      else
        local deficit = requested - tokens
        retryAfter = math.ceil(deficit / refillRate)
      end

      local ttlMs = math.ceil((capacity / refillRate) * 2)
      redis.call('HMSET', key, 'tokens', tokens, 'lastRefill', lastRefill)
      redis.call('PEXPIRE', key, ttlMs)

      return {allowed, math.floor(tokens), retryAfter}
    `;

    /**
     * Sliding window log using a sorted set.
     * Scores = timestamps, members = unique request IDs.
     *
     * KEYS[1] = key
     * ARGV[1] = window start (ms, requests before this are expired)
     * ARGV[2] = current timestamp (ms)
     * ARGV[3] = unique request ID
     * ARGV[4] = window duration (ms, for TTL)
     *
     * Returns: [count_in_window]
     */
    this.slidingWindowLogScript = `
      local key = KEYS[1]
      local windowStart = tonumber(ARGV[1])
      local now = tonumber(ARGV[2])
      local requestId = ARGV[3]
      local windowMs = tonumber(ARGV[4])

      redis.call('ZREMRANGEBYSCORE', key, '-inf', windowStart)
      redis.call('ZADD', key, now, requestId)
      redis.call('PEXPIRE', key, windowMs)

      local count = redis.call('ZCARD', key)
      return {count}
    `;
  }

  async increment(key, windowMs) {
    try {
      const result = await this.client.eval(
        this.incrementScript,
        1,
        key,
        windowMs
      );

      return {
        count: result[0],
        ttlMs: Math.max(result[1], 0),
      };
    } catch (err) {
      throw new StorageError(`Redis increment failed for key: ${key}`, err);
    }
  }

  async tokenBucketConsume(key, capacity, refillRatePerMs, tokensRequested = 1) {
    try {
      const result = await this.client.eval(
        this.tokenBucketScript,
        1,
        key,
        capacity,
        refillRatePerMs,
        Date.now(),
        tokensRequested
      );

      return {
        allowed: result[0] === 1,
        remaining: result[1],
        retryAfterMs: result[2],
      };
    } catch (err) {
      throw new StorageError(`Redis token bucket failed for key: ${key}`, err);
    }
  }

  async slidingWindowLog(key, windowMs, requestId) {
    try {
      const now = Date.now();
      const windowStart = now - windowMs;

      const result = await this.client.eval(
        this.slidingWindowLogScript,
        1,
        key,
        windowStart,
        now,
        requestId,
        windowMs
      );

      return { count: result[0] };
    } catch (err) {
      throw new StorageError(`Redis sliding window failed for key: ${key}`, err);
    }
  }

  async get(key) {
    try {
      const data = await this.client.hgetall(key);
      if (!data || Object.keys(data).length === 0) return null;

      const pttl = await this.client.pttl(key);
      return { ...data, ttlMs: Math.max(pttl, 0) };
    } catch (err) {
      throw new StorageError(`Redis get failed for key: ${key}`, err);
    }
  }

  async reset(key) {
    try {
      await this.client.del(key);
    } catch (err) {
      throw new StorageError(`Redis reset failed for key: ${key}`, err);
    }
  }

  async delete(key) {
    return this.reset(key);
  }
}

module.exports = RedisStore;
```

### `src/algorithms/fixedWindow.js`

```javascript
/**
 * Fixed Window Counter
 *
 * Simplest algorithm. Divides time into fixed windows and counts
 * requests per window.
 *
 * Weakness: "boundary burst" — a client can send 2x the limit by
 * sending max requests at the end of one window and max at the start
 * of the next. Use sliding window if this matters for your use case.
 *
 * Best for: simple APIs, internal tools, cases where boundary bursts
 * are acceptable.
 */
class FixedWindow {
  constructor(store, { limit, windowMs, keyPrefix = 'fw' }) {
    if (!limit || limit < 1) throw new Error('limit must be >= 1');
    if (!windowMs || windowMs < 1) throw new Error('windowMs must be >= 1');

    this.store = store;
    this.limit = limit;
    this.windowMs = windowMs;
    this.keyPrefix = keyPrefix;
  }

  _buildKey(identifier) {
    return `${this.keyPrefix}:${identifier}`;
  }

  async check(identifier) {
    const key = this._buildKey(identifier);
    const { count, ttlMs } = await this.store.increment(key, this.windowMs);

    const remaining = Math.max(0, this.limit - count);
    const allowed = count <= this.limit;
    const resetAt = new Date(Date.now() + ttlMs);

    return {
      allowed,
      limit: this.limit,
      remaining,
      count,
      resetAt,
      retryAfterMs: allowed ? 0 : ttlMs,
      algorithm: 'fixed-window',
    };
  }

  async reset(identifier) {
    await this.store.reset(this._buildKey(identifier));
  }
}

module.exports = FixedWindow;
```

### `src/algorithms/slidingWindow.js`

```javascript
const crypto = require('crypto');

/**
 * Sliding Window — two implementations:
 *
 * 1. Sliding Window Counter (approximate, memory-efficient)
 *    Blends the current and previous window counts using a weighted
 *    average based on how far into the current window we are.
 *    ~0.003% error rate in practice. Great for most use cases.
 *
 * 2. Sliding Window Log (exact, higher memory usage)
 *    Stores a timestamp for every request. Exact accuracy but O(n)
 *    memory per key. Use for low-volume, high-precision needs.
 */

class SlidingWindowCounter {
  constructor(store, { limit, windowMs, keyPrefix = 'swc' }) {
    if (!limit || limit < 1) throw new Error('limit must be >= 1');
    if (!windowMs || windowMs < 1) throw new Error('windowMs must be >= 1');

    this.store = store;
    this.limit = limit;
    this.windowMs = windowMs;
    this.keyPrefix = keyPrefix;
  }

  _buildKeys(identifier) {
    const now = Date.now();
    const currentWindow = Math.floor(now / this.windowMs);
    const previousWindow = currentWindow - 1;

    return {
      currentKey: `${this.keyPrefix}:${identifier}:${currentWindow}`,
      previousKey: `${this.keyPrefix}:${identifier}:${previousWindow}`,
      currentWindowStart: currentWindow * this.windowMs,
      now,
    };
  }

  async check(identifier) {
    const { currentKey, previousKey, currentWindowStart, now } =
      this._buildKeys(identifier);

    // Fetch previous window count (read-only, no increment)
    const prevEntry = await this.store.get(previousKey);
    const prevCount = prevEntry ? prevEntry.count : 0;

    // How far into the current window are we? (0.0 to 1.0)
    const windowProgress = (now - currentWindowStart) / this.windowMs;

    // Weight previous window's contribution by how much of it
    // is still "in scope" of our sliding window
    const weightedPrevCount = prevCount * (1 - windowProgress);

    // Increment current window
    const { count: currentCount, ttlMs } = await this.store.increment(
      currentKey,
      this.windowMs * 2 // Keep previous window data around
    );

    const estimatedCount = weightedPrevCount + currentCount;
    const allowed = estimatedCount <= this.limit;
    const remaining = Math.max(0, Math
