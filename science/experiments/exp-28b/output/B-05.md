# Production Rate Limiter in Node.js

Let me build this properly — working code first, then an honest accounting of what's missing before this is production-safe.

## Architecture Decisions (and why)

I'm implementing **sliding window log** backed by Redis. Here's why I rejected the alternatives:

| Algorithm | Rejected Because |
|---|---|
| Fixed window | Burst vulnerability at window boundaries (2x traffic spike) |
| Token bucket | Complex distributed state, harder to reason about under failure |
| Leaky bucket | Queuing adds latency, wrong for most API use cases |
| Sliding window counter | Approximation — fine for most cases, but I'll show you the exact version first |

### `src/redisClient.js`

```javascript
const Redis = require('ioredis');
const logger = require('./logger');

/**
 * Redis client with:
 * - Connection retry with exponential backoff
 * - Circuit breaker state tracking
 * - Explicit timeout on every command
 *
 * OPERATIONAL GAP FLAGGED: See bottom of file.
 */

let circuitOpen = false;
let circuitOpenedAt = null;
const CIRCUIT_RESET_MS = 30_000; // 30 seconds

function createRedisClient(options = {}) {
  const config = {
    host: process.env.REDIS_HOST || 'localhost',
    port: parseInt(process.env.REDIS_PORT || '6379', 10),
    password: process.env.REDIS_PASSWORD || undefined,
    db: parseInt(process.env.REDIS_DB || '0', 10),

    // Critical: without this, a hung Redis blocks your entire event loop
    commandTimeout: 500,
    connectTimeout: 2000,

    // Retry with backoff — don't hammer a recovering Redis
    retryStrategy(times) {
      if (times > 10) {
        logger.error('Redis: max retries exceeded, stopping reconnection');
        return null; // Stop retrying
      }
      const delay = Math.min(100 * Math.pow(2, times), 5000);
      logger.warn(`Redis: retry attempt ${times}, waiting ${delay}ms`);
      return delay;
    },

    // Don't queue commands while disconnected — fail fast
    enableOfflineQueue: false,

    lazyConnect: true,
    ...options,
  };

  const client = new Redis(config);

  client.on('connect', () => {
    logger.info('Redis: connection established');
    circuitOpen = false;
    circuitOpenedAt = null;
  });

  client.on('error', (err) => {
    logger.error('Redis: connection error', { error: err.message });
    tripCircuit();
  });

  client.on('close', () => {
    logger.warn('Redis: connection closed');
  });

  return client;
}

function tripCircuit() {
  if (!circuitOpen) {
    circuitOpen = true;
    circuitOpenedAt = Date.now();
    logger.error('Redis circuit breaker: OPEN — rate limiter will fail open');
  }
}

function isCircuitOpen() {
  if (!circuitOpen) return false;

  // Auto-reset after timeout — attempt to recover
  if (Date.now() - circuitOpenedAt > CIRCUIT_RESET_MS) {
    logger.info('Redis circuit breaker: attempting reset (half-open)');
    circuitOpen = false;
    circuitOpenedAt = null;
    return false;
  }

  return true;
}

module.exports = { createRedisClient, isCircuitOpen, tripCircuit };

### `scripts/lua/slidingWindow.lua`

```lua
--[[
  Sliding Window Log — Atomic via Lua
  
  Why Lua? A check-then-act implemented as two separate Redis commands
  has a race condition: two requests can both pass the check before
  either increments the counter. Lua scripts execute atomically.

  KEYS[1] = rate limit key (e.g., "rl:user:123")
  ARGV[1] = current timestamp (milliseconds)
  ARGV[2] = window size (milliseconds)
  ARGV[3] = max requests allowed
  ARGV[4] = TTL for the key (seconds)

  Returns: { allowed (0|1), current_count, oldest_timestamp_in_window }
--]]

local key = KEYS[1]
local now = tonumber(ARGV[1])
local window = tonumber(ARGV[2])
local limit = tonumber(ARGV[3])
local ttl = tonumber(ARGV[4])

local window_start = now - window

-- Remove all entries outside the current window
redis.call('ZREMRANGEBYSCORE', key, '-inf', window_start)

-- Count remaining entries in window
local current_count = redis.call('ZCARD', key)

if current_count < limit then
  -- Add current request with timestamp as both score and member
  -- Using now + random suffix to handle sub-millisecond requests
  -- that would otherwise collide on the same score
  local member = now .. ':' .. redis.call('INCR', key .. ':seq')
  redis.call('ZADD', key, now, member)
  redis.call('EXPIRE', key, ttl)
  
  -- Clean up sequence counter (it's just a helper)
  redis.call('EXPIRE', key .. ':seq', ttl)
  
  return { 1, current_count + 1, window_start }
else
  -- Get the oldest entry to tell the client when they can retry
  local oldest = redis.call('ZRANGE', key, 0, 0, 'WITHSCORES')
  local oldest_ts = 0
  if #oldest > 0 then
    oldest_ts = tonumber(oldest[2])
  end
  
  return { 0, current_count, oldest_ts }
end
```

### `src/strategies/slidingWindowLog.js`

```javascript
const fs = require('fs');
const path = require('path');
const { RedisUnavailableError } = require('../errors');
const { isCircuitOpen, tripCircuit } = require('../redisClient');
const logger = require('./logger');

const luaScript = fs.readFileSync(
  path.join(__dirname, '../../scripts/lua/slidingWindow.lua'),
  'utf8'
);

/**
 * Sliding Window Log Rate Limiter
 *
 * Exact algorithm — tracks every request timestamp.
 * Memory cost: O(requests_per_window) per key.
 *
 * For 1000 req/hour per user, each entry ~50 bytes = ~50KB per user.
 * At 10,000 active users = ~500MB. Plan your Redis memory accordingly.
 */
class SlidingWindowLog {
  constructor(redisClient, options = {}) {
    this.redis = redisClient;
    this.windowMs = options.windowMs || 60_000;       // 1 minute default
    this.maxRequests = options.maxRequests || 100;
    this.keyPrefix = options.keyPrefix || 'rl';

    // Load script SHA for EVALSHA (more efficient than EVAL on every call)
    this.scriptSha = null;
    this._loadScript();
  }

  async _loadScript() {
    try {
      this.scriptSha = await this.redis.script('LOAD', luaScript);
      logger.info('Rate limiter Lua script loaded', { sha: this.scriptSha });
    } catch (err) {
      logger.error('Failed to load Lua script', { error: err.message });
      // Will fall back to EVAL
    }
  }

  _buildKey(identifier) {
    // Namespace clearly — you'll thank yourself when debugging Redis
    return `${this.keyPrefix}:${identifier}`;
  }

  async check(identifier) {
    if (isCircuitOpen()) {
      // POLICY DECISION: fail open (allow request) when Redis is down.
      // This is the right default for most APIs — availability > strict limiting.
      // Some use cases (fraud prevention, paid quotas) should fail CLOSED.
      // Make this configurable. See operational gaps.
      logger.warn('Rate limiter: circuit open, failing open', { identifier });
      return this._buildOpenCircuitResult(identifier);
    }

    const key = this._buildKey(identifier);
    const now = Date.now();
    const ttlSeconds = Math.ceil(this.windowMs / 1000) + 1;

    try {
      let result;

      if (this.scriptSha) {
        try {
          result = await this.redis.evalsha(
            this.scriptSha,
            1,
            key,
            now,
            this.windowMs,
            this.maxRequests,
            ttlSeconds
          );
        } catch (err) {
          if (err.message.includes('NOSCRIPT')) {
            // Script was flushed (Redis restart) — reload and retry
            await this._loadScript();
            result = await this.redis.eval(
              luaScript,
              1,
              key,
              now,
              this.windowMs,
              this.maxRequests,
              ttlSeconds
            );
          } else {
            throw err;
          }
        }
      } else {
        result = await this.redis.eval(
          luaScript,
          1,
          key,
          now,
          this.windowMs,
          this.maxRequests,
          ttlSeconds
        );
      }

      const [allowed, currentCount, oldestTimestamp] = result;

      const resetAfterMs = oldestTimestamp
        ? Math.max(0, oldestTimestamp + this.windowMs - now)
        : this.windowMs;

      return {
        allowed: allowed === 1,
        limit: this.maxRequests,
        remaining: Math.max(0, this.maxRequests - currentCount),
        currentCount,
        resetAfterMs,
        resetAt: new Date(now + resetAfterMs).toISOString(),
        identifier,
        algorithm: 'sliding-window-log',
      };
    } catch (err) {
      logger.error('Rate limiter Redis error', {
        identifier,
        error: err.message,
        stack: err.stack,
      });

      tripCircuit();

      // Fail open — see note above
      return this._buildOpenCircuitResult(identifier);
    }
  }

  _buildOpenCircuitResult(identifier) {
    return {
      allowed: true,
      limit: this.maxRequests,
      remaining: this.maxRequests,
      currentCount: 0,
      resetAfterMs: this.windowMs,
      resetAt: new Date(Date.now() + this.windowMs).toISOString(),
      identifier,
      algorithm: 'sliding-window-log',
      degraded: true, // Signal to callers that this result is not authoritative
    };
  }
}

module.exports = SlidingWindowLog;
```

### `src/strategies/slidingWindowCounter.js`

```javascript
/**
 * Sliding Window Counter (Approximate)
 *
 * Uses two fixed windows (current + previous) and weights the previous
 * window based on how far through the current window we are.
 *
 * Formula: count = prev_count * (1 - elapsed/window) + curr_count
 *
 * Trade-off: ~0.003% error rate vs O(1) memory per key (vs O(n) for log).
 * Use this when you have millions of unique keys and memory is a concern.
 * Cloudflare uses this approach at their scale.
 */
class SlidingWindowCounter {
  constructor(redisClient, options = {}) {
    this.redis = redisClient;
    this.windowMs = options.windowMs || 60_000;
    this.maxRequests = options.maxRequests || 100;
    this.keyPrefix = options.keyPrefix || 'rlc';
  }

  _getWindowKeys(identifier) {
    const now = Date.now();
    const windowSize = this.windowMs;
    const currentWindow = Math.floor(now / windowSize);
    const prevWindow = currentWindow - 1;

    return {
      currentKey: `${this.keyPrefix}:${identifier}:${currentWindow}`,
      prevKey: `${this.keyPrefix}:${identifier}:${prevWindow}`,
      elapsed: now - currentWindow * windowSize,
      now,
    };
  }

  async check(identifier) {
    const { currentKey, prevKey, elapsed, now } = this._getWindowKeys(identifier);
    const ttlSeconds = Math.ceil((this.windowMs * 2) / 1000);

    // Pipeline both reads — single round trip
    const pipeline = this.redis.pipeline();
    pipeline.get(prevKey);
    pipeline.get(currentKey);
    const [[, prevCount], [, currCount]] = await pipeline.exec();

    const prev = parseInt(prevCount || '0', 10);
    const curr = parseInt(currCount || '0', 10);

    // Weighted estimate of requests in the sliding window
    const weight = 1 - elapsed / this.windowMs;
    const estimated = Math.ceil(prev * weight) + curr;

    if (estimated >= this.maxRequests) {
      const resetAfterMs = this.windowMs - elapsed;
      return {
        allowed: false,
        limit: this.maxRequests,
        remaining: 0,
        currentCount: estimated,
        resetAfterMs,
        resetAt: new Date(now + resetAfterMs).toISOString(),
        identifier,
        algorithm: 'sliding-window-counter',
      };
