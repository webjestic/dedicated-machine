## 2. Memory Store — `src/stores/memoryStore.js`

```javascript
/**
 * In-memory store for rate limiter data.
 * Suitable for single-instance applications.
 */
class MemoryStore {
  constructor() {
    this.store = new Map();
    this._startCleanupInterval();
  }

  /**
   * Increment the request count for a key.
   * @param {string} key - Unique identifier (e.g., IP address)
   * @param {number} windowMs - Time window in milliseconds
   * @returns {{ count: number, resetTime: number }}
   */
  async increment(key, windowMs) {
    const now = Date.now();
    const record = this.store.get(key);

    if (!record || now >= record.resetTime) {
      // Create a new window
      const newRecord = {
        count: 1,
        resetTime: now + windowMs,
        createdAt: now,
      };
      this.store.set(key, newRecord);
      return { count: 1, resetTime: newRecord.resetTime };
    }

    record.count += 1;
    return { count: record.count, resetTime: record.resetTime };
  }

  /**
   * Get current record for a key without incrementing.
   */
  async get(key) {
    const record = this.store.get(key);
    if (!record || Date.now() >= record.resetTime) return null;
    return record;
  }

  /**
   * Reset a specific key.
   */
  async reset(key) {
    this.store.delete(key);
  }

  /**
   * Periodically clean up expired entries to prevent memory leaks.
   */
  _startCleanupInterval() {
    this._cleanupTimer = setInterval(() => {
      const now = Date.now();
      for (const [key, record] of this.store.entries()) {
        if (now >= record.resetTime) {
          this.store.delete(key);
        }
      }
    }, 60_000); // Run every 60 seconds

    // Don't block process exit
    if (this._cleanupTimer.unref) {
      this._cleanupTimer.unref();
    }
  }

  destroy() {
    clearInterval(this._cleanupTimer);
    this.store.clear();
  }
}

module.exports = MemoryStore;
```

/**
 * Redis-backed store for distributed rate limiting.
 * Suitable for multi-instance / clustered applications.
 */
class RedisStore {
  /**
   * @param {import('ioredis').Redis} redisClient - An ioredis client instance
   */
  constructor(redisClient) {
    if (!redisClient) {
      throw new Error("RedisStore requires an ioredis client instance.");
    }
    this.client = redisClient;
  }

  /**
   * Atomically increment the request count using a Lua script.
   * This ensures thread-safety in distributed environments.
   */
  async increment(key, windowMs) {
    const windowSeconds = Math.ceil(windowMs / 1000);

    // Lua script for atomic increment + TTL set
    const luaScript = `
      local current = redis.call('INCR', KEYS[1])
      if current == 1 then
        redis.call('EXPIRE', KEYS[1], ARGV[1])
      end
      local ttl = redis.call('TTL', KEYS[1])
      return {current, ttl}
    `;

    const [count, ttl] = await this.client.eval(
      luaScript,
      1,
      key,
      windowSeconds
    );

    const resetTime = Date.now() + ttl * 1000;
    return { count: parseInt(count), resetTime };
  }

  async get(key) {
    const [count, ttl] = await Promise.all([
      this.client.get(key),
      this.client.ttl(key),
    ]);

    if (!count || ttl < 0) return null;

    return {
      count: parseInt(count),
      resetTime: Date.now() + ttl * 1000,
    };
  }

  async reset(key) {
    await this.client.del(key);
  }
}

module.exports = RedisStore;

## 4. Fixed Window Strategy — `src/strategies/fixedWindow.js`

/**
 * Fixed Window Strategy
 *
 * Counts requests in fixed time windows (e.g., 0-60s, 60-120s).
 * Simple but can allow 2x the limit at window boundaries.
 */
class FixedWindowStrategy {
  /**
   * @param {object} store - MemoryStore or RedisStore
   * @param {number} windowMs - Window size in milliseconds
   * @param {number} maxRequests - Max requests allowed per window
   */
  constructor(store, windowMs, maxRequests) {
    this.store = store;
    this.windowMs = windowMs;
    this.maxRequests = maxRequests;
  }

  async check(key) {
    const { count, resetTime } = await this.store.increment(
      key,
      this.windowMs
    );

    const remaining = Math.max(0, this.maxRequests - count);
    const allowed = count <= this.maxRequests;

    return {
      allowed,
      count,
      remaining,
      resetTime,
      limit: this.maxRequests,
    };
  }
}

module.exports = FixedWindowStrategy;

## 5. Sliding Window Strategy — `src/strategies/slidingWindow.js`

```javascript
/**
 * Sliding Window Strategy
 *
 * Tracks individual request timestamps within the window.
 * More accurate than fixed window — no boundary spike issues.
 * Higher memory usage for high-traffic keys.
 */
class SlidingWindowStrategy {
  /**
   * @param {number} windowMs - Window size in milliseconds
   * @param {number} maxRequests - Max requests allowed per window
   */
  constructor(windowMs, maxRequests) {
    this.windowMs = windowMs;
    this.maxRequests = maxRequests;
    this.store = new Map(); // key -> array of timestamps
    this._startCleanupInterval();
  }

  async check(key) {
    const now = Date.now();
    const windowStart = now - this.windowMs;

    // Get existing timestamps and filter out expired ones
    const timestamps = (this.store.get(key) || []).filter(
      (ts) => ts > windowStart
    );

    const count = timestamps.length;
    const allowed = count < this.maxRequests;

    if (allowed) {
      timestamps.push(now);
      this.store.set(key, timestamps);
    }

    // Reset time = when the oldest request in the window expires
    const oldestTimestamp = timestamps[0] || now;
    const resetTime = oldestTimestamp + this.windowMs;

    return {
      allowed,
      count: allowed ? count + 1 : count,
      remaining: Math.max(0, this.maxRequests - timestamps.length),
      resetTime,
      limit: this.maxRequests,
    };
  }

  _startCleanupInterval() {
    this._cleanupTimer = setInterval(() => {
      const windowStart = Date.now() - this.windowMs;
      for (const [key, timestamps] of this.store.entries()) {
        const valid = timestamps.filter((ts) => ts > windowStart);
        if (valid.length === 0) {
          this.store.delete(key);
        } else {
          this.store.set(key, valid);
        }
      }
    }, 60_000);

    if (this._cleanupTimer.unref) {
      this._cleanupTimer.unref();
    }
  }

  destroy() {
    clearInterval(this._cleanupTimer);
    this.store.clear();
  }
}

module.exports = SlidingWindowStrategy;
```

## 6. Token Bucket Strategy — `src/strategies/tokenBucket.js`

```javascript
/**
 * Token Bucket Strategy
 *
 * Allows bursting up to `capacity` requests, then refills at `refillRate`
 * tokens per second. Great for APIs that want to allow short bursts.
 */
class TokenBucketStrategy {
  /**
   * @param {number} capacity - Max tokens (burst size)
   * @param {number} refillRate - Tokens added per second
   */
  constructor(capacity, refillRate) {
    this.capacity = capacity;
    this.refillRate = refillRate; // tokens per second
    this.buckets = new Map(); // key -> { tokens, lastRefill }
    this._startCleanupInterval();
  }

  async check(key) {
    const now = Date.now();
    let bucket = this.buckets.get(key);

    if (!bucket) {
      bucket = { tokens: this.capacity, lastRefill: now };
    }

    // Calculate tokens to add based on elapsed time
    const elapsed = (now - bucket.lastRefill) / 1000; // seconds
    const tokensToAdd = elapsed * this.refillRate;
    bucket.tokens = Math.min(this.capacity, bucket.tokens + tokensToAdd);
    bucket.lastRefill = now;

    const allowed = bucket.tokens >= 1;

    if (allowed) {
      bucket.tokens -= 1;
    }

    this.buckets.set(key, bucket);

    // Estimate reset time: when will 1 token be available?
    const timeUntilNextToken =
      bucket.tokens < 1 ? ((1 - bucket.tokens) / this.refillRate) * 1000 : 0;

    return {
      allowed,
      tokens: Math.floor(bucket.tokens),
      remaining: Math.floor(bucket.tokens),
      resetTime: now + timeUntilNextToken,
      limit: this.capacity,
    };
  }

  _startCleanupInterval() {
    this._cleanupTimer = setInterval(() => {
      const now = Date.now();
      const idleThreshold = 3600_000; // Remove buckets idle for 1 hour
      for (const [key, bucket] of this.buckets.entries()) {
        if (now - bucket.lastRefill > idleThreshold) {
          this.buckets.delete(key);
        }
      }
    }, 300_000); // Every 5 minutes

    if (this._cleanupTimer.unref) {
      this._cleanupTimer.unref();
    }
  }

  destroy() {
    clearInterval(this._cleanupTimer);
    this.buckets.clear();
  }
}

module.exports = TokenBucketStrategy;
```

## 7. Core Rate Limiter — `src/rateLimiter.js`

```javascript
const MemoryStore = require("./stores/memoryStore");
const FixedWindowStrategy = require("./strategies/fixedWindow");
const SlidingWindowStrategy = require("./strategies/slidingWindow");
const TokenBucketStrategy = require("./strategies/tokenBucket");

/**
 * @typedef {Object} RateLimiterOptions
 * @property {number}   [windowMs=60000]       - Time window in ms (fixed/sliding)
 * @property {number}   [maxRequests=100]       - Max requests per window
 * @property {string}   [strategy='fixedWindow'] - 'fixedWindow' | 'slidingWindow' | 'tokenBucket'
 * @property {number}   [capacity=10]           - Token bucket capacity
 * @property {number}   [refillRate=1]          - Token bucket refill rate (tokens/sec)
 * @property {object}   [store]                 - Custom store (MemoryStore or RedisStore)
 * @property {Function} [keyGenerator]          - Custom key generation function
 * @property {Function} [onLimitReached]        - Callback when limit is exceeded
 * @property {boolean}  [skipFailedRequests]    - Don't count failed responses
 * @property {boolean}  [skipSuccessfulRequests]- Don't count successful responses
 */

class RateLimiter {
  constructor(options = {}) {
    this.options = {
      windowMs: 60_000,
      maxRequests: 100,
      strategy: "fixedWindow",
      capacity: 10,
      refillRate: 1,
      keyGenerator: (req) => req.ip || req.connection?.remoteAddress || "global",
      onLimitReached: null,
      skipFailedRequests: false,
      skipSuccessfulRequests: false,
      ...options,
    };

    this.store = options.store || new MemoryStore();
    this.strategy = this._createStrategy();
  }

  _createStrategy() {
    const { strategy, windowMs, maxRequests, capacity, refillRate } =
      this.options;

    switch (strategy) {
      case "fixedWindow":
        return new FixedWindowStrategy(this.store, windowMs, maxRequests);

      case "slidingWindow":
        // Sliding window manages its own store internally
        return new SlidingWindowStrategy(windowMs, maxRequests);

      case "tokenBucket":
        return new TokenBucketStrategy(capacity, refillRate);

      default:
        throw new Error(
          `Unknown strategy: "${strategy}". Use 'fixedWindow', 'slidingWindow', or 'tokenBucket'.`
        );
    }
  }

  /**
   * Check if a request should be allowed.
   * @param {string} key - The identifier for this client
   * @returns {Promise<object>} Rate limit result
   */
  async check(key) {
    return this.strategy.check(key);
  }

   * Manually reset the limit for a key.
   * @param {string} key
   */
  async reset(key) {
    if (this.store.reset) {
      await this.store.reset(key);
    }
  }

  async getStatus(key) {
