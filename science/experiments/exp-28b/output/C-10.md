```javascript
// src/stores/memoryStore.js

class MemoryStore {
  constructor() {
    this.store = new Map();
    this._startCleanup();
  }

  async get(key) {
    const entry = this.store.get(key);
    if (!entry) return null;
    if (Date.now() > entry.expiresAt) {
      this.store.delete(key);
      return null;
    }
    return entry.data;
  }

  async set(key, data, ttlMs) {
    this.store.set(key, {
      data,
      expiresAt: Date.now() + ttlMs,
    });
  }

  async delete(key) {
    this.store.delete(key);
  }

  async increment(key, ttlMs) {
    const entry = this.store.get(key);
    const now = Date.now();

    if (!entry || now > entry.expiresAt) {
      this.store.set(key, { data: 1, expiresAt: now + ttlMs });
      return 1;
    }

    entry.data += 1;
    return entry.data;
  }

  // Cleanup expired keys every 60 seconds
  _startCleanup() {
    this._cleanupInterval = setInterval(() => {
      const now = Date.now();
      for (const [key, entry] of this.store.entries()) {
        if (now > entry.expiresAt) {
          this.store.delete(key);
        }
      }
    }, 60_000);

    // Don't block process exit
    if (this._cleanupInterval.unref) {
      this._cleanupInterval.unref();
    }
  }

  destroy() {
    clearInterval(this._cleanupInterval);
    this.store.clear();
  }
}

module.exports = MemoryStore;
```

```javascript
class RedisStore {
  /**
   * @param {import('ioredis').Redis} client - ioredis client instance
   */
  constructor(client) {
    if (!client) throw new Error('Redis client is required');
    this.client = client;
  }

  async get(key) {
    const value = await this.client.get(key);
    return value ? JSON.parse(value) : null;
  }

  async set(key, data, ttlMs) {
    await this.client.set(key, JSON.stringify(data), 'PX', ttlMs);
  }

  async delete(key) {
    await this.client.del(key);
  }

  async increment(key, ttlMs) {
    const pipeline = this.client.pipeline();
    pipeline.incr(key);
    pipeline.pexpire(key, ttlMs);
    const results = await pipeline.exec();
    return results[0][1]; // INCR result
  }

  // Atomic sliding window using Lua script
  async slidingWindowIncrement(key, windowMs, now) {
    const script = `
      local key = KEYS[1]
      local now = tonumber(ARGV[1])
      local window = tonumber(ARGV[2])
      local clearBefore = now - window

      redis.call('ZREMRANGEBYSCORE', key, '-inf', clearBefore)
      redis.call('ZADD', key, now, now)
      redis.call('PEXPIRE', key, window)
      return redis.call('ZCARD', key)
    `;
    return await this.client.eval(script, 1, key, now, windowMs);
  }

  async slidingWindowCount(key, windowMs, now) {
    const clearBefore = now - windowMs;
    await this.client.zremrangebyscore(key, '-inf', clearBefore);
    return await this.client.zcard(key);
  }
}

module.exports = RedisStore;
```

```javascript
// src/strategies/fixedWindow.js

class FixedWindowStrategy {
  /**
   * @param {object} options
   * @param {number} options.windowMs   - Window size in milliseconds
   * @param {number} options.maxRequests - Max requests per window
   * @param {object} options.store      - Store instance
   */
  constructor({ windowMs, maxRequests, store }) {
    this.windowMs = windowMs;
    this.maxRequests = maxRequests;
    this.store = store;
  }

  async check(key) {
    const count = await this.store.increment(key, this.windowMs);
    const remaining = Math.max(0, this.maxRequests - count);
    const allowed = count <= this.maxRequests;

    return {
      allowed,
      count,
      remaining,
      resetAfterMs: this.windowMs,
      strategy: 'fixed-window',
    };
  }

  async reset(key) {
    await this.store.delete(key);
  }
}

module.exports = FixedWindowStrategy;
```

## 4. Sliding Window Strategy

```javascript
// src/strategies/slidingWindow.js

class SlidingWindowStrategy {
  /**
   * @param {object} options
   * @param {number} options.windowMs    - Window size in milliseconds
   * @param {number} options.maxRequests - Max requests per window
   * @param {object} options.store       - Store instance
   */
  constructor({ windowMs, maxRequests, store }) {
    this.windowMs = windowMs;
    this.maxRequests = maxRequests;
    this.store = store;
  }

  async check(key) {
    const now = Date.now();

    let count;

    // Use Redis atomic Lua script if available
    if (typeof this.store.slidingWindowIncrement === 'function') {
      count = await this.store.slidingWindowIncrement(key, this.windowMs, now);
    } else {
      count = await this._memoryCheck(key, now);
    }

    const remaining = Math.max(0, this.maxRequests - count);
    const allowed = count <= this.maxRequests;

    return {
      allowed,
      count,
      remaining,
      resetAfterMs: this.windowMs,
      strategy: 'sliding-window',
    };
  }

  async _memoryCheck(key, now) {
    const windowStart = now - this.windowMs;
    let timestamps = (await this.store.get(key)) || [];

    // Remove expired timestamps
    timestamps = timestamps.filter((ts) => ts > windowStart);

    // Add current request
    timestamps.push(now);

    await this.store.set(key, timestamps, this.windowMs);
    return timestamps.length;
  }

  async reset(key) {
    await this.store.delete(key);
  }
}

module.exports = SlidingWindowStrategy;
```

## 5. Token Bucket Strategy

```javascript
// src/strategies/tokenBucket.js

class TokenBucketStrategy {
  /**
   * @param {object} options
   * @param {number} options.capacity       - Max tokens in bucket
   * @param {number} options.refillRate     - Tokens added per second
   * @param {number} options.tokensPerRequest - Tokens consumed per request (default: 1)
   * @param {object} options.store          - Store instance
   */
  constructor({ capacity, refillRate, tokensPerRequest = 1, store }) {
    this.capacity = capacity;
    this.refillRate = refillRate; // tokens per second
    this.tokensPerRequest = tokensPerRequest;
    this.store = store;
  }

  async check(key) {
    const now = Date.now();
    let bucket = await this.store.get(key);

    if (!bucket) {
      // New bucket starts full
      bucket = { tokens: this.capacity, lastRefill: now };
    }

    // Calculate tokens to add based on elapsed time
    const elapsed = (now - bucket.lastRefill) / 1000; // seconds
    const tokensToAdd = elapsed * this.refillRate;
    bucket.tokens = Math.min(this.capacity, bucket.tokens + tokensToAdd);
    bucket.lastRefill = now;

    const allowed = bucket.tokens >= this.tokensPerRequest;

    if (allowed) {
      bucket.tokens -= this.tokensPerRequest;
    }

    // TTL: time to refill from 0 to capacity
    const ttlMs = (this.capacity / this.refillRate) * 1000;
    await this.store.set(key, bucket, ttlMs);

    const msUntilNextToken = allowed
      ? 0
      : ((this.tokensPerRequest - bucket.tokens) / this.refillRate) * 1000;

    return {
      allowed,
      tokens: Math.floor(bucket.tokens),
      remaining: Math.floor(bucket.tokens),
      resetAfterMs: Math.ceil(msUntilNextToken),
      strategy: 'token-bucket',
    };
  }

  async reset(key) {
    await this.store.delete(key);
  }
}

module.exports = TokenBucketStrategy;
```

## 6. Core Rate Limiter

```javascript
// src/RateLimiter.js

const FixedWindowStrategy = require('./strategies/fixedWindow');
const SlidingWindowStrategy = require('./strategies/slidingWindow');
const TokenBucketStrategy = require('./strategies/tokenBucket');
const MemoryStore = require('./stores/memoryStore');

const STRATEGIES = {
  'fixed-window': FixedWindowStrategy,
  'sliding-window': SlidingWindowStrategy,
  'token-bucket': TokenBucketStrategy,
};

class RateLimiter {
  /**
   * @param {object} options
   * @param {string}   [options.strategy='fixed-window'] - Rate limiting strategy
   * @param {number}   [options.windowMs=60000]          - Window in ms (fixed/sliding)
   * @param {number}   [options.maxRequests=100]         - Max requests per window
   * @param {number}   [options.capacity=10]             - Token bucket capacity
   * @param {number}   [options.refillRate=1]            - Token bucket refill rate/sec
   * @param {number}   [options.tokensPerRequest=1]      - Tokens per request
   * @param {object}   [options.store]                   - Custom store (default: MemoryStore)
   * @param {function} [options.keyGenerator]            - Custom key generator fn(req)
   * @param {function} [options.onLimitReached]          - Callback when limit is hit
   * @param {boolean}  [options.skipFailedRequests=false]- Don't count failed responses
   * @param {number}   [options.statusCode=429]          - HTTP status when limited
   * @param {string}   [options.message]                 - Custom rejection message
   */
  constructor(options = {}) {
    const {
      strategy = 'fixed-window',
      windowMs = 60_000,
      maxRequests = 100,
      capacity = 10,
      refillRate = 1,
      tokensPerRequest = 1,
      store = new MemoryStore(),
      keyGenerator = null,
      onLimitReached = null,
      skipFailedRequests = false,
      statusCode = 429,
      message = 'Too many requests, please try again later.',
    } = options;

    if (!STRATEGIES[strategy]) {
      throw new Error(
        `Unknown strategy: "${strategy}". Choose from: ${Object.keys(STRATEGIES).join(', ')}`
      );
    }

    this.store = store;
    this.keyGenerator = keyGenerator || this._defaultKeyGenerator;
    this.onLimitReached = onLimitReached;
    this.skipFailedRequests = skipFailedRequests;
    this.statusCode = statusCode;
    this.message = message;

    const StrategyClass = STRATEGIES[strategy];
    this.strategy = new StrategyClass({
      windowMs,
      maxRequests,
      capacity,
      refillRate,
      tokensPerRequest,
      store,
    });
  }

  _defaultKeyGenerator(req) {
    return (
      req.ip ||
      req.headers['x-forwarded-for']?.split(',')[0].trim() ||
      req.connection?.remoteAddress ||
      'unknown'
    );
  }

  async check(identifier) {
    return await this.strategy.check(identifier);
  }

  async reset(identifier) {
    return await this.strategy.reset(identifier);
  }

  /**
   * Express/Connect middleware
   */
  middleware() {
    return async (req, res, next) => {
      const key = this.keyGenerator(req);
      let result;

      try {
        result = await this.check(key);
      } catch (err) {
        console.error('[RateLimiter] Store error:', err);
        return next(); // Fail open
      }

      // Set standard rate limit headers
      res.setHeader('X-RateLimit-Strategy', result.strategy);
      res.setHeader('X-RateLimit-Remaining', result.remaining ?? result.tokens ?? 0);
      res.setHeader('X-RateLimit-Reset', Math.ceil(Date.now() / 1000) + Math.ceil((result.resetAfterMs || 0) / 1000));

      if (result.allowed) {
        if (this.skipFailedRequests) {
          const originalEnd = res.end.bind(res);
          res.end = async (...args) => {
            if (res.statusCode >= 400) {
              await this.reset(key).catch(() => {});
            }
            return originalEnd(...args);
          };
        }
        return next();
      }

      // Rate limit exceeded
      res.setHeader('Retry-After', Math.ceil((result.resetAfterMs || 0) / 1000));

      if (this.onLimitReached) {
        this.onLimitReached(req, res, key, result);
      }

      res.status(this.statusCode).json({
        error: this.message,
        retryAfter: Math.ceil((result.resetAfterMs || 0) / 1000),
      });
    };
  }
}

module.exports = RateLimiter;
```
