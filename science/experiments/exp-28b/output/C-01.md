## Stores

### `src/stores/memoryStore.js`

```javascript
/**
 * In-memory store for rate limiter data.
 * Suitable for single-instance applications.
 */
class MemoryStore {
  constructor() {
    this.store = new Map();
    // Periodically clean up expired entries
    this.cleanupInterval = setInterval(() => this.cleanup(), 60_000);
    // Allow process to exit even if interval is running
    this.cleanupInterval.unref();
  }

  /**
   * Get the current record for a key.
   * @param {string} key
   * @returns {object|null}
   */
  async get(key) {
    const record = this.store.get(key);
    if (!record) return null;

    // Return null if the record has expired
    if (Date.now() > record.expiresAt) {
      this.store.delete(key);
      return null;
    }

    return record;
  }

  /**
   * Set a record with a TTL (in seconds).
   * @param {string} key
   * @param {object} value
   * @param {number} ttlSeconds
   */
  async set(key, value, ttlSeconds) {
    this.store.set(key, {
      ...value,
      expiresAt: Date.now() + ttlSeconds * 1000,
    });
  }

  /**
   * Atomically increment a counter for a key.
   * @param {string} key
   * @param {number} ttlSeconds - TTL to set if key is new
   * @returns {number} new count
   */
  async increment(key, ttlSeconds) {
    const record = await this.get(key);

    if (!record) {
      await this.set(key, { count: 1 }, ttlSeconds);
      return 1;
    }

    record.count += 1;
    this.store.set(key, record);
    return record.count;
  }

  /**
   * Delete a key.
   * @param {string} key
   */
  async delete(key) {
    this.store.delete(key);
  }

  /**
   * Remove all expired entries from the store.
   */
  cleanup() {
    const now = Date.now();
    for (const [key, record] of this.store.entries()) {
      if (now > record.expiresAt) {
        this.store.delete(key);
      }
    }
  }

  /**
   * Destroy the store and clear the cleanup interval.
   */
  destroy() {
    clearInterval(this.cleanupInterval);
    this.store.clear();
  }
}

module.exports = MemoryStore;
```

const Redis = require('ioredis');

/**
 * Redis-backed store for rate limiter data.
 * Suitable for distributed / multi-instance applications.
 */
class RedisStore {
  /**
   * @param {object} options - ioredis connection options
   */
  constructor(options = {}) {
    this.client = options.client || new Redis(options);
  }

  /**
   * Get the current record for a key.
   * @param {string} key
   * @returns {object|null}
   */
  async get(key) {
    const data = await this.client.get(key);
    return data ? JSON.parse(data) : null;
  }

  /**
   * Set a record with a TTL (in seconds).
   * @param {string} key
   * @param {object} value
   * @param {number} ttlSeconds
   */
  async set(key, value, ttlSeconds) {
    await this.client.set(key, JSON.stringify(value), 'EX', ttlSeconds);
  }

  /**
   * Atomically increment a counter using a Lua script.
   * Ensures thread-safe increments in a distributed environment.
   * @param {string} key
   * @param {number} ttlSeconds
   * @returns {number} new count
   */
  async increment(key, ttlSeconds) {
    const luaScript = `
      local current = redis.call('INCR', KEYS[1])
      if current == 1 then
        redis.call('EXPIRE', KEYS[1], ARGV[1])
      end
      return current
    `;
    return this.client.eval(luaScript, 1, key, ttlSeconds);
  }

  /**
   * Delete a key.
   * @param {string} key
   */
  async delete(key) {
    await this.client.del(key);
  }

  /**
   * Close the Redis connection.
   */
  async destroy() {
    await this.client.quit();
  }
}

module.exports = RedisStore;

## Strategies

### `src/strategies/fixedWindow.js`

```javascript
/**
 * Fixed Window Strategy
 *
 * Divides time into fixed windows (e.g., every 60s).
 * Allows up to `maxRequests` per window per key.
 *
 * Pros:  Simple, low memory usage.
 * Cons:  Burst traffic possible at window boundaries.
 */
class FixedWindowStrategy {
  /**
   * @param {object} store - Storage backend
   * @param {number} maxRequests - Max requests per window
   * @param {number} windowSizeSeconds - Window duration in seconds
   */
  constructor(store, maxRequests, windowSizeSeconds) {
    this.store = store;
    this.maxRequests = maxRequests;
    this.windowSizeSeconds = windowSizeSeconds;
  }

  /**
   * Check if the request is allowed and update the counter.
   * @param {string} key - Unique identifier (e.g., IP address)
   * @returns {object} result
   */
  async check(key) {
    const windowKey = `fw:${key}:${Math.floor(Date.now() / (this.windowSizeSeconds * 1000))}`;
    const count = await this.store.increment(windowKey, this.windowSizeSeconds);

    const remaining = Math.max(0, this.maxRequests - count);
    const allowed = count <= this.maxRequests;

    return {
      allowed,
      count,
      remaining,
      limit: this.maxRequests,
      resetTime: this.getResetTime(),
    };
  }

  getResetTime() {
    const windowMs = this.windowSizeSeconds * 1000;
    return new Date(Math.ceil(Date.now() / windowMs) * windowMs);
  }
}

module.exports = FixedWindowStrategy;
```

### `src/strategies/slidingWindow.js`

```javascript
/**
 * Sliding Window Strategy
 *
 * Tracks requests in a rolling time window.
 * Smooths out burst traffic at window boundaries.
 *
 * Pros:  More accurate than fixed window.
 * Cons:  Higher memory usage (stores timestamps).
 */
class SlidingWindowStrategy {
  /**
   * @param {object} store - Storage backend
   * @param {number} maxRequests - Max requests per window
   * @param {number} windowSizeSeconds - Window duration in seconds
   */
  constructor(store, maxRequests, windowSizeSeconds) {
    this.store = store;
    this.maxRequests = maxRequests;
    this.windowSizeSeconds = windowSizeSeconds;
  }

  /**
   * Check if the request is allowed and update the timestamp log.
   * @param {string} key
   * @returns {object} result
   */
  async check(key) {
    const storeKey = `sw:${key}`;
    const now = Date.now();
    const windowStart = now - this.windowSizeSeconds * 1000;

    // Retrieve existing record
    let record = await this.store.get(storeKey);
    let timestamps = record ? record.timestamps : [];

    // Remove timestamps outside the current window
    timestamps = timestamps.filter((ts) => ts > windowStart);

    // Add the current request timestamp
    timestamps.push(now);

    const count = timestamps.length;
    const allowed = count <= this.maxRequests;

    // Persist updated timestamps
    await this.store.set(storeKey, { timestamps }, this.windowSizeSeconds);

    const remaining = Math.max(0, this.maxRequests - count);
    const oldestTimestamp = timestamps[0];
    const resetTime = new Date(oldestTimestamp + this.windowSizeSeconds * 1000);

    return {
      allowed,
      count,
      remaining,
      limit: this.maxRequests,
      resetTime,
    };
  }
}

module.exports = SlidingWindowStrategy;
```

### `src/strategies/tokenBucket.js`

```javascript
/**
 * Token Bucket Strategy
 *
 * Tokens are added to a bucket at a fixed rate.
 * Each request consumes one token.
 * Allows controlled bursting up to bucket capacity.
 *
 * Pros:  Handles burst traffic gracefully.
 * Cons:  Slightly more complex state management.
 */
class TokenBucketStrategy {
  /**
   * @param {object} store - Storage backend
   * @param {number} capacity - Maximum tokens in the bucket
   * @param {number} refillRate - Tokens added per second
   */
  constructor(store, capacity, refillRate) {
    this.store = store;
    this.capacity = capacity;
    this.refillRate = refillRate;
  }

  /**
   * Check if a token is available and consume it.
   * @param {string} key
   * @returns {object} result
   */
  async check(key) {
    const storeKey = `tb:${key}`;
    const now = Date.now();

    let record = await this.store.get(storeKey);

    if (!record) {
      // First request: full bucket minus one token
      record = {
        tokens: this.capacity - 1,
        lastRefill: now,
      };
      await this.store.set(storeKey, record, this.getTTL());

      return {
        allowed: true,
        tokens: record.tokens,
        remaining: record.tokens,
        limit: this.capacity,
        resetTime: this.getResetTime(record.tokens),
      };
    }

    // Calculate tokens to add since last refill
    const elapsed = (now - record.lastRefill) / 1000; // seconds
    const tokensToAdd = elapsed * this.refillRate;
    record.tokens = Math.min(this.capacity, record.tokens + tokensToAdd);
    record.lastRefill = now;

    const allowed = record.tokens >= 1;

    if (allowed) {
      record.tokens -= 1;
    }

    await this.store.set(storeKey, record, this.getTTL());

    return {
      allowed,
      tokens: Math.floor(record.tokens),
      remaining: Math.floor(record.tokens),
      limit: this.capacity,
      resetTime: this.getResetTime(record.tokens),
    };
  }

  getTTL() {
    // Keep the record alive long enough to refill the bucket
    return Math.ceil(this.capacity / this.refillRate) + 60;
  }

  getResetTime(currentTokens) {
    const tokensNeeded = 1 - currentTokens;
    if (tokensNeeded <= 0) return new Date();
    const secondsUntilRefill = tokensNeeded / this.refillRate;
    return new Date(Date.now() + secondsUntilRefill * 1000);
  }
}

module.exports = TokenBucketStrategy;
```

## Core Rate Limiter

### `src/rateLimiter.js`

```javascript
const MemoryStore = require('./stores/memoryStore');
const FixedWindowStrategy = require('./strategies/fixedWindow');
const SlidingWindowStrategy = require('./strategies/slidingWindow');
const TokenBucketStrategy = require('./strategies/tokenBucket');

const STRATEGIES = {
  fixedWindow: FixedWindowStrategy,
  slidingWindow: SlidingWindowStrategy,
  tokenBucket: TokenBucketStrategy,
};

/**
 * RateLimiter - Core class that wires together a strategy and a store.
 */
class RateLimiter {
  /**
   * @param {object} options
   * @param {string}  [options.strategy='fixedWindow']  - 'fixedWindow' | 'slidingWindow' | 'tokenBucket'
   * @param {object}  [options.store]                   - Custom store (defaults to MemoryStore)
   * @param {number}  [options.maxRequests=100]         - Max requests (or bucket capacity)
   * @param {number}  [options.windowSizeSeconds=60]    - Window size in seconds
   * @param {number}  [options.refillRate]              - Tokens/sec for tokenBucket (defaults to maxRequests/windowSizeSeconds)
   * @param {Function}[options.keyGenerator]            - Custom key generator fn(req) => string
   * @param {Function}[options.onLimitReached]          - Custom handler fn(req, res, info) called when limit exceeded
   * @param {boolean} [options.skipFailedRequests=false]- Don't count failed responses (4xx/5xx)
   */
  constructor(options = {}) {
    const {
      strategy = 'fixedWindow',
      store = new MemoryStore(),
      maxRequests = 100,
      windowSizeSeconds = 60,
      refillRate,
      keyGenerator = (req) => req.ip || req.connection?.remoteAddress || 'unknown',
      onLimitReached = null,
      skipFailedRequests = false,
    } = options;

    if (!STRATEGIES[strategy]) {
      throw new Error(`Unknown strategy: "${strategy}". Choose from: ${Object.keys(STRATEGIES).join(', ')}`);
    }

    this.keyGenerator = keyGenerator;
    this.onLimitReached = onLimitReached;
    this.skipFailedRequests = skipFailedRequests;
    this.store = store;

    // Instantiate the chosen strategy
    if (strategy === 'tokenBucket') {
      const rate = refillRate || maxRequests
