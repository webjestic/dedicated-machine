## 1. Memory Store

```javascript
// src/stores/memoryStore.js
class MemoryStore {
  constructor() {
    this.store = new Map();
    this._startCleanup();
  }

  async get(key) {
    const entry = this.store.get(key);
    if (!entry) return null;
    if (Date.now() > entry.expiresAt) {
      this.store.delete(key);
      return null;
    }
    return entry.value;
  }

  async set(key, value, ttlMs) {
    this.store.set(key, {
      value,
      expiresAt: Date.now() + ttlMs,
    });
  }

  async delete(key) {
    this.store.delete(key);
  }

  async increment(key, ttlMs) {
    const entry = this.store.get(key);
    const now = Date.now();

    if (!entry || now > entry.expiresAt) {
      this.store.set(key, { value: 1, expiresAt: now + ttlMs });
      return 1;
    }

    entry.value += 1;
    return entry.value;
  }

  // Clean up expired keys every 60 seconds
  _startCleanup() {
    this._cleanupInterval = setInterval(() => {
      const now = Date.now();
      for (const [key, entry] of this.store.entries()) {
        if (now > entry.expiresAt) {
          this.store.delete(key);
        }
      }
    }, 60_000);

    // Don't block process exit
    if (this._cleanupInterval.unref) {
      this._cleanupInterval.unref();
    }
  }

  destroy() {
    clearInterval(this._cleanupInterval);
    this.store.clear();
  }
}

module.exports = MemoryStore;
```

```javascript
// src/stores/redisStore.js
class RedisStore {
  /**
   * @param {import('ioredis').Redis} client - ioredis client instance
   */
  constructor(client) {
    if (!client) throw new Error('Redis client is required');
    this.client = client;
  }

  async get(key) {
    const value = await this.client.get(key);
    return value ? JSON.parse(value) : null;
  }

  async set(key, value, ttlMs) {
    await this.client.set(key, JSON.stringify(value), 'PX', ttlMs);
  }

  async delete(key) {
    await this.client.del(key);
  }

  async increment(key, ttlMs) {
    const pipeline = this.client.pipeline();
    pipeline.incr(key);
    pipeline.pexpire(key, ttlMs);
    const results = await pipeline.exec();
    return results[0][1]; // INCR result
  }

  // Atomic sliding window using Lua script
  async slidingWindowCount(key, windowMs, now) {
    const script = `
      local key = KEYS[1]
      local now = tonumber(ARGV[1])
      local window = tonumber(ARGV[2])
      local cutoff = now - window

      redis.call('ZREMRANGEBYSCORE', key, '-inf', cutoff)
      local count = redis.call('ZCARD', key)
      redis.call('ZADD', key, now, now)
      redis.call('PEXPIRE', key, window)
      return count + 1
    `;
    return await this.client.eval(script, 1, key, now, windowMs);
  }
}

module.exports = RedisStore;
```

## 3. Fixed Window Strategy

```javascript
// src/strategies/fixedWindow.js

/**
 * Fixed Window Rate Limiting
 * Simple but can allow 2x traffic at window boundaries
 */
class FixedWindowStrategy {
  constructor(store, options = {}) {
    const { limit = 100, windowMs = 60_000 } = options;

    if (limit <= 0) throw new Error('limit must be a positive number');
    if (windowMs <= 0) throw new Error('windowMs must be a positive number');

    this.store = store;
    this.limit = limit;
    this.windowMs = windowMs;
  }

  async check(key) {
    const windowKey = `fw:${key}:${Math.floor(Date.now() / this.windowMs)}`;
    const count = await this.store.increment(windowKey, this.windowMs);

    const remaining = Math.max(0, this.limit - count);
    const resetAt = (Math.floor(Date.now() / this.windowMs) + 1) * this.windowMs;

    return {
      allowed: count <= this.limit,
      limit: this.limit,
      remaining,
      count,
      resetAt,
      retryAfter: count > this.limit ? Math.ceil((resetAt - Date.now()) / 1000) : 0,
    };
  }
}

module.exports = FixedWindowStrategy;
```

## 4. Sliding Window Strategy

```javascript
// src/strategies/slidingWindow.js

/**
 * Sliding Window Log Rate Limiting
 * More accurate but uses more memory (stores timestamps)
 */
class SlidingWindowStrategy {
  constructor(store, options = {}) {
    const { limit = 100, windowMs = 60_000 } = options;

    if (limit <= 0) throw new Error('limit must be a positive number');
    if (windowMs <= 0) throw new Error('windowMs must be a positive number');

    this.store = store;
    this.limit = limit;
    this.windowMs = windowMs;
  }

  async check(key) {
    const now = Date.now();
    const windowKey = `sw:${key}`;

    let count;

    // Use Redis atomic Lua script if available
    if (typeof this.store.slidingWindowCount === 'function') {
      count = await this.store.slidingWindowCount(windowKey, this.windowMs, now);
    } else {
      count = await this._inMemorySlide(windowKey, now);
    }

    const remaining = Math.max(0, this.limit - count);

    return {
      allowed: count <= this.limit,
      limit: this.limit,
      remaining,
      count,
      resetAt: now + this.windowMs,
      retryAfter: count > this.limit ? Math.ceil(this.windowMs / 1000) : 0,
    };
  }

  async _inMemorySlide(key, now) {
    const cutoff = now - this.windowMs;
    let timestamps = (await this.store.get(key)) || [];

    // Remove expired timestamps
    timestamps = timestamps.filter((ts) => ts > cutoff);
    timestamps.push(now);

    await this.store.set(key, timestamps, this.windowMs);
    return timestamps.length;
  }
}

module.exports = SlidingWindowStrategy;
```

## 5. Token Bucket Strategy

```javascript
// src/strategies/tokenBucket.js

/**
 * Token Bucket Rate Limiting
 * Allows controlled bursting while maintaining average rate
 */
class TokenBucketStrategy {
  constructor(store, options = {}) {
    const {
      capacity = 100,       // Max tokens (burst size)
      refillRate = 10,      // Tokens added per second
      refillIntervalMs = 1000,
    } = options;

    if (capacity <= 0) throw new Error('capacity must be positive');
    if (refillRate <= 0) throw new Error('refillRate must be positive');

    this.store = store;
    this.capacity = capacity;
    this.refillRate = refillRate;
    this.refillIntervalMs = refillIntervalMs;
  }

  async check(key, tokensRequested = 1) {
    const bucketKey = `tb:${key}`;
    const now = Date.now();

    let bucket = await this.store.get(bucketKey);

    if (!bucket) {
      bucket = {
        tokens: this.capacity,
        lastRefill: now,
      };
    }

    // Calculate tokens to add since last refill
    const elapsed = now - bucket.lastRefill;
    const tokensToAdd = Math.floor((elapsed / this.refillIntervalMs) * this.refillRate);

    bucket.tokens = Math.min(this.capacity, bucket.tokens + tokensToAdd);
    bucket.lastRefill = tokensToAdd > 0 ? now : bucket.lastRefill;

    const allowed = bucket.tokens >= tokensRequested;

    if (allowed) {
      bucket.tokens -= tokensRequested;
    }

    await this.store.set(bucketKey, bucket, this.refillIntervalMs * this.capacity);

    const timeToNextToken = allowed
      ? 0
      : Math.ceil((this.refillIntervalMs / this.refillRate) * (tokensRequested - bucket.tokens));

    return {
      allowed,
      limit: this.capacity,
      remaining: Math.floor(bucket.tokens),
      count: this.capacity - Math.floor(bucket.tokens),
      resetAt: now + timeToNextToken,
      retryAfter: allowed ? 0 : Math.ceil(timeToNextToken / 1000),
    };
  }
}

module.exports = TokenBucketStrategy;
```

## 6. Core Rate Limiter

```javascript
// src/rateLimiter.js
const MemoryStore = require('./stores/memoryStore');
const FixedWindowStrategy = require('./strategies/fixedWindow');
const SlidingWindowStrategy = require('./strategies/slidingWindow');
const TokenBucketStrategy = require('./strategies/tokenBucket');

const STRATEGIES = {
  fixedWindow: FixedWindowStrategy,
  slidingWindow: SlidingWindowStrategy,
  tokenBucket: TokenBucketStrategy,
};

class RateLimiter {
  /**
   * @param {Object} options
   * @param {string}   options.strategy      - 'fixedWindow' | 'slidingWindow' | 'tokenBucket'
   * @param {number}   options.limit         - Max requests per window (or bucket capacity)
   * @param {number}   options.windowMs      - Window duration in ms
   * @param {Object}   [options.store]       - Custom store (default: MemoryStore)
   * @param {Function} [options.keyGenerator]- Custom key function (req) => string
   * @param {boolean}  [options.skipFailed]  - Don't count failed requests
   */
  constructor(options = {}) {
    const {
      strategy = 'fixedWindow',
      limit = 100,
      windowMs = 60_000,
      store = new MemoryStore(),
      keyGenerator = null,
      skipFailed = false,
      ...strategyOptions
    } = options;

    if (!STRATEGIES[strategy]) {
      throw new Error(`Unknown strategy: "${strategy}". Use: ${Object.keys(STRATEGIES).join(', ')}`);
    }

    this.store = store;
    this.skipFailed = skipFailed;
    this.keyGenerator = keyGenerator || this._defaultKeyGenerator;

    this.strategy = new STRATEGIES[strategy](store, {
      limit,
      windowMs,
      ...strategyOptions,
    });
  }

  /**
   * Check rate limit for a given key
   * @param {string} key
   * @returns {Promise<RateLimitResult>}
   */
  async check(key) {
    if (!key || typeof key !== 'string') {
      throw new Error('key must be a non-empty string');
    }
    return this.strategy.check(key);
  }

  /**
   * Reset rate limit for a key
   */
  async reset(key) {
    await this.store.delete(`fw:${key}`);
    await this.store.delete(`sw:${key}`);
    await this.store.delete(`tb:${key}`);
  }

  _defaultKeyGenerator(req) {
    return (
      req.ip ||
      req.headers['x-forwarded-for']?.split(',')[0].trim() ||
      req.connection?.remoteAddress ||
      'unknown'
    );
  }
}

module.exports = RateLimiter;
```

## 7. Express Middleware

```javascript
// src/middleware.js
const RateLimiter = require('./rateLimiter');

/**
 * Creates Express rate limiting middleware
 * @param {Object} options - RateLimiter options + middleware-specific options
 */
function createRateLimitMiddleware(options = {}) {
  const {
    message = 'Too many requests, please try again later.',
    statusCode = 429,
    headers = true,
    keyGenerator,
    onLimitReached,
    skip,
    ...limiterOptions
  } = options;

  const limiter = new RateLimiter({
    ...limiterOptions,
    keyGenerator,
  });

  return async function rateLimitMiddleware(req, res, next) {
    try {
      // Allow skipping certain requests
      if (skip && (await skip(req))) {
        return next();
      }

      const key = keyGenerator
        ? keyGenerator(req)
        : limiter._defaultKeyGenerator(req);

      const result = await limiter.check(key);

      // Set rate limit headers
      if (headers) {
        res.setHeader('X-RateLimit-Limit', result.limit);
        res.setHeader('X-RateLimit-Remaining', result.remaining);
        res.setHeader('X-RateLimit-Reset', Math.ceil(result.resetAt / 1000));

        if (!result.allowed) {
          res.setHeader('Retry-After', result.retryAfter);
        }
      }

      if (!result.allowed) {
        if (onLimitReached) {
          await onLimitReached(req, res, result);
        }

        return res.status(statusCode).json({
          error: message,
          retryAfter: result.retryAfter,
        });
      }

      // Attach result to request for downstream use
      req.rateLimit = result;
      next();
    } catch (err) {
      // Fail open — don't block requests on store errors
      console.error('[RateLimiter] Error:', err.message);
      next();
    }
  };
}

module.exports = { createRateLimitMiddleware };
```
