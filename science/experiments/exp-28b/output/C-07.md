## 1. Memory Store

```javascript
// src/stores/memoryStore.js

class MemoryStore {
  constructor(cleanupInterval = 60000) {
    this.store = new Map();

    // Periodically clean up expired entries
    this.cleanupTimer = setInterval(() => {
      this.cleanup();
    }, cleanupInterval);

    // Allow process to exit even if timer is active
    if (this.cleanupTimer.unref) {
      this.cleanupTimer.unref();
    }
  }

  async get(key) {
    const entry = this.store.get(key);
    if (!entry) return null;

    if (Date.now() > entry.expiresAt) {
      this.store.delete(key);
      return null;
    }

    return entry.value;
  }

  async set(key, value, ttlMs) {
    this.store.set(key, {
      value,
      expiresAt: Date.now() + ttlMs,
    });
  }

  async increment(key, ttlMs) {
    const entry = this.store.get(key);
    const now = Date.now();

    if (!entry || now > entry.expiresAt) {
      this.store.set(key, { value: 1, expiresAt: now + ttlMs });
      return 1;
    }

    entry.value += 1;
    return entry.value;
  }

  async delete(key) {
    this.store.delete(key);
  }

  async reset() {
    this.store.clear();
  }

  cleanup() {
    const now = Date.now();
    for (const [key, entry] of this.store.entries()) {
      if (now > entry.expiresAt) {
        this.store.delete(key);
      }
    }
  }

  destroy() {
    clearInterval(this.cleanupTimer);
    this.store.clear();
  }
}

module.exports = MemoryStore;
```

```javascript
// src/stores/redisStore.js

class RedisStore {
  constructor(redisClient) {
    if (!redisClient) {
      throw new Error('Redis client is required');
    }
    this.client = redisClient;
  }

  async get(key) {
    const value = await this.client.get(key);
    return value ? JSON.parse(value) : null;
  }

  async set(key, value, ttlMs) {
    await this.client.set(key, JSON.stringify(value), {
      PX: ttlMs, // PX = milliseconds
    });
  }

  async increment(key, ttlMs) {
    // Atomic increment using Lua script
    const script = `
      local current = redis.call('GET', KEYS[1])
      if current == false then
        redis.call('SET', KEYS[1], 1, 'PX', ARGV[1])
        return 1
      else
        local newVal = tonumber(current) + 1
        redis.call('SET', KEYS[1], newVal, 'KEEPTTL')
        return newVal
      end
    `;

    return await this.client.eval(script, {
      keys: [key],
      arguments: [String(ttlMs)],
    });
  }

  async delete(key) {
    await this.client.del(key);
  }

  async reset(pattern = '*') {
    const keys = await this.client.keys(pattern);
    if (keys.length > 0) {
      await this.client.del(keys);
    }
  }
}

module.exports = RedisStore;
```

## 3. Fixed Window Strategy

```javascript
// src/strategies/fixedWindow.js

class FixedWindowStrategy {
  /**
   * Fixed Window Rate Limiting
   * Simple counter per time window. Resets at fixed intervals.
   *
   * Pros: Simple, low memory usage
   * Cons: Burst traffic at window boundaries
   */
  constructor(store, options = {}) {
    this.store = store;
    this.windowMs = options.windowMs || 60000;   // 1 minute
    this.maxRequests = options.maxRequests || 100;
  }

  async check(key) {
    const windowKey = this._getWindowKey(key);
    const count = await this.store.increment(windowKey, this.windowMs);
    const remaining = Math.max(0, this.maxRequests - count);
    const resetTime = this._getWindowResetTime();

    return {
      allowed: count <= this.maxRequests,
      count,
      remaining,
      resetTime,
      retryAfter: count > this.maxRequests
        ? Math.ceil((resetTime - Date.now()) / 1000)
        : 0,
    };
  }

  async reset(key) {
    const windowKey = this._getWindowKey(key);
    await this.store.delete(windowKey);
  }

  _getWindowKey(key) {
    const windowId = Math.floor(Date.now() / this.windowMs);
    return `fw:${key}:${windowId}`;
  }

  _getWindowResetTime() {
    const windowId = Math.floor(Date.now() / this.windowMs);
    return (windowId + 1) * this.windowMs;
  }
}

module.exports = FixedWindowStrategy;
```

## 4. Sliding Window Strategy

```javascript
// src/strategies/slidingWindow.js

class SlidingWindowStrategy {
  /**
   * Sliding Window Rate Limiting
   * Tracks requests in a rolling time window.
   *
   * Pros: Smooth rate limiting, no boundary bursts
   * Cons: Higher memory usage (stores timestamps)
   */
  constructor(store, options = {}) {
    this.store = store;
    this.windowMs = options.windowMs || 60000;
    this.maxRequests = options.maxRequests || 100;
  }

  async check(key) {
    const now = Date.now();
    const windowStart = now - this.windowMs;
    const storeKey = `sw:${key}`;

    // Get existing timestamps
    let timestamps = (await this.store.get(storeKey)) || [];

    // Filter out expired timestamps
    timestamps = timestamps.filter((ts) => ts > windowStart);

    // Add current request timestamp
    timestamps.push(now);

    // Save updated timestamps
    await this.store.set(storeKey, timestamps, this.windowMs);

    const count = timestamps.length;
    const remaining = Math.max(0, this.maxRequests - count);

    // Calculate when the oldest request will expire
    const oldestTimestamp = timestamps[0];
    const resetTime = oldestTimestamp + this.windowMs;

    return {
      allowed: count <= this.maxRequests,
      count,
      remaining,
      resetTime,
      retryAfter: count > this.maxRequests
        ? Math.ceil((resetTime - now) / 1000)
        : 0,
    };
  }

  async reset(key) {
    await this.store.delete(`sw:${key}`);
  }
}

module.exports = SlidingWindowStrategy;
```

## 5. Token Bucket Strategy

```javascript
// src/strategies/tokenBucket.js

class TokenBucketStrategy {
  /**
   * Token Bucket Rate Limiting
   * Tokens are added at a fixed rate. Requests consume tokens.
   *
   * Pros: Handles bursts gracefully, smooth over time
   * Cons: More complex state management
   */
  constructor(store, options = {}) {
    this.store = store;
    this.capacity = options.capacity || 100;          // Max tokens
    this.refillRate = options.refillRate || 10;       // Tokens per second
    this.refillIntervalMs = options.refillIntervalMs || 1000;
    this.tokensPerInterval = options.tokensPerInterval || this.refillRate;
  }

  async check(key) {
    const storeKey = `tb:${key}`;
    const now = Date.now();

    let bucket = await this.store.get(storeKey);

    if (!bucket) {
      // Initialize bucket with full capacity
      bucket = {
        tokens: this.capacity,
        lastRefill: now,
      };
    }

    // Calculate tokens to add since last refill
    const elapsed = now - bucket.lastRefill;
    const intervalsElapsed = Math.floor(elapsed / this.refillIntervalMs);
    const tokensToAdd = intervalsElapsed * this.tokensPerInterval;

    if (tokensToAdd > 0) {
      bucket.tokens = Math.min(this.capacity, bucket.tokens + tokensToAdd);
      bucket.lastRefill = bucket.lastRefill + intervalsElapsed * this.refillIntervalMs;
    }

    const allowed = bucket.tokens >= 1;

    if (allowed) {
      bucket.tokens -= 1;
    }

    // TTL: time until bucket would be full again
    const ttlMs = Math.ceil(
      ((this.capacity - bucket.tokens) / this.tokensPerInterval) *
        this.refillIntervalMs
    ) + this.refillIntervalMs;

    await this.store.set(storeKey, bucket, Math.max(ttlMs, this.refillIntervalMs));

    // Calculate retry after (time until next token)
    const retryAfter = allowed
      ? 0
      : Math.ceil(
          (this.refillIntervalMs - (now - bucket.lastRefill)) / 1000
        );

    return {
      allowed,
      count: this.capacity - bucket.tokens,
      remaining: Math.floor(bucket.tokens),
      resetTime: bucket.lastRefill + this.refillIntervalMs,
      retryAfter: Math.max(0, retryAfter),
    };
  }

  async reset(key) {
    await this.store.delete(`tb:${key}`);
  }
}

module.exports = TokenBucketStrategy;
```

## 6. Core Rate Limiter

```javascript
// src/rateLimiter.js

const MemoryStore = require('./stores/memoryStore');
const FixedWindowStrategy = require('./strategies/fixedWindow');
const SlidingWindowStrategy = require('./strategies/slidingWindow');
const TokenBucketStrategy = require('./strategies/tokenBucket');

const STRATEGIES = {
  fixedWindow: FixedWindowStrategy,
  slidingWindow: SlidingWindowStrategy,
  tokenBucket: TokenBucketStrategy,
};

class RateLimiter {
  constructor(options = {}) {
    const {
      strategy = 'fixedWindow',
      store = null,
      keyGenerator = null,
      skip = null,
      onLimitReached = null,
      ...strategyOptions
    } = options;

    // Validate strategy
    if (!STRATEGIES[strategy]) {
      throw new Error(
        `Unknown strategy: "${strategy}". Available: ${Object.keys(STRATEGIES).join(', ')}`
      );
    }

    this.store = store || new MemoryStore();
    this.strategy = new STRATEGIES[strategy](this.store, strategyOptions);
    this.keyGenerator = keyGenerator || this._defaultKeyGenerator;
    this.skip = skip;
    this.onLimitReached = onLimitReached;
  }

  async check(identifier) {
    if (!identifier) {
      throw new Error('Identifier is required');
    }
    return await this.strategy.check(identifier);
  }

  async reset(identifier) {
    if (!identifier) {
      throw new Error('Identifier is required');
    }
    return await this.strategy.reset(identifier);
  }

  // Express/Connect middleware
  middleware() {
    return async (req, res, next) => {
      try {
        // Check if request should be skipped
        if (this.skip && (await this.skip(req))) {
          return next();
        }

        const key = await this.keyGenerator(req);
        const result = await this.check(key);

        // Set rate limit headers
        res.setHeader('X-RateLimit-Limit', this._getLimit());
        res.setHeader('X-RateLimit-Remaining', result.remaining);
        res.setHeader('X-RateLimit-Reset', Math.ceil(result.resetTime / 1000));

        if (!result.allowed) {
          res.setHeader('Retry-After', result.retryAfter);

          if (this.onLimitReached) {
            await this.onLimitReached(req, res, result);
          }

          if (!res.headersSent) {
            return res.status(429).json({
              error: 'Too Many Requests',
              message: `Rate limit exceeded. Try again in ${result.retryAfter} seconds.`,
              retryAfter: result.retryAfter,
            });
          }
          return;
        }

        next();
      } catch (err) {
        next(err);
      }
    };
  }

  _defaultKeyGenerator(req) {
    return (
      req.ip ||
      req.headers['x-forwarded-for']?.split(',')[0].trim() ||
      req.connection?.remoteAddress ||
      'unknown'
    );
  }

  _getLimit() {
    return (
      this.strategy.maxRequests ||
      this.strategy.capacity ||
      'unknown'
    );
  }
}

module.exports = RateLimiter;
```

## 7. Middleware Factory (Convenience)

```javascript
// src/middleware.js

const RateLimiter = require('./rateLimiter');

/**
 * Create rate limiter middleware with sensible defaults
 */
function createRateLimiter(options = {}) {
  const limiter = new RateLimiter(options);
  return limiter.middleware();
}

/**
 * Create multiple rate limiters (e.g., per-route + global)
 */
function createTieredLimiter(tiers = []) {
  const limiters = tiers.map((tier) => new RateLimiter(tier));

  return async (req, res, next) => {
    for (const limiter of limiters) {
      const middleware = limiter.middleware();
      const blocked = await new Promise((resolve) => {
        middleware(req, res, (err) => {
          if (err) resolve(err);
          else resolve(null);
        });

        // If response was sent (blocked), resolve with true
        res.on('finish', () => resolve(true));
      });

      if (blocked === true || res.headersSent) return;
      if (blocked instanceof Error) return next(blocked);
    }
    next();
  };
}

module.exports = { createRateLimiter, createTieredLimiter };
```
