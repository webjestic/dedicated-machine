# Production Rate Limiter in Node.js

## Architecture Overview

I'll build a multi-strategy rate limiter with:
- **Token Bucket** (default) — smooth traffic, allows bursts
- **Fixed Window** — simple, low memory
- **Sliding Window Log** — precise, higher memory
- **Sliding Window Counter** — precise + memory efficient

Redis-backed for distributed environments with in-memory fallback.

### `src/stores/memoryStore.js`

```javascript
/**
 * In-memory store with TTL-based expiration.
 *
 * NOT suitable for multi-process deployments — use RedisStore there.
 * Useful for: single-process servers, testing, local dev.
 *
 * Uses a Map with periodic cleanup to avoid unbounded memory growth.
 */
class MemoryStore {
  constructor(options = {}) {
    this.store = new Map();
    // Run cleanup every 5 minutes by default
    this.cleanupInterval = options.cleanupInterval ?? 5 * 60 * 1000;
    this._startCleanup();
  }

  /**
   * Atomically increment a counter with expiry.
   * Returns { count, ttlMs } after increment.
   */
  async increment(key, windowMs) {
    const now = Date.now();
    const entry = this.store.get(key);

    if (!entry || now >= entry.expiresAt) {
      // New window
      const expiresAt = now + windowMs;
      this.store.set(key, { count: 1, expiresAt });
      return { count: 1, ttlMs: windowMs };
    }

    entry.count += 1;
    return { count: entry.count, ttlMs: entry.expiresAt - now };
  }

  async get(key) {
    const now = Date.now();
    const entry = this.store.get(key);
    if (!entry || now >= entry.expiresAt) return null;
    return { ...entry, ttlMs: entry.expiresAt - now };
  }

  async set(key, value, ttlMs) {
    this.store.set(key, {
      ...value,
      expiresAt: Date.now() + ttlMs,
    });
  }

  async delete(key) {
    this.store.delete(key);
  }

  /**
   * Append to a list and trim entries older than windowMs.
   * Used by sliding window log algorithm.
   */
  async appendToLog(key, timestamp, windowMs) {
    const now = Date.now();
    const cutoff = now - windowMs;
    const entry = this.store.get(key);
    const log = entry ? entry.log.filter((ts) => ts > cutoff) : [];

    log.push(timestamp);
    this.store.set(key, { log, expiresAt: now + windowMs });
    return log.length;
  }

  async getLog(key, windowMs) {
    const now = Date.now();
    const cutoff = now - windowMs;
    const entry = this.store.get(key);
    if (!entry) return [];
    return entry.log.filter((ts) => ts > cutoff);
  }

  async reset(key) {
    this.store.delete(key);
  }

  _startCleanup() {
    this._cleanupTimer = setInterval(() => {
      const now = Date.now();
      for (const [key, entry] of this.store.entries()) {
        if (now >= entry.expiresAt) {
          this.store.delete(key);
        }
      }
    }, this.cleanupInterval);

    // Don't block process exit
    if (this._cleanupTimer.unref) {
      this._cleanupTimer.unref();
    }
  }

  destroy() {
    clearInterval(this._cleanupTimer);
    this.store.clear();
  }
}

module.exports = MemoryStore;
```

### `src/stores/redisStore.js`

```javascript
const { StoreConnectionError } = require('../errors');

/**
 * Redis-backed store using atomic Lua scripts.
 *
 * Why Lua scripts?
 * - Atomic execution: no race conditions between check and increment
 * - Single round-trip: reduces latency
 * - Server-side logic: consistent across all app instances
 *
 * Requires: ioredis (npm install ioredis)
 */
class RedisStore {
  constructor(redisClient, options = {}) {
    if (!redisClient) {
      throw new StoreConnectionError('Redis client is required');
    }
    this.client = redisClient;
    this.keyPrefix = options.keyPrefix ?? 'rl:';

    this._defineScripts();
  }

  _prefixKey(key) {
    return `${this.keyPrefix}${key}`;
  }

  _defineScripts() {
    /**
     * Atomic increment with sliding window support.
     * KEYS[1] = key
     * ARGV[1] = windowMs (TTL)
     * Returns: [count, ttlMs]
     */
    this.incrementScript = `
      local key = KEYS[1]
      local window = tonumber(ARGV[1])
      local count = redis.call('INCR', key)
      if count == 1 then
        redis.call('PEXPIRE', key, window)
      end
      local ttl = redis.call('PTTL', key)
      return {count, ttl}
    `;

    /**
     * Token bucket: get current tokens, refill based on elapsed time.
     * KEYS[1] = key
     * ARGV[1] = capacity
     * ARGV[2] = refillRate (tokens per ms)
     * ARGV[3] = now (ms timestamp)
     * ARGV[4] = requested tokens
     * Returns: [allowed (0/1), remaining, retryAfterMs]
     */
    this.tokenBucketScript = `
      local key = KEYS[1]
      local capacity = tonumber(ARGV[1])
      local refill_rate = tonumber(ARGV[2])
      local now = tonumber(ARGV[3])
      local requested = tonumber(ARGV[4])

      local data = redis.call('HMGET', key, 'tokens', 'last_refill')
      local tokens = tonumber(data[1])
      local last_refill = tonumber(data[2])

      if tokens == nil then
        tokens = capacity
        last_refill = now
      else
        local elapsed = now - last_refill
        local refilled = elapsed * refill_rate
        tokens = math.min(capacity, tokens + refilled)
        last_refill = now
      end

      local allowed = 0
      local retry_after = 0

      if tokens >= requested then
        tokens = tokens - requested
        allowed = 1
      else
        local deficit = requested - tokens
        retry_after = math.ceil(deficit / refill_rate)
      end

      redis.call('HMSET', key, 'tokens', tokens, 'last_refill', last_refill)
      -- TTL: time to refill entire bucket + buffer
      local ttl = math.ceil(capacity / refill_rate) + 1000
      redis.call('PEXPIRE', key, ttl)

      return {allowed, math.floor(tokens), retry_after}
    `;

    /**
     * Sliding window log: add timestamp, count entries in window.
     * KEYS[1] = key
     * ARGV[1] = now (ms)
     * ARGV[2] = windowMs
     * ARGV[3] = limit
     * Returns: [allowed (0/1), count, retryAfterMs]
     */
    this.slidingLogScript = `
      local key = KEYS[1]
      local now = tonumber(ARGV[1])
      local window = tonumber(ARGV[2])
      local limit = tonumber(ARGV[3])
      local cutoff = now - window

      -- Remove expired entries
      redis.call('ZREMRANGEBYSCORE', key, '-inf', cutoff)

      -- Count current entries
      local count = redis.call('ZCARD', key)

      local allowed = 0
      local retry_after = 0

      if count < limit then
        redis.call('ZADD', key, now, now .. '-' .. math.random(1000000))
        allowed = 1
        count = count + 1
      else
        -- Oldest entry + window = when a slot opens
        local oldest = redis.call('ZRANGE', key, 0, 0, 'WITHSCORES')
        if oldest[2] then
          retry_after = math.ceil(tonumber(oldest[2]) + window - now)
        end
      end

      redis.call('PEXPIRE', key, window)
      return {allowed, count, retry_after}
    `;
  }

  async increment(key, windowMs) {
    try {
      const result = await this.client.eval(
        this.incrementScript,
        1,
        this._prefixKey(key),
        windowMs
      );
      return { count: result[0], ttlMs: result[1] };
    } catch (err) {
      throw new StoreConnectionError(`Redis increment failed: ${err.message}`, err);
    }
  }

  async tokenBucketConsume(key, capacity, refillRatePerMs, requested = 1) {
    try {
      const result = await this.client.eval(
        this.tokenBucketScript,
        1,
        this._prefixKey(key),
        capacity,
        refillRatePerMs,
        Date.now(),
        requested
      );
      return {
        allowed: result[0] === 1,
        remaining: result[1],
        retryAfterMs: result[2],
      };
    } catch (err) {
      throw new StoreConnectionError(`Redis token bucket failed: ${err.message}`, err);
    }
  }

  async slidingLogConsume(key, windowMs, limit) {
    try {
      const result = await this.client.eval(
        this.slidingLogScript,
        1,
        this._prefixKey(key),
        Date.now(),
        windowMs,
        limit
      );
      return {
        allowed: result[0] === 1,
        count: result[1],
        retryAfterMs: result[2],
      };
    } catch (err) {
      throw new StoreConnectionError(`Redis sliding log failed: ${err.message}`, err);
    }
  }

  async get(key) {
    try {
      const data = await this.client.hgetall(this._prefixKey(key));
      return Object.keys(data).length ? data : null;
    } catch (err) {
      throw new StoreConnectionError(`Redis get failed: ${err.message}`, err);
    }
  }

  async reset(key) {
    try {
      await this.client.del(this._prefixKey(key));
    } catch (err) {
      throw new StoreConnectionError(`Redis reset failed: ${err.message}`, err);
    }
  }

  async ping() {
    try {
      const result = await this.client.ping();
      return result === 'PONG';
    } catch {
      return false;
    }
  }
}

module.exports = RedisStore;
```

### `src/algorithms/tokenBucket.js`

```javascript
/**
 * Token Bucket Algorithm
 *
 * Mental model: A bucket holds tokens. Each request consumes tokens.
 * Tokens refill at a constant rate. Allows controlled bursting.
 *
 * Best for: APIs where you want to allow short bursts but enforce
 *           an average rate over time. Most real-world APIs use this.
 *
 * Properties:
 * - Allows burst up to `capacity`
 * - Average rate enforced by `refillRate`
 * - Smooth: doesn't hard-reset at window boundaries
 */
class TokenBucket {
  /**
   * @param {object} options
   * @param {number} options.capacity       - Max tokens (burst size)
   * @param {number} options.refillRate     - Tokens added per second
   * @param {object} options.store          - Storage backend
   */
  constructor(options) {
    this.capacity = options.capacity;
    this.refillRate = options.refillRate;
    this.refillRatePerMs = options.refillRate / 1000;
    this.store = options.store;
  }

  async consume(key, tokens = 1) {
    // Redis store has native atomic token bucket support
    if (this.store.tokenBucketConsume) {
      return this.store.tokenBucketConsume(
        key,
        this.capacity,
        this.refillRatePerMs,
        tokens
      );
    }

    // Memory store fallback — still correct, single-process only
    return this._memoryConsume(key, tokens);
  }

  async _memoryConsume(key, requested) {
    const now = Date.now();
    const entry = await this.store.get(key);

    let currentTokens;
    let lastRefill;

    if (!entry) {
      currentTokens = this.capacity;
      lastRefill = now;
    } else {
      const elapsed = now - entry.lastRefill;
      const refilled = elapsed * this.refillRatePerMs;
      currentTokens = Math.min(this.capacity, entry.tokens + refilled);
      lastRefill = now;
    }

    if (currentTokens >= requested) {
      const remaining = currentTokens - requested;
      const ttlMs = Math.ceil(this.capacity / this.refillRatePerMs);
      await this.store.set(key, { tokens: remaining, lastRefill }, ttlMs);
      return { allowed: true, remaining: Math.floor(remaining), retryAfterMs: 0 };
    }

    // Not enough tokens — calculate wait time
    const deficit = requested - currentTokens;
    const retryAfterMs = Math.ceil(deficit / this.refillRatePerMs);

    // Still save updated token count (don't waste the refill calculation)
    const ttlMs = Math.ceil(this.capacity / this.refillRatePerMs);
    await this.store.set(key, { tokens: currentTokens, lastRefill }, ttlMs);
