## Phase 1: Requirements Design

### What problem are we actually solving?

A rate limiter protects a service from:
- **Abuse** — intentional flooding
- **Bugs** — client loops gone wrong
- **Cascading failures** — one service hammering another
- **Cost overruns** — metered APIs, DB connections, etc.

### Decision Matrix — Algorithm Selection

| Algorithm | Memory | Burst Handling | Accuracy | Distributed-Safe | Verdict |
|---|---|---|---|---|---|
| Fixed Window | Low | Poor (boundary burst) | Low | Easy | ❌ Too naive |
| Sliding Window Log | High | Excellent | High | Hard | ❌ Memory cost |
| Sliding Window Counter | Medium | Good | Good | Medium | ✅ Good balance |
| Token Bucket | Medium | Configurable | High | Hard | ✅ Best for APIs |
| Leaky Bucket | Low | None (smoothing) | High | Medium | ⚠️ Too restrictive |

**Decision: Sliding Window Counter** as primary (Redis-backed, distributed-safe), with **Token Bucket** available as a strategy. Sliding window is the right default for HTTP APIs — it prevents boundary bursts, is memory-efficient, and maps cleanly to Redis atomic operations.

### Key Design Decisions

**1. Where does state live?**
- In-memory only → breaks with multiple instances (horizontal scale)
- Redis → correct for production, adds network latency (~1ms)
- Hybrid → in-memory fast path + Redis sync (complex, needed at extreme scale)

**Decision: Redis primary, in-memory fallback with explicit degraded-mode behavior**

**2. What is the rate limit key?**
Options: IP, User ID, API Key, Route, combinations thereof.
**Decision: Pluggable key strategy — caller defines it**

**3. What happens when Redis is down?**
- Fail open (allow all) → protects availability, risks abuse
- Fail closed (deny all) → protects backend, kills availability
- Fail open with circuit breaker + alerting → correct answer

**Decision: Configurable per use case, default fail-open with logging**

**4. What headers do we return?**
Follow the draft IETF standard (`RateLimit-*` headers) + de facto standard (`X-RateLimit-*`).

**5. Multi-tier limits?**
Yes. Real systems need: per-second burst + per-minute sustained + per-day quota.

```javascript
'use strict';

/**
 * Thrown when a rate limit is exceeded.
 * Carries enough metadata to build a proper 429 response.
 */
class RateLimitExceededError extends Error {
  constructor({ limit, remaining, resetAt, retryAfter, tier, key }) {
    super(`Rate limit exceeded on tier "${tier}" for key "${key}"`);
    this.name = 'RateLimitExceededError';
    this.limit = limit;
    this.remaining = remaining;
    this.resetAt = resetAt;       // Unix timestamp (seconds)
    this.retryAfter = retryAfter; // Seconds until next allowed request
    this.tier = tier;
    this.key = key;
  }
}

/**
 * Thrown when the rate limiter store is unavailable.
 * Used to trigger circuit breaker / fallback logic.
 */
class StoreUnavailableError extends Error {
  constructor(message, cause) {
    super(message);
    this.name = 'StoreUnavailableError';
    this.cause = cause;
  }
}

module.exports = { RateLimitExceededError, StoreUnavailableError };

### `src/config.js`

```javascript
'use strict';

/**
 * Configuration schema with validation.
 *
 * A "tier" defines one rate limit window.
 * Multiple tiers are evaluated together — ALL must pass.
 *
 * Example: allow 10 req/sec burst AND 500 req/min sustained AND 10000 req/day quota.
 */

const DEFAULT_TIERS = [
  { name: 'burst',     windowMs: 1_000,        maxRequests: 10  },
  { name: 'sustained', windowMs: 60_000,        maxRequests: 100 },
  { name: 'daily',     windowMs: 86_400_000,    maxRequests: 5000 },
];

function validateTier(tier, index) {
  if (typeof tier.name !== 'string' || !tier.name) {
    throw new Error(`Tier[${index}].name must be a non-empty string`);
  }
  if (typeof tier.windowMs !== 'number' || tier.windowMs <= 0) {
    throw new Error(`Tier[${index}].windowMs must be a positive number`);
  }
  if (typeof tier.maxRequests !== 'number' || tier.maxRequests <= 0) {
    throw new Error(`Tier[${index}].maxRequests must be a positive number`);
  }
}

function buildConfig(options = {}) {
  const config = {
    // Rate limit tiers — evaluated in order, all must pass
    tiers: options.tiers ?? DEFAULT_TIERS,

    // Key extraction: function(req) => string
    // Defaults to IP address
    keyStrategy: options.keyStrategy ?? null,

    // Prefix for all Redis keys — prevents collisions in shared Redis
    keyPrefix: options.keyPrefix ?? 'rl:',

    // What to do when the store is unavailable
    // 'open'  = allow requests (availability > protection)
    // 'closed' = deny requests (protection > availability)
    onStoreError: options.onStoreError ?? 'open',

    // Called on every rate limit decision — use for metrics/logging
    onLimitReached: options.onLimitReached ?? null,

    // Called on store errors — use for alerting
    onStoreUnavailable: options.onStoreUnavailable ?? null,

    // Skip rate limiting for matched requests
    skip: options.skip ?? null,

    // Custom response handler — override default 429 behavior
    handler: options.handler ?? null,

    // Include standard RateLimit headers in response
    headers: options.headers ?? true,

    // Draft IETF header format vs legacy X-RateLimit-* format
    // 'draft-7' | 'legacy' | 'both'
    headerFormat: options.headerFormat ?? 'both',
  };

  // Validate
  if (!Array.isArray(config.tiers) || config.tiers.length === 0) {
    throw new Error('config.tiers must be a non-empty array');
  }
  config.tiers.forEach(validateTier);

  if (!['open', 'closed'].includes(config.onStoreError)) {
    throw new Error('config.onStoreError must be "open" or "closed"');
  }

  if (!['draft-7', 'legacy', 'both'].includes(config.headerFormat)) {
    throw new Error('config.headerFormat must be "draft-7", "legacy", or "both"');
  }

  return config;
}

module.exports = { buildConfig, DEFAULT_TIERS };
```

### `src/strategies/keyStrategy.js`

```javascript
'use strict';

/**
 * Key extraction strategies.
 *
 * The key determines WHO is being rate limited.
 * Getting this wrong is a security vulnerability:
 *   - Too broad (e.g., shared IP behind NAT) → legitimate users blocked
 *   - Too narrow (e.g., spoofable header) → bypass possible
 */

/**
 * Extract IP address from request.
 * Handles proxies via X-Forwarded-For with trust validation.
 *
 * SECURITY NOTE: X-Forwarded-For can be spoofed by clients unless
 * your infrastructure (load balancer / proxy) strips or validates it.
 * Only trust headers your infrastructure sets.
 */
function ipStrategy(req) {
  // Trust X-Forwarded-For only if set by trusted proxy
  // In production: configure this based on your infrastructure
  const forwarded = req.headers['x-forwarded-for'];
  if (forwarded) {
    // X-Forwarded-For: client, proxy1, proxy2
    // The leftmost is the original client (but can be spoofed)
    // The rightmost is the last trusted proxy
    // For most setups behind a single load balancer, take the first entry
    const clientIp = forwarded.split(',')[0].trim();
    if (isValidIp(clientIp)) {
      return `ip:${clientIp}`;
    }
  }

  // Fall back to socket remote address
  const socketIp = req.socket?.remoteAddress ?? req.connection?.remoteAddress;
  return `ip:${socketIp ?? 'unknown'}`;
}

/**
 * Rate limit by authenticated user ID.
 * Requires auth middleware to run first and set req.user.
 */
function userIdStrategy(req) {
  if (!req.user?.id) {
    // Fall back to IP for unauthenticated requests
    return ipStrategy(req);
  }
  return `user:${req.user.id}`;
}

/**
 * Rate limit by API key.
 * Supports Authorization: Bearer <key> and X-API-Key header.
 */
function apiKeyStrategy(req) {
  const authHeader = req.headers['authorization'];
  if (authHeader?.startsWith('Bearer ')) {
    return `apikey:${authHeader.slice(7)}`;
  }

  const apiKey = req.headers['x-api-key'];
  if (apiKey) {
    return `apikey:${apiKey}`;
  }

  // Fall back to IP
  return ipStrategy(req);
}

/**
 * Composite strategy: combine multiple dimensions.
 * Example: per-user per-route limiting.
 */
function compositeStrategy(...strategies) {
  return function (req) {
    return strategies.map(s => s(req)).join('|');
  };
}

/**
 * Route-aware strategy: different limits per endpoint.
 * Wrap another strategy with route information.
 */
function routeAwareStrategy(baseStrategy) {
  return function (req) {
    const route = req.route?.path ?? req.path ?? 'unknown';
    const method = req.method ?? 'GET';
    return `${method}:${route}:${baseStrategy(req)}`;
  };
}

function isValidIp(ip) {
  // Basic validation — IPv4 and IPv6
  const ipv4 = /^(\d{1,3}\.){3}\d{1,3}$/;
  const ipv6 = /^[0-9a-fA-F:]+$/;
  return ipv4.test(ip) || ipv6.test(ip);
}

module.exports = {
  ipStrategy,
  userIdStrategy,
  apiKeyStrategy,
  compositeStrategy,
  routeAwareStrategy,
};
```

### `src/stores/memoryStore.js`

```javascript
'use strict';

/**
 * In-memory store using a sliding window counter algorithm.
 *
 * Uses an LRU-like eviction to bound memory usage.
 * Suitable for:
 *   - Single-instance deployments
 *   - Testing
 *   - Fallback when Redis is unavailable
 *
 * NOT suitable for:
 *   - Multi-instance deployments (each instance has independent state)
 *   - Persistence across restarts
 */

const DEFAULT_MAX_KEYS = 100_000; // Bound memory usage

class MemoryStore {
  constructor(options = {}) {
    this.maxKeys = options.maxKeys ?? DEFAULT_MAX_KEYS;

    // Map<key, { count: number, windowStart: number }>
    // Using Map preserves insertion order for LRU eviction
    this._windows = new Map();

    // Periodic cleanup to prevent memory leaks from expired keys
    this._cleanupInterval = setInterval(
      () => this._cleanup(),
      options.cleanupIntervalMs ?? 60_000
    );

    // Don't prevent process exit
    this._cleanupInterval.unref();
  }

  /**
   * Increment counter for a key within a sliding window.
   *
   * Sliding window algorithm:
   *   - Divide time into fixed windows
   *   - Count requests in current window
   *   - Blend with previous window based on how far into current window we are
   *   - This smooths the boundary burst problem of fixed windows
   *
   * @param {string} key - Rate limit key
   * @param {number} windowMs - Window size in milliseconds
   * @param {number} maxRequests - Maximum requests per window
   * @returns {{ count: number, remaining: number, resetAt: number, allowed: boolean }}
   */
  async increment(key, windowMs, maxRequests) {
    const now = Date.now();
    const windowStart = Math.floor(now / windowMs) * windowMs;
    const windowEnd = windowStart + windowMs;

    const storeKey = `${key}:${windowMs}`;
    const existing = this._windows.get(storeKey);

    let currentCount = 0;
    let previousCount = 0;

    if (existing) {
      if (existing.windowStart === windowStart) {
        // Same window — use current count
        currentCount = existing.count;
        previousCount = existing.previousCount ?? 0;
      } else if (existing.windowStart === windowStart - windowMs) {
        // Previous window — use as previous count for blending
        previousCount = existing.count;
        currentCount = 0;
      }
      // Older windows are irrelevant
    }

    // Sliding window calculation:
    // How far through the current window are we? (0.0 to 1.0)
    const windowProgress = (now - windowStart) / windowMs;

    // Weight the previous window's count by how much of it is still "visible"
    // At the start of a new window (progress=0), previous window fully counts
