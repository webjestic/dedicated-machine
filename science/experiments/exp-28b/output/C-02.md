## 1. Memory Store

```javascript
// src/stores/memoryStore.js

class MemoryStore {
  constructor(cleanupInterval = 60000) {
    this.store = new Map();

    // Periodically clean up expired entries
    this.cleanupTimer = setInterval(() => {
      this.cleanup();
    }, cleanupInterval);

    // Allow process to exit even if timer is running
    if (this.cleanupTimer.unref) {
      this.cleanupTimer.unref();
    }
  }

  async get(key) {
    const entry = this.store.get(key);
    if (!entry) return null;
    if (Date.now() > entry.expiresAt) {
      this.store.delete(key);
      return null;
    }
    return entry.value;
  }

  async set(key, value, ttlMs) {
    this.store.set(key, {
      value,
      expiresAt: Date.now() + ttlMs,
    });
  }

  async increment(key, ttlMs) {
    const entry = this.store.get(key);
    const now = Date.now();

    if (!entry || now > entry.expiresAt) {
      this.store.set(key, { value: 1, expiresAt: now + ttlMs });
      return 1;
    }

    entry.value += 1;
    return entry.value;
  }

  async delete(key) {
    this.store.delete(key);
  }

  async reset() {
    this.store.clear();
  }

  cleanup() {
    const now = Date.now();
    for (const [key, entry] of this.store.entries()) {
      if (now > entry.expiresAt) {
        this.store.delete(key);
      }
    }
  }

  destroy() {
    clearInterval(this.cleanupTimer);
    this.store.clear();
  }
}

module.exports = MemoryStore;
```

```javascript
// src/stores/redisStore.js

class RedisStore {
  constructor(redisClient) {
    if (!redisClient) {
      throw new Error("Redis client is required");
    }
    this.client = redisClient;
  }

  async get(key) {
    const value = await this.client.get(key);
    return value ? JSON.parse(value) : null;
  }

  async set(key, value, ttlMs) {
    await this.client.set(key, JSON.stringify(value), {
      PX: ttlMs, // PX = milliseconds
    });
  }

  async increment(key, ttlMs) {
    // Atomic increment using Lua script
    const script = `
      local current = redis.call('GET', KEYS[1])
      if current == false then
        redis.call('SET', KEYS[1], 1, 'PX', ARGV[1])
        return 1
      else
        return redis.call('INCR', KEYS[1])
      end
    `;
    return await this.client.eval(script, {
      keys: [key],
      arguments: [String(ttlMs)],
    });
  }

  async delete(key) {
    await this.client.del(key);
  }

  async reset(pattern = "*") {
    const keys = await this.client.keys(pattern);
    if (keys.length > 0) {
      await this.client.del(keys);
    }
  }
}

module.exports = RedisStore;
```

## 3. Fixed Window Strategy

```javascript
// src/strategies/fixedWindow.js

class FixedWindowStrategy {
  /**
   * Fixed Window Rate Limiting
   * Simple counter per time window. Resets at fixed intervals.
   *
   * Pros: Simple, low memory usage
   * Cons: Burst traffic possible at window boundaries
   */
  constructor(store, options = {}) {
    this.store = store;
    this.limit = options.limit || 100;
    this.windowMs = options.windowMs || 60000; // 1 minute default
  }

  async check(key) {
    const storeKey = `fw:${key}`;
    const count = await this.store.increment(storeKey, this.windowMs);
    const remaining = Math.max(0, this.limit - count);
    const allowed = count <= this.limit;

    return {
      allowed,
      limit: this.limit,
      remaining,
      count,
      resetTime: new Date(Date.now() + this.windowMs),
      strategy: "fixed-window",
    };
  }

  async reset(key) {
    await this.store.delete(`fw:${key}`);
  }
}

module.exports = FixedWindowStrategy;
```

## 4. Sliding Window Strategy

```javascript
// src/strategies/slidingWindow.js

class SlidingWindowStrategy {
  /**
   * Sliding Window Rate Limiting
   * Tracks individual request timestamps within a rolling window.
   *
   * Pros: More accurate, no boundary burst issues
   * Cons: Higher memory usage (stores timestamps)
   */
  constructor(store, options = {}) {
    this.store = store;
    this.limit = options.limit || 100;
    this.windowMs = options.windowMs || 60000;
  }

  async check(key) {
    const storeKey = `sw:${key}`;
    const now = Date.now();
    const windowStart = now - this.windowMs;

    // Get existing timestamps
    let timestamps = (await this.store.get(storeKey)) || [];

    // Filter out expired timestamps
    timestamps = timestamps.filter((ts) => ts > windowStart);

    const count = timestamps.length;
    const allowed = count < this.limit;

    if (allowed) {
      timestamps.push(now);
      await this.store.set(storeKey, timestamps, this.windowMs);
    }

    const remaining = Math.max(0, this.limit - timestamps.length);

    // Calculate when the oldest request will expire
    const oldestTimestamp = timestamps[0];
    const resetTime = oldestTimestamp
      ? new Date(oldestTimestamp + this.windowMs)
      : new Date(now + this.windowMs);

    return {
      allowed,
      limit: this.limit,
      remaining,
      count: timestamps.length,
      resetTime,
      strategy: "sliding-window",
    };
  }

  async reset(key) {
    await this.store.delete(`sw:${key}`);
  }
}

module.exports = SlidingWindowStrategy;
```

## 5. Token Bucket Strategy

```javascript
// src/strategies/tokenBucket.js

class TokenBucketStrategy {
  /**
   * Token Bucket Rate Limiting
   * Tokens refill at a steady rate. Allows controlled bursting.
   *
   * Pros: Handles burst traffic gracefully, smooth rate limiting
   * Cons: More complex state management
   */
  constructor(store, options = {}) {
    this.store = store;
    this.capacity = options.capacity || 100; // Max tokens
    this.refillRate = options.refillRate || 10; // Tokens per second
    this.refillIntervalMs = options.refillIntervalMs || 1000;
    this.tokensPerInterval =
      options.tokensPerInterval ||
      (this.refillRate * this.refillIntervalMs) / 1000;
  }

  async check(key, tokensRequested = 1) {
    const storeKey = `tb:${key}`;
    const now = Date.now();

    let bucket = await this.store.get(storeKey);

    if (!bucket) {
      // Initialize bucket with full capacity
      bucket = {
        tokens: this.capacity,
        lastRefill: now,
      };
    }

    // Calculate tokens to add based on elapsed time
    const elapsed = now - bucket.lastRefill;
    const intervalsElapsed = elapsed / this.refillIntervalMs;
    const tokensToAdd = intervalsElapsed * this.tokensPerInterval;

    bucket.tokens = Math.min(this.capacity, bucket.tokens + tokensToAdd);
    bucket.lastRefill = now;

    const allowed = bucket.tokens >= tokensRequested;

    if (allowed) {
      bucket.tokens -= tokensRequested;
    }

    // TTL: time to refill empty bucket
    const ttlMs = (this.capacity / this.tokensPerInterval) * this.refillIntervalMs * 2;
    await this.store.set(storeKey, bucket, ttlMs);

    const timeToNextToken =
      bucket.tokens < 1
        ? Math.ceil(
            ((1 - bucket.tokens) / this.tokensPerInterval) *
              this.refillIntervalMs
          )
        : 0;

    return {
      allowed,
      limit: this.capacity,
      remaining: Math.floor(bucket.tokens),
      count: this.capacity - Math.floor(bucket.tokens),
      resetTime: new Date(now + timeToNextToken),
      retryAfterMs: allowed ? 0 : timeToNextToken,
      strategy: "token-bucket",
    };
  }

  async reset(key) {
    await this.store.delete(`tb:${key}`);
  }
}

module.exports = TokenBucketStrategy;
```

## 6. Core Rate Limiter

```javascript
// src/rateLimiter.js

const MemoryStore = require("./stores/memoryStore");
const FixedWindowStrategy = require("./strategies/fixedWindow");
const SlidingWindowStrategy = require("./strategies/slidingWindow");
const TokenBucketStrategy = require("./strategies/tokenBucket");

const STRATEGIES = {
  "fixed-window": FixedWindowStrategy,
  "sliding-window": SlidingWindowStrategy,
  "token-bucket": TokenBucketStrategy,
};

class RateLimiter {
  constructor(options = {}) {
    this.store = options.store || new MemoryStore();
    this.strategy = this._initStrategy(options);
    this.keyGenerator = options.keyGenerator || this._defaultKeyGenerator;
    this.skip = options.skip || (() => false);
    this.onLimitReached = options.onLimitReached || null;
  }

  _initStrategy(options) {
    const strategyName = options.strategy || "fixed-window";
    const StrategyClass = STRATEGIES[strategyName];

    if (!StrategyClass) {
      throw new Error(
        `Unknown strategy: ${strategyName}. Available: ${Object.keys(STRATEGIES).join(", ")}`
      );
    }

    return new StrategyClass(this.store, options);
  }

  _defaultKeyGenerator(req) {
    // Support proxies by checking X-Forwarded-For
    const forwarded = req.headers["x-forwarded-for"];
    const ip = forwarded
      ? forwarded.split(",")[0].trim()
      : req.socket?.remoteAddress || req.ip || "unknown";
    return ip;
  }

  async consume(key, tokens = 1) {
    return await this.strategy.check(key, tokens);
  }

  async reset(key) {
    return await this.strategy.reset(key);
  }

  async check(key) {
    // Peek without consuming (only works accurately for token bucket)
    return await this.strategy.check(key, 0);
  }
}

module.exports = RateLimiter;
module.exports.STRATEGIES = STRATEGIES;
```

## 7. Express Middleware

```javascript
// src/middleware.js

const RateLimiter = require("./rateLimiter");

/**
 * Creates an Express middleware for rate limiting
 */
function createRateLimitMiddleware(options = {}) {
  const limiter = options.limiter || new RateLimiter(options);

  const handler =
    options.handler ||
    function (req, res, result) {
      res.status(429).json({
        error: "Too Many Requests",
        message: "You have exceeded the rate limit. Please try again later.",
        retryAfter: result.resetTime,
        limit: result.limit,
        remaining: result.remaining,
      });
    };

  const headers = options.headers !== false; // Default: send headers

  return async function rateLimitMiddleware(req, res, next) {
    try {
      // Check if this request should be skipped
      if (await limiter.skip(req)) {
        return next();
      }

      const key = await limiter.keyGenerator(req);
      const result = await limiter.consume(key);

      // Set standard rate limit headers
      if (headers) {
        res.setHeader("X-RateLimit-Limit", result.limit);
        res.setHeader("X-RateLimit-Remaining", result.remaining);
        res.setHeader(
          "X-RateLimit-Reset",
          Math.ceil(result.resetTime.getTime() / 1000)
        );
        res.setHeader("X-RateLimit-Strategy", result.strategy);

        if (!result.allowed) {
          res.setHeader(
            "Retry-After",
            Math.ceil((result.resetTime - Date.now()) / 1000)
          );
        }
      }

      if (!result.allowed) {
        // Fire optional callback
        if (limiter.onLimitReached) {
          await limiter.onLimitReached(req, res, result);
        }
        return handler(req, res, result);
      }

      next();
    } catch (err) {
      console.error("[RateLimiter] Error:", err);
      // Fail open — don't block requests if rate limiter errors
      next();
    }
  };
}

/**
 * Creates per-route rate limiters with different configs
 */
function createTieredLimiter(tiers) {
  return tiers.map((tier) => createRateLimitMiddleware(tier));
}

module.exports = { createRateLimitMiddleware, createTieredLimiter };
```
