## 1. Memory Store

```javascript
// src/stores/memoryStore.js

class MemoryStore {
  constructor(cleanupIntervalMs = 60_000) {
    this.store = new Map();

    // Periodically clean up expired entries
    this.cleanupInterval = setInterval(
      () => this.cleanup(),
      cleanupIntervalMs
    );

    // Allow process to exit even if interval is active
    this.cleanupInterval.unref();
  }

  /**
   * Get a record by key
   * @param {string} key
   * @returns {object|null}
   */
  get(key) {
    const record = this.store.get(key);
    if (!record) return null;

    // Return null if expired
    if (record.expiresAt && Date.now() > record.expiresAt) {
      this.store.delete(key);
      return null;
    }

    return record;
  }

  /**
   * Set a record with optional TTL
   * @param {string} key
   * @param {object} value
   * @param {number} ttlMs - Time to live in milliseconds
   */
  set(key, value, ttlMs = null) {
    this.store.set(key, {
      ...value,
      expiresAt: ttlMs ? Date.now() + ttlMs : null,
    });
  }

  /**
   * Delete a record
   * @param {string} key
   */
  delete(key) {
    this.store.delete(key);
  }

  /**
   * Remove all expired entries
   */
  cleanup() {
    const now = Date.now();
    for (const [key, record] of this.store.entries()) {
      if (record.expiresAt && now > record.expiresAt) {
        this.store.delete(key);
      }
    }
  }

  /**
   * Clear all records
   */
  reset() {
    this.store.clear();
  }

  /**
   * Stop the cleanup interval
   */
  destroy() {
    clearInterval(this.cleanupInterval);
  }
}

module.exports = MemoryStore;
```

```javascript
// src/stores/redisStore.js

class RedisStore {
  /**
   * @param {object} redisClient - ioredis or node-redis client
   */
  constructor(redisClient) {
    if (!redisClient) {
      throw new Error("RedisStore requires a Redis client instance");
    }
    this.client = redisClient;
  }

  /**
   * Get a record by key
   * @param {string} key
   * @returns {Promise<object|null>}
   */
  async get(key) {
    const data = await this.client.get(key);
    if (!data) return null;

    try {
      return JSON.parse(data);
    } catch {
      return null;
    }
  }

  /**
   * Set a record with optional TTL
   * @param {string} key
   * @param {object} value
   * @param {number} ttlMs
   */
  async set(key, value, ttlMs = null) {
    const serialized = JSON.stringify(value);

    if (ttlMs) {
      await this.client.set(key, serialized, "PX", ttlMs);
    } else {
      await this.client.set(key, serialized);
    }
  }

  /**
   * Delete a record
   * @param {string} key
   */
  async delete(key) {
    await this.client.del(key);
  }

  /**
   * Atomic increment for fixed window
   * @param {string} key
   * @param {number} ttlMs
   * @returns {Promise<number>} - New count
   */
  async increment(key, ttlMs) {
    const pipeline = this.client.pipeline
      ? this.client.pipeline()
      : this.client.multi();

    pipeline.incr(key);
    pipeline.pexpire(key, ttlMs);

    const results = await pipeline.exec();
    // ioredis returns [[err, value], ...]
    return results[0][1];
  }

  /**
   * Clear all records (use with caution in production)
   */
  async reset() {
    await this.client.flushdb();
  }
}

module.exports = RedisStore;
```

## 3. Fixed Window Strategy

```javascript
// src/strategies/fixedWindow.js

class FixedWindow {
  /**
   * @param {object} options
   * @param {number} options.windowMs     - Window size in milliseconds
   * @param {number} options.maxRequests  - Max requests per window
   * @param {object} options.store        - Storage backend
   */
  constructor({ windowMs, maxRequests, store }) {
    this.windowMs = windowMs;
    this.maxRequests = maxRequests;
    this.store = store;
  }

  /**
   * Check if a request is allowed
   * @param {string} key - Unique identifier (e.g., IP address)
   * @returns {Promise<object>} - { allowed, remaining, resetAt, retryAfter }
   */
  async check(key) {
    const now = Date.now();
    const windowKey = `fw:${key}:${Math.floor(now / this.windowMs)}`;

    let record = await this.store.get(windowKey);

    if (!record) {
      record = { count: 0, windowStart: now };
    }

    const resetAt = record.windowStart + this.windowMs;
    const remaining = Math.max(0, this.maxRequests - record.count - 1);

    if (record.count >= this.maxRequests) {
      return {
        allowed: false,
        remaining: 0,
        resetAt,
        retryAfter: Math.ceil((resetAt - now) / 1000),
        strategy: "fixed-window",
      };
    }

    record.count += 1;
    await this.store.set(windowKey, record, this.windowMs);

    return {
      allowed: true,
      remaining,
      resetAt,
      retryAfter: 0,
      strategy: "fixed-window",
    };
  }

  async reset(key) {
    const now = Date.now();
    const windowKey = `fw:${key}:${Math.floor(now / this.windowMs)}`;
    await this.store.delete(windowKey);
  }
}

module.exports = FixedWindow;
```

## 4. Sliding Window Strategy

```javascript
// src/strategies/slidingWindow.js

class SlidingWindow {
  /**
   * @param {object} options
   * @param {number} options.windowMs     - Window size in milliseconds
   * @param {number} options.maxRequests  - Max requests per window
   * @param {object} options.store        - Storage backend
   */
  constructor({ windowMs, maxRequests, store }) {
    this.windowMs = windowMs;
    this.maxRequests = maxRequests;
    this.store = store;
  }

  /**
   * Check if a request is allowed using sliding window log
   * @param {string} key
   * @returns {Promise<object>}
   */
  async check(key) {
    const now = Date.now();
    const windowStart = now - this.windowMs;
    const storeKey = `sw:${key}`;

    let record = await this.store.get(storeKey);

    if (!record) {
      record = { timestamps: [] };
    }

    // Remove timestamps outside the current window
    record.timestamps = record.timestamps.filter((ts) => ts > windowStart);

    const currentCount = record.timestamps.length;
    const remaining = Math.max(0, this.maxRequests - currentCount - 1);

    if (currentCount >= this.maxRequests) {
      // Retry after oldest request falls outside the window
      const oldestTimestamp = record.timestamps[0];
      const retryAfterMs = oldestTimestamp + this.windowMs - now;

      return {
        allowed: false,
        remaining: 0,
        resetAt: oldestTimestamp + this.windowMs,
        retryAfter: Math.ceil(retryAfterMs / 1000),
        strategy: "sliding-window",
      };
    }

    record.timestamps.push(now);
    await this.store.set(storeKey, record, this.windowMs);

    return {
      allowed: true,
      remaining,
      resetAt: now + this.windowMs,
      retryAfter: 0,
      strategy: "sliding-window",
    };
  }

  async reset(key) {
    await this.store.delete(`sw:${key}`);
  }
}

module.exports = SlidingWindow;
```

## 5. Token Bucket Strategy

```javascript
// src/strategies/tokenBucket.js

class TokenBucket {
  /**
   * @param {object} options
   * @param {number} options.capacity       - Max tokens in the bucket
   * @param {number} options.refillRate     - Tokens added per second
   * @param {object} options.store          - Storage backend
   */
  constructor({ capacity, refillRate, store }) {
    this.capacity = capacity;
    this.refillRate = refillRate; // tokens per second
    this.store = store;
  }

  /**
   * Check if a request is allowed (consumes 1 token)
   * @param {string} key
   * @param {number} tokensRequested - Number of tokens to consume (default: 1)
   * @returns {Promise<object>}
   */
  async check(key, tokensRequested = 1) {
    const now = Date.now();
    const storeKey = `tb:${key}`;

    let bucket = await this.store.get(storeKey);

    if (!bucket) {
      bucket = {
        tokens: this.capacity,
        lastRefill: now,
      };
    }

    // Calculate tokens to add based on elapsed time
    const elapsed = (now - bucket.lastRefill) / 1000; // seconds
    const tokensToAdd = elapsed * this.refillRate;

    bucket.tokens = Math.min(this.capacity, bucket.tokens + tokensToAdd);
    bucket.lastRefill = now;

    const ttlMs = Math.ceil((this.capacity / this.refillRate) * 1000);

    if (bucket.tokens < tokensRequested) {
      const tokensNeeded = tokensRequested - bucket.tokens;
      const retryAfterSeconds = tokensNeeded / this.refillRate;

      await this.store.set(storeKey, bucket, ttlMs);

      return {
        allowed: false,
        remaining: Math.floor(bucket.tokens),
        resetAt: now + retryAfterSeconds * 1000,
        retryAfter: Math.ceil(retryAfterSeconds),
        strategy: "token-bucket",
      };
    }

    bucket.tokens -= tokensRequested;
    await this.store.set(storeKey, bucket, ttlMs);

    return {
      allowed: true,
      remaining: Math.floor(bucket.tokens),
      resetAt: now + (this.capacity - bucket.tokens) / this.refillRate * 1000,
      retryAfter: 0,
      strategy: "token-bucket",
    };
  }

  async reset(key) {
    await this.store.delete(`tb:${key}`);
  }
}

module.exports = TokenBucket;
```

## 6. Core Rate Limiter

```javascript
// src/rateLimiter.js

const MemoryStore = require("./stores/memoryStore");
const FixedWindow = require("./strategies/fixedWindow");
const SlidingWindow = require("./strategies/slidingWindow");
const TokenBucket = require("./strategies/tokenBucket");

const STRATEGIES = {
  "fixed-window": FixedWindow,
  "sliding-window": SlidingWindow,
  "token-bucket": TokenBucket,
};

class RateLimiter {
  /**
   * @param {object} options
   * @param {string}   options.strategy       - 'fixed-window' | 'sliding-window' | 'token-bucket'
   * @param {number}   options.windowMs       - Window duration (fixed/sliding window)
   * @param {number}   options.maxRequests    - Max requests per window (fixed/sliding window)
   * @param {number}   options.capacity       - Bucket capacity (token bucket)
   * @param {number}   options.refillRate     - Tokens per second (token bucket)
   * @param {object}   options.store          - Custom store (default: MemoryStore)
   * @param {function} options.keyGenerator   - Function to extract key from request
   * @param {function} options.onLimitReached - Callback when limit is reached
   * @param {boolean}  options.skipFailedRequests - Don't count failed requests
   * @param {boolean}  options.skipSuccessfulRequests - Don't count successful requests
   */
  constructor(options = {}) {
    this.options = {
      strategy: "sliding-window",
      windowMs: 60_000,        // 1 minute
      maxRequests: 100,
      capacity: 10,
      refillRate: 1,
      store: new MemoryStore(),
      keyGenerator: (req) => req.ip || req.connection?.remoteAddress || "unknown",
      onLimitReached: null,
      skipFailedRequests: false,
      skipSuccessfulRequests: false,
      ...options,
    };

    const StrategyClass = STRATEGIES[this.options.strategy];
    if (!StrategyClass) {
      throw new Error(
        `Unknown strategy: "${this.options.strategy}". ` +
        `Valid options: ${Object.keys(STRATEGIES).join(", ")}`
      );
    }

    this.strategy = new StrategyClass(this.options);
  }

  /**
   * Check if a key is rate limited
   * @param {string} key
   * @returns {Promise<object>}
   */
  async check(key) {
    return this.strategy.check(key);
  }

  /**
   * Manually reset the limit for a key
   * @param {string} key
   */
  async reset(key) {
    return this.strategy.reset(key);
  }

  /**
   * Consume tokens (alias for check with token bucket)
   * @param {string} key
   * @param {number} tokens
   */
  async consume(key, tokens = 1) {
    if (this.options.strategy !== "token-bucket") {
      throw new Error("consume() is only available for the token-bucket strategy");
    }
    return this.strategy.check(key, tokens);
  }
}

module.exports = RateLimiter;
```
