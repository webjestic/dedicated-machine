## 1. Memory Store

```javascript
// src/stores/memoryStore.js

class MemoryStore {
  constructor(cleanupInterval = 60000) {
    this.store = new Map();

    // Periodically clean up expired entries
    this.cleanupTimer = setInterval(() => {
      this.cleanup();
    }, cleanupInterval);

    // Allow Node.js to exit even if timer is active
    if (this.cleanupTimer.unref) {
      this.cleanupTimer.unref();
    }
  }

  async get(key) {
    const entry = this.store.get(key);
    if (!entry) return null;

    if (Date.now() > entry.expiresAt) {
      this.store.delete(key);
      return null;
    }

    return entry.value;
  }

  async set(key, value, ttlMs) {
    this.store.set(key, {
      value,
      expiresAt: Date.now() + ttlMs,
    });
  }

  async increment(key, ttlMs) {
    const entry = this.store.get(key);
    const now = Date.now();

    if (!entry || now > entry.expiresAt) {
      this.store.set(key, { value: 1, expiresAt: now + ttlMs });
      return 1;
    }

    entry.value += 1;
    return entry.value;
  }

  async delete(key) {
    this.store.delete(key);
  }

  async reset() {
    this.store.clear();
  }

  cleanup() {
    const now = Date.now();
    for (const [key, entry] of this.store.entries()) {
      if (now > entry.expiresAt) {
        this.store.delete(key);
      }
    }
  }

  destroy() {
    clearInterval(this.cleanupTimer);
    this.store.clear();
  }
}

module.exports = MemoryStore;
```

class RedisStore {
  /**
   * @param {object} client - ioredis or node-redis client
   */
  constructor(client) {
    if (!client) throw new Error("Redis client is required");
    this.client = client;
  }

  async get(key) {
    const value = await this.client.get(key);
    return value ? JSON.parse(value) : null;
  }

  async set(key, value, ttlMs) {
    await this.client.set(key, JSON.stringify(value), "PX", ttlMs);
  }

  async increment(key, ttlMs) {
    // Atomic increment using Lua script
    const script = `
      local current = redis.call('incr', KEYS[1])
      if current == 1 then
        redis.call('pexpire', KEYS[1], ARGV[1])
      end
      return current
    `;
    return await this.client.eval(script, 1, key, ttlMs);
  }

  async delete(key) {
    await this.client.del(key);
  }

  async reset(pattern = "*") {
    const keys = await this.client.keys(pattern);
    if (keys.length > 0) {
      await this.client.del(...keys);
    }
  }
}

module.exports = RedisStore;

## 3. Fixed Window Strategy

```javascript
// src/strategies/fixedWindow.js

class FixedWindow {
  /**
   * @param {object} store      - Storage backend
   * @param {number} limit      - Max requests per window
   * @param {number} windowMs   - Window size in milliseconds
   */
  constructor(store, limit, windowMs) {
    this.store = store;
    this.limit = limit;
    this.windowMs = windowMs;
  }

  async consume(key) {
    const count = await this.store.increment(key, this.windowMs);
    const remaining = Math.max(0, this.limit - count);
    const allowed = count <= this.limit;

    return {
      allowed,
      limit: this.limit,
      remaining,
      count,
      resetAfter: this.windowMs,
      retryAfter: allowed ? 0 : this.windowMs,
    };
  }

  async reset(key) {
    await this.store.delete(key);
  }
}

module.exports = FixedWindow;

## 4. Sliding Window Strategy

```javascript
// src/strategies/slidingWindow.js

class SlidingWindow {
  /**
   * Uses a sliding log of timestamps for precise rate limiting.
   * @param {object} store      - Storage backend
   * @param {number} limit      - Max requests per window
   * @param {number} windowMs   - Window size in milliseconds
   */
  constructor(store, limit, windowMs) {
    this.store = store;
    this.limit = limit;
    this.windowMs = windowMs;
  }

  async consume(key) {
    const now = Date.now();
    const windowStart = now - this.windowMs;

    // Get existing timestamps
    let timestamps = (await this.store.get(key)) || [];

    // Remove timestamps outside the window
    timestamps = timestamps.filter((ts) => ts > windowStart);

    const count = timestamps.length;
    const allowed = count < this.limit;

    if (allowed) {
      timestamps.push(now);
      await this.store.set(key, timestamps, this.windowMs);
    }

    const remaining = Math.max(0, this.limit - timestamps.length);

    // Calculate when the oldest request expires
    const retryAfter =
      !allowed && timestamps.length > 0
        ? timestamps[0] + this.windowMs - now
        : 0;

    return {
      allowed,
      limit: this.limit,
      remaining,
      count: timestamps.length,
      resetAfter: timestamps.length > 0 ? timestamps[0] + this.windowMs - now : this.windowMs,
      retryAfter,
    };
  }

  async reset(key) {
    await this.store.delete(key);
  }
}

module.exports = SlidingWindow;

## 5. Token Bucket Strategy

```javascript
// src/strategies/tokenBucket.js

class TokenBucket {
  /**
   * Allows bursting up to `capacity` tokens, refilling at `refillRate` per second.
   * @param {object} store        - Storage backend
   * @param {number} capacity     - Max tokens (burst limit)
   * @param {number} refillRate   - Tokens added per second
   */
  constructor(store, capacity, refillRate) {
    this.store = store;
    this.capacity = capacity;
    this.refillRate = refillRate; // tokens per second
  }

  async consume(key, tokensRequested = 1) {
    const now = Date.now();
    let bucket = await this.store.get(key);

    if (!bucket) {
      bucket = {
        tokens: this.capacity,
        lastRefill: now,
      };
    }

    // Refill tokens based on elapsed time
    const elapsed = (now - bucket.lastRefill) / 1000; // seconds
    const tokensToAdd = elapsed * this.refillRate;
    bucket.tokens = Math.min(this.capacity, bucket.tokens + tokensToAdd);
    bucket.lastRefill = now;

    const allowed = bucket.tokens >= tokensRequested;

    if (allowed) {
      bucket.tokens -= tokensRequested;
    }

    // TTL: time until bucket is full (no need to store longer)
    const ttlMs = ((this.capacity - bucket.tokens) / this.refillRate) * 1000 + 5000;
    await this.store.set(key, bucket, Math.max(ttlMs, 5000));

    const retryAfter = allowed
      ? 0
      : Math.ceil(((tokensRequested - bucket.tokens) / this.refillRate) * 1000);

    return {
      allowed,
      limit: this.capacity,
      remaining: Math.floor(bucket.tokens),
      count: this.capacity - Math.floor(bucket.tokens),
      resetAfter: Math.ceil((this.capacity - bucket.tokens) / this.refillRate) * 1000,
      retryAfter,
    };
  }

  async reset(key) {
    await this.store.delete(key);
  }
}

module.exports = TokenBucket;

## 6. Leaky Bucket Strategy

```javascript
// src/strategies/leakyBucket.js

class LeakyBucket {
  /**
   * Processes requests at a fixed rate, queuing up to `capacity`.
   * @param {object} store      - Storage backend
   * @param {number} capacity   - Max queue size
   * @param {number} leakRate   - Requests processed per second
   */
  constructor(store, capacity, leakRate) {
    this.store = store;
    this.capacity = capacity;
    this.leakRate = leakRate; // requests per second
  }

  async consume(key) {
    const now = Date.now();
    let bucket = await this.store.get(key);

    if (!bucket) {
      bucket = { queue: 0, lastLeak: now };
    }

    // Leak requests based on elapsed time
    const elapsed = (now - bucket.lastLeak) / 1000;
    const leaked = Math.floor(elapsed * this.leakRate);
    bucket.queue = Math.max(0, bucket.queue - leaked);
    bucket.lastLeak = now;

    const allowed = bucket.queue < this.capacity;

    if (allowed) {
      bucket.queue += 1;
    }

    const ttlMs = (this.capacity / this.leakRate) * 1000 + 5000;
    await this.store.set(key, bucket, ttlMs);

    const retryAfter = allowed
      ? 0
      : Math.ceil(((bucket.queue - this.capacity + 1) / this.leakRate) * 1000);

    return {
      allowed,
      limit: this.capacity,
      remaining: Math.max(0, this.capacity - bucket.queue),
      count: bucket.queue,
      resetAfter: Math.ceil((bucket.queue / this.leakRate) * 1000),
      retryAfter,
    };
  }

  async reset(key) {
    await this.store.delete(key);
  }
}

module.exports = LeakyBucket;

## 7. Core Rate Limiter

```javascript
// src/RateLimiter.js

const MemoryStore = require("./stores/memoryStore");
const FixedWindow = require("./strategies/fixedWindow");
const SlidingWindow = require("./strategies/slidingWindow");
const TokenBucket = require("./strategies/tokenBucket");
const LeakyBucket = require("./strategies/leakyBucket");

const STRATEGIES = {
  fixedWindow: FixedWindow,
  slidingWindow: SlidingWindow,
  tokenBucket: TokenBucket,
  leakyBucket: LeakyBucket,
};

class RateLimiter {
  /**
   * @param {object} options
   * @param {string}   options.strategy       - 'fixedWindow' | 'slidingWindow' | 'tokenBucket' | 'leakyBucket'
   * @param {number}   options.limit          - Max requests (or capacity for bucket strategies)
   * @param {number}   options.windowMs       - Window in ms (fixedWindow / slidingWindow)
   * @param {number}   options.refillRate     - Tokens/sec (tokenBucket)
   * @param {number}   options.leakRate       - Requests/sec (leakyBucket)
   * @param {object}   options.store          - Custom store (default: MemoryStore)
   * @param {function} options.keyGenerator   - Custom key function (req) => string
   * @param {function} options.onLimitReached - Callback when limit is hit
   * @param {boolean}  options.skipFailedRequests - Don't count failed responses
   * @param {boolean}  options.skipSuccessfulRequests - Don't count successful responses
   */
  constructor(options = {}) {
    const {
      strategy = "fixedWindow",
      limit = 100,
      windowMs = 60000,
      refillRate = 10,
      leakRate = 10,
      store = new MemoryStore(),
      keyGenerator = null,
      onLimitReached = null,
      skipFailedRequests = false,
      skipSuccessfulRequests = false,
    } = options;

    if (!STRATEGIES[strategy]) {
      throw new Error(
        `Unknown strategy: "${strategy}". Valid options: ${Object.keys(STRATEGIES).join(", ")}`
      );
    }

    this.options = options;
    this.store = store;
    this.keyGenerator = keyGenerator || this._defaultKeyGenerator;
    this.onLimitReached = onLimitReached;
    this.skipFailedRequests = skipFailedRequests;
    this.skipSuccessfulRequests = skipSuccessfulRequests;

    // Initialize strategy
    if (strategy === "tokenBucket") {
      this.strategy = new TokenBucket(store, limit, refillRate);
    } else if (strategy === "leakyBucket") {
      this.strategy = new LeakyBucket(store, limit, leakRate);
    } else {
      this.strategy = new STRATEGIES[strategy](store, limit, windowMs);
    }
  }

  _defaultKeyGenerator(identifier) {
    return `rl:${identifier}`;
  }

  /**
   * Check and consume a rate limit token.
   * @param {string} identifier - Unique key (IP, user ID, API key, etc.)
   * @param {number} tokens     - Tokens to consume (tokenBucket only)
   * @returns {Promise<object>} Result object
   */
  async consume(identifier, tokens = 1) {
    const key = this.keyGenerator(identifier);

    try {
      const result = await this.strategy.consume(key, tokens);

      if (!result.allowed && this.onLimitReached) {
        await this.onLimitReached(identifier, result);
      }

      return result;
    } catch (error) {
      // Fail open — allow request if store is unavailable
      console.error("[RateLimiter] Store error, failing open:", error.message);
      return {
        allowed: true,
        limit: 0,
        remaining: 0,
        count: 0,
        resetAfter: 0,
        retryAfter: 0,
        error: error.message,
      };
    }
  }

  /**
   * Reset the rate limit for a specific identifier.
   */
  async
