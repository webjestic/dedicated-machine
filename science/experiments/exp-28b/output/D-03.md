# Production Rate Limiter — Node.js

I'll design this completely before writing a single line of code.

---

## Phase 1: Requirements Design

### What algorithm?

| Algorithm | Pros | Cons |
|---|---|---|
| Fixed Window | Simple, fast | Burst at window edges (2x traffic spike) |
| Sliding Window Log | Exact, fair | High memory (stores every timestamp) |
| Sliding Window Counter | Good approximation, low memory | ~1% error at boundaries |
| Token Bucket | Smooth bursting, intuitive | State sync harder in distributed |
| Leaky Bucket | Strict output rate | Queuing complexity, latency |

**Decision: Sliding Window Counter** — best balance of accuracy, memory efficiency, and distributed-system friendliness. Acceptable for production API rate limiting.

### What storage backend?

**Redis** — atomic operations via Lua scripts, TTL-based key expiration, sub-millisecond latency, survives process restarts. Single source of truth across multiple Node.js instances.

### What are we rate limiting on?

- **Per IP** (default, unauthenticated)
- **Per API key** (authenticated clients, higher limits)
- **Per route** (some endpoints cost more)

### Failure mode?

When Redis is unavailable: **fail open** (allow requests) with logged alert. A rate limiter that takes down your API is worse than one that temporarily allows excess traffic.

/**
 * Rate limit rules.
 *
 * Rules are evaluated in order. First match wins.
 * This lets you express: "API key clients get 1000 req/min,
 * everyone else gets 60 req/min, but /auth endpoints get 10 req/min globally."
 *
 * windowMs   — sliding window duration in milliseconds
 * max        — maximum requests allowed in that window
 * keyPrefix  — Redis key namespace (prevents collisions between rules)
 * cost       — how many tokens this request consumes (default 1)
 *              useful for expensive endpoints (/search costs 5, /ping costs 1)
 */
const rules = [
  {
    name: 'auth-endpoints',
    match: (req) => req.path.startsWith('/auth'),
    windowMs: 15 * 60 * 1000, // 15 minutes
    max: 10,
    keyPrefix: 'rl:auth',
    keyResolver: (req) => req.ip,
  },
  {
    name: 'authenticated-api-key',
    match: (req) => Boolean(req.headers['x-api-key']),
    windowMs: 60 * 1000, // 1 minute
    max: 1000,
    keyPrefix: 'rl:apikey',
    keyResolver: (req) => `apikey:${req.headers['x-api-key']}`,
  },
  {
    name: 'default-ip',
    match: () => true, // catch-all
    windowMs: 60 * 1000, // 1 minute
    max: 60,
    keyPrefix: 'rl:ip',
    keyResolver: (req) => `ip:${req.ip}`,
  },
];

module.exports = rules;

### `src/store/redisStore.js`

```javascript
const { createClient } = require('redis');
const logger = require('../utils/logger');

class RedisStore {
  constructor(options = {}) {
    this.client = createClient({
      socket: {
        host: options.host || process.env.REDIS_HOST || 'localhost',
        port: options.port || process.env.REDIS_PORT || 6379,
        /**
         * reconnectStrategy: exponential backoff capped at 30s.
         * Without this, a Redis blip causes your process to hammer
         * reconnect attempts and spike CPU.
         */
        reconnectStrategy: (retries) => {
          if (retries > 20) {
            logger.error('Redis: max reconnection attempts reached');
            return new Error('Max reconnection attempts reached');
          }
          return Math.min(retries * 100, 30000);
        },
      },
      password: options.password || process.env.REDIS_PASSWORD || undefined,
    });

    this.isConnected = false;

    this.client.on('connect', () => {
      this.isConnected = true;
      logger.info('Redis: connected');
    });

    this.client.on('error', (err) => {
      this.isConnected = false;
      // Log but don't throw — we handle this in the middleware (fail open)
      logger.error('Redis: connection error', { error: err.message });
    });

    this.client.on('reconnecting', () => {
      logger.warn('Redis: reconnecting...');
    });

    /**
     * Lua script for atomic sliding window counter.
     *
     * WHY LUA: Redis executes Lua scripts atomically. Without this,
     * a race condition exists between reading the count and writing
     * the increment — two concurrent requests could both read 99,
     * both decide they're under the 100 limit, and both write 100.
     * That's a classic TOCTOU bug. Lua eliminates it.
     *
     * HOW IT WORKS (Sliding Window Counter approximation):
     * We maintain two buckets: current window and previous window.
     * The weighted count = prev_count * (overlap%) + curr_count
     * This approximates a true sliding window with O(1) memory per key
     * instead of O(n) for a sliding window log.
     *
     * KEYS[1] = current window key  (e.g., "rl:ip:1.2.3.4:1700000060")
     * KEYS[2] = previous window key (e.g., "rl:ip:1.2.3.4:1700000000")
     * ARGV[1] = window size in seconds
     * ARGV[2] = max requests allowed
     * ARGV[3] = current timestamp in milliseconds
     * ARGV[4] = request cost (default 1)
     */
    this.slidingWindowScript = `
      local curr_key    = KEYS[1]
      local prev_key    = KEYS[2]
      local window_ms   = tonumber(ARGV[1])
      local max_req     = tonumber(ARGV[2])
      local now_ms      = tonumber(ARGV[3])
      local cost        = tonumber(ARGV[4])

      -- How far into the current window are we? (0.0 to 1.0)
      local window_start_ms = now_ms - (now_ms % window_ms)
      local elapsed_in_window = now_ms - window_start_ms
      local prev_weight = 1 - (elapsed_in_window / window_ms)

      -- Read previous window count (may not exist = 0)
      local prev_count = tonumber(redis.call('GET', prev_key)) or 0

      -- Weighted estimate of requests in the sliding window
      local weighted_prev = prev_count * prev_weight

      -- Read current window count
      local curr_count = tonumber(redis.call('GET', curr_key)) or 0

      -- Total estimated requests in the sliding window
      local total = weighted_prev + curr_count

      -- Remaining capacity
      local remaining = max_req - total

      if remaining < cost then
        -- REJECT: return current state without incrementing
        return {
          0,                              -- allowed (0 = false)
          math.floor(remaining),          -- remaining
          math.floor(total),              -- current count
          window_start_ms + window_ms     -- reset timestamp (ms)
        }
      end

      -- ALLOW: increment current window counter
      local new_count = redis.call('INCRBY', curr_key, cost)

      -- Set TTL on first write (2 windows to keep prev window alive)
      if new_count == cost then
        local ttl_seconds = math.ceil((window_ms * 2) / 1000)
        redis.call('EXPIRE', curr_key, ttl_seconds)
      end

      return {
        1,                                        -- allowed (1 = true)
        math.floor(remaining - cost),             -- remaining
        math.floor(total + cost),                 -- current count
        window_start_ms + window_ms               -- reset timestamp (ms)
      }
    `;
  }

  async connect() {
    await this.client.connect();
  }

  async disconnect() {
    await this.client.disconnect();
  }

  /**
   * Check and increment rate limit counter.
   *
   * @param {string} currKey  - Redis key for current window
   * @param {string} prevKey  - Redis key for previous window
   * @param {number} windowMs - Window size in milliseconds
   * @param {number} max      - Maximum requests per window
   * @param {number} cost     - Request cost (default 1)
   * @returns {Object} { allowed, remaining, total, resetAt }
   */
  async checkLimit(currKey, prevKey, windowMs, max, cost = 1) {
    const result = await this.client.eval(this.slidingWindowScript, {
      keys: [currKey, prevKey],
      arguments: [
        String(windowMs),
        String(max),
        String(Date.now()),
        String(cost),
      ],
    });

    return {
      allowed: result[0] === 1,
      remaining: Math.max(0, result[1]),
      total: result[2],
      resetAt: result[3], // Unix ms timestamp
    };
  }
}

// Singleton — one Redis connection shared across the app
let instance = null;

function getStore(options) {
  if (!instance) {
    instance = new RedisStore(options);
  }
  return instance;
}

module.exports = { getStore };
```

### `src/algorithms/slidingWindowCounter.js`

```javascript
/**
 * Sliding Window Counter — key generation logic.
 *
 * We need two keys per request:
 *   - Current window key  (the bucket we're writing to now)
 *   - Previous window key (the bucket we weight against)
 *
 * Key format: {prefix}:{identifier}:{windowIndex}
 * windowIndex = Math.floor(now / windowMs)
 *
 * Example with 60s window at timestamp 1700000075000ms:
 *   windowIndex     = floor(1700000075000 / 60000) = 28333334
 *   currKey         = "rl:ip:1.2.3.4:28333334"
 *   prevKey         = "rl:ip:1.2.3.4:28333333"
 */
function buildKeys(prefix, identifier, windowMs) {
  const now = Date.now();
  const windowIndex = Math.floor(now / windowMs);

  const currKey = `${prefix}:${identifier}:${windowIndex}`;
  const prevKey = `${prefix}:${identifier}:${windowIndex - 1}`;

  return { currKey, prevKey };
}

module.exports = { buildKeys };
```

### `src/middleware/rateLimiter.js`

```javascript
const rules = require('../config/rules');
const { getStore } = require('../store/redisStore');
const { buildKeys } = require('../algorithms/slidingWindowCounter');
const logger = require('../utils/logger');

/**
 * Rate limiter middleware factory.
 *
 * Usage:
 *   app.use(createRateLimiter())
 *   app.use('/api', createRateLimiter({ failOpen: true }))
 *
 * @param {Object} options
 * @param {boolean} options.failOpen   - Allow requests when Redis is down (default: true)
 * @param {Function} options.onLimited - Custom handler for rate-limited requests
 */
function createRateLimiter(options = {}) {
  const {
    failOpen = true,
    onLimited = defaultOnLimited,
  } = options;

  const store = getStore();

  return async function rateLimiterMiddleware(req, res, next) {
    // Find the first matching rule
    const rule = rules.find((r) => r.match(req));

    if (!rule) {
      // No rule matched — this shouldn't happen if you have a catch-all,
      // but if it does, fail open and log a warning so you can fix your rules.
      logger.warn('RateLimiter: no matching rule found', {
        path: req.path,
        method: req.method,
      });
      return next();
    }

    const identifier = rule.keyResolver(req);
    const { currKey, prevKey } = buildKeys(rule.keyPrefix, identifier, rule.windowMs);
    const cost = rule.cost || 1;

    let result;

    try {
      if (!store.isConnected) {
        throw new Error('Redis not connected');
      }

      result = await store.checkLimit(
        currKey,
        prevKey,
        rule.windowMs,
        rule.max,
        cost,
      );
    } catch (err) {
      /**
       * FAILURE MODE: Redis is unavailable.
       *
       * failOpen = true  → allow the request, log the error
       * failOpen = false → reject with 503 (use this for high-security endpoints)
       *
       * This is a deliberate product decision, not a code decision.
       * Document which mode you're using and why.
       */
      logger.error('RateLimiter: store error', {
        error: err.message,
        rule: rule.name,
        identifier,
        failOpen,
      });

      if (failOpen) {
        // Set a header so you can detect fail-open in your logs/dashboards
        res.setHeader('X-RateLimit-Bypass', 'store-unavailable');
        return next();
      }

      return res.status(503).json({
        error: 'Service temporarily unavailable',
        retryAfter: 5,
      });
    }

    // Always set rate limit headers — clients need these to implement backoff
    setRateLimitHeaders(res, rule, result);

    if (!result.allowed) {
      logger.warn('RateLimiter: request rejected', {
