# Production Rate Limiter in Node.js

Let me build this properly — working code first, then an honest accounting of what's required to run it safely in production.

## Architecture Decisions (Made Explicitly)

| Decision | Choice | Reason |
|----------|--------|--------|
| Algorithm | Sliding window log | Accurate, no boundary burst problem of fixed window |
| Storage | Redis | Shared state across instances; in-memory won't work at scale |
| Fallback | Fail-open with logging | Prefer availability over false-blocking on Redis failure |
| Key strategy | Configurable (IP, user, API key) | Different resources need different identifiers |
| Headers | X-RateLimit-* standard | Client transparency, debuggability |

```javascript
'use strict';

const { createClient } = require('redis');
const { RedisConnectionError } = require('./errors');

// Module-level singleton — one connection pool per process.
// Do NOT call createRedisClient() on every request.
let client = null;
let isHealthy = false;

/**
 * Returns the shared Redis client, creating it on first call.
 * Subsequent calls return the same instance.
 */
async function getRedisClient(config = {}) {
  if (client) return client;

  const {
    url = process.env.REDIS_URL || 'redis://localhost:6379',
    socket = {
      connectTimeout: 5000,
      // Reconnect with exponential backoff, capped at 30s.
      // This prevents thundering herd when Redis restarts.
      reconnectStrategy: (retries) => {
        if (retries > 20) {
          // Stop retrying after 20 attempts — something is seriously wrong.
          return new Error('Redis reconnect limit reached');
        }
        return Math.min(retries * 100, 30000);
      },
    },
    // Separate read/write timeouts from connection timeout
    commandTimeout: 2000,
  } = config;

  client = createClient({ url, socket, commandTimeout });

  client.on('connect', () => {
    isHealthy = true;
    console.info('[redis] Connected');
  });

  client.on('ready', () => {
    isHealthy = true;
    console.info('[redis] Ready');
  });

  client.on('error', (err) => {
    // Mark unhealthy but do NOT throw here — this is an event emitter callback.
    // The rate limiter will detect isHealthy=false and fail-open.
    isHealthy = false;
    console.error('[redis] Error:', err.message);
  });

  client.on('reconnecting', () => {
    isHealthy = false;
    console.warn('[redis] Reconnecting...');
  });

  client.on('end', () => {
    isHealthy = false;
    console.warn('[redis] Connection closed');
  });

  try {
    await client.connect();
  } catch (err) {
    throw new RedisConnectionError('Failed to connect to Redis on startup', err);
  }

  return client;
}

function getHealthStatus() {
  return { connected: isHealthy, client: client !== null };
}

async function closeRedisClient() {
  if (client) {
    await client.quit();
    client = null;
    isHealthy = false;
  }
}

module.exports = { getRedisClient, getHealthStatus, closeRedisClient };
```

### `src/keyStrategies.js`

```javascript
'use strict';

const { ConfigurationError } = require('./errors');

/**
 * Key strategies determine HOW a caller is identified.
 * Wrong key strategy = wrong rate limiting behavior.
 *
 * WARNING on IP-based keying:
 * - Corporate NAT: thousands of users share one IP → false throttling
 * - IPv6: clients may rotate addresses → ineffective limiting
 * - Proxies: X-Forwarded-For can be spoofed if not validated upstream
 *
 * Prefer user/API key strategies when authentication is available.
 */

function ipKey(req) {
  // Trust proxy headers only if your infrastructure sets them reliably.
  // If you're behind a load balancer, req.ip is already resolved by Express
  // when `app.set('trust proxy', true)` is configured.
  const ip = req.ip || req.connection?.remoteAddress;
  if (!ip) throw new ConfigurationError('Cannot determine client IP');
  return `rl:ip:${ip}`;
}

function userKey(req) {
  // Assumes authentication middleware has run and set req.user
  const userId = req.user?.id;
  if (!userId) throw new ConfigurationError('req.user.id not set — auth middleware must run before rate limiter');
  return `rl:user:${userId}`;
}

function apiKeyStrategy(req) {
  const key = req.headers['x-api-key'] || req.query.api_key;
  if (!key) throw new ConfigurationError('No API key found in request');
  // Hash the key so raw API keys aren't stored in Redis
  const crypto = require('crypto');
  const hashed = crypto.createHash('sha256').update(key).digest('hex').slice(0, 16);
  return `rl:apikey:${hashed}`;
}

function compositeKey(...strategies) {
  return (req) => strategies.map((s) => s(req)).join(':');
}

module.exports = { ipKey, userKey, apiKeyStrategy, compositeKey };
```

### `src/rateLimiter.js`

```javascript
'use strict';

const { RateLimitError } = require('./errors');

/**
 * Sliding Window Log Rate Limiter
 *
 * Algorithm:
 * 1. Store each request timestamp in a Redis sorted set (score = timestamp)
 * 2. On each request:
 *    a. Remove all entries older than windowMs
 *    b. Count remaining entries
 *    c. If count >= limit → reject
 *    d. Otherwise → add current timestamp, set TTL, allow
 *
 * All steps run in a single Lua script for atomicity.
 * Without atomicity, two concurrent requests at the limit could both pass.
 *
 * Memory: O(requests_in_window) per key. For limit=100/min, this is trivial.
 * For limit=10000/min, monitor Redis memory carefully.
 */

// Lua script runs atomically on the Redis server.
// Arguments: key, windowMs, limit, now (milliseconds), uniqueRequestId
const SLIDING_WINDOW_SCRIPT = `
local key = KEYS[1]
local window_ms = tonumber(ARGV[1])
local limit = tonumber(ARGV[2])
local now = tonumber(ARGV[3])
local request_id = ARGV[4]
local window_start = now - window_ms

-- Remove timestamps outside the current window
redis.call('ZREMRANGEBYSCORE', key, '-inf', window_start)

-- Count requests in the current window
local count = redis.call('ZCARD', key)

if count >= limit then
  -- Return current count and oldest timestamp for Retry-After calculation
  local oldest = redis.call('ZRANGE', key, 0, 0, 'WITHSCORES')
  local oldest_ts = oldest[2] and tonumber(oldest[2]) or now
  return {0, count, oldest_ts}
end

-- Add this request with a unique member to handle same-millisecond requests
redis.call('ZADD', key, now, request_id)

-- Set TTL so keys self-clean. Add buffer to avoid race at window boundary.
local ttl_ms = window_ms + 1000
redis.call('PEXPIRE', key, ttl_ms)

return {1, count + 1, 0}
`;

class RateLimiter {
  /**
   * @param {object} options
   * @param {object} options.redisClient - Connected Redis client
   * @param {number} options.limit - Max requests per window
   * @param {number} options.windowMs - Window size in milliseconds
   * @param {Function} options.keyFn - Function(req) => string key
   * @param {boolean} [options.failOpen=true] - Allow requests when Redis is down
   * @param {Function} [options.onLimitReached] - Callback for observability
   * @param {Function} [options.onRedisError] - Callback for Redis errors
   */
  constructor(options) {
    this._validate(options);

    this.redisClient = options.redisClient;
    this.limit = options.limit;
    this.windowMs = options.windowMs;
    this.keyFn = options.keyFn;
    this.failOpen = options.failOpen !== false; // Default: fail-open
    this.onLimitReached = options.onLimitReached || (() => {});
    this.onRedisError = options.onRedisError || ((err) => console.error('[rate-limiter] Redis error:', err));

    // Load the Lua script SHA for EVALSHA (more efficient than EVAL on every call)
    this._scriptSha = null;
    this._loadScript();
  }

  _validate(options) {
    const { ConfigurationError } = require('./errors');
    if (!options.redisClient) throw new ConfigurationError('redisClient is required');
    if (!Number.isInteger(options.limit) || options.limit < 1) {
      throw new ConfigurationError('limit must be a positive integer');
    }
    if (!Number.isInteger(options.windowMs) || options.windowMs < 100) {
      throw new ConfigurationError('windowMs must be an integer >= 100ms');
    }
    if (typeof options.keyFn !== 'function') {
      throw new ConfigurationError('keyFn must be a function');
    }
  }

  async _loadScript() {
    try {
      this._scriptSha = await this.redisClient.scriptLoad(SLIDING_WINDOW_SCRIPT);
    } catch (err) {
      // Non-fatal: will fall back to EVAL
      console.warn('[rate-limiter] Could not preload Lua script, will use EVAL:', err.message);
    }
  }

  /**
   * Check and record a request.
   *
   * @param {object} req - Express request object
   * @returns {Promise<RateLimitResult>}
   */
  async check(req) {
    let key;
    try {
      key = this.keyFn(req);
    } catch (err) {
      // Key extraction failed (e.g., missing auth). Treat as a hard error,
      // not a Redis error — don't fail-open on misconfiguration.
      throw err;
    }

    const now = Date.now();
    // Unique ID per request prevents ZADD collisions at the same millisecond
    const requestId = `${now}-${Math.random().toString(36).slice(2, 9)}`;

    let result;
    try {
      result = await this._runScript(key, now, requestId);
    } catch (err) {
      this.onRedisError(err);

      if (this.failOpen) {
        // Redis is down. Allow the request but signal degraded mode.
        // Log this — sustained fail-open means your rate limiter is off.
        console.warn(`[rate-limiter] Failing open for key=${key} due to Redis error`);
        return this._buildResult({ allowed: true, count: 0, limit: this.limit, degraded: true, key });
      }

      // Fail-closed: reject when Redis is unavailable
      throw new RateLimitError('Rate limiter unavailable', { key });
    }

    const [allowed, count, oldestTs] = result;
    const isAllowed = allowed === 1;

    if (!isAllowed) {
      const retryAfterMs = oldestTs ? oldestTs + this.windowMs - now : this.windowMs;
      this.onLimitReached({ key, count, limit: this.limit, windowMs: this.windowMs });
      return this._buildResult({
        allowed: false,
        count,
        limit: this.limit,
        retryAfterMs: Math.max(0, retryAfterMs),
        key,
      });
    }

    return this._buildResult({ allowed: true, count, limit: this.limit, key });
  }

  async _runScript(key, now, requestId) {
    const args = [String(this.windowMs), String(this.limit), String(now), requestId];

    // Try EVALSHA first (uses cached script), fall back to EVAL
    if (this._scriptSha) {
      try {
        return await this.redisClient.evalSha(this._scriptSha, {
          keys: [key],
          arguments: args,
        });
      } catch (err) {
        // NOSCRIPT means Redis was restarted and lost the cached script
        if (err.message?.includes('NOSCRIPT')) {
          await this._loadScript();
          // Fall through to EVAL
        } else {
          throw err;
        }
      }
    }

    return await this.redisClient.eval(SLIDING_WINDOW_SCRIPT, {
      keys: [key],
      arguments: args,
    });
  }

  _buildResult({ allowed, count, limit, retryAfterMs = 0, degraded = false, key }) {
    const remaining = Math.max(0, limit - count);
    const resetMs = Date.now() + (retryAfterMs || this.windowMs);

    return {
      allowed,
      count,
      remaining,
      limit,
      retryAfterMs,
      resetAt: new Date(resetMs).toISOString(),
      degraded,
      key, // Useful for debugging; strip from responses if key contains PII
    };
  }
}

module.exports = { RateLimiter };
```
