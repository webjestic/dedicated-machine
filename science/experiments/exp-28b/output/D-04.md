# Production Rate Limiter in Node.js

Let me work through this the right way — requirements first, then implementation, then operational gaps.

---

## Phase 1: Requirements Design

Before writing a line of code, I need to establish what this rate limiter must do, what guarantees it makes, and where it's allowed to fail.

### Functional Requirements

| Requirement | Decision |
|---|---|
| Algorithm | **Sliding window log** (most accurate) with **token bucket** for burst support |
| Granularity | Per-IP, per-user, per-API-key, or composite keys |
| Multiple limit tiers | Yes — different limits per route/client tier |
| Distributed support | Yes — Redis as shared state store |
| Fail behavior | **Fail open** (configurable) — don't block traffic if Redis is down |
| Headers returned | `X-RateLimit-Limit`, `X-RateLimit-Remaining`, `X-RateLimit-Reset`, `Retry-After` |
| Penalty system | Yes — repeated violations extend lockout |

### Non-Functional Requirements

- Atomic operations only — no race conditions under concurrent load
- Sub-millisecond overhead on the hot path
- No single point of failure
- Observable — every limit decision must be loggable and traceable
- Configurable without redeployment

### Algorithm Choice: Sliding Window Counter

I'm choosing **sliding window counter** (not fixed window, not pure log):

- Fixed window: allows 2x burst at window boundaries — unacceptable
- Sliding window log: exact but memory-expensive at scale
- **Sliding window counter**: approximation that's accurate within ~0.1%, O(1) memory per key — this is what production systems use

The math: `current_window_count + previous_window_count * overlap_percentage`

### `scripts/rateLimiter.lua`

This is the most critical file. All state mutations happen atomically in a single Lua script — this eliminates race conditions entirely.

```lua
--[[
  Sliding Window Counter Rate Limiter
  
  KEYS[1] = current window key
  KEYS[2] = previous window key
  
  ARGV[1] = current timestamp (ms)
  ARGV[2] = window size (ms)
  ARGV[3] = limit (max requests per window)
  ARGV[4] = request cost (usually 1, higher for expensive endpoints)
  
  Returns: { allowed, current_count, limit, reset_time_ms, retry_after_ms }
--]]

local current_key    = KEYS[1]
local previous_key   = KEYS[2]
local now            = tonumber(ARGV[1])
local window_ms      = tonumber(ARGV[2])
local limit          = tonumber(ARGV[3])
local cost           = tonumber(ARGV[4])

-- Calculate how far we are into the current window (0.0 to 1.0)
local window_start   = math.floor(now / window_ms) * window_ms
local elapsed        = now - window_start
local overlap        = 1 - (elapsed / window_ms)

-- Fetch counts from both windows
local current_count  = tonumber(redis.call('GET', current_key)) or 0
local previous_count = tonumber(redis.call('GET', previous_key)) or 0

-- Weighted count: previous window contributes proportionally
local weighted_count = math.floor(previous_count * overlap) + current_count

-- Decision
local allowed = 0
local new_count = current_count

if weighted_count + cost <= limit then
  -- Increment current window atomically
  new_count = redis.call('INCRBY', current_key, cost)
  
  -- Set TTL on first write to this window key (2 windows to keep previous)
  if new_count == cost then
    redis.call('PEXPIRE', current_key, window_ms * 2)
  end
  
  allowed = 1
end

local reset_time = window_start + window_ms
local retry_after = 0

if allowed == 0 then
  retry_after = window_start + window_ms - now
end

local remaining = math.max(0, limit - (math.floor(previous_count * overlap) + new_count))

return {
  allowed,
  math.floor(previous_count * overlap) + new_count,
  limit,
  reset_time,
  retry_after,
  remaining
}
```

### `src/store/redisStore.js`

```javascript
'use strict';

const Redis = require('ioredis');
const fs = require('fs');
const path = require('path');
const logger = require('../utils/logger');

const LUA_SCRIPT = fs.readFileSync(
  path.join(__dirname, '../../scripts/rateLimiter.lua'),
  'utf8'
);

class RedisStore {
  constructor(options = {}) {
    this.windowMs = options.windowMs || 60_000;
    this.failOpen = options.failOpen !== false; // default: fail open
    this.connected = false;

    this.client = new Redis({
      host: options.host || process.env.REDIS_HOST || '127.0.0.1',
      port: options.port || process.env.REDIS_PORT || 6379,
      password: options.password || process.env.REDIS_PASSWORD,
      
      // Connection resilience
      retryStrategy: (times) => {
        const delay = Math.min(times * 50, 2000);
        logger.warn({ event: 'redis_retry', attempt: times, delay_ms: delay });
        return delay;
      },
      maxRetriesPerRequest: 1,       // fail fast on individual commands
      enableReadyCheck: true,
      lazyConnect: false,
      
      // Timeouts — critical for not blocking request handling
      connectTimeout: 5000,
      commandTimeout: 100,           // 100ms max — rate check must be fast
    });

    // Load script SHA for EVALSHA (more efficient than EVAL every time)
    this.scriptSha = null;

    this.client.on('connect', () => {
      this.connected = true;
      logger.info({ event: 'redis_connected' });
      this._loadScript();
    });

    this.client.on('error', (err) => {
      this.connected = false;
      logger.error({ event: 'redis_error', error: err.message });
    });

    this.client.on('close', () => {
      this.connected = false;
      logger.warn({ event: 'redis_disconnected' });
    });
  }

  async _loadScript() {
    try {
      this.scriptSha = await this.client.script('LOAD', LUA_SCRIPT);
      logger.info({ event: 'lua_script_loaded', sha: this.scriptSha });
    } catch (err) {
      logger.error({ event: 'lua_script_load_failed', error: err.message });
    }
  }

  /**
   * Build time-bucketed Redis keys for current and previous windows.
   * Using two keys lets us implement sliding window without storing per-request logs.
   */
  _buildWindowKeys(identifier) {
    const now = Date.now();
    const windowMs = this.windowMs;
    const currentWindow = Math.floor(now / windowMs);
    const previousWindow = currentWindow - 1;

    return {
      currentKey: `rl:${identifier}:${currentWindow}`,
      previousKey: `rl:${identifier}:${previousWindow}`,
      now,
    };
  }

  /**
   * @returns {Object} { allowed, count, limit, resetTime, retryAfter, remaining }
   */
  async increment(identifier, limit, cost = 1) {
    const { currentKey, previousKey, now } = this._buildWindowKeys(identifier);

    try {
      let result;

      // Prefer EVALSHA (cached script) over EVAL (sends full script each time)
      if (this.scriptSha) {
        try {
          result = await this.client.evalsha(
            this.scriptSha,
            2,                    // number of keys
            currentKey,
            previousKey,
            now,
            this.windowMs,
            limit,
            cost
          );
        } catch (err) {
          // Script was flushed (e.g., Redis restart) — reload and fall back to EVAL
          if (err.message.includes('NOSCRIPT')) {
            await this._loadScript();
            result = await this._evalFallback(currentKey, previousKey, now, limit, cost);
          } else {
            throw err;
          }
        }
      } else {
        result = await this._evalFallback(currentKey, previousKey, now, limit, cost);
      }

      return this._parseResult(result);

    } catch (err) {
      logger.error({
        event: 'redis_increment_failed',
        identifier,
        error: err.message,
      });

      // Fail open: allow request through but flag it
      if (this.failOpen) {
        return this._failOpenResult(limit);
      }

      throw err;
    }
  }

  async _evalFallback(currentKey, previousKey, now, limit, cost) {
    return this.client.eval(
      LUA_SCRIPT,
      2,
      currentKey,
      previousKey,
      now,
      this.windowMs,
      limit,
      cost
    );
  }

  _parseResult([allowed, count, limit, resetTime, retryAfter, remaining]) {
    return {
      allowed: allowed === 1,
      count: Number(count),
      limit: Number(limit),
      resetTime: Number(resetTime),
      retryAfter: Number(retryAfter),
      remaining: Number(remaining),
      storeError: false,
    };
  }

  _failOpenResult(limit) {
    return {
      allowed: true,
      count: 0,
      limit,
      resetTime: Date.now() + this.windowMs,
      retryAfter: 0,
      remaining: limit,
      storeError: true,   // caller can log/alert on this
    };
  }

  async reset(identifier) {
    const { currentKey, previousKey } = this._buildWindowKeys(identifier);
    await this.client.del(currentKey, previousKey);
  }

  async close() {
    await this.client.quit();
  }
}

module.exports = RedisStore;
```

### `src/store/memoryStore.js`

Local fallback for single-instance deployments or testing. Not suitable for multi-instance production.

```javascript
'use strict';

/**
 * In-memory store implementing the same sliding window counter algorithm.
 * 
 * PRODUCTION WARNING: This store is not shared across processes.
 * Use only for: single-instance deployments, local dev, or testing.
 */
class MemoryStore {
  constructor(options = {}) {
    this.windowMs = options.windowMs || 60_000;
    this.windows = new Map();

    // Periodic cleanup to prevent unbounded memory growth
    this._cleanupInterval = setInterval(
      () => this._cleanup(),
      this.windowMs * 2
    );

    // Don't keep process alive just for cleanup
    this._cleanupInterval.unref();
  }

  _getWindowKey(identifier, windowIndex) {
    return `${identifier}:${windowIndex}`;
  }

  async increment(identifier, limit, cost = 1) {
    const now = Date.now();
    const windowMs = this.windowMs;
    const currentWindowIndex = Math.floor(now / windowMs);
    const previousWindowIndex = currentWindowIndex - 1;

    const currentKey = this._getWindowKey(identifier, currentWindowIndex);
    const previousKey = this._getWindowKey(identifier, previousWindowIndex);

    const currentCount = this.windows.get(currentKey) || 0;
    const previousCount = this.windows.get(previousKey) || 0;

    const windowStart = currentWindowIndex * windowMs;
    const elapsed = now - windowStart;
    const overlap = 1 - elapsed / windowMs;

    const weightedCount = Math.floor(previousCount * overlap) + currentCount;
    const allowed = weightedCount + cost <= limit;

    let newCount = currentCount;
    if (allowed) {
      newCount = currentCount + cost;
      this.windows.set(currentKey, newCount);
    }

    const resetTime = windowStart + windowMs;
    const retryAfter = allowed ? 0 : resetTime - now;
    const remaining = Math.max(
      0,
      limit - (Math.floor(previousCount * overlap) + newCount)
    );

    return {
      allowed,
      count: Math.floor(previousCount * overlap) + newCount,
      limit,
      resetTime,
      retryAfter,
      remaining,
      storeError: false,
    };
  }

  async reset(identifier) {
    const now = Date.now();
    const windowMs = this.windowMs;
    const currentWindowIndex = Math.floor(now / windowMs);

    this.windows.delete(this._getWindowKey(identifier, currentWindowIndex));
    this.windows.delete(this._getWindowKey(identifier, currentWindowIndex - 1));
  }

  _cleanup() {
    const now = Date.now();
    const currentWindowIndex = Math.floor(now / this.windowMs);

    for (const key of this.windows.keys()) {
      const parts = key.split(':');
      const windowIndex = parseInt(parts[parts.length - 1], 10);
      if (windowIndex < currentWindowIndex - 1) {
        this.windows.delete(key);
      }
    }
  }

  async close() {
    clearInterval(this._cleanupInterval);
    this.windows.clear();
  }
}

module.exports = MemoryStore;
```
