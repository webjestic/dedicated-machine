# Production Rate Limiter in Node.js

Let me build this properly — working code first, then an honest accounting of what "production-ready" actually requires.

## Architecture Decision

I'll implement **sliding window log** with a Redis backend. Fixed window has boundary burst problems. Token bucket is great but harder to reason about for most teams. Sliding window log is accurate and auditable.

### `scripts/redis-rate-limiter.lua`

This runs atomically in Redis — no race conditions, no multi/exec complexity.

```lua
-- KEYS[1]: the rate limit key (e.g., "rl:user:123")
-- ARGV[1]: current timestamp in milliseconds
-- ARGV[2]: window size in milliseconds
-- ARGV[3]: max requests allowed in window
-- ARGV[4]: TTL for the key in milliseconds (slightly > window to handle clock skew)
--
-- Returns: { allowed (0|1), current_count, oldest_timestamp_in_window }

local key        = KEYS[1]
local now        = tonumber(ARGV[1])
local window_ms  = tonumber(ARGV[2])
local limit      = tonumber(ARGV[3])
local ttl_ms     = tonumber(ARGV[4])

-- Remove all entries outside the current window
local window_start = now - window_ms
redis.call('ZREMRANGEBYSCORE', key, '-inf', window_start)

-- Count remaining entries (requests in current window)
local count = redis.call('ZCARD', key)

if count < limit then
  -- Add current request with score = timestamp
  -- Use timestamp + random suffix as member to handle same-millisecond requests
  local member = now .. ':' .. redis.call('INCR', key .. ':seq')
  redis.call('ZADD', key, now, member)
  -- Reset TTL on every write
  redis.call('PEXPIRE', key, ttl_ms)
  return { 1, count + 1, window_start }
else
  -- Do NOT add the request — just return denied
  redis.call('PEXPIRE', key, ttl_ms)
  return { 0, count, window_start }
end
```

### `src/RedisStore.js`

```javascript
'use strict';

const fs   = require('fs');
const path = require('path');
const { StoreUnavailableError } = require('./errors');

const LUA_SCRIPT = fs.readFileSync(
  path.join(__dirname, '../scripts/redis-rate-limiter.lua'),
  'utf8'
);

class RedisStore {
  /**
   * @param {object} options
   * @param {object} options.client         - ioredis or node-redis v4 client
   * @param {string} [options.keyPrefix]    - Namespace prefix for all keys
   * @param {number} [options.clockSkewMs]  - Extra TTL buffer for clock skew (default: 1000ms)
   */
  constructor({ client, keyPrefix = 'rl:', clockSkewMs = 1000 } = {}) {
    if (!client) {
      throw new Error('RedisStore requires a Redis client');
    }

    this.client      = client;
    this.keyPrefix   = keyPrefix;
    this.clockSkewMs = clockSkewMs;
    this._scriptSha  = null;

    // Pre-load the script on construction so first request isn't slow
    this._loadScript();
  }

  async _loadScript() {
    try {
      this._scriptSha = await this.client.scriptLoad(LUA_SCRIPT);
    } catch (err) {
      // Non-fatal at construction time — will fall back to EVAL
      console.warn('[RateLimiter] Could not pre-load Lua script:', err.message);
    }
  }

  /**
   * Execute the Lua script, falling back from EVALSHA to EVAL on cache miss.
   * Redis can flush the script cache on restart; EVALSHA will return NOSCRIPT.
   */
  async _runScript(key, nowMs, windowMs, limit) {
    const ttlMs = windowMs + this.clockSkewMs;
    const args  = [String(nowMs), String(windowMs), String(limit), String(ttlMs)];

    const execute = async (useSha) => {
      if (useSha && this._scriptSha) {
        return this.client.evalsha(this._scriptSha, 1, key, ...args);
      }
      return this.client.eval(LUA_SCRIPT, 1, key, ...args);
    };

    try {
      return await execute(true);
    } catch (err) {
      // NOSCRIPT means the script was flushed from Redis cache
      if (err.message && err.message.includes('NOSCRIPT')) {
        this._scriptSha = await this.client.scriptLoad(LUA_SCRIPT);
        return execute(true);
      }
      throw err;
    }
  }

  /**
   * @param {string} identifier  - Unique key (user ID, IP, API key, etc.)
   * @param {number} windowMs    - Window size in milliseconds
   * @param {number} limit       - Max requests per window
   * @returns {Promise<{allowed: boolean, count: number, remaining: number, resetAfterMs: number}>}
   */
  async increment(identifier, windowMs, limit) {
    const key = `${this.keyPrefix}${identifier}`;
    const now = Date.now();

    let result;
    try {
      result = await this._runScript(key, now, windowMs, limit);
    } catch (err) {
      throw new StoreUnavailableError(err);
    }

    const [allowed, count] = result;
    const remaining        = Math.max(0, limit - count);
    // Earliest point at which one more request would be allowed
    const resetAfterMs     = allowed ? windowMs : windowMs;

    return {
      allowed:      allowed === 1,
      count:        count,
      remaining:    remaining,
      resetAfterMs: resetAfterMs,
      limit:        limit,
    };
  }

  async close() {
    // Caller owns the client lifecycle; this is a no-op by design.
    // If you want the store to own the client, pass `ownsClient: true`.
  }
}

module.exports = RedisStore;
```

### `src/MemoryStore.js`

For local development, testing, and single-instance deployments. **Not safe for multi-process deployments.**

```javascript
'use strict';

class MemoryStore {
  constructor({ cleanupIntervalMs = 60_000 } = {}) {
    // Map<key, SortedList<timestamp>>
    // We use a plain array and sort — acceptable for low-volume local use
    this._windows = new Map();

    // Periodic cleanup to prevent unbounded memory growth
    this._cleanupTimer = setInterval(
      () => this._cleanup(),
      cleanupIntervalMs
    ).unref(); // unref so this doesn't keep the process alive
  }

  async increment(identifier, windowMs, limit) {
    const now          = Date.now();
    const windowStart  = now - windowMs;

    if (!this._windows.has(identifier)) {
      this._windows.set(identifier, []);
    }

    const timestamps = this._windows.get(identifier);

    // Evict expired entries
    while (timestamps.length > 0 && timestamps[0] <= windowStart) {
      timestamps.shift();
    }

    const count   = timestamps.length;
    const allowed = count < limit;

    if (allowed) {
      timestamps.push(now);
    }

    return {
      allowed,
      count:        allowed ? count + 1 : count,
      remaining:    Math.max(0, limit - (allowed ? count + 1 : count)),
      resetAfterMs: windowMs,
      limit,
    };
  }

  _cleanup() {
    const now = Date.now();
    for (const [key, timestamps] of this._windows.entries()) {
      // Remove keys with no recent activity (older than 1 hour)
      if (timestamps.length === 0 || now - timestamps[timestamps.length - 1] > 3_600_000) {
        this._windows.delete(key);
      }
    }
  }

  async close() {
    clearInterval(this._cleanupTimer);
    this._windows.clear();
  }
}

module.exports = MemoryStore;
```

### `src/RateLimiter.js`

```javascript
'use strict';

const { ConfigurationError } = require('./errors');

/**
 * A sliding-window rate limiter with pluggable storage backends.
 *
 * Supports multiple named rules so you can apply different limits
 * to different tiers (e.g., free vs. paid) without instantiating
 * multiple limiters.
 */
class RateLimiter {
  /**
   * @param {object} options
   * @param {object}   options.store               - RedisStore or MemoryStore
   * @param {number}   options.windowMs            - Default window in ms
   * @param {number}   options.limit               - Default max requests per window
   * @param {string}   [options.keyPrefix]         - Additional prefix for key namespacing
   * @param {Function} [options.onLimitReached]    - Hook called when a request is denied
   * @param {Function} [options.onStoreError]      - Hook called on store errors
   * @param {string}   [options.failOpen]          - If true, allow requests when store is down
   */
  constructor({
    store,
    windowMs,
    limit,
    keyPrefix       = '',
    onLimitReached  = null,
    onStoreError    = null,
    failOpen        = false,
  } = {}) {
    this._validateConfig({ store, windowMs, limit });

    this.store          = store;
    this.windowMs       = windowMs;
    this.limit          = limit;
    this.keyPrefix      = keyPrefix;
    this.onLimitReached = onLimitReached;
    this.onStoreError   = onStoreError;
    this.failOpen       = failOpen;
  }

  _validateConfig({ store, windowMs, limit }) {
    if (!store || typeof store.increment !== 'function') {
      throw new ConfigurationError('store must implement increment(identifier, windowMs, limit)');
    }
    if (!Number.isInteger(windowMs) || windowMs <= 0) {
      throw new ConfigurationError('windowMs must be a positive integer');
    }
    if (!Number.isInteger(limit) || limit <= 0) {
      throw new ConfigurationError('limit must be a positive integer');
    }
  }

  /**
   * Check and record a request for the given identifier.
   *
   * @param {string} identifier  - The entity being rate-limited (user ID, IP, etc.)
   * @param {object} [overrides] - Per-call overrides for windowMs and limit (e.g., per-tier)
   * @returns {Promise<RateLimitResult>}
   */
  async check(identifier, { windowMs, limit } = {}) {
    if (!identifier || typeof identifier !== 'string') {
      throw new ConfigurationError('identifier must be a non-empty string');
    }

    const effectiveWindow = windowMs ?? this.windowMs;
    const effectiveLimit  = limit    ?? this.limit;
    const storeKey        = this.keyPrefix
      ? `${this.keyPrefix}:${identifier}`
      : identifier;

    let result;
    try {
      result = await this.store.increment(storeKey, effectiveWindow, effectiveLimit);
    } catch (err) {
      if (this.onStoreError) {
        this.onStoreError(err, { identifier });
      }

      // Fail-open: allow the request but signal the error
      if (this.failOpen) {
        return {
          allowed:      true,
          count:        0,
          remaining:    effectiveLimit,
          resetAfterMs: effectiveWindow,
          limit:        effectiveLimit,
          storeError:   err,
        };
      }

      // Fail-closed: re-throw so the caller can return 503
      throw err;
    }

    if (!result.allowed && this.onLimitReached) {
      this.onLimitReached({ identifier, count: result.count, limit: effectiveLimit });
    }

    return result;
  }

  async close() {
    return this.store.close();
  }
}

module.exports = RateLimiter;
```
