```javascript
// src/stores/memoryStore.js
/**
 * In-memory store using a Map.
 * Good for: single-process apps, testing, development.
 * Not for: multi-instance deployments (no shared state).
 *
 * Uses a cleanup interval to prevent unbounded memory growth.
 */
class MemoryStore {
  constructor(cleanupIntervalMs = 60_000) {
    this.store = new Map();
    this.timers = new Map();

    // Periodic cleanup of expired keys to prevent memory leaks
    this.cleanupInterval = setInterval(
      () => this._cleanup(),
      cleanupIntervalMs
    );

    // Don't block process exit
    if (this.cleanupInterval.unref) {
      this.cleanupInterval.unref();
    }
  }

  async get(key) {
    const entry = this.store.get(key);
    if (!entry) return null;

    // Lazy expiration check
    if (entry.expiresAt && Date.now() > entry.expiresAt) {
      this.store.delete(key);
      return null;
    }

    return entry.value;
  }

  async set(key, value, ttlMs) {
    this.store.set(key, {
      value,
      expiresAt: ttlMs ? Date.now() + ttlMs : null,
    });
  }

  async increment(key, ttlMs) {
    const entry = this.store.get(key);
    const now = Date.now();

    if (!entry || (entry.expiresAt && now > entry.expiresAt)) {
      // Key doesn't exist or expired — start fresh
      this.store.set(key, {
        value: 1,
        expiresAt: ttlMs ? now + ttlMs : null,
      });
      return { count: 1, isNew: true };
    }

    entry.value += 1;
    return { count: entry.value, isNew: false };
  }

  async delete(key) {
    this.store.delete(key);
  }

  async exists(key) {
    const value = await this.get(key);
    return value !== null;
  }

  _cleanup() {
    const now = Date.now();
    for (const [key, entry] of this.store.entries()) {
      if (entry.expiresAt && now > entry.expiresAt) {
        this.store.delete(key);
      }
    }
  }

  destroy() {
    clearInterval(this.cleanupInterval);
    this.store.clear();
  }
}

module.exports = MemoryStore;
```

```javascript
// src/stores/redisStore.js
/**
 * Redis-backed store for distributed rate limiting.
 * Uses atomic Lua scripts to prevent race conditions.
 *
 * Requires: ioredis
 * Good for: multi-instance, high-throughput production systems.
 */
const { StoreConnectionError } = require('../errors');

// Atomic increment with TTL — prevents race conditions
// between GET/SET operations
const INCREMENT_SCRIPT = `
  local key = KEYS[1]
  local ttl = tonumber(ARGV[1])
  local current = redis.call('INCR', key)
  if current == 1 then
    redis.call('PEXPIRE', key, ttl)
  end
  return {current, redis.call('PTTL', key)}
`;

// Atomic token bucket update
const TOKEN_BUCKET_SCRIPT = `
  local key = KEYS[1]
  local capacity = tonumber(ARGV[1])
  local refillRate = tonumber(ARGV[2])
  local now = tonumber(ARGV[3])
  local requested = tonumber(ARGV[4])

  local bucket = redis.call('HMGET', key, 'tokens', 'lastRefill')
  local tokens = tonumber(bucket[1])
  local lastRefill = tonumber(bucket[2])

  if tokens == nil then
    tokens = capacity
    lastRefill = now
  end

  -- Refill tokens based on elapsed time
  local elapsed = math.max(0, now - lastRefill)
  local refilled = math.floor(elapsed * refillRate / 1000)
  tokens = math.min(capacity, tokens + refilled)

  local allowed = 0
  if tokens >= requested then
    tokens = tokens - requested
    allowed = 1
  end

  redis.call('HMSET', key, 'tokens', tokens, 'lastRefill', now)
  redis.call('PEXPIRE', key, math.ceil(capacity / refillRate * 1000) + 1000)

  return {allowed, tokens}
`;

class RedisStore {
  constructor(redisClient) {
    if (!redisClient) {
      throw new StoreConnectionError('Redis client is required');
    }
    this.client = redisClient;
    this._defineScripts();
  }

  _defineScripts() {
    // Pre-load scripts for performance
    this.incrementScript = INCREMENT_SCRIPT;
    this.tokenBucketScript = TOKEN_BUCKET_SCRIPT;
  }

  async increment(key, ttlMs) {
    try {
      const result = await this.client.eval(
        this.incrementScript,
        1,      // number of keys
        key,
        ttlMs
      );
      return {
        count: result[0],
        ttlMs: result[1],
        isNew: result[0] === 1,
      };
    } catch (err) {
      throw new StoreConnectionError(`Redis increment failed: ${err.message}`);
    }
  }

  async get(key) {
    try {
      const value = await this.client.get(key);
      return value ? JSON.parse(value) : null;
    } catch (err) {
      throw new StoreConnectionError(`Redis get failed: ${err.message}`);
    }
  }

  async set(key, value, ttlMs) {
    try {
      const serialized = JSON.stringify(value);
      if (ttlMs) {
        await this.client.set(key, serialized, 'PX', ttlMs);
      } else {
        await this.client.set(key, serialized);
      }
    } catch (err) {
      throw new StoreConnectionError(`Redis set failed: ${err.message}`);
    }
  }

  async delete(key) {
    try {
      await this.client.del(key);
    } catch (err) {
      throw new StoreConnectionError(`Redis delete failed: ${err.message}`);
    }
  }

  async tokenBucketUpdate(key, capacity, refillRate, requested = 1) {
    try {
      const result = await this.client.eval(
        this.tokenBucketScript,
        1,
        key,
        capacity,
        refillRate,
        Date.now(),
        requested
      );
      return {
        allowed: result[0] === 1,
        remaining: result[1],
      };
    } catch (err) {
      throw new StoreConnectionError(
        `Redis token bucket update failed: ${err.message}`
      );
    }
  }
}

module.exports = RedisStore;
```

## Rate Limiting Algorithms

```javascript
// src/algorithms/fixedWindow.js
/**
 * Fixed Window Algorithm
 *
 * How it works:
 *   Divides time into fixed windows (e.g., 0-60s, 60-120s).
 *   Counts requests per window. Resets at window boundary.
 *
 * Pros:  Simple, memory efficient, predictable resets.
 * Cons:  Boundary spike problem — a client can make 2x the limit
 *        by sending max requests at the end of one window and
 *        the start of the next.
 *
 * Use when: Simplicity matters and boundary spikes are acceptable.
 */
class FixedWindowLimiter {
  constructor(store, options = {}) {
    const {
      windowMs = 60_000,   // 1 minute
      maxRequests = 100,
      keyPrefix = 'fw',
    } = options;

    this.store = store;
    this.windowMs = windowMs;
    this.maxRequests = maxRequests;
    this.keyPrefix = keyPrefix;
  }

  _getWindowKey(identifier) {
    // Align key to window boundary so all clients in the same
    // window share the same expiry
    const windowId = Math.floor(Date.now() / this.windowMs);
    return `${this.keyPrefix}:${identifier}:${windowId}`;
  }

  async check(identifier) {
    const key = this._getWindowKey(identifier);
    const { count, isNew } = await this.store.increment(key, this.windowMs);

    const remaining = Math.max(0, this.maxRequests - count);
    const resetAt = this._getWindowResetTime();
    const allowed = count <= this.maxRequests;

    return {
      allowed,
      count,
      remaining,
      resetAt,
      windowMs: this.windowMs,
      maxRequests: this.maxRequests,
      algorithm: 'fixed-window',
    };
  }

  _getWindowResetTime() {
    const windowId = Math.floor(Date.now() / this.windowMs);
    return new Date((windowId + 1) * this.windowMs);
  }
}

module.exports = FixedWindowLimiter;
```

```javascript
// src/algorithms/slidingWindow.js
/**
 * Sliding Window Log Algorithm
 *
 * How it works:
 *   Stores a timestamp log for each request. On each check,
 *   prunes entries older than the window and counts what remains.
 *
 * Pros:  No boundary spike problem. Smooth, accurate limiting.
 * Cons:  Higher memory usage (stores per-request timestamps).
 *        Not ideal for very high-volume keys.
 *
 * Use when: Accuracy is critical and request volume per key is moderate.
 */
class SlidingWindowLimiter {
  constructor(store, options = {}) {
    const {
      windowMs = 60_000,
      maxRequests = 100,
      keyPrefix = 'sw',
    } = options;

    this.store = store;
    this.windowMs = windowMs;
    this.maxRequests = maxRequests;
    this.keyPrefix = keyPrefix;
  }

  _getKey(identifier) {
    return `${this.keyPrefix}:${identifier}`;
  }

  async check(identifier) {
    const key = this._getKey(identifier);
    const now = Date.now();
    const windowStart = now - this.windowMs;

    // Fetch existing log
    let log = (await this.store.get(key)) || [];

    // Prune expired timestamps
    log = log.filter((timestamp) => timestamp > windowStart);

    const count = log.length;
    const allowed = count < this.maxRequests;

    if (allowed) {
      log.push(now);
      await this.store.set(key, log, this.windowMs);
    }

    const oldest = log[0] || now;
    const resetAt = new Date(oldest + this.windowMs);

    return {
      allowed,
      count: allowed ? count + 1 : count,
      remaining: Math.max(0, this.maxRequests - log.length),
      resetAt,
      windowMs: this.windowMs,
      maxRequests: this.maxRequests,
      algorithm: 'sliding-window',
    };
  }
}

module.exports = SlidingWindowLimiter;
```

/**
 * Token Bucket Algorithm
 *
 * How it works:
 *   A bucket holds up to `capacity` tokens. Tokens are added
 *   at `refillRate` tokens/second. Each request consumes one token.
 *   If the bucket is empty, the request is rejected.
 *
 * Pros:  Handles bursts gracefully (up to bucket capacity).
 *        Smooth long-term rate enforcement.
 * Cons:  Slightly more complex state management.
 *
 * Use when: You want to allow short bursts while enforcing
 *           an average rate (e.g., API clients).
 */
class TokenBucketLimiter {
  constructor(store, options = {}) {
    const {
      capacity = 100,        // max tokens (burst size)
      refillRate = 10,       // tokens added per second
      keyPrefix = 'tb',
      tokensPerRequest = 1,
    } = options;

    this.capacity = capacity;
    this.refillRate = refillRate;
    this.keyPrefix = keyPrefix;
    this.tokensPerRequest = tokensPerRequest;

    return `${this.keyPrefix}:${identifier}`;
