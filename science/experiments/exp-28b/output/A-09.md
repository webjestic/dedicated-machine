## Storage Backends

```javascript
// src/storage/memoryStore.js

/**
 * In-memory store with TTL-based expiration.
 *
 * Good for: single-process apps, local dev, testing.
 * Not for: multi-instance deployments (use Redis instead).
 *
 * Uses a Map + periodic cleanup instead of per-key timeouts
 * to avoid timer explosion under high cardinality (many unique keys).
 */
class MemoryStore {
  constructor({ cleanupIntervalMs = 60_000 } = {}) {
    this._store = new Map();
    this._cleanupInterval = setInterval(
      () => this._cleanup(),
      cleanupIntervalMs
    );
    // Don't block process exit
    this._cleanupInterval.unref();
  }

  /**
   * Atomically increment a counter and set TTL on first write.
   * Returns { count, ttlMs } — the current count and remaining TTL.
   */
  async increment(key, windowMs) {
    const now = Date.now();
    const existing = this._store.get(key);

    if (existing && now < existing.expiresAt) {
      existing.count += 1;
      return {
        count: existing.count,
        ttlMs: existing.expiresAt - now,
      };
    }

    // New window
    const entry = { count: 1, expiresAt: now + windowMs };
    this._store.set(key, entry);
    return { count: 1, ttlMs: windowMs };
  }

  async get(key) {
    const now = Date.now();
    const entry = this._store.get(key);
    if (!entry || now >= entry.expiresAt) return null;
    return { count: entry.count, ttlMs: entry.expiresAt - now };
  }

  async set(key, value, ttlMs) {
    this._store.set(key, {
      ...value,
      expiresAt: Date.now() + ttlMs,
    });
  }

  async delete(key) {
    this._store.delete(key);
  }

  async reset(key) {
    this._store.delete(key);
  }

  _cleanup() {
    const now = Date.now();
    for (const [key, entry] of this._store) {
      if (now >= entry.expiresAt) {
        this._store.delete(key);
      }
    }
  }

  // Graceful shutdown
  destroy() {
    clearInterval(this._cleanupInterval);
    this._store.clear();
  }
}

module.exports = MemoryStore;
```

```javascript
// src/storage/redisStore.js
const { StorageError } = require('../errors');

/**
 * Redis-backed store for distributed rate limiting.
 *
 * Uses Lua scripts for atomic operations — critical to prevent
 * race conditions when multiple instances hit the same key.
 *
 * Accepts any ioredis-compatible client.
 */
class RedisStore {
  constructor(redisClient, { keyPrefix = 'rl:' } = {}) {
    if (!redisClient) throw new Error('Redis client is required');
    this._client = redisClient;
    this._keyPrefix = keyPrefix;

    // Lua script for atomic increment + TTL set on first write.
    // KEYS[1] = key, ARGV[1] = windowMs (in seconds for Redis TTL)
    this._incrementScript = `
      local key = KEYS[1]
      local window = tonumber(ARGV[1])
      local current = redis.call('GET', key)

      if current == false then
        redis.call('SET', key, 1, 'PX', window)
        return {1, window}
      else
        local count = redis.call('INCR', key)
        local ttl = redis.call('PTTL', key)
        return {count, ttl}
      end
    `;

    // Lua script for sliding window log
    // KEYS[1] = key, ARGV[1] = now (ms), ARGV[2] = windowMs, ARGV[3] = limit
    this._slidingWindowScript = `
      local key = KEYS[1]
      local now = tonumber(ARGV[1])
      local window = tonumber(ARGV[2])
      local limit = tonumber(ARGV[3])
      local clearBefore = now - window

      redis.call('ZREMRANGEBYSCORE', key, '-inf', clearBefore)
      local count = redis.call('ZCARD', key)

      if count < limit then
        redis.call('ZADD', key, now, now .. '-' .. math.random(1, 1000000))
        redis.call('PEXPIRE', key, window)
        return {count + 1, 1}
      else
        local oldest = redis.call('ZRANGE', key, 0, 0, 'WITHSCORES')
        local resetAt = 0
        if oldest[2] then
          resetAt = tonumber(oldest[2]) + window
        end
        return {count, 0, resetAt}
      end
    `;
  }

  _prefixKey(key) {
    return `${this._keyPrefix}${key}`;
  }

  async increment(key, windowMs) {
    try {
      const result = await this._client.eval(
        this._incrementScript,
        1,
        this._prefixKey(key),
        windowMs
      );
      return { count: result[0], ttlMs: result[1] };
    } catch (err) {
      throw new StorageError(`Redis increment failed for key: ${key}`, err);
    }
  }

  async slidingWindowCheck(key, windowMs, limit) {
    try {
      const now = Date.now();
      const result = await this._client.eval(
        this._slidingWindowScript,
        1,
        this._prefixKey(key),
        now,
        windowMs,
        limit
      );
      return {
        count: result[0],
        allowed: result[1] === 1,
        resetAt: result[2] ? Number(result[2]) : null,
      };
    } catch (err) {
      throw new StorageError(`Redis sliding window failed for key: ${key}`, err);
    }
  }

  async get(key) {
    try {
      const [value, ttl] = await Promise.all([
        this._client.get(this._prefixKey(key)),
        this._client.pttl(this._prefixKey(key)),
      ]);
      if (value === null) return null;
      return { count: parseInt(value, 10), ttlMs: ttl };
    } catch (err) {
      throw new StorageError(`Redis get failed for key: ${key}`, err);
    }
  }

  async set(key, value, ttlMs) {
    try {
      await this._client.set(
        this._prefixKey(key),
        JSON.stringify(value),
        'PX',
        ttlMs
      );
    } catch (err) {
      throw new StorageError(`Redis set failed for key: ${key}`, err);
    }
  }

  async delete(key) {
    try {
      await this._client.del(this._prefixKey(key));
    } catch (err) {
      throw new StorageError(`Redis delete failed for key: ${key}`, err);
    }
  }

  async reset(key) {
    return this.delete(key);
  }
}

module.exports = RedisStore;
```

## Rate Limiting Algorithms

```javascript
// src/algorithms/fixedWindow.js

/**
 * Fixed Window Counter
 *
 * Simplest algorithm. Counts requests in fixed time buckets.
 *
 * Pros: Simple, low memory, predictable reset times.
 * Cons: Boundary burst problem — a client can send 2x the limit
 *       by hitting the end of one window and start of the next.
 *
 * Use when: simplicity matters and boundary bursts are acceptable.
 */
class FixedWindow {
  constructor(store) {
    this._store = store;
  }

  async check({ key, limit, windowMs }) {
    const { count, ttlMs } = await this._store.increment(key, windowMs);
    const remaining = Math.max(0, limit - count);
    const allowed = count <= limit;
    const resetAt = Date.now() + ttlMs;

    return {
      allowed,
      count,
      remaining,
      limit,
      resetAt,
      retryAfterMs: allowed ? 0 : ttlMs,
    };
  }
}

module.exports = FixedWindow;
```

```javascript
// src/algorithms/slidingWindow.js

/**
 * Sliding Window Log
 *
 * Tracks exact timestamps of each request in a sorted set.
 * Counts only requests within [now - windowMs, now].
 *
 * Pros: No boundary burst problem. Most accurate.
 * Cons: Higher memory usage (stores each request timestamp).
 *       O(log N) per request due to sorted set operations.
 *
 * Use when: accuracy is critical (auth endpoints, payment APIs).
 */
class SlidingWindow {
  constructor(store) {
    this._store = store;

    // Sliding window requires sorted set support
    if (typeof store.slidingWindowCheck !== 'function') {
      throw new Error(
        'SlidingWindow algorithm requires a store with slidingWindowCheck() support (use RedisStore)'
      );
    }
  }

  async check({ key, limit, windowMs }) {
    const result = await this._store.slidingWindowCheck(key, windowMs, limit);
    const remaining = Math.max(0, limit - result.count);
    const retryAfterMs =
      !result.allowed && result.resetAt
        ? Math.max(0, result.resetAt - Date.now())
        : 0;

    return {
      allowed: result.allowed,
      count: result.count,
      remaining: result.allowed ? remaining : 0,
      limit,
      resetAt: result.resetAt || Date.now() + windowMs,
      retryAfterMs,
    };
  }
}

module.exports = SlidingWindow;
```

```javascript
// src/algorithms/tokenBucket.js

/**
 * Token Bucket
 *
 * Tokens accumulate at a fixed rate up to a max capacity.
 * Each request consumes one token. Requests are rejected when empty.
 *
 * Pros: Handles bursts gracefully (up to bucket capacity).
 *       Smooth rate enforcement over time.
 * Cons: Slightly more complex state (tokens + last refill time).
 *
 * Use when: you want to allow short bursts but enforce average rate.
 *           Great for APIs consumed by automated clients.
 *
 * @param {number} options.capacity   - Max tokens (burst size)
 * @param {number} options.refillRate - Tokens added per second
 */
class TokenBucket {
  constructor(store) {
    this._store = store;
  }

  async check({ key, capacity, refillRate, windowMs }) {
    const now = Date.now();
    const storeKey = `tb:${key}`;

    // Load existing bucket state
    const existing = await this._store.get(storeKey);

    let tokens;
    let lastRefill;

    if (!existing) {
      // First request — full bucket
      tokens = capacity;
      lastRefill = now;
    } else {
      const data = existing.count !== undefined
        ? existing
        : JSON.parse(existing);

      lastRefill = data.lastRefill || now;
      const elapsedSeconds = (now - lastRefill) / 1000;
      const tokensToAdd = elapsedSeconds * refillRate;

      tokens = Math.min(capacity, (data.tokens || 0) + tokensToAdd);
      lastRefill = now;
    }

    const allowed = tokens >= 1;

    if (allowed) {
      tokens -= 1;
    }

    // Persist updated bucket — TTL based on time to refill from empty
    const ttlMs = Math.ceil((capacity / refillRate) * 1000) + windowMs;
    await this._store.set(
      storeKey,
      { tokens, lastRefill },
      ttlMs
    );

    const tokensUntilRefill = allowed ? 0 : (1 - tokens) / refillRate;

    return {
      allowed,
      count: capacity - Math.floor(tokens),
      remaining: Math.floor(tokens),
      limit: capacity,
      resetAt: now + tokensUntilRefill * 1000,
      retryAfterMs: allowed ? 0 : Math.ceil(tokensUntilRefill * 1000),
    };
  }
}

module.exports = TokenBucket;
```

```javascript
// src/algorithms/leakyBucket.js

/**
 * Leaky Bucket
 *
 * Requests enter a queue and are processed at a fixed rate.
 * Overflow (queue full) requests are rejected immediately.
 *
 * Pros: Enforces a perfectly smooth output rate.
 *       Protects downstream services from any burst.
 * Cons: Can introduce latency if you actually queue requests.
 *       Here we approximate it — tracking queue depth + drain rate.
 *
 * Use when: you need strict output rate control (e.g., calling
 *           a third-party API with hard rate limits).
 *
 * @param {number} options.capacity  - Max queue depth
 * @param {number} options.leakRate  - Requests drained per second
 */
class LeakyBucket {
  constructor(store) {
    this._store = store;
  }

  async check({ key, capacity, leakRate, windowMs }) {
    const now = Date.now();
    const storeKey = `lb:${key}`;

    const existing = await this._store.get(storeKey);

    let queueSize;
    let lastLeak;

    if (!existing) {
      queueSize = 0;
      lastLeak = now;
    } else {
      const data = existing.count !== undefined
        ? existing
        : JSON.parse(existing);

      lastLeak = data.lastLeak || now;
      const elapsedSeconds = (now -
