## Storage Backends

### Memory Store

```javascript
// src/storage/memoryStore.js
'use strict';

/**
 * In-process memory store.
 * Fast, zero dependencies — but does NOT share state across
 * multiple processes or pods. Use Redis in production clusters.
 */
class MemoryStore {
  constructor({ cleanupIntervalMs = 60_000 } = {}) {
    // key → { value, expiresAt }
    this._store = new Map();

    // Periodic cleanup so we don't leak memory on high-cardinality keys
    this._cleanupTimer = setInterval(
      () => this._cleanup(),
      cleanupIntervalMs
    ).unref(); // unref so the timer doesn't keep the process alive
  }

  /**
   * Atomically increment a counter and set TTL on first write.
   * Returns the new value.
   */
  async increment(key, windowMs) {
    const now = Date.now();
    const entry = this._store.get(key);

    if (!entry || now >= entry.expiresAt) {
      // First request in this window
      this._store.set(key, { value: 1, expiresAt: now + windowMs });
      return { count: 1, ttlMs: windowMs };
    }

    entry.value += 1;
    return { count: entry.value, ttlMs: entry.expiresAt - now };
  }

  async get(key) {
    const now = Date.now();
    const entry = this._store.get(key);
    if (!entry || now >= entry.expiresAt) return null;
    return { value: entry.value, ttlMs: entry.expiresAt - now };
  }

  async set(key, value, ttlMs) {
    this._store.set(key, { value, expiresAt: Date.now() + ttlMs });
  }

  async delete(key) {
    this._store.delete(key);
  }

  /**
   * Sliding window log: store an array of timestamps per key.
   */
  async getLog(key) {
    const entry = this._store.get(key);
    if (!entry) return [];
    return entry.value;
  }

  async setLog(key, timestamps, windowMs) {
    this._store.set(key, {
      value: timestamps,
      expiresAt: Date.now() + windowMs,
    });
  }

  _cleanup() {
    const now = Date.now();
    for (const [key, entry] of this._store) {
      if (now >= entry.expiresAt) {
        this._store.delete(key);
      }
    }
  }

  destroy() {
    clearInterval(this._cleanupTimer);
    this._store.clear();
  }
}

module.exports = MemoryStore;
```

### Redis Store

```javascript
// src/storage/redisStore.js
'use strict';

const { StorageError } = require('../errors');

/**
 * Redis-backed store for distributed rate limiting.
 *
 * Requires an ioredis-compatible client passed in from the caller
 * so the caller controls connection lifecycle.
 *
 * All counter operations use Lua scripts to guarantee atomicity
 * without MULTI/EXEC overhead.
 */
class RedisStore {
  constructor(redisClient, { keyPrefix = 'rl:' } = {}) {
    if (!redisClient) throw new Error('RedisStore requires a redis client');
    this._client = redisClient;
    this._prefix = keyPrefix;

    // Lua script: increment key, set TTL only on first write
    // Returns [new_count, ttl_ms]
    this._incrementScript = `
      local key    = KEYS[1]
      local window = tonumber(ARGV[1])
      local now    = tonumber(ARGV[2])

      local count = redis.call('INCR', key)
      if count == 1 then
        redis.call('PEXPIRE', key, window)
      end
      local ttl = redis.call('PTTL', key)
      return {count, ttl}
    `;

    // Lua script for sliding window log operations
    this._slidingWindowScript = `
      local key       = KEYS[1]
      local now       = tonumber(ARGV[1])
      local windowMs  = tonumber(ARGV[2])
      local limit     = tonumber(ARGV[3])
      local cutoff    = now - windowMs

      -- Remove timestamps outside the window
      redis.call('ZREMRANGEBYSCORE', key, '-inf', cutoff)

      local count = redis.call('ZCARD', key)

      if count < limit then
        -- Add current timestamp (use now+count as score tiebreaker)
        redis.call('ZADD', key, now, now .. '-' .. count)
        redis.call('PEXPIRE', key, windowMs)
        return {count + 1, 0}   -- 0 = not limited
      end

      local oldest = redis.call('ZRANGE', key, 0, 0, 'WITHSCORES')
      local retryAfterMs = windowMs - (now - tonumber(oldest[2]))
      return {count, retryAfterMs}  -- retryAfterMs > 0 = limited
    `;
  }

  _key(key) {
    return `${this._prefix}${key}`;
  }

  async increment(key, windowMs) {
    try {
      const result = await this._client.eval(
        this._incrementScript,
        1,
        this._key(key),
        windowMs,
        Date.now()
      );
      return { count: result[0], ttlMs: result[1] };
    } catch (err) {
      throw new StorageError(`Redis increment failed for key: ${key}`, err);
    }
  }

  async slidingWindowCheck(key, windowMs, limit) {
    try {
      const result = await this._client.eval(
        this._slidingWindowScript,
        1,
        this._key(key),
        Date.now(),
        windowMs,
        limit
      );
      return { count: result[0], retryAfterMs: result[1] };
    } catch (err) {
      throw new StorageError(`Redis sliding window failed for key: ${key}`, err);
    }
  }

  async get(key) {
    try {
      const [value, ttl] = await Promise.all([
        this._client.get(this._key(key)),
        this._client.pttl(this._key(key)),
      ]);
      if (value === null) return null;
      return { value: JSON.parse(value), ttlMs: ttl };
    } catch (err) {
      throw new StorageError(`Redis get failed for key: ${key}`, err);
    }
  }

  async set(key, value, ttlMs) {
    try {
      await this._client.set(
        this._key(key),
        JSON.stringify(value),
        'PX',
        ttlMs
      );
    } catch (err) {
      throw new StorageError(`Redis set failed for key: ${key}`, err);
    }
  }

  async delete(key) {
    try {
      await this._client.del(this._key(key));
    } catch (err) {
      throw new StorageError(`Redis delete failed for key: ${key}`, err);
    }
  }
}

module.exports = RedisStore;
```

## Rate Limiting Algorithms

### Fixed Window

```javascript
// src/algorithms/fixedWindow.js
'use strict';

/**
 * Fixed Window Counter
 *
 * Simple and fast. Weakness: burst traffic at window boundaries
 * (up to 2x the limit in a short period straddling two windows).
 *
 * Best for: simple APIs, internal services, non-critical endpoints.
 */
class FixedWindow {
  /**
   * @param {object} store   - MemoryStore or RedisStore
   * @param {object} options
   * @param {number} options.limit     - Max requests per window
   * @param {number} options.windowMs  - Window size in milliseconds
   */
  constructor(store, { limit, windowMs }) {
    if (!Number.isInteger(limit) || limit < 1)
      throw new TypeError('limit must be a positive integer');
    if (!Number.isInteger(windowMs) || windowMs < 1)
      throw new TypeError('windowMs must be a positive integer');

    this.store = store;
    this.limit = limit;
    this.windowMs = windowMs;
    this.algorithm = 'fixed-window';
  }

  /**
   * @returns {{ allowed: boolean, count: number, remaining: number,
   *             resetMs: number, retryAfterMs: number }}
   */
  async check(key) {
    const { count, ttlMs } = await this.store.increment(key, this.windowMs);

    const allowed = count <= this.limit;
    const remaining = Math.max(0, this.limit - count);
    const resetMs = ttlMs;
    const retryAfterMs = allowed ? 0 : resetMs;

    return { allowed, count, remaining, resetMs, retryAfterMs };
  }

  async reset(key) {
    await this.store.delete(key);
  }
}

module.exports = FixedWindow;
```

### Sliding Window Log

```javascript
// src/algorithms/slidingWindow.js
'use strict';

/**
 * Sliding Window Log
 *
 * Tracks exact timestamps of each request. No boundary burst problem.
 * Memory cost: O(limit) per key — acceptable for most limits.
 *
 * Best for: auth endpoints, payment APIs, anything needing precision.
 */
class SlidingWindow {
  constructor(store, { limit, windowMs }) {
    if (!Number.isInteger(limit) || limit < 1)
      throw new TypeError('limit must be a positive integer');
    if (!Number.isInteger(windowMs) || windowMs < 1)
      throw new TypeError('windowMs must be a positive integer');

    this.store = store;
    this.limit = limit;
    this.windowMs = windowMs;
    this.algorithm = 'sliding-window';
  }

  async check(key) {
    // Redis store has an optimized atomic Lua implementation
    if (this.store.slidingWindowCheck) {
      return this._checkRedis(key);
    }
    return this._checkMemory(key);
  }

  async _checkRedis(key) {
    const { count, retryAfterMs } = await this.store.slidingWindowCheck(
      key,
      this.windowMs,
      this.limit
    );

    const allowed = retryAfterMs === 0;
    const remaining = Math.max(0, this.limit - count);

    return {
      allowed,
      count,
      remaining,
      resetMs: this.windowMs,
      retryAfterMs: allowed ? 0 : retryAfterMs,
    };
  }

  async _checkMemory(key) {
    const now = Date.now();
    const cutoff = now - this.windowMs;

    // Get existing log, prune old entries
    let log = await this.store.getLog(key);
    log = log.filter((ts) => ts > cutoff);

    const count = log.length;
    const allowed = count < this.limit;

    if (allowed) {
      log.push(now);
      await this.store.setLog(key, log, this.windowMs);
    }

    const remaining = Math.max(0, this.limit - log.length);
    const retryAfterMs = allowed
      ? 0
      : this.windowMs - (now - log[0]); // time until oldest entry expires

    return {
      allowed,
      count: log.length,
      remaining,
      resetMs: this.windowMs,
      retryAfterMs: Math.max(0, retryAfterMs),
    };
  }

  async reset(key) {
    await this.store.delete(key);
  }
}

module.exports = SlidingWindow;
```

### Token Bucket

```javascript
// src/algorithms/tokenBucket.js
'use strict';

/**
 * Token Bucket
 *
 * Tokens refill at a steady rate. Allows controlled bursting
 * (up to `capacity` tokens at once). Smooth, natural feel.
 *
 * Best for: APIs that want to allow short bursts but enforce
 * a sustained throughput limit (e.g., 100 req/s sustained,
 * burst up to 200).
 */
class TokenBucket {
  /**
   * @param {object} store
   * @param {object} options
   * @param {number} options.capacity      - Max tokens (burst ceiling)
   * @param {number} options.refillRate    - Tokens added per second
   * @param {number} options.refillIntervalMs - How often to refill
   */
  constructor(store, { capacity, refillRate, refillIntervalMs = 1000 }) {
    if (!Number.isFinite(capacity) || capacity < 1)
      throw new TypeError('capacity must be a positive number');
    if (!Number.isFinite(refillRate) || refillRate <= 0)
      throw new TypeError('refillRate must be a positive number');

    this.store = store;
    this.capacity = capacity;
    this.refillRate = refillRate;         // tokens per second
    this.refillIntervalMs = refillIntervalMs;
    this.algorithm = 'token-bucket';
  }

  async check(key, tokensRequested = 1) {
    const now = Date.now();
    const stored = await this.store.get(key);

    let tokens;
    let lastRefill;

    if (!stored) {
      // New key: start full
      tokens = this.capacity;
      lastRefill = now;
    } else {
      ({ tokens, lastRefill } = stored.value);

      // Refill based on elapsed time
      const elapsedSec = (now - lastRefill) / 1000;
      const newTokens = elapsedSec * this.refillRate;
      tokens = Math.min(this.capacity, tokens + newTokens);
      last
