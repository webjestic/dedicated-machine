# Production Rate Limiter in Node.js

Let me build this properly — working code first, then an honest operational assessment.

## Architecture Decision

I'll implement a **sliding window log** algorithm backed by Redis. It's more accurate than fixed window (no boundary bursts) and more memory-efficient than pure token bucket for most API use cases.

```
Request → Identify Key → Check Redis → Allow/Deny → Record + Return Headers
```

### `src/RedisStore.js`

```javascript
'use strict';

const { StoreConnectionError } = require('./errors');

/**
 * Sliding window log via Redis sorted sets.
 *
 * Key schema:  rl:{namespace}:{key}
 * Value:       sorted set where score = timestamp (ms), member = unique request ID
 *
 * Why sorted sets?
 *   - O(log N) insert
 *   - O(log N) range delete (ZREMRANGEBYSCORE)
 *   - O(1) count (ZCARD)
 *   - Atomic via Lua script — no race conditions
 */
class RedisStore {
  /**
   * @param {object} opts
   * @param {import('ioredis').Redis} opts.client   - ioredis instance (caller owns lifecycle)
   * @param {string}  [opts.namespace='rl']         - key prefix
   * @param {number}  [opts.keyExpireBuffer=5000]   - extra TTL buffer in ms beyond window
   */
  constructor({ client, namespace = 'rl', keyExpireBuffer = 5000 } = {}) {
    if (!client) {
      throw new Error('RedisStore requires an ioredis client instance');
    }

    this.client = client;
    this.namespace = namespace;
    this.keyExpireBuffer = keyExpireBuffer;

    // Lua script: atomic sliding window check-and-record
    // Returns: [current_count, oldest_timestamp_in_window | -1]
    this._luaScript = `
      local key        = KEYS[1]
      local now        = tonumber(ARGV[1])
      local windowMs   = tonumber(ARGV[2])
      local limit      = tonumber(ARGV[3])
      local requestId  = ARGV[4]
      local expireSecs = tonumber(ARGV[5])
      local windowStart = now - windowMs

      -- Remove entries outside the current window
      redis.call('ZREMRANGEBYSCORE', key, '-inf', windowStart)

      -- Count remaining entries
      local count = redis.call('ZCARD', key)

      -- Record this request regardless of allow/deny
      -- Caller decides whether to use the count
      redis.call('ZADD', key, now, requestId)
      redis.call('EXPIRE', key, expireSecs)

      -- Return count BEFORE this request, plus oldest timestamp for retry-after
      local oldest = redis.call('ZRANGE', key, 0, 0, 'WITHSCORES')
      local oldestTs = -1
      if oldest and #oldest > 0 then
        oldestTs = tonumber(oldest[2])
      end

      return {count, oldestTs}
    `;

    // SHA for EVALSHA (loaded lazily)
    this._scriptSha = null;
  }

  /**
   * Load Lua script into Redis script cache.
   * Called once on first use; safe to call multiple times.
   */
  async _ensureScriptLoaded() {
    if (this._scriptSha) return;
    try {
      this._scriptSha = await this.client.script('LOAD', this._luaScript);
    } catch (err) {
      throw new StoreConnectionError('Failed to load Lua script into Redis', err);
    }
  }

  /**
   * Increment and return the current request count for a key within a window.
   *
   * @param {string} key          - rate limit key (e.g. IP or user ID)
   * @param {number} windowMs     - sliding window size in milliseconds
   * @param {number} limit        - maximum allowed requests in window
   * @returns {Promise<{count: number, oldestTs: number, allowed: boolean, retryAfterMs: number}>}
   */
  async increment(key, windowMs, limit) {
    await this._ensureScriptLoaded();

    const redisKey = `${this.namespace}:${key}`;
    const now = Date.now();
    const requestId = `${now}-${Math.random().toString(36).slice(2, 9)}`;
    const expireSecs = Math.ceil((windowMs + this.keyExpireBuffer) / 1000);

    let result;
    try {
      result = await this.client.evalsha(
        this._scriptSha,
        1,                    // number of keys
        redisKey,             // KEYS[1]
        now,                  // ARGV[1]
        windowMs,             // ARGV[2]
        limit,                // ARGV[3]
        requestId,            // ARGV[4]
        expireSecs            // ARGV[5]
      );
    } catch (err) {
      // Script cache miss (Redis restart) — reload and retry once
      if (err.message && err.message.includes('NOSCRIPT')) {
        this._scriptSha = null;
        await this._ensureScriptLoaded();
        result = await this.client.evalsha(
          this._scriptSha,
          1,
          redisKey,
          now,
          windowMs,
          limit,
          requestId,
          expireSecs
        );
      } else {
        throw new StoreConnectionError('Redis evalsha failed', err);
      }
    }

    const countBeforeThisRequest = parseInt(result[0], 10);
    const oldestTs = parseInt(result[1], 10);

    // count includes the request we just added, so allowed if count was < limit before
    const allowed = countBeforeThisRequest < limit;

    // If denied, how long until the oldest entry expires out of the window?
    let retryAfterMs = 0;
    if (!allowed && oldestTs !== -1) {
      retryAfterMs = Math.max(0, oldestTs + windowMs - now);
    }

    return {
      count: countBeforeThisRequest + 1,  // total including this request
      remaining: Math.max(0, limit - countBeforeThisRequest - 1),
      allowed,
      retryAfterMs,
      resetAtMs: now + windowMs,          // approximate window reset
    };
  }

  /**
   * Peek at current count without recording a request.
   * Useful for monitoring endpoints.
   */
  async peek(key, windowMs) {
    const redisKey = `${this.namespace}:${key}`;
    const windowStart = Date.now() - windowMs;

    try {
      await this.client.zremrangebyscore(redisKey, '-inf', windowStart);
      const count = await this.client.zcard(redisKey);
      return { count };
    } catch (err) {
      throw new StoreConnectionError('Redis peek failed', err);
    }
  }

  /**
   * Manually reset a key (admin use, testing).
   */
  async reset(key) {
    const redisKey = `${this.namespace}:${key}`;
    try {
      await this.client.del(redisKey);
    } catch (err) {
      throw new StoreConnectionError('Redis reset failed', err);
    }
  }

  async ping() {
    try {
      const result = await this.client.ping();
      return result === 'PONG';
    } catch {
      return false;
    }
  }
}

module.exports = RedisStore;
```

### `src/MemoryStore.js`

```javascript
'use strict';

/**
 * In-process sliding window store.
 *
 * Use cases:
 *   - Unit tests (no Redis dependency)
 *   - Single-instance development servers
 *   - Fallback when Redis is unavailable (see failOpen option)
 *
 * NOT suitable for multi-instance production deployments.
 * State is not shared across processes.
 */
class MemoryStore {
  constructor() {
    // Map<key, Array<timestamp>>
    this._windows = new Map();

    // Periodic cleanup to prevent unbounded memory growth
    this._cleanupInterval = setInterval(() => this._cleanup(), 60_000);

    // Allow GC / clean shutdown
    if (this._cleanupInterval.unref) {
      this._cleanupInterval.unref();
    }
  }

  async increment(key, windowMs, limit) {
    const now = Date.now();
    const windowStart = now - windowMs;

    if (!this._windows.has(key)) {
      this._windows.set(key, []);
    }

    const timestamps = this._windows.get(key);

    // Evict expired entries
    const valid = timestamps.filter(ts => ts > windowStart);
    const countBefore = valid.length;

    valid.push(now);
    this._windows.set(key, valid);

    const allowed = countBefore < limit;
    const retryAfterMs = !allowed && valid.length > 0
      ? Math.max(0, valid[0] + windowMs - now)
      : 0;

    return {
      count: countBefore + 1,
      remaining: Math.max(0, limit - countBefore - 1),
      allowed,
      retryAfterMs,
      resetAtMs: now + windowMs,
    };
  }

  async peek(key, windowMs) {
    const now = Date.now();
    const windowStart = now - windowMs;
    const timestamps = this._windows.get(key) || [];
    const count = timestamps.filter(ts => ts > windowStart).length;
    return { count };
  }

  async reset(key) {
    this._windows.delete(key);
  }

  async ping() {
    return true;
  }

  _cleanup() {
    const now = Date.now();
    for (const [key, timestamps] of this._windows.entries()) {
      // Use a generous 1-hour window for cleanup — store doesn't know the window size
      const valid = timestamps.filter(ts => ts > now - 3_600_000);
      if (valid.length === 0) {
        this._windows.delete(key);
      } else {
        this._windows.set(key, valid);
      }
    }
  }

  destroy() {
    clearInterval(this._cleanupInterval);
    this._windows.clear();
  }
}

module.exports = MemoryStore;
```

### `src/RateLimiter.js`

```javascript
'use strict';

const { ConfigurationError } = require('./errors');

/**
 * Core rate limiter — store-agnostic.
 *
 * Supports:
 *   - Multiple named rules (e.g. global, per-user, per-endpoint)
 *   - Custom key extraction
 *   - Fail-open or fail-closed on store errors
 *   - Skip conditions (health checks, internal IPs, etc.)
 */
class RateLimiter {
  /**
   * @param {object} opts
   * @param {object}   opts.store              - RedisStore or MemoryStore instance
   * @param {number}   opts.limit              - max requests per window
   * @param {number}   opts.windowMs           - window size in milliseconds
   * @param {Function} [opts.keyGenerator]     - (req) => string
   * @param {Function} [opts.skip]             - (req) => boolean — skip rate limiting
   * @param {boolean}  [opts.failOpen=false]   - allow requests when store errors occur
   * @param {Function} [opts.onLimitReached]   - (req, result) => void — hook for logging/alerting
   * @param {Function} [opts.onStoreError]     - (err, req) => void — hook for alerting
   * @param {object}   [opts.logger]           - logger with .warn() and .error()
   */
  constructor(opts = {}) {
    this._validateConfig(opts);

    this.store = opts.store;
    this.limit = opts.limit;
    this.windowMs = opts.windowMs;
    this.keyGenerator = opts.keyGenerator || defaultKeyGenerator;
    this.skip = opts.skip || (() => false);
    this.failOpen = opts.failOpen === true;
    this.onLimitReached = opts.onLimitReached || null;
    this.onStoreError = opts.onStoreError || null;
    this.logger = opts.logger || console;
  }

  _validateConfig({ store, limit, windowMs }) {
    if (!store) throw new ConfigurationError('store is required');
    if (!Number.isInteger(limit) || limit < 1) {
      throw new ConfigurationError('limit must be a positive integer');
    }
    if (!Number.isInteger(windowMs) || windowMs < 100) {
      throw new ConfigurationError('windowMs must be an integer >= 100ms');
    }
  }

  /**
   * Check and record a request.
   *
   * @param {object} req - Express request (or any object your keyGenerator understands)
   * @returns {Promise<RateLimitResult>}
   */
  async check(req) {
    if (await this.skip(req)) {
      return { allowed: true, skipped: true, count: 0, remaining: this.limit };
    }

    const key = await this.keyGenerator(req);

    let result;
    try {
      result = await this.store.increment(key, this.windowMs, this.limit);
    } catch (err) {
      if (this.onStoreError) {
        this.onStoreError(err, req);
      } else {
        this.logger.error('[RateLimiter] Store error', {
          error: err.message,
          key,
          failOpen: this.failOpen,
        });
      }

      // Fail-open: allow the request but signal the error
      if (this.failOpen) {
        return {
          allowed: true,
          storeError: true,
          count: 0,
          remaining: this.limit,
          retryAfterMs: 0,
          resetAtMs: Date.now() + this.windowMs,
        };
      }

      // Fail-closed: deny the request
      return {
        allowed: false,
        storeError: true,
        count: 0,
        remaining
