```javascript
// src/storage/memoryStore.js

/**
 * In-memory store with TTL-based expiry.
 * Good for: single-process apps, testing, development.
 * Not good for: multi-instance deployments (no shared state).
 */
class MemoryStore {
  constructor({ cleanupIntervalMs = 60_000 } = {}) {
    // Map<key, { value, expiresAt }>
    this._store = new Map();

    // Periodically sweep expired keys so memory doesn't grow unbounded.
    // This is a background task — don't let it crash the process.
    this._cleanupTimer = setInterval(() => {
      this._cleanup();
    }, cleanupIntervalMs).unref(); // .unref() so it doesn't block process exit
  }

  /**
   * Get a value. Returns null if missing or expired.
   */
  async get(key) {
    const entry = this._store.get(key);
    if (!entry) return null;

    if (Date.now() > entry.expiresAt) {
      this._store.delete(key);
      return null;
    }

    return entry.value;
  }

  /**
   * Set a value with a TTL in milliseconds.
   */
  async set(key, value, ttlMs) {
    this._store.set(key, {
      value,
      expiresAt: Date.now() + ttlMs,
    });
  }

  /**
   * Atomically increment a counter, initializing to 1 if it doesn't exist.
   * Returns the new value.
   */
  async increment(key, ttlMs) {
    const entry = this._store.get(key);
    const now = Date.now();

    if (!entry || now > entry.expiresAt) {
      // Key doesn't exist or has expired — start a fresh window
      this._store.set(key, { value: 1, expiresAt: now + ttlMs });
      return { count: 1, ttlMs };
    }

    entry.value += 1;
    return { count: entry.value, ttlMs: entry.expiresAt - now };
  }

  /**
   * Delete a key.
   */
  async delete(key) {
    this._store.delete(key);
  }

  /**
   * Get remaining TTL in milliseconds for a key.
   * Returns -1 if key doesn't exist.
   */
  async ttl(key) {
    const entry = this._store.get(key);
    if (!entry) return -1;
    return Math.max(0, entry.expiresAt - Date.now());
  }

  /**
   * Remove all expired entries. Called by the cleanup timer.
   */
  _cleanup() {
    const now = Date.now();
    for (const [key, entry] of this._store) {
      if (now > entry.expiresAt) {
        this._store.delete(key);
      }
    }
  }

  /**
   * Graceful shutdown — clear the cleanup timer.
   */
  destroy() {
    clearInterval(this._cleanupTimer);
    this._store.clear();
  }
}

module.exports = MemoryStore;
```

```javascript
// src/storage/redisStore.js
const { StorageError } = require('../errors');

/**
 * Redis-backed store using atomic Lua scripts.
 * Good for: multi-instance deployments, production.
 *
 * Requires an ioredis-compatible client passed in.
 * We don't create the client here — caller manages the connection lifecycle.
 */
class RedisStore {
  constructor(redisClient) {
    if (!redisClient) {
      throw new Error('RedisStore requires a Redis client instance');
    }
    this._client = redisClient;

    // Lua script for atomic increment with TTL initialization.
    // This is critical — a non-atomic read-increment-write would have race conditions
    // under concurrent load. Lua scripts execute atomically in Redis.
    this._incrementScript = `
      local key = KEYS[1]
      local ttl_ms = tonumber(ARGV[1])
      local current = redis.call('GET', key)
      
      if current == false then
        redis.call('SET', key, 1, 'PX', ttl_ms)
        return {1, ttl_ms}
      end
      
      local new_val = redis.call('INCR', key)
      local remaining_ttl = redis.call('PTTL', key)
      
      -- If TTL was lost somehow (shouldn't happen, but be defensive)
      if remaining_ttl < 0 then
        redis.call('PEXPIRE', key, ttl_ms)
        remaining_ttl = ttl_ms
      end
      
      return {new_val, remaining_ttl}
    `;

    // Lua script for token bucket — atomically get/update bucket state
    this._tokenBucketScript = `
      local key = KEYS[1]
      local capacity = tonumber(ARGV[1])
      local refill_rate = tonumber(ARGV[2])  -- tokens per ms
      local requested = tonumber(ARGV[3])
      local now = tonumber(ARGV[4])
      local ttl_ms = tonumber(ARGV[5])
      
      local data = redis.call('HMGET', key, 'tokens', 'last_refill')
      local tokens = tonumber(data[1])
      local last_refill = tonumber(data[2])
      
      if tokens == nil then
        tokens = capacity
        last_refill = now
      end
      
      -- Refill tokens based on elapsed time
      local elapsed = now - last_refill
      local new_tokens = math.min(capacity, tokens + elapsed * refill_rate)
      
      local allowed = 0
      if new_tokens >= requested then
        new_tokens = new_tokens - requested
        allowed = 1
      end
      
      redis.call('HMSET', key, 'tokens', new_tokens, 'last_refill', now)
      redis.call('PEXPIRE', key, ttl_ms)
      
      return {allowed, math.floor(new_tokens)}
    `;
  }

  async increment(key, ttlMs) {
    try {
      const result = await this._client.eval(
        this._incrementScript,
        1,       // number of keys
        key,     // KEYS[1]
        ttlMs    // ARGV[1]
      );
      return { count: result[0], ttlMs: result[1] };
    } catch (err) {
      throw new StorageError(`Redis increment failed for key: ${key}`, err);
    }
  }

  async get(key) {
    try {
      const value = await this._client.get(key);
      return value !== null ? JSON.parse(value) : null;
    } catch (err) {
      throw new StorageError(`Redis get failed for key: ${key}`, err);
    }
  }

  async set(key, value, ttlMs) {
    try {
      await this._client.set(key, JSON.stringify(value), 'PX', ttlMs);
    } catch (err) {
      throw new StorageError(`Redis set failed for key: ${key}`, err);
    }
  }

  async delete(key) {
    try {
      await this._client.del(key);
    } catch (err) {
      throw new StorageError(`Redis delete failed for key: ${key}`, err);
    }
  }

  async ttl(key) {
    try {
      return await this._client.pttl(key);
    } catch (err) {
      throw new StorageError(`Redis TTL failed for key: ${key}`, err);
    }
  }

  async tokenBucketCheck(key, capacity, refillRate, requested, ttlMs) {
    try {
      const now = Date.now();
      const result = await this._client.eval(
        this._tokenBucketScript,
        1,
        key,
        capacity,
        refillRate,
        requested,
        now,
        ttlMs
      );
      return { allowed: result[0] === 1, remainingTokens: result[1] };
    } catch (err) {
      throw new StorageError(`Redis token bucket check failed for key: ${key}`, err);
    }
  }
}

module.exports = RedisStore;
```

## Algorithms

```javascript
// src/algorithms/fixedWindow.js

/**
 * Fixed Window Counter
 *
 * How it works:
 *   - Divide time into fixed windows (e.g., every 60s)
 *   - Count requests per window per key
 *   - Reject if count exceeds limit
 *
 * Pros: Simple, low memory, easy to reason about
 * Cons: Boundary problem — a client can send 2x the limit by straddling
 *       two windows (burst at end of window + start of next)
 *
 * Use when: simplicity matters and you can tolerate the boundary burst
 */
class FixedWindow {
  constructor(store) {
    this._store = store;
  }

  /**
   * @param {string} key       - Unique identifier (e.g., IP, user ID)
   * @param {number} limit     - Max requests per window
   * @param {number} windowMs  - Window size in milliseconds
   * @returns {Promise<RateLimitResult>}
   */
  async check(key, limit, windowMs) {
    // Align key to the current window boundary so all requests in the
    // same window share the same counter key
    const windowStart = Math.floor(Date.now() / windowMs) * windowMs;
    const windowKey = `fw:${key}:${windowStart}`;

    const { count, ttlMs } = await this._store.increment(windowKey, windowMs);

    const remaining = Math.max(0, limit - count);
    const allowed = count <= limit;
    const resetAt = new Date(windowStart + windowMs);

    return {
      allowed,
      limit,
      remaining,
      count,
      resetAt,
      retryAfterMs: allowed ? 0 : ttlMs,
      algorithm: 'fixed-window',
    };
  }
}

module.exports = FixedWindow;
```

```javascript
// src/algorithms/slidingWindow.js

/**
 * Sliding Window Log
 *
 * How it works:
 *   - Store a sorted set of request timestamps per key
 *   - On each request, remove timestamps older than the window
 *   - Count remaining entries — reject if over limit
 *
 * Pros: Precise, no boundary burst problem
 * Cons: Memory grows with request count (O(requests) per key)
 *       More expensive than fixed window
 *
 * Use when: precision matters and you have reasonable request rates
 *
 * Note: For Redis, use a ZSET with timestamp as score.
 *       For memory store, we maintain a sorted array.
 */
class SlidingWindow {
  constructor(store) {
    this._store = store;
    // For memory store: Map<key, number[]> of timestamps
    this._logs = new Map();
  }

  async check(key, limit, windowMs) {
    const now = Date.now();
    const windowStart = now - windowMs;
    const logKey = `sw:${key}`;

    // Get existing timestamps, filter out expired ones
    let timestamps = (await this._store.get(logKey)) || [];
    timestamps = timestamps.filter((ts) => ts > windowStart);

    const count = timestamps.length;
    const allowed = count < limit;

    if (allowed) {
      timestamps.push(now);
    }

    // TTL: keep the log alive for one full window after the last request
    await this._store.set(logKey, timestamps, windowMs);

    const remaining = Math.max(0, limit - timestamps.length);

    // When can they retry? When the oldest request falls outside the window
    let retryAfterMs = 0;
    if (!allowed && timestamps.length > 0) {
      const oldestInWindow = timestamps[0];
      retryAfterMs = oldestInWindow + windowMs - now;
    }

    return {
      allowed,
      limit,
      remaining,
      count: timestamps.length,
      resetAt: new Date(now + windowMs),
      retryAfterMs: Math.max(0, retryAfterMs),
      algorithm: 'sliding-window',
    };
  }
}

module.exports = SlidingWindow;
```

/**
 * Token Bucket
 *
 * How it works:
 *   - Each key has a bucket with a max capacity of tokens
 *   - Tokens refill at a constant rate (e.g., 10/second)
 *   - Each request consumes tokens
 *   - Reject if insufficient tokens
 *
 * Pros: Handles bursts gracefully (up to bucket capacity)
 *       Smooth rate limiting
 * Cons: More complex state, harder to reason about exact limits
 *
 * Use when: you want to allow short bursts but enforce average rate
 */
class TokenBucket {
  constructor(store) {
    this._store = store;
    // For memory store: Map<key, { tokens, lastRefill }>
    this._buckets = new Map();
  }

  /**
   * @param {string} key          - Unique identifier
   * @param {number} capacity     - Max tokens (burst size)
   * @param {number} refillRate   - Tokens added per second
   * @param {number} requested    - Tokens this request costs (default 1)
   */
  async check(key, capacity, refillRate, requested = 1) {
    const now = Date.now();
    const bucketKey = `tb:${key}`;
    const refillRatePerMs = refillRate / 1000;

    // If using Redis store with Lua support, delegate to atomic script
    if (typeof this._store.tokenBucketCheck === 'function') {
      const ttlMs = (capacity / refillRate) * 1000 * 2; // 2x fill time as TTL
      const result = await this._store.tokenBucketCheck(
        bucketKey,
        capacity,
        refillRatePerMs,
        requested,
        ttlMs
      );
