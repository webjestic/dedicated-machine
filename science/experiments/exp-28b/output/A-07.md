## Storage Backends

```javascript
// src/storage/memoryStore.js

/**
 * In-memory store with TTL-based expiration.
 *
 * Good for: single-process apps, testing, development.
 * Not for: multi-instance deployments (state is not shared).
 *
 * Uses a Map with lazy expiration + a periodic sweep to prevent
 * unbounded memory growth under high cardinality key spaces.
 */
class MemoryStore {
  constructor({ sweepIntervalMs = 60_000 } = {}) {
    this._store = new Map();
    this._sweepInterval = setInterval(
      () => this._sweep(),
      sweepIntervalMs
    ).unref(); // unref so the timer doesn't keep the process alive
  }

  /**
   * Atomically increment a counter and set TTL on first write.
   * Returns { count, ttlMs } — the current count and remaining TTL.
   */
  async increment(key, windowMs) {
    const now = Date.now();
    const entry = this._store.get(key);

    if (!entry || now >= entry.expiresAt) {
      // First request in this window
      const expiresAt = now + windowMs;
      this._store.set(key, { count: 1, expiresAt });
      return { count: 1, ttlMs: windowMs };
    }

    entry.count += 1;
    return { count: entry.count, ttlMs: entry.expiresAt - now };
  }

  async get(key) {
    const now = Date.now();
    const entry = this._store.get(key);
    if (!entry || now >= entry.expiresAt) return null;
    return { count: entry.count, ttlMs: entry.expiresAt - now };
  }

  async set(key, value, ttlMs) {
    this._store.set(key, {
      ...value,
      expiresAt: Date.now() + ttlMs,
    });
  }

  async delete(key) {
    this._store.delete(key);
  }

  async reset(key) {
    this._store.delete(key);
  }

  /** Remove expired entries to prevent memory leaks */
  _sweep() {
    const now = Date.now();
    for (const [key, entry] of this._store) {
      if (now >= entry.expiresAt) {
        this._store.delete(key);
      }
    }
  }

  destroy() {
    clearInterval(this._sweepInterval);
    this._store.clear();
  }
}

module.exports = MemoryStore;
```

```javascript
// src/storage/redisStore.js
const { StorageError } = require('../errors');

/**
 * Redis-backed store using atomic Lua scripts.
 *
 * Why Lua? Redis executes Lua scripts atomically — no race conditions
 * between CHECK and INCREMENT across concurrent requests or instances.
 *
 * Requires: ioredis client passed in (we don't own the connection lifecycle).
 */
class RedisStore {
  constructor(redisClient, { keyPrefix = 'rl:' } = {}) {
    if (!redisClient) {
      throw new Error('RedisStore requires a redis client instance');
    }
    this._client = redisClient;
    this._keyPrefix = keyPrefix;

    // Lua script: increment counter, set expiry on first write
    // Returns [count, ttl_in_ms]
    this._incrementScript = `
      local key     = KEYS[1]
      local windowMs = tonumber(ARGV[1])
      local now     = tonumber(ARGV[2])

      local count = redis.call('INCR', key)

      if count == 1 then
        -- First request: set expiry in milliseconds
        redis.call('PEXPIRE', key, windowMs)
      end

      local ttl = redis.call('PTTL', key)
      return {count, ttl}
    `;

    // Lua script for sliding window log
    // Removes old entries, adds new one, returns count
    this._slidingWindowScript = `
      local key       = KEYS[1]
      local now       = tonumber(ARGV[1])
      local windowMs  = tonumber(ARGV[2])
      local requestId = ARGV[3]
      local cutoff    = now - windowMs

      -- Remove timestamps outside the window
      redis.call('ZREMRANGEBYSCORE', key, '-inf', cutoff)

      -- Add current request
      redis.call('ZADD', key, now, requestId)

      -- Set key expiry to window size
      redis.call('PEXPIRE', key, windowMs)

      -- Return current count
      local count = redis.call('ZCARD', key)
      return count
    `;
  }

  _prefixKey(key) {
    return `${this._keyPrefix}${key}`;
  }

  async increment(key, windowMs) {
    try {
      const prefixedKey = this._prefixKey(key);
      const now = Date.now();
      const result = await this._client.eval(
        this._incrementScript,
        1,
        prefixedKey,
        windowMs,
        now
      );
      return { count: result[0], ttlMs: result[1] };
    } catch (err) {
      throw new StorageError('Redis increment failed', err);
    }
  }

  async slidingWindowIncrement(key, windowMs, requestId) {
    try {
      const prefixedKey = this._prefixKey(key);
      const now = Date.now();
      const count = await this._client.eval(
        this._slidingWindowScript,
        1,
        prefixedKey,
        now,
        windowMs,
        requestId
      );
      return { count, ttlMs: windowMs };
    } catch (err) {
      throw new StorageError('Redis sliding window increment failed', err);
    }
  }

  async get(key) {
    try {
      const prefixedKey = this._prefixKey(key);
      const [count, ttl] = await Promise.all([
        this._client.get(prefixedKey),
        this._client.pttl(prefixedKey),
      ]);
      if (count === null) return null;
      return { count: parseInt(count, 10), ttlMs: ttl };
    } catch (err) {
      throw new StorageError('Redis get failed', err);
    }
  }

  async reset(key) {
    try {
      await this._client.del(this._prefixKey(key));
    } catch (err) {
      throw new StorageError('Redis reset failed', err);
    }
  }
}

module.exports = RedisStore;
```

## Rate Limiting Algorithms

```javascript
// src/algorithms/fixedWindow.js

/**
 * Fixed Window Counter
 *
 * How it works:
 *   - Divide time into fixed windows (e.g., 0-60s, 60-120s)
 *   - Count requests per window per key
 *   - Reject if count exceeds limit
 *
 * Pros:  Simple, memory efficient, fast
 * Cons:  Boundary burst problem — a client can send 2x limit
 *        by sending max requests at end of window N and start of window N+1
 *
 * Use when: simplicity matters and boundary bursts are acceptable
 */
class FixedWindow {
  constructor(store) {
    this._store = store;
  }

  async check({ key, limit, windowMs }) {
    const { count, ttlMs } = await this._store.increment(key, windowMs);

    const allowed = count <= limit;
    const remaining = Math.max(0, limit - count);
    const resetAt = new Date(Date.now() + ttlMs);

    return {
      allowed,
      count,
      remaining,
      limit,
      resetAt,
      retryAfterMs: allowed ? 0 : ttlMs,
      algorithm: 'fixed-window',
    };
  }
}

module.exports = FixedWindow;
```

```javascript
// src/algorithms/slidingWindow.js
const { randomUUID } = require('crypto');

/**
 * Sliding Window Log
 *
 * How it works:
 *   - Store a timestamp log of every request
 *   - On each request, purge entries older than windowMs
 *   - Count remaining entries — reject if over limit
 *
 * Pros:  Precise — no boundary burst problem
 * Cons:  Memory scales with request volume (O(limit) per key)
 *        More expensive than fixed window
 *
 * Use when: accuracy is critical (e.g., financial APIs, auth endpoints)
 */
class SlidingWindow {
  constructor(store) {
    this._store = store;
    this._isRedis = typeof store.slidingWindowIncrement === 'function';
  }

  async check({ key, limit, windowMs }) {
    let count;

    if (this._isRedis) {
      // Use atomic Lua script for Redis
      const requestId = `${Date.now()}-${randomUUID()}`;
      const result = await this._store.slidingWindowIncrement(
        key,
        windowMs,
        requestId
      );
      count = result.count;
    } else {
      // In-memory implementation
      count = await this._memoryCheck(key, windowMs);
    }

    const allowed = count <= limit;
    const remaining = Math.max(0, limit - count);

    return {
      allowed,
      count,
      remaining,
      limit,
      resetAt: new Date(Date.now() + windowMs),
      retryAfterMs: allowed ? 0 : windowMs,
      algorithm: 'sliding-window',
    };
  }

  async _memoryCheck(key, windowMs) {
    const now = Date.now();
    const cutoff = now - windowMs;

    const entry = await this._store.get(key);
    const log = (entry && entry.log) ? entry.log.filter(ts => ts > cutoff) : [];

    log.push(now);

    await this._store.set(key, { log }, windowMs);
    return log.length;
  }
}

module.exports = SlidingWindow;
```

```javascript
// src/algorithms/tokenBucket.js

/**
 * Token Bucket
 *
 * How it works:
 *   - Each key has a bucket with capacity N tokens
 *   - Tokens refill at a constant rate (e.g., 10/second)
 *   - Each request consumes 1 token
 *   - Reject if bucket is empty
 *
 * Pros:  Handles bursts gracefully (up to bucket capacity)
 *        Smooth rate limiting over time
 * Cons:  More complex state (tokens + last refill timestamp)
 *
 * Use when: you want to allow short bursts while enforcing average rate
 *           (e.g., API clients that batch requests occasionally)
 */
class TokenBucket {
  constructor(store) {
    this._store = store;
  }

  async check({ key, capacity, refillRate, refillIntervalMs = 1000, tokensPerRequest = 1 }) {
    const now = Date.now();
    const entry = await this._store.get(key);

    let tokens;
    let lastRefill;

    if (!entry) {
      // First request — full bucket
      tokens = capacity;
      lastRefill = now;
    } else {
      tokens = entry.tokens;
      lastRefill = entry.lastRefill;

      // Calculate tokens to add based on elapsed time
      const elapsed = now - lastRefill;
      const intervalsElapsed = Math.floor(elapsed / refillIntervalMs);
      const tokensToAdd = intervalsElapsed * refillRate;

      if (tokensToAdd > 0) {
        tokens = Math.min(capacity, tokens + tokensToAdd);
        lastRefill = lastRefill + intervalsElapsed * refillIntervalMs;
      }
    }

    const allowed = tokens >= tokensPerRequest;

    if (allowed) {
      tokens -= tokensPerRequest;
    }

    // TTL: time until bucket would be full again (for cleanup)
    const ttlMs = Math.ceil((capacity / refillRate) * refillIntervalMs) * 2;
    await this._store.set(key, { tokens, lastRefill }, ttlMs);

    // Time until next token is available
    const tokensNeeded = tokensPerRequest - tokens;
    const retryAfterMs = allowed
      ? 0
      : Math.ceil((tokensNeeded / refillRate) * refillIntervalMs);

    return {
      allowed,
      tokens: Math.floor(tokens),
      remaining: Math.floor(tokens),
      capacity,
      retryAfterMs,
      resetAt: new Date(now + retryAfterMs),
      algorithm: 'token-bucket',
    };
  }
}

module.exports = TokenBucket;
```
