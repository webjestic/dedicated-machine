# Production Rate Limiter in Node.js

Let me build this properly — working code first, then an honest accounting of what's required to run it safely in production.

## Architecture Decisions (Made Explicitly)

Before code, here are the decisions baked in and why:

| Decision | Choice | Rationale |
|---|---|---|
| Algorithm | Sliding window log | Accurate, no boundary burst problem of fixed window |
| Storage | Redis | Shared state across instances; in-memory fails in multi-node |
| Atomicity | Lua script | Prevents race conditions between check and increment |
| Identification | Pluggable (IP, API key, user ID) | Different resources need different identifiers |
| Response | RFC 7807 + standard headers | Interoperability with clients and API gateways |

### `src/errors/RateLimitError.js`

```javascript
'use strict';

/**
 * Structured error thrown when a rate limit is exceeded.
 * Carries everything needed to build a correct HTTP response.
 */
class RateLimitError extends Error {
  /**
   * @param {object} params
   * @param {string} params.key        - The rate limit key that was exceeded
   * @param {number} params.limit      - The configured limit
   * @param {number} params.windowMs   - The window duration in milliseconds
   * @param {number} params.retryAfter - Seconds until the client may retry
   * @param {string} [params.message]  - Human-readable message
   */
  constructor({ key, limit, windowMs, retryAfter, message }) {
    super(message ?? `Rate limit exceeded for key: ${key}`);
    this.name = 'RateLimitError';
    this.key = key;
    this.limit = limit;
    this.windowMs = windowMs;
    this.retryAfter = retryAfter;

    // Maintains proper stack trace in V8
    if (Error.captureStackTrace) {
      Error.captureStackTrace(this, RateLimitError);
    }
  }
}

module.exports = { RateLimitError };
```

### `src/store/redisStore.js`

```javascript
'use strict';

const Redis = require('ioredis');

/**
 * Redis-backed store for rate limiter state.
 *
 * ATOMICITY: All check-and-increment operations are performed inside
 * a single Lua script. Redis executes Lua atomically — no other command
 * can interleave between the read and write. This eliminates the TOCTOU
 * race condition that would exist with separate GET/SET calls.
 *
 * FAILURE MODE: If Redis is unreachable, the store throws. The middleware
 * layer decides whether to fail open or closed — that policy does not
 * belong here.
 */
class RedisStore {
  /**
   * @param {object} options
   * @param {import('ioredis').Redis} [options.client]     - Existing Redis client
   * @param {object}                 [options.redisConfig] - ioredis constructor options
   * @param {number}                 [options.connectTimeoutMs=2000]
   */
  constructor({ client, redisConfig = {}, connectTimeoutMs = 2000 } = {}) {
    if (client) {
      this._client = client;
    } else {
      this._client = new Redis({
        // Sensible production defaults — override via redisConfig
        enableOfflineQueue: false,   // fail fast; don't queue commands during outage
        connectTimeout: connectTimeoutMs,
        maxRetriesPerRequest: 1,     // one retry, then throw — don't hang requests
        lazyConnect: true,
        ...redisConfig,
      });
    }

    this._client.on('error', (err) => {
      // Surface errors to the process; caller decides how to handle
      // In production: pipe this to your structured logger
      console.error('[RedisStore] Redis client error:', err.message);
    });

    // Pre-load Lua scripts onto the client for reuse
    this._scripts = this._defineScripts();
  }

  // ---------------------------------------------------------------------------
  // Lua Scripts
  // ---------------------------------------------------------------------------

  /**
   * Define all Lua scripts used by this store.
   * Each script is registered as a named command on the Redis client via
   * ioredis's defineCommand(), which handles SHA caching and EVALSHA/EVAL
   * fallback automatically.
   */
  _defineScripts() {
    /**
     * SLIDING WINDOW LOG script
     *
     * Algorithm:
     *   1. Remove all entries older than (now - windowMs)
     *   2. Count remaining entries
     *   3. If count < limit: add current timestamp, return [0, count+1, ttl]
     *   4. If count >= limit: return [1, count, ttl]  (1 = rejected)
     *
     * KEYS[1] = rate limit key (e.g., "rl:user:42")
     * ARGV[1] = current timestamp in milliseconds
     * ARGV[2] = window size in milliseconds
     * ARGV[3] = max requests allowed in window
     * ARGV[4] = TTL for the key in seconds (window + buffer)
     *
     * Returns: [rejected (0|1), current_count, oldest_entry_ts_or_0]
     */
    const slidingWindowLogScript = `
      local key        = KEYS[1]
      local now        = tonumber(ARGV[1])
      local window_ms  = tonumber(ARGV[2])
      local limit      = tonumber(ARGV[3])
      local ttl_sec    = tonumber(ARGV[4])
      local window_start = now - window_ms

      -- Remove timestamps outside the current window
      redis.call('ZREMRANGEBYSCORE', key, '-inf', window_start)

      -- Count requests in the current window
      local count = redis.call('ZCARD', key)

      if count < limit then
        -- Allow: record this request with timestamp as both score and member
        -- Using now + a random suffix to handle sub-millisecond duplicate timestamps
        local member = now .. ':' .. redis.call('INCR', key .. ':seq')
        redis.call('ZADD', key, now, member)
        redis.call('EXPIRE', key, ttl_sec)
        -- Also expire the sequence counter
        redis.call('EXPIRE', key .. ':seq', ttl_sec)
        return {0, count + 1, 0}
      else
        -- Reject: find when the oldest entry in the window expires
        local oldest = redis.call('ZRANGE', key, 0, 0, 'WITHSCORES')
        local oldest_ts = 0
        if #oldest > 0 then
          oldest_ts = tonumber(oldest[2])
        end
        return {1, count, oldest_ts}
      end
    `;

    /**
     * SLIDING WINDOW COUNTER script (fixed window approximation)
     *
     * More memory-efficient than the log. Uses two counters:
     * current window and previous window, weighted by overlap.
     *
     * Estimated count = prev_count * (1 - elapsed/window) + curr_count
     *
     * KEYS[1] = current window key
     * KEYS[2] = previous window key
     * ARGV[1] = current timestamp in milliseconds
     * ARGV[2] = window size in milliseconds
     * ARGV[3] = max requests allowed
     *
     * Returns: [rejected (0|1), estimated_count]
     */
    const slidingWindowCounterScript = `
      local curr_key   = KEYS[1]
      local prev_key   = KEYS[2]
      local now        = tonumber(ARGV[1])
      local window_ms  = tonumber(ARGV[2])
      local limit      = tonumber(ARGV[3])

      local curr_count = tonumber(redis.call('GET', curr_key) or 0)
      local prev_count = tonumber(redis.call('GET', prev_key) or 0)

      -- How far into the current window are we? (0.0 to 1.0)
      local window_start = math.floor(now / window_ms) * window_ms
      local elapsed = now - window_start
      local weight = 1 - (elapsed / window_ms)

      local estimated = math.floor(prev_count * weight) + curr_count

      if estimated < limit then
        local new_count = redis.call('INCR', curr_key)
        if new_count == 1 then
          -- First request in this window; set TTL
          redis.call('PEXPIRE', curr_key, window_ms * 2)
        end
        return {0, estimated + 1}
      else
        return {1, estimated}
      end
    `;

    /**
     * TOKEN BUCKET script
     *
     * Allows controlled bursting. Tokens refill at a steady rate.
     * Useful for APIs where occasional bursts are acceptable.
     *
     * KEYS[1] = token bucket key
     * ARGV[1] = current timestamp in milliseconds
     * ARGV[2] = bucket capacity (max tokens)
     * ARGV[3] = refill rate (tokens per millisecond)
     * ARGV[4] = tokens required for this request (usually 1)
     * ARGV[5] = TTL in seconds
     *
     * Returns: [rejected (0|1), remaining_tokens_floor, next_refill_ms]
     */
    const tokenBucketScript = `
      local key          = KEYS[1]
      local now          = tonumber(ARGV[1])
      local capacity     = tonumber(ARGV[2])
      local refill_rate  = tonumber(ARGV[3])
      local tokens_req   = tonumber(ARGV[4])
      local ttl_sec      = tonumber(ARGV[5])

      local bucket = redis.call('HMGET', key, 'tokens', 'last_refill')
      local tokens      = tonumber(bucket[1])
      local last_refill = tonumber(bucket[2])

      if tokens == nil then
        -- First request: initialize full bucket
        tokens = capacity
        last_refill = now
      else
        -- Refill tokens based on elapsed time
        local elapsed = now - last_refill
        local refill  = elapsed * refill_rate
        tokens = math.min(capacity, tokens + refill)
        last_refill = now
      end

      if tokens >= tokens_req then
        tokens = tokens - tokens_req
        redis.call('HMSET', key, 'tokens', tokens, 'last_refill', last_refill)
        redis.call('EXPIRE', key, ttl_sec)
        return {0, math.floor(tokens), 0}
      else
        -- Calculate ms until enough tokens are available
        local tokens_needed = tokens_req - tokens
        local wait_ms = math.ceil(tokens_needed / refill_rate)
        redis.call('HMSET', key, 'tokens', tokens, 'last_refill', last_refill)
        redis.call('EXPIRE', key, ttl_sec)
        return {1, math.floor(tokens), wait_ms}
      end
    `;

    // Register scripts as named commands on the Redis client
    this._client.defineCommand('slidingWindowLog', {
      numberOfKeys: 1,
      lua: slidingWindowLogScript,
    });

    this._client.defineCommand('slidingWindowCounter', {
      numberOfKeys: 2,
      lua: slidingWindowCounterScript,
    });

    this._client.defineCommand('tokenBucket', {
      numberOfKeys: 1,
      lua: tokenBucketScript,
    });

    return { slidingWindowLogScript, slidingWindowCounterScript, tokenBucketScript };
  }

  // ---------------------------------------------------------------------------
  // Public API
  // ---------------------------------------------------------------------------

  /**
   * Check and increment using the sliding window log algorithm.
   *
   * @param {string} key
   * @param {number} limit      - Max requests per window
   * @param {number} windowMs   - Window size in milliseconds
   * @returns {Promise<{allowed: boolean, count: number, retryAfterMs: number}>}
   */
  async checkSlidingWindowLog(key, limit, windowMs) {
    const now = Date.now();
    const ttlSec = Math.ceil((windowMs / 1000) * 1.1); // 10% buffer

    const [rejected, count, oldestTs] = await this._client.slidingWindowLog(
      key,
      now,
      windowMs,
      limit,
      ttlSec
    );

    const retryAfterMs = rejected
      ? Math.max(0, oldestTs + windowMs - now)
      : 0;

    return {
      allowed: rejected === 0,
      count: Number(count),
      retryAfterMs: Number(retryAfterMs),
    };
  }

  /**
   * Check and increment using the sliding window counter algorithm.
   *
   * @param {string} key
   * @param {number} limit
   * @param {number} windowMs
   * @returns {Promise<{allowed: boolean, count: number, retryAfterMs: number}>}
   */
  async checkSlidingWindowCounter(key, limit, windowMs) {
    const now = Date.now();
    const windowStart = Math.floor(now / windowMs) * windowMs;
    const currKey = `${key}:${windowStart}`;
    const prevKey = `${key}:${windowStart - windowMs}`;

    const [rejected, count] = await this._client.slidingWindowCounter(
      currKey,
      prevKey,
      now,
      windowMs,
      limit
    );

    // Approximate: retry at the start of the next window
    const retryAfterMs = rejected
      ? windowStart + windowMs - now
      : 0;

    return {
      allowed: rejected === 0,
      count: Number(count),
      retryAfterMs: Number(retryAfterMs),
    };
