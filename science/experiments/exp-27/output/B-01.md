### `src/crypto/KeyPair.js`

```javascript
'use strict';

const crypto = require('crypto');
const EC = require('elliptic').ec;

// secp256k1 — same curve Bitcoin uses
const ec = new EC('secp256k1');

class KeyPair {
  /**
   * Generate a brand-new keypair.
   * Returns { privateKey (hex), publicKey (hex compressed) }
   */
  static generate() {
    const keyPair = ec.genKeyPair();
    return new KeyPair(
      keyPair.getPrivate('hex'),
      keyPair.getPublic(true, 'hex')   // compressed public key
    );
  }

  /**
   * Restore a KeyPair from an existing private key hex string.
   */
  static fromPrivateKey(privateKeyHex) {
    const keyPair = ec.keyFromPrivate(privateKeyHex, 'hex');
    return new KeyPair(
      privateKeyHex,
      keyPair.getPublic(true, 'hex')
    );
  }

  constructor(privateKey, publicKey) {
    this.privateKey = privateKey;
    this.publicKey  = publicKey;
  }

  /**
   * Sign arbitrary data (hex string or Buffer).
   * Returns DER-encoded signature as hex.
   */
  sign(dataHash) {
    const keyPair = ec.keyFromPrivate(this.privateKey, 'hex');
    const signature = keyPair.sign(dataHash);
    return signature.toDER('hex');
  }

  /**
   * Derive a wallet address from a public key.
   * Real Bitcoin does Base58Check(RIPEMD160(SHA256(pubkey))).
   * We do a simplified but still cryptographically sound version.
   */
  getAddress() {
    return KeyPair.publicKeyToAddress(this.publicKey);
  }

  static publicKeyToAddress(publicKeyHex) {
    // SHA-256 then RIPEMD-160, prefixed with '0x' for readability
    const sha256Hash = crypto
      .createHash('sha256')
      .update(Buffer.from(publicKeyHex, 'hex'))
      .digest();

    const ripemd160Hash = crypto
      .createHash('ripemd160')
      .update(sha256Hash)
      .digest('hex');

    return '0x' + ripemd160Hash;
  }

  /**
   * Verify a signature against a public key and data hash.
   */
  static verify(publicKeyHex, dataHash, signatureHex) {
    try {
      const key = ec.keyFromPublic(publicKeyHex, 'hex');
      return key.verify(dataHash, signatureHex);
    } catch (err) {
      // Malformed signature or key — treat as invalid
      return false;
    }
  }
}

module.exports = { KeyPair, ec };
```

### `src/crypto/MerkleTree.js`

```javascript
'use strict';

const crypto = require('crypto');

/**
 * Binary Merkle Tree over transaction IDs.
 *
 * Properties:
 *  - Deterministic: same set of txids always produces the same root.
 *  - Tamper-evident: changing any tx changes the root.
 *  - Efficient proofs: O(log n) inclusion proofs.
 */
class MerkleTree {
  constructor(leaves) {
    if (!leaves || leaves.length === 0) {
      this.root = this._hash('');
      this.layers = [];
      return;
    }

    // Hash each leaf if not already a 64-char hex string
    const hashedLeaves = leaves.map(leaf =>
      leaf.length === 64 ? leaf : this._hash(leaf)
    );

    this.layers = this._buildTree(hashedLeaves);
    this.root   = this.layers[this.layers.length - 1][0];
  }

  _hash(data) {
    // Double-SHA256 like Bitcoin
    const first  = crypto.createHash('sha256').update(data).digest();
    const second = crypto.createHash('sha256').update(first).digest('hex');
    return second;
  }

  _hashPair(left, right) {
    // Canonical ordering: always hash smaller || larger to prevent
    // second-preimage attacks on unordered pairs
    const combined = left + right;
    return this._hash(combined);
  }

  _buildTree(leaves) {
    const layers = [leaves];
    let current  = leaves;

    while (current.length > 1) {
      const next = [];

      for (let i = 0; i < current.length; i += 2) {
        const left  = current[i];
        // If odd number of nodes, duplicate the last one (Bitcoin convention)
        const right = current[i + 1] || current[i];
        next.push(this._hashPair(left, right));
      }

      layers.push(next);
      current = next;
    }

    return layers;
  }

  /**
   * Generate an inclusion proof for the leaf at `index`.
   * Returns an array of { position: 'left'|'right', hash } objects.
   */
  getProof(index) {
    const proof = [];
    let idx = index;

    for (let i = 0; i < this.layers.length - 1; i++) {
      const layer     = this.layers[i];
      const isRight   = idx % 2 === 0;
      const siblingIdx = isRight ? idx + 1 : idx - 1;
      const sibling   = layer[siblingIdx] || layer[idx]; // duplicate if missing

      proof.push({
        position: isRight ? 'right' : 'left',
        hash: sibling,
      });

      idx = Math.floor(idx / 2);
    }

    return proof;
  }

  /**
   * Verify an inclusion proof.
   */
  static verify(leafHash, proof, root) {
    let computed = leafHash;

    for (const { position, hash } of proof) {
      const combined = position === 'right'
        ? computed + hash
        : hash + computed;

      const first  = crypto.createHash('sha256').update(combined).digest();
      computed     = crypto.createHash('sha256').update(first).digest('hex');
    }

    return computed === root;
  }

  getRoot() {
    return this.root;
  }
}

module.exports = MerkleTree;
```

### `src/core/Transaction.js`

```javascript
'use strict';

const crypto  = require('crypto');
const { v4: uuidv4 } = require('uuid');
const { KeyPair }    = require('../crypto/KeyPair');

/**
 * UTXO-based transaction model (simplified Bitcoin-style).
 *
 * Each transaction consumes one or more UTXOs (inputs) and
 * produces one or more new UTXOs (outputs).
 */
class TransactionInput {
  constructor({ txOutputId, txOutputIndex, amount, address, publicKey }) {
    this.txOutputId    = txOutputId;    // ID of the transaction containing the UTXO
    this.txOutputIndex = txOutputIndex; // Index of the output in that transaction
    this.amount        = amount;        // Cached amount (for validation convenience)
    this.address       = address;       // Spender's address
    this.publicKey     = publicKey;     // Spender's public key (for sig verification)
    this.signature     = null;          // Set when signing
  }

  /**
   * The data that gets signed: txOutputId + txOutputIndex.
   * We do NOT sign the amount — it's derived from the UTXO set.
   */
  getSignatureData() {
    return this._hash(`${this.txOutputId}${this.txOutputIndex}`);
  }

  _hash(data) {
    return crypto.createHash('sha256').update(data).digest('hex');
  }

  sign(keyPair) {
    this.signature = keyPair.sign(this.getSignatureData());
  }

  validate() {
    if (!this.signature) {
      throw new Error(`Input ${this.txOutputId}:${this.txOutputIndex} is unsigned`);
    }
    const valid = KeyPair.verify(
      this.publicKey,
      this.getSignatureData(),
      this.signature
    );
    if (!valid) {
      throw new Error(`Invalid signature on input ${this.txOutputId}:${this.txOutputIndex}`);
    }
    return true;
  }
}

class TransactionOutput {
  constructor({ amount, address }) {
    this.amount  = amount;
    this.address = address;
  }
}

class Transaction {
  constructor({ id, inputs, outputs, timestamp, type = 'TRANSFER' }) {
    this.id        = id || Transaction.generateId(inputs, outputs);
    this.inputs    = inputs  || [];
    this.outputs   = outputs || [];
    this.timestamp = timestamp || Date.now();
    this.type      = type;   // 'TRANSFER' | 'COINBASE'
  }

  static generateId(inputs, outputs) {
    const data = JSON.stringify({ inputs, outputs, nonce: uuidv4() });
    return crypto.createHash('sha256').update(data).digest('hex');
  }

  /**
   * Create a standard transfer transaction.
   * Handles change output automatically.
   */
  static create({ senderKeyPair, recipientAddress, amount, utxos }) {
    const senderAddress = senderKeyPair.getAddress();

    // Select UTXOs (simple greedy selection)
    const { selectedUTXOs, totalInput } = Transaction._selectUTXOs(utxos, amount, senderAddress);

    if (totalInput < amount) {
      throw new Error(`Insufficient funds. Available: ${totalInput}, Required: ${amount}`);
    }

    // Build inputs
    const inputs = selectedUTXOs.map(utxo => new TransactionInput({
      txOutputId:    utxo.txOutputId,
      txOutputIndex: utxo.txOutputIndex,
      amount:        utxo.amount,
      address:       senderAddress,
      publicKey:     senderKeyPair.publicKey,
    }));

    // Build outputs
    const outputs = [new TransactionOutput({ amount, address: recipientAddress })];

    // Change output
    const change = totalInput - amount;
    if (change > 0) {
      outputs.push(new TransactionOutput({ amount: change, address: senderAddress }));
    }

    const tx = new Transaction({ inputs, outputs });

    // Sign all inputs
    inputs.forEach(input => input.sign(senderKeyPair));

    return tx;
  }

  /**
   * Create a coinbase transaction (mining reward, no inputs).
   */
  static createCoinbase({ minerAddress, reward, blockHeight }) {
    const output = new TransactionOutput({ amount: reward, address: minerAddress });
    const tx = new Transaction({
      id:      crypto.createHash('sha256').update(`coinbase:${blockHeight}:${minerAddress}`).digest('hex'),
      inputs:  [],
      outputs: [output],
      type:    'COINBASE',
    });
    return tx;
  }

  /**
   * Greedy UTXO selection — pick smallest UTXOs first to consolidate dust,
   * switch to largest-first once we have enough.
   */
  static _selectUTXOs(utxos, targetAmount, address) {
    const available = utxos.filter(u => u.address === address);
    available.sort((a, b) => a.amount - b.amount);

    const selectedUTXOs = [];
    let totalInput = 0;

    for (const utxo of available) {
      selectedUTXOs.push(utxo);
      totalInput += utxo.amount;
      if (totalInput >= targetAmount) break;
    }

    return { selectedUTXOs, totalInput };
  }

  /**
   * Validate transaction structure and signatures.
   * Does NOT validate against UTXO set — that's Blockchain's job.
   */
  validate() {
    if (this.type === 'COINBASE') {
      if (this.inputs.length !== 0) {
        throw new Error('Coinbase transaction must have no inputs');
      }
      if (this.outputs.length !== 1) {
