## src/crypto/KeyPair.js

```javascript
'use strict';

const crypto = require('crypto');
const { ec: EC } = require('elliptic');

// secp256k1 — same curve Bitcoin uses
const ec = new EC('secp256k1');

/**
 * All cryptographic primitives live here.
 * We use secp256k1 for signing and SHA-256 for hashing.
 * Never roll your own crypto — we delegate to battle-tested libs.
 */

/**
 * Hash any data deterministically.
 * Always returns a hex string.
 */
function sha256(data) {
  return crypto
    .createHash('sha256')
    .update(typeof data === 'string' ? data : JSON.stringify(data))
    .digest('hex');
}

/**
 * Double-SHA256 — used for block hashes (Bitcoin-style).
 * Protects against length-extension attacks.
 */
function doubleSha256(data) {
  const first = crypto
    .createHash('sha256')
    .update(typeof data === 'string' ? data : JSON.stringify(data))
    .digest();
  return crypto.createHash('sha256').update(first).digest('hex');
}

/**
 * Generate a new secp256k1 key pair.
 * Returns { privateKey, publicKey } as hex strings.
 */
function generateKeyPair() {
  const keyPair = ec.genKeyPair();
  return {
    privateKey: keyPair.getPrivate('hex'),
    publicKey: keyPair.getPublic('hex'),
  };
}

/**
 * Derive public key from private key.
 * Useful for wallet recovery from seed.
 */
function getPublicKey(privateKeyHex) {
  const keyPair = ec.keyFromPrivate(privateKeyHex, 'hex');
  return keyPair.getPublic('hex');
}

/**
 * Sign data with a private key.
 * Returns DER-encoded signature as hex.
 */
function sign(data, privateKeyHex) {
  const keyPair = ec.keyFromPrivate(privateKeyHex, 'hex');
  const hash = sha256(data);
  const signature = keyPair.sign(hash);
  return signature.toDER('hex');
}

/**
 * Verify a signature against data and a public key.
 * Returns boolean — never throws on invalid sig.
 */
function verify(data, signatureHex, publicKeyHex) {
  try {
    const key = ec.keyFromPublic(publicKeyHex, 'hex');
    const hash = sha256(data);
    return key.verify(hash, signatureHex);
  } catch {
    return false;
  }
}

/**
 * Derive a wallet address from a public key.
 * Address = last 20 bytes of SHA-256(publicKey), hex-encoded.
 * Real production systems use RIPEMD-160 + Base58Check.
 */
function publicKeyToAddress(publicKeyHex) {
  const hash = sha256(publicKeyHex);
  // Take last 40 hex chars = 20 bytes
  return '0x' + hash.slice(-40);
}

module.exports = {
  sha256,
  doubleSha256,
  generateKeyPair,
  getPublicKey,
  sign,
  verify,
  publicKeyToAddress,
};
```

## src/crypto/MerkleTree.js

```javascript
'use strict';

const { sha256 } = require('./KeyPair');

/**
 * Merkle Tree implementation.
 *
 * A Merkle tree lets us:
 * 1. Commit to a set of transactions with a single hash (the root).
 * 2. Prove a transaction is in a block with O(log n) hashes (proof).
 * 3. Detect any tampering with any transaction.
 *
 * If the number of leaves is odd, we duplicate the last leaf
 * (Bitcoin-compatible behavior).
 */
class MerkleTree {
  constructor(items) {
    if (!items || items.length === 0) {
      this.root = sha256('empty');
      this.layers = [];
      return;
    }

    // Leaf layer: hash each item
    const leaves = items.map((item) =>
      sha256(typeof item === 'string' ? item : JSON.stringify(item))
    );

    this.layers = this._buildTree(leaves);
    this.root = this.layers[this.layers.length - 1][0];
  }

  _buildTree(leaves) {
    const layers = [leaves];
    let current = leaves;

    while (current.length > 1) {
      const next = [];

      // Duplicate last element if odd count
      if (current.length % 2 !== 0) {
        current = [...current, current[current.length - 1]];
      }

      for (let i = 0; i < current.length; i += 2) {
        // Concatenate and hash pairs
        next.push(sha256(current[i] + current[i + 1]));
      }

      layers.push(next);
      current = next;
    }

    return layers;
  }

  getRoot() {
    return this.root;
  }

  /**
   * Generate a Merkle proof for item at index.
   * Returns an array of { hash, position } objects.
   * position: 'left' | 'right' — which side the sibling is on.
   */
  getProof(index) {
    if (this.layers.length === 0) return [];

    const proof = [];
    let currentIndex = index;

    for (let i = 0; i < this.layers.length - 1; i++) {
      const layer = this.layers[i];

      // Handle odd-length layers
      const layerWithDup =
        layer.length % 2 !== 0 ? [...layer, layer[layer.length - 1]] : layer;

      const isRightNode = currentIndex % 2 === 1;
      const siblingIndex = isRightNode ? currentIndex - 1 : currentIndex + 1;

      proof.push({
        hash: layerWithDup[siblingIndex],
        position: isRightNode ? 'left' : 'right',
      });

      currentIndex = Math.floor(currentIndex / 2);
    }

    return proof;
  }

  /**
   * Verify a Merkle proof.
   * Static so callers don't need to reconstruct the tree.
   */
  static verify(itemHash, proof, root) {
    let computedHash = itemHash;

    for (const { hash, position } of proof) {
      if (position === 'left') {
        computedHash = sha256(hash + computedHash);
      } else {
        computedHash = sha256(computedHash + hash);
      }
    }

    return computedHash === root;
  }
}

module.exports = MerkleTree;
```

## src/core/Transaction.js

```javascript
'use strict';

const { sha256, sign, verify, publicKeyToAddress } = require('../crypto/KeyPair');

/**
 * Transaction types supported by this chain.
 */
const TxType = {
  TRANSFER: 'TRANSFER',   // Standard value transfer
  COINBASE: 'COINBASE',   // Block reward — no signature required
};

/**
 * A Transaction represents a signed intent to move value
 * from one address to another.
 *
 * Structure:
 *   - id:        SHA-256 hash of the transaction content
 *   - type:      TxType
 *   - from:      sender address (null for coinbase)
 *   - to:        recipient address
 *   - amount:    value in base units (integer, no floats)
 *   - fee:       miner fee in base units
 *   - nonce:     sender's transaction count — prevents replay attacks
 *   - timestamp: creation time
 *   - publicKey: sender's public key (for signature verification)
 *   - signature: secp256k1 DER signature over the tx hash
 */
class Transaction {
  constructor({ type, from, to, amount, fee, nonce, timestamp, publicKey, data }) {
    this.type = type || TxType.TRANSFER;
    this.from = from || null;
    this.to = to;
    this.amount = amount;
    this.fee = fee || 0;
    this.nonce = nonce || 0;
    this.timestamp = timestamp || Date.now();
    this.publicKey = publicKey || null;
    this.data = data || null; // arbitrary payload for future smart contract use
    this.signature = null;

    // ID is computed over all fields except signature
    this.id = this._computeId();
  }

  /**
   * Canonical serialization for hashing and signing.
   * Must be deterministic — same fields, same order, always.
   */
  _getSignablePayload() {
    return JSON.stringify({
      type: this.type,
      from: this.from,
      to: this.to,
      amount: this.amount,
      fee: this.fee,
      nonce: this.nonce,
      timestamp: this.timestamp,
      data: this.data,
    });
  }

  _computeId() {
    return sha256(this._getSignablePayload());
  }

  /**
   * Sign the transaction with the sender's private key.
   * Must be called before submitting to mempool.
   */
  sign(privateKey) {
    if (this.type === TxType.COINBASE) {
      throw new Error('Coinbase transactions cannot be signed');
    }
    this.signature = sign(this._getSignablePayload(), privateKey);
    return this;
  }

  /**
   * Full validation of a transaction.
   * Returns { valid: boolean, error: string|null }
   */
  validate() {
    // Coinbase transactions have special rules
    if (this.type === TxType.COINBASE) {
      if (this.amount <= 0) return { valid: false, error: 'Coinbase amount must be positive' };
      if (!this.to) return { valid: false, error: 'Coinbase must have recipient' };
      return { valid: true, error: null };
    }

    // Standard transfer validation
    if (!this.from) return { valid: false, error: 'Missing sender' };
    if (!this.to) return { valid: false, error: 'Missing recipient' };
    if (this.from === this.to) return { valid: false, error: 'Cannot send to self' };
    if (!Number.isInteger(this.amount) || this.amount <= 0) {
      return { valid: false, error: 'Amount must be a positive integer' };
    }
    if (!Number.isInteger(this.fee) || this.fee < 0) {
      return { valid: false, error: 'Fee must be a non-negative integer' };
    }
    if (!this.publicKey) return { valid: false, error: 'Missing public key' };
    if (!this.signature) return { valid: false, error: 'Missing signature' };

    // Verify the address matches the public key
    const derivedAddress = publicKeyToAddress(this.publicKey);
    if (derivedAddress !== this.from) {
      return { valid: false, error: 'Public key does not match sender address' };
    }

    // Verify the cryptographic signature
    const sigValid = verify(this._getSignablePayload(), this.signature, this.publicKey);
    if (!sigValid) {
      return { valid: false, error: 'Invalid signature' };
    }

    return { valid: true, error: null };
  }

  /**
   * Create a coinbase transaction (block reward).
   */
  static createCoinbase(minerAddress, reward, blockHeight) {
    return new Transaction({
      type: TxType.COINBASE,
      from: null,
      to: minerAddress,
      amount: reward,
      fee: 0,
      nonce: blockHeight,
      data: `Block ${blockHeight} reward`,
    });
  }

  toJSON() {
    return {
      id: this.id,
      type: this.type,
      from: this.from,
      to: this.to,
      amount: this.amount,
      fee: this.fee,
      nonce: this.nonce,
      timestamp: this.timestamp,
      publicKey: this.publicKey,
      signature: this.signature,
      data: this.data,
    };
  }

  static fromJSON(obj) {
    const tx = new Transaction(obj);
    tx.id = obj.id;
    tx.signature = obj.signature;
    return tx;
  }
}

module.exports = { Transaction, TxType };
```
