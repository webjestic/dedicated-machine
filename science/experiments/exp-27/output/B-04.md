## src/crypto/KeyPair.js

```javascript
'use strict';

const { ec: EC } = require('elliptic');
const crypto = require('crypto');

// secp256k1 — same curve Bitcoin uses
const ec = new EC('secp256k1');

/**
 * Low-level cryptographic primitives.
 * All public keys are stored as compressed hex strings (33 bytes → 66 hex chars).
 * All private keys are stored as 32-byte hex strings.
 * Signatures are DER-encoded hex strings.
 */
class KeyPair {
  constructor() {
    this._keyPair = ec.genKeyPair();
  }

  /**
   * Restore a KeyPair from an existing private key hex string.
   */
  static fromPrivateKey(privateKeyHex) {
    const instance = Object.create(KeyPair.prototype);
    instance._keyPair = ec.keyFromPrivate(privateKeyHex, 'hex');
    return instance;
  }

  get privateKey() {
    return this._keyPair.getPrivate('hex');
  }

  get publicKey() {
    // Compressed public key (33 bytes)
    return this._keyPair.getPublic(true, 'hex');
  }

  /**
   * Sign arbitrary data. Data is SHA-256 hashed before signing.
   * Returns a DER-encoded signature as a hex string.
   */
  sign(data) {
    const hash = KeyPair.hash(data);
    const signature = this._keyPair.sign(hash);
    return signature.toDER('hex');
  }

  /**
   * Verify a DER-encoded hex signature against a public key and data.
   */
  static verify(publicKeyHex, data, signatureHex) {
    try {
      const key = ec.keyFromPublic(publicKeyHex, 'hex');
      const hash = KeyPair.hash(data);
      return key.verify(hash, signatureHex);
    } catch {
      return false;
    }
  }

  /**
   * Deterministic SHA-256 hash of any value.
   * Objects are JSON-stringified with sorted keys for determinism.
   */
  static hash(data) {
    const content =
      typeof data === 'object' ? JSON.stringify(KeyPair.sortObject(data)) : String(data);
    return crypto.createHash('sha256').update(content).digest('hex');
  }

  /**
   * Recursively sort object keys so hashing is deterministic regardless
   * of insertion order.
   */
  static sortObject(obj) {
    if (typeof obj !== 'object' || obj === null) return obj;
    if (Array.isArray(obj)) return obj.map(KeyPair.sortObject);
    return Object.keys(obj)
      .sort()
      .reduce((acc, key) => {
        acc[key] = KeyPair.sortObject(obj[key]);
        return acc;
      }, {});
  }

  /**
   * Derive a wallet address from a public key.
   * Real chains do HASH160 (SHA256 → RIPEMD160) + checksum + Base58Check.
   * We use a simplified but still collision-resistant version.
   */
  static publicKeyToAddress(publicKeyHex) {
    const sha256 = crypto.createHash('sha256').update(Buffer.from(publicKeyHex, 'hex')).digest();
    const ripemd160 = crypto.createHash('ripemd160').update(sha256).digest('hex');
    return '1' + ripemd160; // '1' prefix mimics mainnet P2PKH addresses
  }
}

module.exports = { KeyPair, ec };
```

## src/crypto/MerkleTree.js

```javascript
'use strict';

const crypto = require('crypto');

/**
 * Binary Merkle Tree implementation.
 *
 * Used to compute the Merkle root of a block's transactions.
 * This gives us:
 *  - O(log n) proof of inclusion (SPV proofs)
 *  - Any change to any transaction changes the root → block hash changes
 */
class MerkleTree {
  constructor(leaves) {
    if (!leaves || leaves.length === 0) {
      // Empty tree — use a well-known null hash
      this.root = '0'.repeat(64);
      return;
    }

    // Hash each leaf if not already a 64-char hex string
    let level = leaves.map((leaf) =>
      typeof leaf === 'string' && leaf.length === 64
        ? leaf
        : MerkleTree.hash(leaf)
    );

    while (level.length > 1) {
      level = MerkleTree.buildLevel(level);
    }

    this.root = level[0];
  }

  static buildLevel(nodes) {
    const next = [];
    for (let i = 0; i < nodes.length; i += 2) {
      const left = nodes[i];
      // Duplicate last node if odd number (Bitcoin convention)
      const right = i + 1 < nodes.length ? nodes[i + 1] : nodes[i];
      next.push(MerkleTree.hash(left + right));
    }
    return next;
  }

  static hash(data) {
    // Double SHA-256 (Bitcoin convention)
    const first = crypto.createHash('sha256').update(data).digest();
    return crypto.createHash('sha256').update(first).digest('hex');
  }

  /**
   * Generate a Merkle proof (array of sibling hashes + directions)
   * for the leaf at the given index.
   */
  static generateProof(leaves, index) {
    let level = leaves.map((leaf) =>
      typeof leaf === 'string' && leaf.length === 64 ? leaf : MerkleTree.hash(leaf)
    );

    const proof = [];

    while (level.length > 1) {
      const isRightNode = index % 2 === 1;
      const siblingIndex = isRightNode ? index - 1 : index + 1;

      if (siblingIndex < level.length) {
        proof.push({
          hash: level[siblingIndex],
          direction: isRightNode ? 'left' : 'right',
        });
      }

      index = Math.floor(index / 2);
      level = MerkleTree.buildLevel(level);
    }

    return proof;
  }

  /**
   * Verify a Merkle proof.
   */
  static verifyProof(leafHash, proof, root) {
    let current = leafHash;
    for (const { hash, direction } of proof) {
      current =
        direction === 'left'
          ? MerkleTree.hash(hash + current)
          : MerkleTree.hash(current + hash);
    }
    return current === root;
  }
}

module.exports = MerkleTree;
```

## src/core/Transaction.js

```javascript
'use strict';

const { v4: uuidv4 } = require('uuid');
const { KeyPair } = require('../crypto/KeyPair');
const { COINBASE_AMOUNT } = require('../../config');

/**
 * UTXO-based transaction model (simplified Bitcoin-style).
 *
 * Structure:
 *   inputs  → references to previous unspent outputs
 *   outputs → new UTXOs being created
 *
 * Each input must be signed by the owner of the referenced UTXO.
 * The sum of input amounts must equal the sum of output amounts
 * (miners collect fees implicitly via coinbase).
 */
class Transaction {
  constructor({ id, inputs, outputs, timestamp } = {}) {
    this.id = id || uuidv4();
    this.timestamp = timestamp || Date.now();
    this.inputs = inputs || [];   // Array of TxInput
    this.outputs = outputs || []; // Array of TxOutput
  }

  /**
   * Compute the signing hash for this transaction.
   * We sign over id + outputs to prevent malleability.
   */
  get signingHash() {
    return KeyPair.hash({
      id: this.id,
      outputs: this.outputs,
    });
  }

  /**
   * Sign all inputs that belong to the given wallet's public key.
   */
  signInputs(wallet) {
    this.inputs.forEach((input) => {
      if (input.address === wallet.address) {
        input.signature = wallet.sign(this.signingHash);
      }
    });
  }

  /**
   * Validate all inputs have valid signatures.
   */
  isValid(utxoSet) {
    // Coinbase transactions have no inputs to validate
    if (this.isCoinbase()) return this.validateCoinbase();

    // Check for duplicate inputs (double-spend within same tx)
    const inputIds = this.inputs.map((i) => `${i.txId}:${i.outputIndex}`);
    if (new Set(inputIds).size !== inputIds.length) {
      console.error('Transaction has duplicate inputs');
      return false;
    }

    let inputTotal = 0;
    for (const input of this.inputs) {
      // Look up the UTXO being spent
      const utxo = utxoSet.get(input.txId, input.outputIndex);
      if (!utxo) {
        console.error(`UTXO not found: ${input.txId}:${input.outputIndex}`);
        return false;
      }

      // Verify the signature
      if (!KeyPair.verify(utxo.address, this.signingHash, input.signature)) {
        console.error('Invalid input signature');
        return false;
      }

      inputTotal += utxo.amount;
    }

    const outputTotal = this.outputs.reduce((sum, o) => sum + o.amount, 0);

    // Input total must be >= output total (difference is miner fee)
    if (inputTotal < outputTotal) {
      console.error(`Input total (${inputTotal}) < output total (${outputTotal})`);
      return false;
    }

    return true;
  }

  isCoinbase() {
    return this.inputs.length === 0;
  }

  validateCoinbase() {
    if (this.outputs.length !== 1) return false;
    if (this.outputs[0].amount !== COINBASE_AMOUNT) return false;
    return true;
  }

  /**
   * Miner fee = inputs total - outputs total.
   * Requires the UTXO set to look up input amounts.
   */
  fee(utxoSet) {
    if (this.isCoinbase()) return 0;
    const inputTotal = this.inputs.reduce((sum, input) => {
      const utxo = utxoSet.get(input.txId, input.outputIndex);
      return sum + (utxo ? utxo.amount : 0);
    }, 0);
    const outputTotal = this.outputs.reduce((sum, o) => sum + o.amount, 0);
    return inputTotal - outputTotal;
  }

  toJSON() {
    return {
      id: this.id,
      timestamp: this.timestamp,
      inputs: this.inputs,
      outputs: this.outputs,
    };
  }

  static fromJSON(data) {
    return new Transaction(data);
  }

  /**
   * Create a coinbase transaction (block reward for miner).
   */
  static createCoinbase(minerAddress, blockIndex) {
    return new Transaction({
      id: KeyPair.hash(`coinbase:${minerAddress}:${blockIndex}:${Date.now()}`),
      inputs: [],
      outputs: [
        {
          address: minerAddress,
          amount: COINBASE_AMOUNT,
          index: 0,
        },
      ],
    });
  }
}

module.exports = Transaction;
```
