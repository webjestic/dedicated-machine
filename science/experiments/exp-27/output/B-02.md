## src/crypto/KeyPair.js

```javascript
'use strict';

/**
 * Cryptographic key pair management using secp256k1 (same curve as Bitcoin/Ethereum).
 * 
 * Why secp256k1:
 * - Battle-tested in production blockchain systems
 * - Efficient signature verification
 * - Deterministic signatures via RFC 6979 (no random nonce leakage)
 */

const { ec: EC } = require('elliptic');
const crypto = require('crypto');

const ec = new EC('secp256k1');

class KeyPair {
  constructor(privateKeyHex = null) {
    if (privateKeyHex) {
      this.keyPair = ec.keyFromPrivate(privateKeyHex, 'hex');
    } else {
      this.keyPair = ec.genKeyPair();
    }
  }

  /**
   * Returns compressed public key (33 bytes) as hex.
   * Compressed keys save space and are standard in production systems.
   */
  getPublicKey() {
    return this.keyPair.getPublic(true, 'hex');
  }

  getPrivateKey() {
    return this.keyPair.getPrivate('hex');
  }

  /**
   * Sign arbitrary data. Input is hashed with SHA-256 before signing.
   * Returns DER-encoded signature as hex string.
   * 
   * IMPORTANT: Never sign raw user input — always hash first.
   * This prevents length-extension attacks and normalizes input size.
   */
  sign(data) {
    const hash = KeyPair.hash(data);
    const signature = this.keyPair.sign(hash);
    return signature.toDER('hex');
  }

  /**
   * Static verification — does not require private key.
   * This is what nodes use to validate transactions.
   */
  static verify(publicKeyHex, data, signatureHex) {
    try {
      const key = ec.keyFromPublic(publicKeyHex, 'hex');
      const hash = KeyPair.hash(data);
      return key.verify(hash, signatureHex);
    } catch (err) {
      // Malformed keys/signatures should fail gracefully, not throw
      return false;
    }
  }

  /**
   * Derive a wallet address from a public key.
   * 
   * Production pattern (simplified Bitcoin-style):
   * 1. SHA-256 the compressed public key
   * 2. RIPEMD-160 the result (hash160)
   * 3. Return hex — in production you'd add version byte + Base58Check
   */
  static publicKeyToAddress(publicKeyHex) {
    const sha256 = crypto
      .createHash('sha256')
      .update(Buffer.from(publicKeyHex, 'hex'))
      .digest();

    const ripemd160 = crypto
      .createHash('ripemd160')
      .update(sha256)
      .digest('hex');

    return ripemd160;
  }

  static hash(data) {
    const normalized = typeof data === 'string' ? data : JSON.stringify(data);
    return crypto.createHash('sha256').update(normalized).digest('hex');
  }

  static fromPublicKey(publicKeyHex) {
    return ec.keyFromPublic(publicKeyHex, 'hex');
  }
}

module.exports = { KeyPair, ec };
```

## src/crypto/MerkleTree.js

```javascript
'use strict';

/**
 * Merkle Tree implementation for transaction integrity.
 *
 * Why Merkle Trees:
 * - O(log n) proof of inclusion (SPV clients don't need full block)
 * - Any tampered transaction invalidates the root
 * - Standard in Bitcoin, Ethereum, and virtually all production chains
 *
 * Construction: pair and hash leaves bottom-up.
 * If odd number of nodes at a level, duplicate the last node (Bitcoin convention).
 */

const crypto = require('crypto');

class MerkleTree {
  constructor(transactions) {
    if (!transactions || transactions.length === 0) {
      this.root = this._hash('');
      this.layers = [];
      return;
    }

    // Leaf nodes are hashes of serialized transactions
    const leaves = transactions.map(tx =>
      this._hash(typeof tx === 'string' ? tx : JSON.stringify(tx))
    );

    this.layers = this._buildTree(leaves);
    this.root = this.layers[this.layers.length - 1][0];
  }

  _buildTree(leaves) {
    const layers = [leaves];
    let currentLayer = leaves;

    while (currentLayer.length > 1) {
      currentLayer = this._buildLayer(currentLayer);
      layers.push(currentLayer);
    }

    return layers;
  }

  _buildLayer(nodes) {
    const layer = [];

    for (let i = 0; i < nodes.length; i += 2) {
      const left = nodes[i];
      // Duplicate last node if odd count — Bitcoin convention
      const right = i + 1 < nodes.length ? nodes[i + 1] : nodes[i];
      layer.push(this._hash(left + right));
    }

    return layer;
  }

  /**
   * Generate a Merkle proof for a transaction at a given index.
   * Returns array of { hash, position } objects.
   * Verifier can recompute root without knowing other transactions.
   */
  getProof(index) {
    if (this.layers.length === 0) return [];

    const proof = [];
    let currentIndex = index;

    for (let i = 0; i < this.layers.length - 1; i++) {
      const layer = this.layers[i];
      const isRightNode = currentIndex % 2 === 1;
      const siblingIndex = isRightNode ? currentIndex - 1 : currentIndex + 1;

      if (siblingIndex < layer.length) {
        proof.push({
          hash: layer[siblingIndex],
          position: isRightNode ? 'left' : 'right'
        });
      }

      currentIndex = Math.floor(currentIndex / 2);
    }

    return proof;
  }

  /**
   * Verify a Merkle proof without the full tree.
   * Used by light clients (SPV nodes).
   */
  static verify(leafData, proof, root) {
    const tree = new MerkleTree([]);
    let hash = tree._hash(
      typeof leafData === 'string' ? leafData : JSON.stringify(leafData)
    );

    for (const { hash: siblingHash, position } of proof) {
      if (position === 'left') {
        hash = tree._hash(siblingHash + hash);
      } else {
        hash = tree._hash(hash + siblingHash);
      }
    }

    return hash === root;
  }

  getRoot() {
    return this.root;
  }

  _hash(data) {
    // Double-SHA256 — same as Bitcoin, prevents length-extension attacks
    const first = crypto.createHash('sha256').update(data).digest();
    return crypto.createHash('sha256').update(first).digest('hex');
  }
}

module.exports = MerkleTree;
```

## src/core/Transaction.js

```javascript
'use strict';

/**
 * UTXO-based transaction model (Bitcoin-style).
 *
 * Why UTXO over account model:
 * - Transactions are stateless — easier to verify in parallel
 * - No replay attacks across forks (each UTXO can only be spent once)
 * - Better privacy (each output can use a fresh address)
 *
 * Structure:
 *   inputs:  references to existing UTXOs being consumed
 *   outputs: new UTXOs being created
 *
 * Validation rules enforced here:
 *   1. Inputs must reference existing, unspent UTXOs
 *   2. Sum(inputs) >= Sum(outputs) — difference is miner fee
 *   3. Each input must be signed by the UTXO owner
 *   4. No double-spending within the same transaction
 */

const { v4: uuidv4 } = require('uuid');
const { KeyPair } = require('../crypto/KeyPair');

const COINBASE_AMOUNT = 50; // Block reward in base units

class TransactionInput {
  constructor({ txOutputId, txOutputIndex, amount, address, signature = null }) {
    this.txOutputId = txOutputId;       // ID of the transaction containing the UTXO
    this.txOutputIndex = txOutputIndex; // Index of the output in that transaction
    this.amount = amount;               // Cached for convenience — must match UTXO
    this.address = address;             // Public key of the UTXO owner
    this.signature = signature;         // Signature over the output reference
  }

  /**
   * Sign this input using the owner's private key.
   * We sign the UTXO reference (id + index), not the amount —
   * the amount is committed to by referencing the specific UTXO.
   */
  sign(keyPair, transactionId) {
    // Verify the signer owns this input
    if (keyPair.getPublicKey() !== this.address) {
      throw new Error(
        `Cannot sign input: key mismatch. Expected ${this.address}`
      );
    }
    this.signature = keyPair.sign(this._signingData(transactionId));
  }

  verify(transactionId) {
    return KeyPair.verify(
      this.address,
      this._signingData(transactionId),
      this.signature
    );
  }

  /**
   * The data being signed commits to:
   * - The transaction ID (prevents signature reuse across transactions)
   * - The specific UTXO being spent (txOutputId + index)
   */
  _signingData(transactionId) {
    return JSON.stringify({
      transactionId,
      txOutputId: this.txOutputId,
      txOutputIndex: this.txOutputIndex
    });
  }
}

class TransactionOutput {
  constructor({ amount, address }) {
    this.amount = amount;
    this.address = address; // Recipient's public key (compressed hex)
  }
}

class Transaction {
  constructor({ id, inputs, outputs, timestamp, type = 'REGULAR' }) {
    this.id = id || uuidv4();
    this.inputs = inputs.map(i =>
      i instanceof TransactionInput ? i : new TransactionInput(i)
    );
    this.outputs = outputs.map(o =>
      o instanceof TransactionOutput ? o : new TransactionOutput(o)
    );
    this.timestamp = timestamp || Date.now();
    this.type = type; // 'REGULAR' | 'COINBASE'
  }

  /**
   * Create a coinbase transaction — the block reward paid to the miner.
   * Coinbase has no inputs (coins are created from nothing, up to the reward limit).
   * In Bitcoin, coinbase also carries arbitrary data (e.g., Satoshi's genesis message).
   */
  static createCoinbase(minerAddress, blockIndex) {
    const output = new TransactionOutput({
      amount: COINBASE_AMOUNT,
      address: minerAddress
    });

    return new Transaction({
      id: uuidv4(),
      inputs: [{
        txOutputId: '',
        txOutputIndex: blockIndex, // Ensures uniqueness per block
        amount: COINBASE_AMOUNT,
        address: '',
        signature: 'coinbase'
      }],
      outputs: [output],
      type: 'COINBASE'
    });
  }

  /**
   * Create and sign a regular transaction.
   *
   * @param {KeyPair} senderKeyPair - Sender's key pair
   * @param {string} recipientAddress - Recipient's public key
   * @param {number} amount - Amount to send
   * @param {Array} utxoSet - Available UTXOs for the sender
   * @returns {Transaction}
   */
  static create(senderKeyPair, recipientAddress, amount, utxoSet) {
    const senderAddress = senderKeyPair.getPublicKey();

    // Find UTXOs owned by sender
    const senderUTXOs = utxoSet.filter(u => u.address === senderAddress);
    const totalAvailable = senderUTXOs.reduce((sum, u) => sum + u.amount, 0);

    if (totalAvailable < amount) {
      throw new Error(
        `Insufficient funds. Available: ${totalAvailable}, Required: ${amount}`
      );
    }

    // Greedy UTXO selection — collect until we have enough
    let accumulated = 0;
    const selectedUTXOs = [];

    for (const utxo of senderUTXOs) {
      if (accumulated >= amount) break;
      selectedUTXOs.push(utxo);
      accumulated += utxo.amount;
    }

    // Build inputs from selected UTXOs
    const inputs = selectedUTXOs.map(utxo => new TransactionInput({
      txOutputId: utxo.txOutputId,
      txOutputIndex: utxo.txOutputIndex,
      amount: utxo.amount,
      address: senderAddress
    }));

    // Build outputs: recipient gets `amount`, sender gets change
    const outputs = [
      new TransactionOutput({ amount, address: recipientAddress })
    ];

    const change = accumulated - amount;
    if (change > 0) {
      outputs.push(new TransactionOutput({ amount: change, address: senderAddress }));
    }

    const tx = new Transaction({ id: uuidv4(), inputs, outputs });
