## src/utils/crypto.js

```javascript
/**
 * Cryptographic primitives.
 *
 * Design decisions:
 * - SHA-256 via Node's built-in crypto (no external dep, FIPS-auditable)
 * - Double-SHA256 for block/tx hashing (Bitcoin-style, prevents length extension)
 * - secp256k1 for signatures (same curve as Bitcoin/Ethereum, hardware wallet compatible)
 */

import { createHash, randomBytes } from 'crypto';
import secp256k1 from 'secp256k1';

/**
 * Double-SHA256. Used for all block and transaction hashing.
 * Single SHA256 is vulnerable to length-extension attacks.
 */
export function hash256(data) {
  const buf = Buffer.isBuffer(data) ? data : Buffer.from(data, 'utf8');
  return createHash('sha256')
    .update(createHash('sha256').update(buf).digest())
    .digest('hex');
}

/**
 * Single SHA256. Used internally for Merkle tree intermediate nodes.
 */
export function sha256(data) {
  const buf = Buffer.isBuffer(data) ? data : Buffer.from(data, 'utf8');
  return createHash('sha256').update(buf).digest('hex');
}

/**
 * RIPEMD160(SHA256(data)) — produces 20-byte address hash.
 * This is the standard Bitcoin address derivation step.
 */
export function hash160(data) {
  const buf = Buffer.isBuffer(data) ? data : Buffer.from(data, 'utf8');
  const sha = createHash('sha256').update(buf).digest();
  return createHash('ripemd160').update(sha).digest('hex');
}

/**
 * Generate a new secp256k1 private key.
 * Loops until a valid key is found (probability of invalid key is ~1/2^128).
 */
export function generatePrivateKey() {
  let privKey;
  do {
    privKey = randomBytes(32);
  } while (!secp256k1.privateKeyVerify(privKey));
  return privKey;
}

/**
 * Derive compressed public key from private key.
 */
export function getPublicKey(privateKey) {
  const privBuf = Buffer.isBuffer(privateKey)
    ? privateKey
    : Buffer.from(privateKey, 'hex');
  return Buffer.from(secp256k1.publicKeyCreate(privBuf, true));
}

/**
 * Sign a 32-byte message hash.
 * Returns { r, s, recovery } as hex strings.
 *
 * CRITICAL: The message must be a hash, not raw data.
 * Signing raw data leaks information about the private key.
 */
export function sign(messageHash, privateKey) {
  const msgBuf = Buffer.isBuffer(messageHash)
    ? messageHash
    : Buffer.from(messageHash, 'hex');
  const privBuf = Buffer.isBuffer(privateKey)
    ? privateKey
    : Buffer.from(privateKey, 'hex');

  if (msgBuf.length !== 32) {
    throw new Error('Message must be exactly 32 bytes (a hash)');
  }

  const { signature, recovery } = secp256k1.ecdsaSign(msgBuf, privBuf);
  return {
    signature: Buffer.from(signature).toString('hex'),
    recovery,
  };
}

/**
 * Verify a signature against a public key.
 */
export function verify(messageHash, signatureHex, publicKeyHex) {
  try {
    const msgBuf = Buffer.from(messageHash, 'hex');
    const sigBuf = Buffer.from(signatureHex, 'hex');
    const pubBuf = Buffer.from(publicKeyHex, 'hex');
    return secp256k1.ecdsaVerify(sigBuf, msgBuf, pubBuf);
  } catch {
    // secp256k1 throws on malformed input — treat as invalid
    return false;
  }
}

/**
 * Recover public key from signature + recovery id.
 * Used to verify transactions without storing the full public key in the UTXO set.
 */
export function recoverPublicKey(messageHash, signatureHex, recovery) {
  const msgBuf = Buffer.from(messageHash, 'hex');
  const sigBuf = Buffer.from(signatureHex, 'hex');
  return Buffer.from(secp256k1.ecdsaRecover(sigBuf, recovery, msgBuf, true)).toString('hex');
}
```

```javascript
/**
 * Input validation utilities.
 *
 * Every external input (RPC, P2P) goes through these validators before
 * touching any business logic. This is the first line of DoS defense.
 */

export const MAX_BLOCK_SIZE_BYTES = 1_000_000; // 1 MB
export const MAX_TX_SIZE_BYTES = 100_000;       // 100 KB
export const MAX_INPUTS_PER_TX = 1000;
export const MAX_OUTPUTS_PER_TX = 1000;
export const MAX_SCRIPT_SIZE = 10_000;
export const MAX_MONEY = 21_000_000 * 100_000_000; // 21M coins in satoshis

export function isValidHex(str, expectedLength = null) {
  if (typeof str !== 'string') return false;
  if (!/^[0-9a-f]+$/i.test(str)) return false;
  if (expectedLength !== null && str.length !== expectedLength) return false;
  return true;
}

export function isValidAmount(amount) {
  if (typeof amount !== 'number' && typeof amount !== 'bigint') return false;
  const n = BigInt(amount);
  return n > 0n && n <= BigInt(MAX_MONEY);
}

export function isValidAddress(address) {
  if (typeof address !== 'string') return false;
  // 40-char hex (20-byte hash160)
  return isValidHex(address, 40);
}

export function isPositiveInteger(n) {
  return Number.isInteger(n) && n > 0;
}

/**
 * Validate raw transaction object shape before deserialization.
 * Prevents prototype pollution and oversized inputs.
 */
export function validateRawTransaction(raw) {
  const errors = [];

  if (!raw || typeof raw !== 'object' || Array.isArray(raw)) {
    return ['Transaction must be a plain object'];
  }

  if (!Array.isArray(raw.inputs)) errors.push('inputs must be array');
  if (!Array.isArray(raw.outputs)) errors.push('outputs must be array');

  if (raw.inputs?.length > MAX_INPUTS_PER_TX) {
    errors.push(`Too many inputs: max ${MAX_INPUTS_PER_TX}`);
  }
  if (raw.outputs?.length > MAX_OUTPUTS_PER_TX) {
    errors.push(`Too many outputs: max ${MAX_OUTPUTS_PER_TX}`);
  }
  if (raw.outputs?.length === 0) {
    errors.push('Transaction must have at least one output');
  }

  return errors;
}

/**
 * Validate raw block object shape.
 */
export function validateRawBlock(raw) {
  const errors = [];

  if (!raw || typeof raw !== 'object') return ['Block must be an object'];

  const requiredFields = ['version', 'previousHash', 'merkleRoot',
                          'timestamp', 'difficulty', 'nonce', 'transactions'];
  for (const field of requiredFields) {
    if (!(field in raw)) errors.push(`Missing field: ${field}`);
  }

  if (!Array.isArray(raw.transactions)) {
    errors.push('transactions must be array');
  }

  if (raw.transactions?.length === 0) {
    errors.push('Block must have at least one transaction (coinbase)');
  }

  const serialized = JSON.stringify(raw);
  if (serialized.length > MAX_BLOCK_SIZE_BYTES) {
    errors.push(`Block exceeds max size: ${MAX_BLOCK_SIZE_BYTES} bytes`);
  }

  return errors;
}
```

/**
 * Merkle Tree implementation.
 *
 * Used to:
 * 1. Compute the merkle root stored in block headers
 * 2. Generate inclusion proofs (SPV / light client support)
 *
 * Design: Bitcoin-style binary Merkle tree.
 * Duplicate last node when count is odd (matches Bitcoin behavior).
 *
 * KNOWN VULNERABILITY (flagged, not fixed here):
 * Bitcoin's Merkle tree has a known CVE where certain transaction sets
 * produce the same root. A production system should use a commitment
 * scheme that includes the tree depth. Flagged for future hardening.
 */

import { sha256 } from '../utils/crypto.js';

export class MerkleTree {
  constructor(leaves) {
    if (!leaves || leaves.length === 0) {
      throw new Error('MerkleTree requires at least one leaf');
    }
    this.leaves = leaves.map(l => sha256(l));
    this.layers = this._buildLayers(this.leaves);
  }

  _buildLayers(leaves) {
    const layers = [leaves];
    let current = leaves;

    while (current.length > 1) {
      const next = [];
      for (let i = 0; i < current.length; i += 2) {
        const left = current[i];
        // Duplicate last node if odd count
        const right = current[i + 1] ?? current[i];
        next.push(sha256(left + right));
      }
      layers.push(next);
      current = next;
    }

    return layers;
  }

  getRoot() {
    return this.layers[this.layers.length - 1][0];
  }

  /**
   * Generate a Merkle proof for the leaf at `index`.
   * Returns array of { hash, position: 'left'|'right' }.
   *
   * A verifier can recompute the root from the leaf + proof
   * without having all other leaves.
   */
  getProof(index) {
    const proof = [];
    let currentIndex = index;

    for (let layer = 0; layer < this.layers.length - 1; layer++) {
      const currentLayer = this.layers[layer];
      const isRightNode = currentIndex % 2 === 1;
      const siblingIndex = isRightNode ? currentIndex - 1 : currentIndex + 1;

      if (siblingIndex < currentLayer.length) {
        proof.push({
          hash: currentLayer[siblingIndex],
          position: isRightNode ? 'left' : 'right',
        });
      } else {
        // Odd node — sibling is itself
        proof.push({
          hash: currentLayer[currentIndex],
          position: isRightNode ? 'left' : 'right',
        });
      }

      currentIndex = Math.floor(currentIndex / 2);
    }

    return proof;
  }

    let hash = sha256(leafData);
