```javascript
// src/utils/crypto.js
const crypto = require('crypto');
const EC = require('elliptic').ec;

const ec = new EC('secp256k1'); // Same curve as Bitcoin

/**
 * Creates a SHA-256 hash of the given data
 */
function createHash(data) {
  return crypto
    .createHash('sha256')
    .update(typeof data === 'string' ? data : JSON.stringify(data))
    .digest('hex');
}

/**
 * Creates a double SHA-256 hash (more secure, used in Bitcoin)
 */
function createDoubleHash(data) {
  const firstHash = createHash(data);
  return createHash(firstHash);
}

/**
 * Generates a new key pair for wallet
 */
function generateKeyPair() {
  const keyPair = ec.genKeyPair();
  return {
    privateKey: keyPair.getPrivate('hex'),
    publicKey: keyPair.getPublic('hex'),
  };
}

/**
 * Signs data with a private key
 */
function signData(data, privateKeyHex) {
  const keyPair = ec.keyFromPrivate(privateKeyHex, 'hex');
  const hash = createHash(data);
  const signature = keyPair.sign(hash);
  return signature.toDER('hex');
}

/**
 * Verifies a signature against public key
 */
function verifySignature(data, signature, publicKeyHex) {
  try {
    const keyPair = ec.keyFromPublic(publicKeyHex, 'hex');
    const hash = createHash(data);
    return keyPair.verify(hash, signature);
  } catch (error) {
    return false;
  }
}

/**
 * Creates a Merkle root from an array of transactions
 */
function createMerkleRoot(transactions) {
  if (transactions.length === 0) return createHash('');
  if (transactions.length === 1) return createHash(transactions[0]);

  const hashes = transactions.map((tx) => createHash(tx));
  return buildMerkleTree(hashes);
}

function buildMerkleTree(hashes) {
  if (hashes.length === 1) return hashes[0];

  const newLevel = [];
  for (let i = 0; i < hashes.length; i += 2) {
    const left = hashes[i];
    const right = i + 1 < hashes.length ? hashes[i + 1] : hashes[i]; // Duplicate last if odd
    newLevel.push(createHash(left + right));
  }

  return buildMerkleTree(newLevel);
}

module.exports = {
  createHash,
  createDoubleHash,
  generateKeyPair,
  signData,
  verifySignature,
  createMerkleRoot,
  ec,
};
```

```javascript
// src/Transaction.js
const { createHash, signData, verifySignature } = require('./utils/crypto');
const { v4: uuidv4 } = require('uuid');

class TransactionInput {
  constructor(txOutputId, txOutputIndex, amount, fromAddress) {
    this.txOutputId = txOutputId;       // Reference to previous transaction output
    this.txOutputIndex = txOutputIndex; // Index of the output in previous transaction
    this.amount = amount;
    this.fromAddress = fromAddress;
    this.signature = null;
  }

  sign(privateKey) {
    const dataToSign = this.getSignableData();
    this.signature = signData(dataToSign, privateKey);
  }

  getSignableData() {
    return JSON.stringify({
      txOutputId: this.txOutputId,
      txOutputIndex: this.txOutputIndex,
      amount: this.amount,
      fromAddress: this.fromAddress,
    });
  }

  isValid() {
    if (!this.signature) return false;
    return verifySignature(
      this.getSignableData(),
      this.signature,
      this.fromAddress
    );
  }
}

class TransactionOutput {
  constructor(toAddress, amount) {
    this.id = uuidv4();
    this.toAddress = toAddress;
    this.amount = amount;
    this.spent = false;
  }
}

class Transaction {
  constructor(fromAddress, toAddress, amount, fee = 0) {
    this.id = uuidv4();
    this.fromAddress = fromAddress; // null for coinbase (mining reward)
    this.toAddress = toAddress;
    this.amount = amount;
    this.fee = fee;
    this.timestamp = Date.now();
    this.inputs = [];
    this.outputs = [];
    this.signature = null;
    this.type = fromAddress === null ? 'COINBASE' : 'REGULAR';
  }

  /**
   * Creates a hash of the transaction data
   */
  calculateHash() {
    return createHash({
      id: this.id,
      fromAddress: this.fromAddress,
      toAddress: this.toAddress,
      amount: this.amount,
      fee: this.fee,
      timestamp: this.timestamp,
      type: this.type,
    });
  }

  /**
   * Signs the transaction with sender's private key
   */
  sign(privateKey) {
    if (this.type === 'COINBASE') {
      throw new Error('Cannot sign a coinbase transaction');
    }
    this.signature = signData(this.calculateHash(), privateKey);
  }

  /**
   * Validates the transaction
   */
  isValid() {
    // Coinbase transactions are always valid (mining rewards)
    if (this.type === 'COINBASE') {
      return this.amount > 0 && this.toAddress !== null;
    }

    if (!this.signature) {
      throw new Error('Transaction has no signature');
    }

    if (!this.fromAddress) {
      throw new Error('Transaction has no sender address');
    }

    if (this.amount <= 0) {
      throw new Error('Transaction amount must be greater than 0');
    }

    if (this.fee < 0) {
      throw new Error('Transaction fee cannot be negative');
    }

    if (!verifySignature(this.calculateHash(), this.signature, this.fromAddress)) {
      throw new Error('Transaction signature is invalid');
    }

    return true;
  }

  toJSON() {
    return {
      id: this.id,
      fromAddress: this.fromAddress,
      toAddress: this.toAddress,
      amount: this.amount,
      fee: this.fee,
      timestamp: this.timestamp,
      type: this.type,
      signature: this.signature,
    };
  }

  static createCoinbase(minerAddress, reward) {
    const tx = new Transaction(null, minerAddress, reward);
    tx.type = 'COINBASE';
    return tx;
  }
}

module.exports = { Transaction, TransactionInput, TransactionOutput };
```

## 4. Block Class

```javascript
// src/Block.js
const { createHash, createMerkleRoot } = require('./utils/crypto');

class Block {
  constructor(index, transactions, previousHash = '', difficulty = 4) {
    this.index = index;
    this.timestamp = Date.now();
    this.transactions = transactions;
    this.previousHash = previousHash;
    this.difficulty = difficulty;
    this.nonce = 0;
    this.merkleRoot = this.calculateMerkleRoot();
    this.hash = this.calculateHash();
    this.minedAt = null;
  }

  /**
   * Calculates the Merkle root of all transactions
   */
  calculateMerkleRoot() {
    const txIds = this.transactions.map((tx) =>
      typeof tx === 'object' ? tx.id || JSON.stringify(tx) : tx
    );
    return createMerkleRoot(txIds);
  }

  /**
   * Calculates the hash of the block
   */
  calculateHash() {
    return createHash({
      index: this.index,
      timestamp: this.timestamp,
      merkleRoot: this.merkleRoot,
      previousHash: this.previousHash,
      difficulty: this.difficulty,
      nonce: this.nonce,
    });
  }

  /**
   * Proof of Work mining algorithm
   * Finds a hash that starts with the required number of zeros
   */
  mine() {
    const target = '0'.repeat(this.difficulty);
    const startTime = Date.now();

    console.log(`\n⛏️  Mining block #${this.index}...`);
    console.log(`   Difficulty: ${this.difficulty} (target: ${target}...)`);

    while (!this.hash.startsWith(target)) {
      this.nonce++;
      this.hash = this.calculateHash();
    }

    this.minedAt = Date.now();
    const duration = (this.minedAt - startTime) / 1000;

    console.log(`   ✅ Block mined!`);
    console.log(`   Hash: ${this.hash}`);
    console.log(`   Nonce: ${this.nonce}`);
    console.log(`   Time: ${duration.toFixed(3)}s`);

    return this;
  }

  /**
   * Validates the block's hash and proof of work
   */
  isValid() {
    // Check hash integrity
    if (this.hash !== this.calculateHash()) {
      return { valid: false, reason: 'Block hash is invalid' };
    }

    // Check proof of work
    const target = '0'.repeat(this.difficulty);
    if (!this.hash.startsWith(target)) {
      return { valid: false, reason: 'Block does not meet difficulty target' };
    }

    // Check Merkle root
    if (this.merkleRoot !== this.calculateMerkleRoot()) {
      return { valid: false, reason: 'Merkle root is invalid' };
    }

    return { valid: true, reason: 'Block is valid' };
  }

  /**
   * Validates all transactions in the block
   */
  validateTransactions() {
    for (const tx of this.transactions) {
      try {
        if (!tx.isValid()) {
          return { valid: false, reason: `Transaction ${tx.id} is invalid` };
        }
      } catch (error) {
        return { valid: false, reason: error.message };
      }
    }
    return { valid: true, reason: 'All transactions are valid' };
  }

  toJSON() {
    return {
      index: this.index,
      timestamp: this.timestamp,
      transactions: this.transactions.map((tx) =>
        typeof tx.toJSON === 'function' ? tx.toJSON() : tx
      ),
      previousHash: this.previousHash,
      hash: this.hash,
      merkleRoot: this.merkleRoot,
      nonce: this.nonce,
      difficulty: this.difficulty,
      minedAt: this.minedAt,
    };
  }

  static get GENESIS_DATA() {
    return {
      index: 0,
      timestamp: 1700000000000,
      previousHash: '0'.repeat(64),
      difficulty: 1,
      nonce: 0,
      data: 'Genesis Block - The Beginning',
    };
  }
}

module.exports = Block;
```

const Block = require('./Block');
const { Transaction } = require('./Transaction');

class Blockchain {
  constructor(options = {}) {
    this.chain = [];
    this.pendingTransactions = [];
    this.difficulty = options.difficulty || 3;
    this.miningReward = options.miningReward || 50;
    this.maxTransactionsPerBlock = options.maxTransactionsPerBlock || 10;
    this.blockTime = options.blockTime || 10000; // Target 10 seconds per block
    this.difficultyAdjustmentInterval = options.difficultyAdjustmentInterval || 10;
    this.transactionFeePercent = options.transactionFeePercent || 0.01;

    // UTXO (Unspent Transaction Output) set
    this.utxoSet = new Map();

    // Initialize with genesis block
    this.chain.push(this.createGenesisBlock());
  }

  /**
   * Creates the genesis (first) block
   */
  createGenesisBlock() {
    const genesisData = Block.GENESIS_DATA;
    const block = new Block(0, [], genesisData.previousHash, 1);
    block.timestamp = genesisData.timestamp;
    block.hash = block.calculateHash();
    block.minedAt = genesisData.timestamp;
    console.log('🌱 Genesis block created');
    return block;
  }

  /**
   * Returns the latest block in the chain
   */
  getLatestBlock() {
    return this.chain[this.chain.length - 1];
  }

  /**
   * Adds a transaction to the pending pool
   */
  addTransaction(transaction) {
    // Validate transaction
    try {
      transaction.isValid();
    } catch (error) {
      throw new Error(`Invalid transaction: ${error.message}`);
    }

    // Check for duplicate transactions
    const isDuplicate = this.pendingTransactions.some(
      (tx) => tx.id === transaction.id
    );
    if (isDuplicate) {
      throw new Error('Duplicate transaction detected');
    }

    // Check sender balance (skip for coinbase)
    if (transaction.type !== 'COINBASE') {
      const senderBalance = this.getBalance(transaction.fromAddress);
      const totalCost = transaction.amount + transaction.fee;

      if (senderBalance < totalCost) {
        throw new Error(
          `Insufficient balance. Required: ${totalCost}, Available: ${senderBalance}`
        );
      }
    }

    this.pendingTransactions.push(transaction);
    console.log(`📝 Transaction added to pool: ${transaction.id.slice(0, 8)}...`);
    return transaction.id;
  }

   * Mines pending transactions and adds a new block
