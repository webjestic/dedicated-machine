## src/crypto/KeyPair.js

```javascript
'use strict';

const crypto = require('crypto');
const { ec: EC } = require('elliptic');

// secp256k1 — same curve Bitcoin uses
const ec = new EC('secp256k1');

class KeyPair {
  /**
   * Generate a brand new keypair.
   * Returns a KeyPair instance wrapping the elliptic key object.
   */
  static generate() {
    const key = ec.genKeyPair();
    return new KeyPair(key);
  }

  /**
   * Restore a KeyPair from a hex-encoded private key.
   */
  static fromPrivateKey(privateKeyHex) {
    const key = ec.keyFromPrivate(privateKeyHex, 'hex');
    return new KeyPair(key);
  }

  /**
   * Restore a public-key-only KeyPair (verify-only, cannot sign).
   */
  static fromPublicKey(publicKeyHex) {
    const key = ec.keyFromPublic(publicKeyHex, 'hex');
    return new KeyPair(key);
  }

  constructor(ecKey) {
    this._key = ecKey;
  }

  get privateKeyHex() {
    return this._key.getPrivate('hex');
  }

  get publicKeyHex() {
    // Compressed public key — 33 bytes, standard for Bitcoin-style addresses
    return this._key.getPublic(true, 'hex');
  }

  /**
   * Derive a wallet address from the public key.
   * Address = Base58Check( RIPEMD160( SHA256( pubKey ) ) )
   * We approximate with hex-encoded double-hash for simplicity while
   * keeping the cryptographic properties intact.
   */
  get address() {
    return KeyPair.publicKeyToAddress(this.publicKeyHex);
  }

  /**
   * Sign arbitrary data. Returns DER-encoded signature as hex.
   * @param {string|Buffer} data
   */
  sign(data) {
    const hash = KeyPair.hash(data);
    const signature = this._key.sign(hash);
    return signature.toDER('hex');
  }

  /**
   * Verify a DER-encoded hex signature against data and this key's public key.
   */
  verify(data, signatureHex) {
    const hash = KeyPair.hash(data);
    return this._key.verify(hash, signatureHex);
  }

  // ─── Static helpers ──────────────────────────────────────────────────────────

  /**
   * Verify a signature given raw public key, data, and signature.
   */
  static verify(publicKeyHex, data, signatureHex) {
    try {
      const key = ec.keyFromPublic(publicKeyHex, 'hex');
      const hash = KeyPair.hash(data);
      return key.verify(hash, signatureHex);
    } catch {
      return false;
    }
  }

  /**
   * SHA-256 hash returning a Buffer. Accepts string or Buffer.
   */
  static hash(data) {
    const input = Buffer.isBuffer(data) ? data : Buffer.from(data, 'utf8');
    return crypto.createHash('sha256').update(input).digest();
  }

  /**
   * Double-SHA-256 — used for block and transaction hashing (Bitcoin convention).
   */
  static hash256(data) {
    const first = crypto.createHash('sha256').update(data).digest();
    return crypto.createHash('sha256').update(first).digest('hex');
  }

  /**
   * Derive an address from a compressed public key hex string.
   * SHA256 → RIPEMD160 → hex string (20 bytes = 40 hex chars)
   */
  static publicKeyToAddress(publicKeyHex) {
    const pubKeyBuffer = Buffer.from(publicKeyHex, 'hex');
    const sha256 = crypto.createHash('sha256').update(pubKeyBuffer).digest();
    const ripemd160 = crypto.createHash('ripemd160').update(sha256).digest('hex');
    // Prefix with version byte '00' (mainnet P2PKH)
    return '00' + ripemd160;
  }

  toJSON() {
    return {
      publicKey: this.publicKeyHex,
      address: this.address,
    };
  }
}

module.exports = { KeyPair, ec };
```

## src/crypto/MerkleTree.js

```javascript
'use strict';

const { KeyPair } = require('./KeyPair');

/**
 * Binary Merkle Tree implementation.
 *
 * Properties:
 *  - Leaves are double-SHA256 hashes of serialized transactions
 *  - Odd number of leaves: duplicate the last leaf (Bitcoin convention)
 *  - Returns the Merkle root as a hex string
 *  - Supports proof generation and verification for SPV clients
 */
class MerkleTree {
  /**
   * @param {string[]} leaves - Array of hex strings (transaction IDs / hashes)
   */
  constructor(leaves) {
    if (!leaves || leaves.length === 0) {
      // Empty block gets a zero hash root
      this.root = '0'.repeat(64);
      this.layers = [];
      return;
    }

    this.layers = [leaves.slice()];
    this._build();
    this.root = this.layers[this.layers.length - 1][0];
  }

  _build() {
    let currentLayer = this.layers[0];

    while (currentLayer.length > 1) {
      const nextLayer = [];

      for (let i = 0; i < currentLayer.length; i += 2) {
        const left = currentLayer[i];
        // Duplicate last node if odd count (Bitcoin convention)
        const right = i + 1 < currentLayer.length ? currentLayer[i + 1] : left;
        nextLayer.push(this._hashPair(left, right));
      }

      this.layers.push(nextLayer);
      currentLayer = nextLayer;
    }
  }

  _hashPair(left, right) {
    const combined = left + right;
    return KeyPair.hash256(Buffer.from(combined, 'hex'));
  }

  /**
   * Generate a Merkle proof for a leaf at a given index.
   * Returns an array of { hash, position } objects ('left' | 'right').
   * A verifier can recompute the root from the leaf + proof.
   */
  generateProof(leafIndex) {
    if (this.layers.length === 0) return [];

    const proof = [];
    let index = leafIndex;

    for (let i = 0; i < this.layers.length - 1; i++) {
      const layer = this.layers[i];
      const isRightNode = index % 2 === 1;
      const siblingIndex = isRightNode ? index - 1 : index + 1;

      if (siblingIndex < layer.length) {
        proof.push({
          hash: layer[siblingIndex],
          position: isRightNode ? 'left' : 'right',
        });
      } else {
        // Duplicate — sibling is self
        proof.push({
          hash: layer[index],
          position: isRightNode ? 'left' : 'right',
        });
      }

      index = Math.floor(index / 2);
    }

    return proof;
  }

  /**
   * Verify a Merkle proof.
   * @param {string} leafHash
   * @param {Array} proof - from generateProof()
   * @param {string} expectedRoot
   */
  static verifyProof(leafHash, proof, expectedRoot) {
    let computedHash = leafHash;

    for (const { hash, position } of proof) {
      const combined =
        position === 'left'
          ? hash + computedHash
          : computedHash + hash;

      computedHash = KeyPair.hash256(Buffer.from(combined, 'hex'));
    }

    return computedHash === expectedRoot;
  }

  static computeRoot(txIds) {
    const tree = new MerkleTree(txIds);
    return tree.root;
  }
}

module.exports = { MerkleTree };
```

'use strict';

const { v4: uuidv4 } = require('uuid');
const { KeyPair } = require('../crypto/KeyPair');
const config = require('../../config');

/**
 * UTXO-based transaction model (similar to Bitcoin).
 *
 * A transaction consumes unspent outputs (inputs) and creates new outputs.
 * Each input references a previous transaction output and carries a signature
 * proving the spender controls the referenced output's address.
 *
 * Special case: coinbase transactions have no inputs and are created by miners.
 */
class Transaction {
  /**
   * @param {TxInput[]}  inputs
   * @param {TxOutput[]} outputs
   * @param {string}     [id]       - omit to auto-generate
   * @param {number}     [timestamp]
   */
  constructor(inputs, outputs, id = null, timestamp = null) {
    this.id        = id || Transaction.generateId(inputs, outputs);
    this.inputs    = inputs;
    this.outputs   = outputs;
    this.timestamp = timestamp || Date.now();
  }

  // ─── Factory methods ──────────────────────────────────────────────────────────

  /**
   * Create a coinbase transaction (mining reward).
   * No inputs — the miner receives COINBASE_AMOUNT to their address.
   */
  static createCoinbase(minerAddress, blockHeight) {
    const reward = Transaction.getBlockReward(blockHeight);

    const output = new TxOutput(reward, minerAddress);
    // Coinbase input encodes block height as data (prevents duplicate txids)
    const input  = new TxInput(null, -1, `coinbase:${blockHeight}:${uuidv4()}`, null);

    const tx = new Transaction([input], [output]);
    tx.isCoinbase = true;
    return tx;
  }

   * Create and sign a standard transaction.
   *
   * @param {KeyPair}     senderKeyPair
   * @param {string}      recipientAddress
   * @param {number}      amount

   * @param {number}      fee
   */

    if (totalInput < totalNeeded) {
      throw new Error(
        `Insufficient funds. Available: ${totalInput}, Required: ${totalNeeded}`
      );

    );

    const change  = totalInput - totalNeeded;
