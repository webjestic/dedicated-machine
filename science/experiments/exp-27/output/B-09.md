## src/crypto/KeyPair.js

```javascript
'use strict';

/**
 * Cryptographic key pair management using secp256k1 (same curve as Bitcoin/Ethereum).
 * 
 * In production you would also want:
 * - BIP-32 HD key derivation
 * - BIP-39 mnemonic support
 * - Hardware wallet (HSM) integration
 * - Key encryption at rest
 */

const { ec: EC } = require('elliptic');
const crypto = require('crypto');

// secp256k1 is the industry standard for blockchain key pairs
const ec = new EC('secp256k1');

class KeyPair {
  /**
   * Generate a new random key pair.
   * Uses Node.js crypto.randomBytes under the hood via elliptic,
   * which sources entropy from the OS CSPRNG.
   */
  static generate() {
    const keyPair = ec.genKeyPair();
    return new KeyPair(keyPair);
  }

  /**
   * Restore a key pair from a hex-encoded private key.
   * @param {string} privateKeyHex
   */
  static fromPrivateKey(privateKeyHex) {
    const keyPair = ec.keyFromPrivate(privateKeyHex, 'hex');
    return new KeyPair(keyPair);
  }

  /**
   * Restore a public-key-only pair for signature verification.
   * Used when you have a sender's address but not their private key.
   * @param {string} publicKeyHex
   */
  static fromPublicKey(publicKeyHex) {
    const keyPair = ec.keyFromPublic(publicKeyHex, 'hex');
    return new KeyPair(keyPair);
  }

  constructor(ecKeyPair) {
    this._keyPair = ecKeyPair;
  }

  /**
   * The public key in uncompressed hex form (130 hex chars, prefix 04).
   * This is what gets broadcast on the network and used as an address basis.
   */
  get publicKey() {
    return this._keyPair.getPublic('hex');
  }

  /**
   * The private key in hex form. NEVER log or transmit this.
   */
  get privateKey() {
    return this._keyPair.getPrivate('hex');
  }

  /**
   * Derive a wallet address from the public key.
   * We use a double-hash (SHA256 then RIPEMD160-equivalent via SHA256)
   * similar to Bitcoin's P2PKH scheme, then take the last 20 bytes.
   * 
   * Production systems add a version byte and Base58Check encoding.
   */
  get address() {
    const pubKeyBuffer = Buffer.from(this.publicKey, 'hex');
    const sha256Hash = crypto.createHash('sha256').update(pubKeyBuffer).digest();
    const ripemd160Hash = crypto.createHash('ripemd160').update(sha256Hash).digest();
    return ripemd160Hash.toString('hex');
  }

  /**
   * Sign arbitrary data (typically a transaction hash).
   * Returns DER-encoded signature as hex string.
   * 
   * IMPORTANT: We hash the data before signing. Never sign raw user data.
   * @param {string} dataHex - hex-encoded data to sign
   * @returns {string} DER-encoded signature hex
   */
  sign(dataHex) {
    const hash = crypto.createHash('sha256').update(dataHex, 'hex').digest();
    const signature = this._keyPair.sign(hash);
    return signature.toDER('hex');
  }

  /**
   * Verify a signature against data and this key pair's public key.
   * @param {string} dataHex - original data that was signed
   * @param {string} signatureHex - DER-encoded signature
   * @returns {boolean}
   */
  verify(dataHex, signatureHex) {
    try {
      const hash = crypto.createHash('sha256').update(dataHex, 'hex').digest();
      return this._keyPair.verify(hash, signatureHex);
    } catch {
      // Malformed signature bytes should not crash the node
      return false;
    }
  }

  toJSON() {
    return {
      publicKey: this.publicKey,
      address: this.address,
      // Never include privateKey in serialization
    };
  }
}

module.exports = { KeyPair, ec };
```

## src/crypto/MerkleTree.js

```javascript
'use strict';

/**
 * Binary Merkle Tree implementation.
 *
 * The Merkle root is stored in each block header, allowing:
 * - O(log n) transaction inclusion proofs (SPV clients)
 * - Tamper detection: changing any transaction changes the root
 * - Efficient partial block validation
 *
 * Production improvements:
 * - Merkle proofs / inclusion proof generation
 * - Sparse Merkle Trees for state tries (like Ethereum's Patricia trie)
 */

const crypto = require('crypto');

class MerkleTree {
  /**
   * @param {string[]} leaves - array of hex strings (typically transaction IDs)
   */
  constructor(leaves) {
    if (!leaves || leaves.length === 0) {
      // Empty block: use a zero hash as the root
      this.root = '0'.repeat(64);
      this.layers = [];
      return;
    }

    // Hash each leaf to ensure uniform 32-byte inputs at every level
    const hashedLeaves = leaves.map(leaf => this._hash(leaf));
    this.layers = this._buildTree(hashedLeaves);
    this.root = this.layers[this.layers.length - 1][0];
  }

  /**
   * Build the tree bottom-up.
   * Each layer halves the number of nodes by hashing pairs.
   * If a layer has an odd number of nodes, the last node is duplicated
   * (Bitcoin convention).
   */
  _buildTree(leaves) {
    const layers = [leaves];
    let currentLayer = leaves;

    while (currentLayer.length > 1) {
      const nextLayer = [];

      for (let i = 0; i < currentLayer.length; i += 2) {
        const left = currentLayer[i];
        // Duplicate last node if odd count (Bitcoin convention)
        const right = i + 1 < currentLayer.length
          ? currentLayer[i + 1]
          : currentLayer[i];

        nextLayer.push(this._hashPair(left, right));
      }

      layers.push(nextLayer);
      currentLayer = nextLayer;
    }

    return layers;
  }

  /**
   * Generate a Merkle proof for a leaf at the given index.
   * The proof is an array of { hash, direction } objects.
   * A verifier can recompute the root using only the leaf and this proof.
   *
   * @param {number} leafIndex
   * @returns {Array<{hash: string, direction: 'left'|'right'}>}
   */
  getProof(leafIndex) {
    if (this.layers.length === 0) return [];

    const proof = [];
    let index = leafIndex;

    for (let i = 0; i < this.layers.length - 1; i++) {
      const layer = this.layers[i];
      const isRightNode = index % 2 === 1;
      const siblingIndex = isRightNode ? index - 1 : index + 1;

      if (siblingIndex < layer.length) {
        proof.push({
          hash: layer[siblingIndex],
          direction: isRightNode ? 'left' : 'right',
        });
      }

      // Move up to parent index
      index = Math.floor(index / 2);
    }

    return proof;
  }

  /**
   * Verify a Merkle proof.
   * @param {string} leaf - original leaf value (will be hashed)
   * @param {Array<{hash: string, direction: string}>} proof
   * @param {string} expectedRoot
   * @returns {boolean}
   */
  static verify(leaf, proof, expectedRoot) {
    let computedHash = MerkleTree._staticHash(leaf);

    for (const { hash, direction } of proof) {
      if (direction === 'left') {
        computedHash = MerkleTree._staticHashPair(hash, computedHash);
      } else {
        computedHash = MerkleTree._staticHashPair(computedHash, hash);
      }
    }

    return computedHash === expectedRoot;
  }

  _hash(data) {
    return MerkleTree._staticHash(data);
  }

  _hashPair(left, right) {
    return MerkleTree._staticHashPair(left, right);
  }

  static _staticHash(data) {
    return crypto.createHash('sha256').update(data).digest('hex');
  }

  static _staticHashPair(left, right) {
    // Double-SHA256 at internal nodes (Bitcoin convention, prevents length extension attacks)
    const combined = left + right;
    const first = crypto.createHash('sha256').update(combined).digest();
    return crypto.createHash('sha256').update(first).digest('hex');
  }
}

module.exports = MerkleTree;
```

## src/core/Transaction.js

```javascript
'use strict';

/**
 * UTXO-based transaction model (similar to Bitcoin).
 *
 * Each transaction:
 * - Consumes one or more UTXOs as inputs (proving ownership via signature)
 * - Creates one or more new UTXOs as outputs
 * - The difference between input value and output value is the miner fee
 *
 * Production additions needed:
 * - Script system (P2PKH, P2SH, multisig, timelocks)
 * - Segregated Witness (SegWit) for transaction malleability fix
 * - Replace-By-Fee (RBF) signaling
 * - Locktime / sequence numbers
 */

const crypto = require('crypto');
const { v4: uuidv4 } = require('uuid');

// Coinbase transactions reward miners. This is the block subsidy.
// In Bitcoin this halves every 210,000 blocks.
const COINBASE_AMOUNT = 50;

class TransactionOutput {
  /**
   * @param {string} address - recipient's wallet address
   * @param {number} amount - value in base units (like satoshis)
   */
  constructor(address, amount) {
    this.address = address;
    this.amount = amount;
  }
}

class TransactionInput {
  /**
   * References a previous transaction output (UTXO) and provides
   * a signature proving the spender owns it.
   *
   * @param {string} txOutId - transaction ID containing the UTXO
   * @param {number} txOutIndex - index of the output in that transaction
   * @param {string} signature - signature over the transaction data
   * @param {string} publicKey - spender's public key
   */
  constructor(txOutId, txOutIndex, signature = '', publicKey = '') {
    this.txOutId = txOutId;
    this.txOutIndex = txOutIndex;
    this.signature = signature;
    this.publicKey = publicKey;
  }
}

class Transaction {
  constructor() {
    this.id = '';
    this.inputs = [];
    this.outputs = [];
    this.timestamp = Date.now();
  }

  /**
   * The transaction ID is a hash of all inputs and outputs.
   * It does NOT include signatures (to allow signing after ID computation).
   * 
   * NOTE: This is a simplification. Bitcoin uses a more complex serialization.
   * SegWit separates witness data to fix transaction malleability.
   */
  computeId() {
    const inputData = this.inputs
      .map(i => i.txOutId + i.txOutIndex)
      .join('');
    const outputData = this.outputs
      .map(o => o.address + o.amount)
      .join('');

    this.id = crypto
      .createHash('sha256')
      .update(inputData + outputData + this.timestamp)
      .digest('hex');

    return this.id;
  }

  /**
   * The data that gets signed for each input.
   * Each input signs the transaction ID to bind it to this specific transaction.
   * This prevents an attacker from replaying inputs in a different transaction.
   */
  getSigningData() {
    return this.id;
  }

  /**
   * Sign all inputs using the provided key pair.
   * In a multi-input transaction where inputs come from different addresses,
   * each input would need to be signed by its respective owner.
   * 
   * @param {KeyPair} keyPair
   */
  signInputs(keyPair) {
    this.inputs = this.inputs.map(input => {
      input.signature = keyPair.sign(this.id);
      input.publicKey = keyPair.publicKey;
      return input;
    });
  }

  /**
   * Validate all inputs have valid signatures.
   * @param {Map<string, TransactionOutput>} utxoSet - current unspent outputs
   * @returns {{ valid: boolean, error?: string }}
   */
  validate(utxoSet) {
    // Validate structure
    if (!this.id || !Array.isArray(this.inputs) || !Array.isArray(this.outputs)) {
      return { valid: false, error: 'Malformed transaction structure' };
    }

    // Validate each input's signature
    for (const input of this.inputs) {
      const referencedOutput = utxoSet.get(`${input.txOutId}:${input.txOutIndex}`);

      if (!
