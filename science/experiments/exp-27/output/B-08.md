```javascript
// src/crypto/KeyPair.js
'use strict';

const { ec: EC } = require('elliptic');
const crypto = require('crypto');

// Use secp256k1 - same curve as Bitcoin
const ec = new EC('secp256k1');

class KeyPair {
  /**
   * Generate a new cryptographic key pair
   */
  static generate() {
    const keyPair = ec.genKeyPair();
    return new KeyPair(keyPair);
  }

  /**
   * Restore a KeyPair from a private key hex string
   */
  static fromPrivateKey(privateKeyHex) {
    const keyPair = ec.keyFromPrivate(privateKeyHex, 'hex');
    return new KeyPair(keyPair);
  }

  /**
   * Restore a KeyPair from a public key hex string (verify-only)
   */
  static fromPublicKey(publicKeyHex) {
    const keyPair = ec.keyFromPublic(publicKeyHex, 'hex');
    return new KeyPair(keyPair);
  }

  constructor(ecKeyPair) {
    this._keyPair = ecKeyPair;
  }

  /**
   * Get compressed public key as hex string
   */
  getPublicKey() {
    return this._keyPair.getPublic(true, 'hex');
  }

  /**
   * Get private key as hex string
   */
  getPrivateKey() {
    if (!this._keyPair.getPrivate()) {
      throw new Error('No private key available in this KeyPair');
    }
    return this._keyPair.getPrivate('hex');
  }

  /**
   * Derive wallet address from public key
   * Address = RIPEMD160(SHA256(publicKey)) encoded as hex
   * In production you'd add version bytes and Base58Check encoding
   */
  getAddress() {
    const publicKeyBuffer = Buffer.from(this.getPublicKey(), 'hex');
    
    // SHA256 first
    const sha256Hash = crypto
      .createHash('sha256')
      .update(publicKeyBuffer)
      .digest();
    
    // Then RIPEMD160
    const ripemd160Hash = crypto
      .createHash('ripemd160')
      .update(sha256Hash)
      .digest('hex');
    
    return ripemd160Hash;
  }

  /**
   * Sign data - returns DER-encoded signature as hex
   */
  sign(dataHex) {
    if (!this._keyPair.getPrivate()) {
      throw new Error('Cannot sign without private key');
    }
    
    const signature = this._keyPair.sign(dataHex);
    return signature.toDER('hex');
  }

  /**
   * Verify a signature against this key pair's public key
   */
  verify(dataHex, signatureHex) {
    try {
      return this._keyPair.verify(dataHex, signatureHex);
    } catch {
      return false;
    }
  }

  /**
   * Static verify - verify with any public key
   */
  static verify(dataHex, signatureHex, publicKeyHex) {
    try {
      const keyPair = ec.keyFromPublic(publicKeyHex, 'hex');
      return keyPair.verify(dataHex, signatureHex);
    } catch {
      return false;
    }
  }

  toJSON() {
    return {
      publicKey: this.getPublicKey(),
      address: this.getAddress(),
    };
  }
}

module.exports = { KeyPair, ec };
```

```javascript
// src/crypto/MerkleTree.js
'use strict';

const crypto = require('crypto');

/**
 * Merkle Tree implementation for transaction integrity verification.
 * 
 * The Merkle root allows light clients to verify transaction inclusion
 * without downloading the full block - this is fundamental to SPV.
 */
class MerkleTree {
  constructor(leaves) {
    if (!leaves || leaves.length === 0) {
      throw new Error('MerkleTree requires at least one leaf');
    }
    
    // Hash all leaves if they aren't already hashed
    this.leaves = leaves.map(leaf => 
      typeof leaf === 'string' && leaf.length === 64 
        ? leaf 
        : MerkleTree.hash(leaf)
    );
    
    this.layers = [this.leaves];
    this._buildTree();
  }

  static hash(data) {
    // Double SHA256 - same as Bitcoin
    const first = crypto
      .createHash('sha256')
      .update(typeof data === 'string' ? data : JSON.stringify(data))
      .digest();
    
    return crypto
      .createHash('sha256')
      .update(first)
      .digest('hex');
  }

  _buildTree() {
    let currentLayer = this.leaves;
    
    while (currentLayer.length > 1) {
      const nextLayer = [];
      
      for (let i = 0; i < currentLayer.length; i += 2) {
        const left = currentLayer[i];
        // If odd number of nodes, duplicate the last one (Bitcoin standard)
        const right = currentLayer[i + 1] || currentLayer[i];
        
        nextLayer.push(MerkleTree.hash(left + right));
      }
      
      this.layers.push(nextLayer);
      currentLayer = nextLayer;
    }
  }

  /**
   * Get the Merkle root - this goes into the block header
   */
  getRoot() {
    return this.layers[this.layers.length - 1][0];
  }

  /**
   * Generate a Merkle proof for a leaf at given index.
   * Proof allows verification of inclusion without full tree.
   */
  getProof(leafIndex) {
    if (leafIndex < 0 || leafIndex >= this.leaves.length) {
      throw new Error(`Invalid leaf index: ${leafIndex}`);
    }
    
    const proof = [];
    let index = leafIndex;
    
    for (let i = 0; i < this.layers.length - 1; i++) {
      const layer = this.layers[i];
      const isRightNode = index % 2 === 1;
      const siblingIndex = isRightNode ? index - 1 : index + 1;
      
      if (siblingIndex < layer.length) {
        proof.push({
          hash: layer[siblingIndex],
          position: isRightNode ? 'left' : 'right',
        });
      }
      
      index = Math.floor(index / 2);
    }
    
    return proof;
  }

  /**
   * Verify a Merkle proof
   */
  static verifyProof(leafHash, proof, root) {
    let computedHash = leafHash;
    
    for (const { hash, position } of proof) {
      if (position === 'left') {
        computedHash = MerkleTree.hash(hash + computedHash);
      } else {
        computedHash = MerkleTree.hash(computedHash + hash);
      }
    }
    
    return computedHash === root;
  }

  /**
   * Convenience: compute root from an array of transaction IDs
   */
  static computeRoot(txIds) {
    if (!txIds || txIds.length === 0) {
      return '0'.repeat(64);
    }
    const tree = new MerkleTree(txIds);
    return tree.getRoot();
  }
}

module.exports = MerkleTree;

## Step 4: Transaction System (UTXO Model)

```javascript
// src/core/Transaction.js
'use strict';

const crypto = require('crypto');
const { KeyPair } = require('../crypto/KeyPair');

/**
 * Transaction Input - references a previous UTXO
 */
class TxInput {
  constructor({ txId, outputIndex, signature, publicKey }) {
    this.txId = txId;           // Hash of the transaction being spent
    this.outputIndex = outputIndex; // Which output of that transaction
    this.signature = signature || null;
    this.publicKey = publicKey || null;
  }

  /**
   * Sign this input with the owner's private key
   * Signs the transaction hash (commitment to all outputs)
   */
  sign(keyPair, txHash) {
    this.publicKey = keyPair.getPublicKey();
    this.signature = keyPair.sign(txHash);
  }

  /**
   * Verify the signature on this input
   */
  verify(txHash) {
    if (!this.signature || !this.publicKey) return false;
    return KeyPair.verify(txHash, this.signature, this.publicKey);
  }

  toJSON() {
    return {
      txId: this.txId,
      outputIndex: this.outputIndex,
      signature: this.signature,
      publicKey: this.publicKey,
    };
  }
}

/**
 * Transaction Output - defines who can spend and how much
 */
class TxOutput {
  constructor({ amount, address }) {
    if (!Number.isInteger(amount) || amount <= 0) {
      throw new Error(`Invalid output amount: ${amount}`);
    }
    if (!address || typeof address !== 'string') {
      throw new Error('Output requires a valid address');
    }
    
    this.amount = amount;   // In satoshis
    this.address = address; // Recipient's address (hash of public key)
  }

  toJSON() {
    return {
      amount: this.amount,
      address: this.address,
    };
  }
}

/**
 * Transaction - transfers value between addresses using UTXO model
 */
class Transaction {
  constructor({ id, inputs, outputs, timestamp, type }) {
    this.id = id || null;
    this.inputs = (inputs || []).map(i => 
      i instanceof TxInput ? i : new TxInput(i)
    );
    this.outputs = (outputs || []).map(o => 
      o instanceof TxOutput ? o : new TxOutput(o)
    );
    this.timestamp = timestamp || Date.now();
    this.type = type || 'TRANSFER'; // 'TRANSFER' | 'COINBASE'
  }

  /**
   * Compute the transaction ID - hash of all inputs and outputs
   * Note: computed WITHOUT signatures so ID is stable
   */
  computeId() {
    const data = {
      inputs: this.inputs.map(i => ({
        txId: i.txId,
        outputIndex: i.outputIndex,
      })),
      outputs: this.outputs.map(o => o.toJSON()),
      timestamp: this.timestamp,
      type: this.type,
    };
    
    const serialized = JSON.stringify(data);
    
    // Double SHA256
    const first = crypto.createHash('sha256').update(serialized).digest();
    return crypto.createHash('sha256').update(first).digest('hex');
  }

  /**
   * Get the signing hash - what inputs actually sign
   * This is the commitment to outputs (prevents output tampering)
   */
  getSigningHash() {
    const data = {
      inputs: this.inputs.map(i => ({
        txId: i.txId,
        outputIndex: i.outputIndex,
      })),
      outputs: this.outputs.map(o => o.toJSON()),
