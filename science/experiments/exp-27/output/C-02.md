# Production Blockchain Implementation in Node.js

Let me start by identifying what a production deployment actually requires that typical "build me a blockchain" specs omit, then build accordingly.

## Production Gap Analysis

Before writing a line of code, here is what production requires that the stated requirements do not mention:

| Gap | Risk if Ignored | My Decision |
|-----|----------------|-------------|
| **Persistent storage** | Node restart loses entire chain | Implement LevelDB persistence |
| **P2P networking** | Single node, not a blockchain | Implement libp2p-style TCP mesh |
| **Mempool management** | No transaction ordering, spam vectors | Implement fee-priority mempool with eviction |
| **Fork resolution** | Chain splits permanently | Implement longest-chain rule with reorganization |
| **Wallet / key management** | No way to sign transactions | Implement secp256k1 HD wallet |
| **DoS protection** | Trivially crashed by peers | Rate limiting, message size caps |
| **Merkle tree** | Can't prove tx inclusion without full block | Implement Merkle root in block header |
| **Difficulty adjustment** | Fixed difficulty breaks under variable hashrate | Implement epoch-based retargeting |
| **Transaction validation** | Double-spend possible | UTXO set with full script validation |
| **API surface** | No way to interact with node | REST + WebSocket API |
| **Structured logging** | Impossible to operate in production | Winston with log levels |
| **Graceful shutdown** | Data corruption on SIGTERM | Drain handlers, flush storage |
| **Configuration management** | Hardcoded values break deployments | Environment-driven config |
| **Test coverage** | Regressions ship silently | Unit + integration tests |

**Flagged for requester (out of scope for this implementation but required before mainnet):**
- BFT/PoS consensus (this implements Proof of Work)
- Slashing conditions
- Light client / SPV support
- Smart contract execution engine (EVM-compatible)
- Formal security audit
- Checkpoint infrastructure to prevent long-range attacks

### `src/crypto/hash.js`

```javascript
'use strict';

const crypto = require('crypto');

/**
 * All hashing goes through this module.
 * Using double-SHA256 (Bitcoin-style) for block and transaction IDs.
 * This prevents length-extension attacks that single SHA256 is vulnerable to.
 */

/**
 * Double-SHA256 hash. Returns hex string.
 * @param {string|Buffer} data
 * @returns {string} 64-character hex string
 */
function sha256d(data) {
  const first = crypto.createHash('sha256').update(data).digest();
  return crypto.createHash('sha256').update(first).digest('hex');
}

/**
 * Single SHA256. Returns hex string.
 * Used for non-consensus-critical operations (e.g., peer message IDs).
 * @param {string|Buffer} data
 * @returns {string}
 */
function sha256(data) {
  return crypto.createHash('sha256').update(data).digest('hex');
}

/**
 * RIPEMD160(SHA256(data)) - used for address derivation (Bitcoin-style P2PKH).
 * @param {Buffer} data
 * @returns {Buffer}
 */
function hash160(data) {
  const sha = crypto.createHash('sha256').update(data).digest();
  return crypto.createHash('ripemd160').update(sha).digest();
}

module.exports = { sha256d, sha256, hash160 };
```

### `src/crypto/keys.js`

```javascript
'use strict';

const { ec: EC } = require('elliptic');
const crypto = require('crypto');
const { hash160 } = require('./hash');

// secp256k1 - same curve as Bitcoin/Ethereum
const ec = new EC('secp256k1');

/**
 * Generate a new secp256k1 key pair.
 * Returns private key as hex, public key as compressed hex.
 */
function generateKeyPair() {
  const keyPair = ec.genKeyPair();
  return {
    privateKey: keyPair.getPrivate('hex'),
    publicKey: keyPair.getPublic(true, 'hex'), // compressed
  };
}

/**
 * Derive public key from private key.
 * @param {string} privateKeyHex
 * @returns {string} compressed public key hex
 */
function getPublicKey(privateKeyHex) {
  const keyPair = ec.keyFromPrivate(privateKeyHex, 'hex');
  return keyPair.getPublic(true, 'hex');
}

/**
 * Derive address from public key.
 * Address = Base58Check(version || RIPEMD160(SHA256(pubkey)))
 * We use a simplified hex encoding here; production would use Base58Check.
 * 
 * PRODUCTION GAP FLAGGED: Full Base58Check encoding with checksum should be
 * implemented before user-facing deployment to prevent address transcription errors.
 * 
 * @param {string} publicKeyHex
 * @returns {string}
 */
function publicKeyToAddress(publicKeyHex) {
  const pubKeyBuffer = Buffer.from(publicKeyHex, 'hex');
  const h160 = hash160(pubKeyBuffer);
  // Prefix with version byte 0x00 (mainnet P2PKH)
  const versioned = Buffer.concat([Buffer.from([0x00]), h160]);
  return versioned.toString('hex');
}

/**
 * Sign data with a private key.
 * Returns DER-encoded signature as hex.
 * @param {string} dataHex - hex string of data to sign
 * @param {string} privateKeyHex
 * @returns {string} DER signature hex
 */
function sign(dataHex, privateKeyHex) {
  const keyPair = ec.keyFromPrivate(privateKeyHex, 'hex');
  const signature = keyPair.sign(dataHex);
  return signature.toDER('hex');
}

/**
 * Verify a signature.
 * @param {string} dataHex - hex string of signed data
 * @param {string} signatureHex - DER signature hex
 * @param {string} publicKeyHex - compressed public key hex
 * @returns {boolean}
 */
function verify(dataHex, signatureHex, publicKeyHex) {
  try {
    const keyPair = ec.keyFromPublic(publicKeyHex, 'hex');
    return keyPair.verify(dataHex, signatureHex);
  } catch {
    // Any malformed input is treated as invalid signature
    return false;
  }
}

module.exports = { generateKeyPair, getPublicKey, publicKeyToAddress, sign, verify };
```

 *

 *   - inputs: array of { txId, outputIndex, publicKey, signature }
 *   - outputs: array of { address, amount }

 *   - id: double-SHA256 of canonical serialization

class TransactionInput {
  constructor({ txId, outputIndex, publicKey, signature = null }) {
    this.txId = txId;
    this.outputIndex = outputIndex;
    this.publicKey = publicKey;
    this.signature = signature; // null before signing
  }
}

  constructor({ address, amount }) {
    if (typeof amount !== 'bigint') {

    this.address = address;
