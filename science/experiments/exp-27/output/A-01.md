## src/Block.js

```javascript
const CryptoJS = require('crypto-js');

class Block {
  constructor({ index, timestamp, transactions, previousHash, nonce = 0, hash = null }) {
    this.index       = index;
    this.timestamp   = timestamp || Date.now();
    this.transactions = transactions || [];
    this.previousHash = previousHash;
    this.nonce       = nonce;
    this.hash        = hash || this.calculateHash();
  }

  // ─── Hashing ────────────────────────────────────────────────────────────────

  calculateHash() {
    return CryptoJS.SHA256(
      this.index +
      this.previousHash +
      this.timestamp +
      JSON.stringify(this.transactions) +
      this.nonce
    ).toString();
  }

  // ─── Proof of Work ──────────────────────────────────────────────────────────

  /**
   * Mine the block by finding a hash that starts with `difficulty` zeros.
   * @param {number} difficulty
   */
  mine(difficulty) {
    const target = '0'.repeat(difficulty);

    while (!this.hash.startsWith(target)) {
      this.nonce++;
      this.hash = this.calculateHash();
    }

    console.log(`⛏  Block #${this.index} mined | nonce: ${this.nonce} | hash: ${this.hash}`);
    return this;
  }

  // ─── Validation ─────────────────────────────────────────────────────────────

  isValid() {
    return this.hash === this.calculateHash();
  }

  // ─── Genesis ────────────────────────────────────────────────────────────────

  static genesis() {
    return new Block({
      index        : 0,
      timestamp    : 1_700_000_000_000,
      transactions : [],
      previousHash : '0'.repeat(64),
      nonce        : 0,
      hash         : 'GENESIS_BLOCK_HASH',
    });
  }

  toJSON() {
    return {
      index        : this.index,
      timestamp    : this.timestamp,
      transactions : this.transactions,
      previousHash : this.previousHash,
      nonce        : this.nonce,
      hash         : this.hash,
    };
  }
}

module.exports = Block;
```

## src/Transaction.js

```javascript
const { v4: uuidv4 } = require('uuid');
const CryptoJS       = require('crypto-js');
const elliptic       = require('elliptic');

const ec = new elliptic.ec('secp256k1');

class Transaction {
  constructor({ id, from, to, amount, timestamp, signature = null }) {
    this.id        = id        || uuidv4();
    this.from      = from;       // public key (hex)
    this.to        = to;
    this.amount    = amount;
    this.timestamp = timestamp || Date.now();
    this.signature = signature;
  }

  // ─── Hash ───────────────────────────────────────────────────────────────────

  calculateHash() {
    return CryptoJS.SHA256(
      this.from + this.to + this.amount + this.timestamp
    ).toString();
  }

  // ─── Sign ───────────────────────────────────────────────────────────────────

  /**
   * Sign the transaction with the sender's private key.
   * @param {elliptic.ec.KeyPair} keyPair
   */
  sign(keyPair) {
    if (keyPair.getPublic('hex') !== this.from) {
      throw new Error('You cannot sign transactions for other wallets!');
    }
    this.signature = keyPair.sign(this.calculateHash(), 'base64').toDER('hex');
    return this;
  }

  // ─── Verify ─────────────────────────────────────────────────────────────────

  isValid() {
    // Coinbase / reward transactions have no sender
    if (this.from === null) return true;

    if (!this.signature || this.signature.length === 0) {
      throw new Error(`Transaction ${this.id} has no signature.`);
    }

    const publicKey = ec.keyFromPublic(this.from, 'hex');
    return publicKey.verify(this.calculateHash(), this.signature);
  }

  // ─── Coinbase (mining reward) ────────────────────────────────────────────────

  static createCoinbase({ to, amount }) {
    return new Transaction({
      id        : uuidv4(),
      from      : null,
      to,
      amount,
      timestamp : Date.now(),
      signature : 'COINBASE',
    });
  }

  toJSON() {
    return {
      id        : this.id,
      from      : this.from,
      to        : this.to,
      amount    : this.amount,
      timestamp : this.timestamp,
      signature : this.signature,
    };
  }
}

module.exports = { Transaction, ec };
```

## src/Wallet.js

```javascript
const { ec }         = require('./Transaction');
const { Transaction } = require('./Transaction');

class Wallet {
  constructor() {
    this.keyPair   = ec.genKeyPair();
    this.publicKey = this.keyPair.getPublic('hex');
    this.balance   = 0;
  }

  // ─── Create a signed transaction ────────────────────────────────────────────

  /**
   * @param {string}     to          - recipient public key
   * @param {number}     amount
   * @param {Blockchain} blockchain  - used to look up current balance
   */
  createTransaction({ to, amount, blockchain }) {
    this.balance = blockchain.getBalanceOf(this.publicKey);

    if (amount > this.balance) {
      throw new Error(
        `Insufficient balance. Available: ${this.balance}, requested: ${amount}`
      );
    }

    const tx = new Transaction({
      from   : this.publicKey,
      to,
      amount,
    });

    tx.sign(this.keyPair);
    return tx;
  }

  toString() {
    return `
Wallet
  Public Key : ${this.publicKey}
  Balance    : ${this.balance}
    `.trim();
  }
}

module.exports = Wallet;
```

## src/TransactionPool.js

```javascript
class TransactionPool {
  constructor() {
    this.transactions = [];
  }

  // ─── Add ────────────────────────────────────────────────────────────────────

  add(transaction) {
    // Replace existing transaction from the same sender (update)
    const existing = this.transactions.findIndex(
      (t) => t.id === transaction.id
    );

    if (existing !== -1) {
      this.transactions[existing] = transaction;
    } else {
      this.transactions.push(transaction);
    }
  }

  // ─── Validate ───────────────────────────────────────────────────────────────

  validTransactions() {
    return this.transactions.filter((tx) => {
      try {
        return tx.isValid();
      } catch {
        return false;
      }
    });
  }

  // ─── Clear ──────────────────────────────────────────────────────────────────

  clear(confirmedTransactions) {
    const confirmedIds = new Set(confirmedTransactions.map((t) => t.id));
    this.transactions  = this.transactions.filter(
      (t) => !confirmedIds.has(t.id)
    );
  }

  size() {
    return this.transactions.length;
  }
}

module.exports = TransactionPool;
```

## src/Blockchain.js

```javascript
const Block           = require('./Block');
const { Transaction } = require('./Transaction');

const MINING_REWARD   = 50;
const INITIAL_BALANCE = 1000;

class Blockchain {
  constructor(difficulty = 3) {
    this.chain      = [Block.genesis()];
    this.difficulty = difficulty;
  }

  // ─── Accessors ──────────────────────────────────────────────────────────────

  get latestBlock() {
    return this.chain[this.chain.length - 1];
  }

  get height() {
    return this.chain.length - 1; // genesis = 0
  }

  // ─── Mining ─────────────────────────────────────────────────────────────────

  /**
   * Mine a new block containing the given transactions plus a coinbase reward.
   * @param {Transaction[]} transactions
   * @param {string}        minerAddress  - public key of the miner
   */
  mineBlock(transactions, minerAddress) {
    // Validate all transactions before including them
    const validTxs = transactions.filter((tx) => {
      try {
        return tx.isValid();
      } catch {
        return false;
      }
    });

    // Add mining reward
    const coinbase = Transaction.createCoinbase({
      to     : minerAddress,
      amount : MINING_REWARD,
    });
    validTxs.unshift(coinbase);

    const block = new Block({
      index        : this.chain.length,
      transactions : validTxs.map((t) => t.toJSON()),
      previousHash : this.latestBlock.hash,
    });

    block.mine(this.difficulty);
    this.chain.push(block);
    return block;
  }

  // ─── Balance ─────────────────────────────────────────────────────────────────

  /**
   * Calculate the balance of an address by replaying all transactions.
   * @param {string} address - public key
   */
  getBalanceOf(address) {
    let balance = INITIAL_BALANCE;

    for (const block of this.chain) {
      for (const tx of block.transactions) {
        if (tx.from === address) balance -= tx.amount;
        if (tx.to   === address) balance += tx.amount;
      }
    }

    return balance;
  }

  // ─── Transaction history ────────────────────────────────────────────────────

  getTransactionsOf(address) {
    const result = [];

    for (const block of this.chain) {
      for (const tx of block.transactions) {
        if (tx.from === address || tx.to === address) {
          result.push({ ...tx, blockIndex: block.index });
        }
      }
    }

    return result;
  }

  // ─── Validation ─────────────────────────────────────────────────────────────

  isChainValid() {
    for (let i = 1; i < this.chain.length; i++) {
      const current  = this.chain[i];
      const previous = this.chain[i - 1];

      // 1. Hash integrity
      if (!current.isValid()) {
        console.error(`❌ Block #${current.index} has an invalid hash.`);
        return false;
      }

      // 2. Chain linkage
      if (current.previousHash !== previous.hash) {
        console.error(`❌ Block #${current.index} is not linked to the previous block.`);
        return false;
      }

      // 3. Proof-of-work
      if (!current.hash.startsWith('0'.repeat(this.difficulty))) {
        console.error(`❌ Block #${current.index} does not meet difficulty.`);
        return false;
      }
    }

    return true;
  }

  // ─── Replace chain (consensus) ──────────────────────────────────────────────

  /**
   * Replace this chain with a longer valid chain (longest-chain rule).
   * @param {Block[]} newChain
   */
  replaceChain(newChain) {
    if (newChain.length <= this.chain.length) {
      console.log('Received chain is not longer. Not replacing.');
      return false;
    }

    const tempChain = new Blockchain(this.difficulty);
    tempChain.chain = newChain;

    if (!tempChain.isChainValid()) {
      console.log('Received chain is invalid. Not replacing.');
      return false;
    }

    console.log('✅ Replacing chain with a longer valid chain.');
    this.chain = newChain;
    return true;
  }

  // ─── Serialization ──────────────────────────────────────────────────────────

  toJSON() {
    return {
      height     : this.height,
      difficulty : this.difficulty,
      chain      : this.chain.map((b) => b.toJSON()),
    };
  }
}

module.exports = { Blockchain, MINING_REWARD, INITIAL_BALANCE };
```
