# Production Blockchain Implementation in Node.js

Let me be direct about what you've asked for and what it actually requires.

## Gap Analysis First

Before writing a line of code, here's what "build me a blockchain in Node.js" leaves unstated that production requires:

**Unstated but Required:**
| Gap | Decision Made | Rationale |
|-----|--------------|-----------|
| Consensus mechanism | Proof of Work (Nakamoto-style) | Deterministic, well-understood, no validator set coordination needed |
| P2P transport | TCP via net module + custom protocol | No external dependencies, full control |
| Persistence | LevelDB (leveldown) | Industry standard for blockchain storage |
| Mempool management | Priority queue by fee rate | Without this, nodes diverge on block content |
| Fork resolution | Longest chain (most cumulative work) | Required for any PoW system |
| Transaction model | UTXO | Prevents double-spend without global state lock |
| Wallet/signing | secp256k1 ECDSA | Bitcoin-compatible, battle-tested |
| API surface | REST + WebSocket | Operators need introspection |
| Block serialization | Custom binary + JSON RPC | Deterministic hashing requires canonical encoding |
| DoS protection | Rate limiting, max message size | Without this, one peer kills your node |
| Chain checkpointing | Not implemented — flagged | Required for sync performance at scale |
| Light client support | Not implemented — flagged | Requires Merkle proofs, separate effort |
| Finality | Probabilistic only — flagged | No BFT finality; 6-block confirmation convention |

## src/crypto/hash.js

```javascript
'use strict';

const { createHash } = require('crypto');

/**
 * Double-SHA256 — the same construction Bitcoin uses.
 * Using double hash defends against length-extension attacks.
 * Input can be string or Buffer; output is always hex string.
 */
function sha256d(data) {
  const buf = Buffer.isBuffer(data) ? data : Buffer.from(data, 'utf8');
  const first = createHash('sha256').update(buf).digest();
  return createHash('sha256').update(first).digest('hex');
}

/**
 * Single SHA256 — used for transaction IDs and Merkle leaves
 * where double-hash is not required.
 */
function sha256(data) {
  const buf = Buffer.isBuffer(data) ? data : Buffer.from(data, 'utf8');
  return createHash('sha256').update(buf).digest('hex');
}

/**
 * RIPEMD160(SHA256(data)) — used for address derivation.
 * This is the standard Pay-to-Public-Key-Hash construction.
 */
function hash160(data) {
  const buf = Buffer.isBuffer(data) ? data : Buffer.from(data, 'utf8');
  const sha = createHash('sha256').update(buf).digest();
  return createHash('ripemd160').update(sha).digest('hex');
}

/**
 * Merkle root of an array of hex-string hashes.
 *
 * Production note: Bitcoin's Merkle tree has a known vulnerability
 * where duplicate last elements can produce the same root for
 * different transaction sets. We guard against this explicitly.
 */
function merkleRoot(hashes) {
  if (!hashes || hashes.length === 0) {
    return sha256d('');
  }

  if (hashes.length === 1) {
    return hashes[0];
  }

  let level = [...hashes];

  while (level.length > 1) {
    const next = [];

    for (let i = 0; i < level.length; i += 2) {
      const left = level[i];
      // Guard: do NOT duplicate the last element if odd count.
      // Instead, pair it with a zero hash to make manipulation harder.
      const right = i + 1 < level.length
        ? level[i + 1]
        : '0'.repeat(64);

      next.push(sha256d(left + right));
    }

    level = next;
  }

  return level[0];
}

module.exports = { sha256d, sha256, hash160, merkleRoot };
```

## src/crypto/wallet.js

```javascript
'use strict';

const { randomBytes, createHash } = require('crypto');
const secp256k1 = require('secp256k1');
const { hash160, sha256d } = require('./hash');

/**
 * Wallet — key generation, address derivation, signing, verification.
 *
 * Address format: base58check(version_byte + hash160(pubkey))
 * This is compatible with Bitcoin P2PKH addresses.
 *
 * Production gaps flagged:
 * - No HD wallet (BIP32) support
 * - No BIP39 mnemonic support
 * - Private keys held in memory; production requires HSM or encrypted keystore
 * - No key rotation mechanism
 */

const BASE58_ALPHABET =
  '123456789ABCDEFGHJKLMNPQRSTUVWXYZabcdefghijkmnopqrstuvwxyz';

const ADDRESS_VERSION = 0x00; // mainnet P2PKH

function base58Encode(buf) {
  let num = BigInt('0x' + buf.toString('hex'));
  const zero = BigInt(0);
  const base = BigInt(58);

  let result = '';
  while (num > zero) {
    const remainder = num % base;
    num = num / base;
    result = BASE58_ALPHABET[Number(remainder)] + result;
  }

  // Preserve leading zero bytes
  for (const byte of buf) {
    if (byte !== 0) break;
    result = '1' + result;
  }

  return result;
}

function base58Decode(str) {
  let num = BigInt(0);
  const base = BigInt(58);

  for (const char of str) {
    const idx = BASE58_ALPHABET.indexOf(char);
    if (idx === -1) throw new Error(`Invalid base58 character: ${char}`);
    num = num * base + BigInt(idx);
  }

  let hex = num.toString(16);
  if (hex.length % 2) hex = '0' + hex;

  const buf = Buffer.from(hex, 'hex');

  // Restore leading zeros
  const leadingZeros = [...str].filter((c, i) => c === '1' && i === 0).length;
  const prefix = Buffer.alloc(leadingZeros);

  return Buffer.concat([prefix, buf]);
}

function base58CheckEncode(payload) {
  const checksum = Buffer.from(
    sha256d(payload).slice(0, 8), // first 4 bytes = 8 hex chars
    'hex'
  );
  return base58Encode(Buffer.concat([payload, checksum]));
}

function base58CheckDecode(str) {
  const buf = base58Decode(str);
  const payload = buf.slice(0, -4);
  const checksum = buf.slice(-4).toString('hex');
  const expected = sha256d(payload).slice(0, 8);

  if (checksum !== expected) {
    throw new Error('Invalid checksum in base58check address');
  }

  return payload;
}

/**
 * Generate a new private key.
 * Loops until we get a valid secp256k1 scalar — typically first try.
 */
function generatePrivateKey() {
  let privKey;
  do {
    privKey = randomBytes(32);
  } while (!secp256k1.privateKeyVerify(privKey));
  return privKey;
}

/**
 * Derive compressed public key from private key.
 */
function getPublicKey(privateKey) {
  return Buffer.from(secp256k1.publicKeyCreate(privateKey, true));
}

/**
 * Derive address from public key.
 * address = base58check(0x00 || RIPEMD160(SHA256(pubkey)))
 */
function getAddress(publicKey) {
  const pubkeyHash = hash160(publicKey);
  const payload = Buffer.concat([
    Buffer.from([ADDRESS_VERSION]),
    Buffer.from(pubkeyHash, 'hex'),
  ]);
  return base58CheckEncode(payload);
}

/**
 * Validate an address by decoding and verifying checksum + version.
 */
function validateAddress(address) {
  try {
    const payload = base58CheckDecode(address);
    return payload[0] === ADDRESS_VERSION && payload.length === 21;
  } catch {
    return false;
  }
}

/**
 * Sign a message hash with a private key.
 * Returns DER-encoded signature as hex string.
 *
 * IMPORTANT: secp256k1.ecdsaSign uses RFC6979 deterministic k-value.
 * This is critical — random k reuse leaks the private key.
 */
function sign(messageHashHex, privateKey) {
  const msgBuf = Buffer.from(messageHashHex, 'hex');
  const { signature, recovery } = secp256k1.ecdsaSign(msgBuf, privateKey);
  return {
    signature: Buffer.from(signature).toString('hex'),
    recovery,
  };
}

/**
 * Verify a signature against a public key and message hash.
 */
function verify(messageHashHex, signatureHex, publicKeyHex) {
  try {
    const msgBuf = Buffer.from(messageHashHex, 'hex');
    const sigBuf = Buffer.from(signatureHex, 'hex');
    const pubBuf = Buffer.from(publicKeyHex, 'hex');
    return secp256k1.ecdsaVerify(sigBuf, msgBuf, pubBuf);
  } catch {
    return false;
  }
}

/**
 * Create a complete wallet object.
 * In production: never log or serialize the privateKey.
 */
function createWallet() {
  const privateKey = generatePrivateKey();
  const publicKey = getPublicKey(privateKey);
  const address = getAddress(publicKey);

  return {
    privateKey, // Buffer — keep secret
    publicKey: publicKey.toString('hex'),
    address,
  };
}

module.exports = {
  createWallet,
  generatePrivateKey,
  getPublicKey,
  getAddress,
  validateAddress,
  sign,
  verify,
  base58CheckEncode,
  base58CheckDecode,
};
```

 * UTXO Transaction Model

 * - Parallel validation: UTXOs are independent, no global nonce ordering
 * - No replay attacks across forks without extra work
 * - Natural support for multi-input/multi-output
 * - Easier to prove inclusion/exclusion with Merkle proofs
 *
 * Transaction structure:
 * {
 *   id: string (hash of canonical serialization)
 *   version: number
 *   inputs: [{txid, outputIndex, scriptSig: {pubkey, signature}}]
 *   outputs: [{address, amount}]
 *   timestamp: number
 *   fee: number (computed: sum(inputs) - sum(outputs), validated by node)
 * }
 *
 * Coinbase transactions:
 * - Single input with txid = '0'.repeat(64), outputIndex = -1
 * - No scriptSig required
 * - Amount = block reward + fees
 *
 * Production gaps flagged:
 * - No script system (only P2PKH supported)
 * - No SegWit / transaction malleability fix
 * - No timelock (nLockTime) support
 * - No RBF (replace-by-fee) signaling
 * - No multi-signature support
