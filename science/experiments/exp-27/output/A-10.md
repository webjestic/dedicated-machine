const CryptoJS = require('crypto-js');

/**
 * SHA-256 hash utility
 */
const sha256 = (...args) => {
  const data = args.map(arg =>
    typeof arg === 'object' ? JSON.stringify(arg) : String(arg)
  ).join('');

  return CryptoJS.SHA256(data).toString(CryptoJS.enc.Hex);
};

module.exports = { sha256 };
```

## `src/utils/merkle.js`

```javascript
const { sha256 } = require('./hash');

/**
 * Build a Merkle Tree and return the root hash
 * Used to summarize all transactions in a block
 */
const getMerkleRoot = (transactions) => {
  if (!transactions || transactions.length === 0) {
    return sha256('');
  }

  let hashes = transactions.map(tx =>
    sha256(JSON.stringify(tx))
  );

  while (hashes.length > 1) {
    const nextLevel = [];

    for (let i = 0; i < hashes.length; i += 2) {
      const left = hashes[i];
      const right = hashes[i + 1] || hashes[i]; // Duplicate last if odd
      nextLevel.push(sha256(left + right));
    }

    hashes = nextLevel;
  }

  return hashes[0];
};

module.exports = { getMerkleRoot };
```

## `src/Transaction.js`

```javascript
const { sha256 } = require('./utils/hash');
const { v4: uuidv4 } = require('uuid');

class Transaction {
  /**
   * @param {string} fromAddress - Sender's public key (null for mining rewards)
   * @param {string} toAddress   - Recipient's public key
   * @param {number} amount      - Amount to transfer
   * @param {number} fee         - Transaction fee for miners
   */
  constructor(fromAddress, toAddress, amount, fee = 0) {
    this.id          = uuidv4();
    this.fromAddress = fromAddress;
    this.toAddress   = toAddress;
    this.amount      = amount;
    this.fee         = fee;
    this.timestamp   = Date.now();
    this.signature   = null;
  }

  /**
   * Calculate hash of transaction data (used for signing)
   */
  calculateHash() {
    return sha256(
      this.id,
      this.fromAddress,
      this.toAddress,
      this.amount,
      this.fee,
      this.timestamp
    );
  }

  /**
   * Sign the transaction with the sender's key pair
   * @param {Object} keyPair - elliptic key pair object
   */
  sign(keyPair) {
    const publicKey = keyPair.getPublic('hex');

    if (publicKey !== this.fromAddress) {
      throw new Error('You cannot sign transactions for other wallets!');
    }

    const hash = this.calculateHash();
    const sig  = keyPair.sign(hash, 'base64');
    this.signature = sig.toDER('hex');
  }

  /**
   * Verify transaction signature
   * @param {Object} ec - elliptic EC instance
   */
  isValid(ec) {
    // Mining reward transactions have no sender
    if (this.fromAddress === null) return true;

    if (!this.signature || this.signature.length === 0) {
      throw new Error('No signature in this transaction!');
    }

    const publicKey = ec.keyFromPublic(this.fromAddress, 'hex');
    return publicKey.verify(this.calculateHash(), this.signature);
  }

  toJSON() {
    return {
      id:          this.id,
      fromAddress: this.fromAddress,
      toAddress:   this.toAddress,
      amount:      this.amount,
      fee:         this.fee,
      timestamp:   this.timestamp,
      signature:   this.signature,
    };
  }
}

module.exports = Transaction;
```

## `src/Wallet.js`

```javascript
const EC      = require('elliptic').ec;
const { sha256 } = require('./utils/hash');
const Transaction = require('./Transaction');

const ec = new EC('secp256k1'); // Same curve as Bitcoin

class Wallet {
  constructor() {
    this.keyPair    = ec.genKeyPair();
    this.publicKey  = this.keyPair.getPublic('hex');
    this.privateKey = this.keyPair.getPrivate('hex');
  }

  /**
   * Restore wallet from existing private key
   */
  static fromPrivateKey(privateKey) {
    const wallet    = new Wallet();
    wallet.keyPair  = ec.keyFromPrivate(privateKey, 'hex');
    wallet.publicKey  = wallet.keyPair.getPublic('hex');
    wallet.privateKey = wallet.keyPair.getPrivate('hex');
    return wallet;
  }

  /**
   * Create and sign a transaction
   */
  createTransaction(toAddress, amount, fee, blockchain) {
    const balance = blockchain.getBalanceOfAddress(this.publicKey);

    if (balance < amount + fee) {
      throw new Error(
        `Insufficient balance. Available: ${balance}, Required: ${amount + fee}`
      );
    }

    const tx = new Transaction(this.publicKey, toAddress, amount, fee);
    tx.sign(this.keyPair);
    return tx;
  }

  getBalance(blockchain) {
    return blockchain.getBalanceOfAddress(this.publicKey);
  }

  toString() {
    return `Wallet {
  Public Key : ${this.publicKey}
  Private Key: ${this.privateKey}
}`;
  }
}

module.exports = { Wallet, ec };
```

## `src/MemPool.js`

```javascript
/**
 * Memory Pool — holds pending transactions waiting to be mined
 */
class MemPool {
  constructor() {
    this.transactions = [];
  }

  addTransaction(transaction) {
    this.transactions.push(transaction);
  }

  /**
   * Select top N transactions sorted by fee (highest fee first)
   */
  getTopTransactions(limit = 10) {
    return [...this.transactions]
      .sort((a, b) => b.fee - a.fee)
      .slice(0, limit);
  }

  /**
   * Remove confirmed transactions from pool
   */
  removeConfirmed(confirmedTxIds) {
    this.transactions = this.transactions.filter(
      tx => !confirmedTxIds.includes(tx.id)
    );
  }

  size() {
    return this.transactions.length;
  }

  clear() {
    this.transactions = [];
  }
}

module.exports = MemPool;
```

## `src/Block.js`

```javascript
const { sha256 }      = require('./utils/hash');
const { getMerkleRoot } = require('./utils/merkle');

class Block {
  /**
   * @param {number}        index        - Block height
   * @param {string}        previousHash - Hash of previous block
   * @param {Transaction[]} transactions - Array of transactions
   * @param {number}        difficulty   - Mining difficulty
   */
  constructor(index, previousHash, transactions, difficulty = 2) {
    this.index        = index;
    this.previousHash = previousHash;
    this.timestamp    = Date.now();
    this.transactions = transactions;
    this.difficulty   = difficulty;
    this.nonce        = 0;
    this.merkleRoot   = getMerkleRoot(transactions);
    this.hash         = this.calculateHash();
  }

  calculateHash() {
    return sha256(
      this.index,
      this.previousHash,
      this.timestamp,
      this.merkleRoot,
      this.difficulty,
      this.nonce
    );
  }

  /**
   * Proof-of-Work mining
   * Increment nonce until hash starts with required number of zeros
   */
  mine(difficulty) {
    const target = '0'.repeat(difficulty);
    let attempts = 0;

    console.log(`⛏  Mining block ${this.index}...`);
    const startTime = Date.now();

    while (!this.hash.startsWith(target)) {
      this.nonce++;
      this.hash = this.calculateHash();
      attempts++;
    }

    const elapsed = ((Date.now() - startTime) / 1000).toFixed(2);
    console.log(
      `✅ Block ${this.index} mined! Nonce: ${this.nonce} | ` +
      `Attempts: ${attempts} | Time: ${elapsed}s | Hash: ${this.hash}`
    );
  }

  /**
   * Validate block integrity
   */
  hasValidTransactions(ec) {
    for (const tx of this.transactions) {
      if (!tx.isValid(ec)) return false;
    }
    return true;
  }

  toJSON() {
    return {
      index:        this.index,
      previousHash: this.previousHash,
      timestamp:    this.timestamp,
      transactions: this.transactions.map(tx => tx.toJSON ? tx.toJSON() : tx),
      difficulty:   this.difficulty,
      nonce:        this.nonce,
      merkleRoot:   this.merkleRoot,
      hash:         this.hash,
    };
  }
}

module.exports = Block;
```

## `src/Blockchain.js`

```javascript
const Block       = require('./Block');
const Transaction = require('./Transaction');
const MemPool     = require('./MemPool');
const { ec }      = require('./Wallet');

const MINING_REWARD    = 50;   // Coins rewarded to miner
const INITIAL_DIFFICULTY = 3;  // Leading zeros required
const DIFFICULTY_ADJUSTMENT_INTERVAL = 5;   // Blocks
const TARGET_BLOCK_TIME = 10000; // 10 seconds in ms

class Blockchain {
  constructor() {
    this.chain      = [this.createGenesisBlock()];
    this.difficulty = INITIAL_DIFFICULTY;
    this.memPool    = new MemPool();
    this.miningReward = MINING_REWARD;
  }

  // ─── Genesis Block ────────────────────────────────────────────────────────

  createGenesisBlock() {
    const genesis = new Block(0, '0'.repeat(64), [], 0);
    genesis.hash  = genesis.calculateHash();
    console.log('🌱 Genesis block created:', genesis.hash);
    return genesis;
  }

  // ─── Chain Helpers ────────────────────────────────────────────────────────

  getLatestBlock() {
    return this.chain[this.chain.length - 1];
  }

  getBlockByIndex(index) {
    return this.chain[index] || null;
  }

  getBlockByHash(hash) {
    return this.chain.find(b => b.hash === hash) || null;
  }

  // ─── Difficulty Adjustment ────────────────────────────────────────────────

  /**
   * Adjust difficulty every N blocks to maintain target block time
   */
  adjustDifficulty() {
    const lastBlock = this.getLatestBlock();

    if (
      lastBlock.index % DIFFICULTY_ADJUSTMENT_INTERVAL !== 0 ||
      lastBlock.index === 0
    ) {
      return this.difficulty;
    }

    const prevAdjustmentBlock =
      this.chain[this.chain.length - DIFFICULTY_ADJUSTMENT_INTERVAL];

    const timeExpected = TARGET_BLOCK_TIME * DIFFICULTY_ADJUSTMENT_INTERVAL;
    const timeTaken    = lastBlock.timestamp - prevAdjustmentBlock.timestamp;

    if (timeTaken < timeExpected / 2) {
      console.log(`📈 Difficulty increased to ${this.difficulty + 1}`);
      return this.difficulty + 1;
    } else if (timeTaken > timeExpected * 2) {
      console.log(`📉 Difficulty decreased to ${Math.max(1, this.difficulty - 1)}`);
      return Math.max(1, this.difficulty - 1);
    }

    return this.difficulty;
  }

  // ─── Mining ───────────────────────────────────────────────────────────────

  /**
   * Mine pending transactions from mempool
   * @param {string} minerAddress - Address to receive mining reward
   */
  minePendingTransactions(minerAddress) {
    // Validate all pending transactions
    const validTransactions = this.memPool
      .getTopTransactions(10)
      .filter(tx => {
        try {
          return tx.isValid(ec);
        } catch (e) {
          console.warn(`⚠️  Invalid transaction skipped: ${e.message}`);
          return false;
        }
      });

    // Add coinbase (mining reward) transaction
    const totalFees = validTransactions.reduce((sum, tx) => sum + tx.fee, 0);
    const reward    = new Transaction(null, minerAddress, this.miningReward + totalFees);
    validTransactions.unshift(reward);

    // Adjust difficulty
    this.difficulty = this.adjustDifficulty();

    // Create and mine new block
    const block = new Block(
      this.chain.length,
      this.getLatestBlock().hash,
      validTransactions,
      this.difficulty
    );

    block.mine(this.difficulty);
    this.chain.push(block);

    // Remove confirmed transactions from mempool
    const confirmedIds = validTransactions.map(tx => tx.id);
    this.memPool.removeConfirmed(confirmedIds);

    console.log(`💰 Mining reward of ${this.miningReward + totalFees} sent to ${minerAddress.slice(0, 20)}...`);

    return block;
  }

  // ─── Transactions ─────────────────────────────────────────────────────────

  addTransaction(transaction) {
    if (!transaction.fromAddress || !transaction.toAddress) {
      throw new Error('Transaction must include from and to address!');
    }

    if (!transaction.isValid(ec)) {
      throw new Error('Cannot add invalid transaction to the mempool!');
    }

    if (transaction.amount <= 0) {
      throw new Error('Transaction amount must be positive!');
    }

    const senderBalance = this.getBalanceOfAddress(transaction.fromAddress);
    if (senderBalance < transaction.amount + transaction.fee) {
      throw new Error(
        `Insufficient balance! Available: ${senderBalance}, ` +
        `Required: ${transaction.amount + transaction.fee}`
      );
    }

    this.memPool.addTransaction(transaction);
    console.log(`📨 Transaction added to mempool: ${transaction.id}`);
    return transaction;
  }
