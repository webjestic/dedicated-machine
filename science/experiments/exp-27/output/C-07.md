### `src/crypto/hash.js`

```javascript
'use strict';

const crypto = require('crypto');

/**
 * All hashing goes through this module.
 * Using Node's built-in crypto — no third-party hash library needed.
 * SHA-256 is used throughout, double-SHA-256 for block/tx IDs
 * (same rationale as Bitcoin: length-extension attack resistance).
 */

/**
 * Single SHA-256.
 * @param {string|Buffer} data
 * @returns {string} hex digest
 */
function sha256(data) {
  return crypto
    .createHash('sha256')
    .update(data)
    .digest('hex');
}

/**
 * Double SHA-256 — used for block hashes and transaction IDs.
 * @param {string|Buffer} data
 * @returns {string} hex digest
 */
function hash256(data) {
  const first = crypto.createHash('sha256').update(data).digest();
  return crypto.createHash('sha256').update(first).digest('hex');
}

/**
 * Merkle root of an array of hex hash strings.
 * Returns the single hash if only one element.
 * Returns a zero hash for empty arrays (empty block body).
 *
 * NOTE: Bitcoin duplicates the last element when the count is odd.
 * We do the same for compatibility, but flag it: this is a known
 * CVE-class issue (CVE-2012-2459 style) if not handled carefully.
 * In production you would use a different tree construction
 * (e.g., RFC 6962 / Certificate Transparency style) to avoid it.
 */
function merkleRoot(hashes) {
  if (!hashes || hashes.length === 0) {
    return '0'.repeat(64);
  }

  let level = [...hashes];

  while (level.length > 1) {
    const next = [];
    for (let i = 0; i < level.length; i += 2) {
      const left = level[i];
      // Duplicate last element if odd count — see note above
      const right = level[i + 1] ?? level[i];
      next.push(hash256(left + right));
    }
    level = next;
  }

  return level[0];
}

/**
 * Verify a merkle proof for a single transaction.
 * @param {string} txHash - the tx hash to verify
 * @param {Array<{hash: string, position: 'left'|'right'}>} proof
 * @param {string} root - expected merkle root
 * @returns {boolean}
 */
function verifyMerkleProof(txHash, proof, root) {
  let current = txHash;
  for (const { hash, position } of proof) {
    if (position === 'left') {
      current = hash256(hash + current);
    } else {
      current = hash256(current + hash);
    }
  }
  return current === root;
}

module.exports = { sha256, hash256, merkleRoot, verifyMerkleProof };
```

### `src/crypto/keys.js`

```javascript
'use strict';

const { ec: EC } = require('elliptic');
const { hash256 } = require('./hash');

// secp256k1 — same curve as Bitcoin/Ethereum
const ec = new EC('secp256k1');

/**
 * Generate a new key pair.
 * In production: private keys MUST be stored in a hardware security module
 * (HSM) or at minimum an encrypted keystore (e.g., PBKDF2 + AES-256-GCM).
 * Returning raw private key bytes here is acceptable for a node's
 * operational wallet but never for user-facing custody.
 *
 * @returns {{ privateKey: string, publicKey: string, address: string }}
 */
function generateKeyPair() {
  const keyPair = ec.genKeyPair();
  const privateKey = keyPair.getPrivate('hex');
  const publicKey = keyPair.getPublic('hex'); // uncompressed, 65 bytes
  const address = publicKeyToAddress(publicKey);
  return { privateKey, publicKey, address };
}

/**
 * Derive address from public key.
 * We use hash256(publicKey) truncated to 40 hex chars (20 bytes).
 * Production systems use a proper address encoding (Base58Check, bech32)
 * with a checksum to prevent typo-induced fund loss.
 * Flagged: implement bech32 encoding before user-facing deployment.
 *
 * @param {string} publicKey - hex encoded
 * @returns {string} 40-char hex address
 */
function publicKeyToAddress(publicKey) {
  return hash256(publicKey).slice(0, 40);
}

/**
 * Sign data with a private key.
 * Returns DER-encoded signature as hex string.
 * @param {string} dataHash - hex string of the hash to sign
 * @param {string} privateKey - hex string
 * @returns {string} hex-encoded DER signature
 */
function sign(dataHash, privateKey) {
  const keyPair = ec.keyFromPrivate(privateKey, 'hex');
  const signature = keyPair.sign(dataHash);
  return signature.toDER('hex');
}

/**
 * Verify a signature.
 * @param {string} dataHash - hex string of the hash that was signed
 * @param {string} signature - hex-encoded DER signature
 * @param {string} publicKey - hex string
 * @returns {boolean}
 */
function verify(dataHash, signature, publicKey) {
  try {
    const keyPair = ec.keyFromPublic(publicKey, 'hex');
    return keyPair.verify(dataHash, signature);
  } catch {
    // Any malformed input returns false — never throws to caller
    return false;
  }
}

/**
 * Recover public key from private key (for wallet operations).
 * @param {string} privateKey - hex string
 * @returns {string} hex-encoded public key
 */
function privateKeyToPublicKey(privateKey) {
  const keyPair = ec.keyFromPrivate(privateKey, 'hex');
  return keyPair.getPublic('hex');
}

module.exports = {
  generateKeyPair,
  publicKeyToAddress,
  privateKeyToPublicKey,
  sign,
  verify,
};
```
