## `src/Transaction.js`

```javascript
const { v4: uuidv4 } = require('uuid');
const crypto = require('crypto');

class Transaction {
  constructor(fromAddress, toAddress, amount, fee = 0) {
    this.id = uuidv4();
    this.fromAddress = fromAddress; // null for coinbase (mining reward)
    this.toAddress = toAddress;
    this.amount = amount;
    this.fee = fee;
    this.timestamp = Date.now();
    this.signature = null;
  }

  /**
   * Creates a SHA-256 hash of the transaction data
   */
  calculateHash() {
    return crypto
      .createHash('sha256')
      .update(
        this.id +
        this.fromAddress +
        this.toAddress +
        this.amount +
        this.fee +
        this.timestamp
      )
      .digest('hex');
  }

  /**
   * Signs the transaction with the sender's private key
   * @param {EC.KeyPair} signingKey - elliptic key pair
   */
  sign(signingKey) {
    if (signingKey.getPublic('hex') !== this.fromAddress) {
      throw new Error('You cannot sign transactions for other wallets!');
    }

    const hash = this.calculateHash();
    const sig = signingKey.sign(hash, 'base64');
    this.signature = sig.toDER('hex');
  }

  /**
   * Validates the transaction signature and data integrity
   */
  isValid(ec) {
    // Coinbase transaction (mining reward) — no signature needed
    if (this.fromAddress === null) return true;

    if (!this.signature || this.signature.length === 0) {
      throw new Error('No signature in this transaction');
    }

    if (this.amount <= 0) {
      throw new Error('Transaction amount must be greater than 0');
    }

    if (this.fee < 0) {
      throw new Error('Transaction fee cannot be negative');
    }

    const publicKey = ec.keyFromPublic(this.fromAddress, 'hex');
    return publicKey.verify(this.calculateHash(), this.signature);
  }

  toJSON() {
    return {
      id: this.id,
      fromAddress: this.fromAddress,
      toAddress: this.toAddress,
      amount: this.amount,
      fee: this.fee,
      timestamp: this.timestamp,
      signature: this.signature,
    };
  }
}

module.exports = Transaction;
```

## `src/Block.js`

```javascript
const crypto = require('crypto');

class Block {
  constructor(index, timestamp, transactions, previousHash = '', difficulty = 3) {
    this.index = index;
    this.timestamp = timestamp;
    this.transactions = transactions;
    this.previousHash = previousHash;
    this.difficulty = difficulty;
    this.nonce = 0;
    this.hash = this.calculateHash();
    this.merkleRoot = this.calculateMerkleRoot();
  }

  /**
   * Calculates SHA-256 hash of the block
   */
  calculateHash() {
    return crypto
      .createHash('sha256')
      .update(
        this.index +
        this.previousHash +
        this.timestamp +
        JSON.stringify(this.transactions) +
        this.nonce +
        this.difficulty
      )
      .digest('hex');
  }

  /**
   * Calculates Merkle root of all transactions
   */
  calculateMerkleRoot() {
    if (this.transactions.length === 0) return '';

    let hashes = this.transactions.map(tx =>
      crypto.createHash('sha256').update(JSON.stringify(tx)).digest('hex')
    );

    while (hashes.length > 1) {
      if (hashes.length % 2 !== 0) {
        // Duplicate last hash if odd number
        hashes.push(hashes[hashes.length - 1]);
      }

      const newHashes = [];
      for (let i = 0; i < hashes.length; i += 2) {
        newHashes.push(
          crypto
            .createHash('sha256')
            .update(hashes[i] + hashes[i + 1])
            .digest('hex')
        );
      }
      hashes = newHashes;
    }

    return hashes[0];
  }

  /**
   * Proof-of-Work: mines the block until hash meets difficulty target
   */
  mineBlock(difficulty) {
    const target = Array(difficulty + 1).join('0');
    const startTime = Date.now();

    console.log(`⛏️  Mining block ${this.index}...`);

    while (this.hash.substring(0, difficulty) !== target) {
      this.nonce++;
      this.hash = this.calculateHash();
    }

    const elapsed = ((Date.now() - startTime) / 1000).toFixed(2);
    console.log(
      `✅ Block ${this.index} mined in ${elapsed}s | Nonce: ${this.nonce} | Hash: ${this.hash}`
    );
  }

  /**
   * Validates block integrity
   */
  hasValidTransactions(ec) {
    for (const tx of this.transactions) {
      if (!tx.isValid(ec)) {
        return false;
      }
    }
    return true;
  }

  toJSON() {
    return {
      index: this.index,
      timestamp: this.timestamp,
      transactions: this.transactions.map(tx =>
        typeof tx.toJSON === 'function' ? tx.toJSON() : tx
      ),
      previousHash: this.previousHash,
      hash: this.hash,
      merkleRoot: this.merkleRoot,
      nonce: this.nonce,
      difficulty: this.difficulty,
    };
  }
}

module.exports = Block;
```

## `src/Mempool.js`

```javascript
class Mempool {
  constructor() {
    this.transactions = [];
  }

  /**
   * Adds a validated transaction to the mempool
   */
  addTransaction(transaction, ec) {
    if (!transaction.fromAddress || !transaction.toAddress) {
      throw new Error('Transaction must include from and to address');
    }

    if (!transaction.isValid(ec)) {
      throw new Error('Cannot add invalid transaction to mempool');
    }

    if (this.transactions.find(tx => tx.id === transaction.id)) {
      throw new Error('Transaction already exists in mempool');
    }

    this.transactions.push(transaction);
    console.log(`📥 Transaction ${transaction.id} added to mempool`);
    return transaction;
  }

  /**
   * Returns top N transactions sorted by fee (highest first)
   */
  getTransactionsForBlock(limit = 10) {
    return [...this.transactions]
      .sort((a, b) => b.fee - a.fee)
      .slice(0, limit);
  }

  /**
   * Removes confirmed transactions from the mempool
   */
  removeConfirmedTransactions(confirmedTxIds) {
    const before = this.transactions.length;
    this.transactions = this.transactions.filter(
      tx => !confirmedTxIds.includes(tx.id)
    );
    console.log(
      `🗑️  Removed ${before - this.transactions.length} confirmed transactions from mempool`
    );
  }

  getSize() {
    return this.transactions.length;
  }

  getAll() {
    return this.transactions;
  }
}

module.exports = Mempool;
```

## `src/Blockchain.js`

```javascript
const EC = require('elliptic').ec;
const Block = require('./Block');
const Transaction = require('./Transaction');
const Mempool = require('./Mempool');

const ec = new EC('secp256k1'); // Same curve as Bitcoin

class Blockchain {
  constructor() {
    this.chain = [this.createGenesisBlock()];
    this.difficulty = 3;
    this.miningReward = 50;
    this.mempool = new Mempool();
    this.blockSizeLimit = 10; // Max transactions per block
  }

  /**
   * Creates the first block in the chain
   */
  createGenesisBlock() {
    const genesis = new Block(0, Date.now(), [], '0', 1);
    genesis.hash = genesis.calculateHash();
    console.log('🌱 Genesis block created');
    return genesis;
  }

  getLatestBlock() {
    return this.chain[this.chain.length - 1];
  }

  /**
   * Mines pending transactions and rewards the miner
   */
  minePendingTransactions(miningRewardAddress) {
    // Create coinbase (reward) transaction
    const rewardTx = new Transaction(null, miningRewardAddress, this.miningReward);

    // Get transactions from mempool
    const txsToMine = this.mempool.getTransactionsForBlock(this.blockSizeLimit - 1);

    // Calculate total fees for miner
    const totalFees = txsToMine.reduce((sum, tx) => sum + tx.fee, 0);
    rewardTx.amount += totalFees;

    const blockTransactions = [rewardTx, ...txsToMine];

    const block = new Block(
      this.chain.length,
      Date.now(),
      blockTransactions,
      this.getLatestBlock().hash,
      this.difficulty
    );

    block.mineBlock(this.difficulty);

    // Validate all transactions before adding
    if (!block.hasValidTransactions(ec)) {
      throw new Error('Block contains invalid transactions');
    }

    this.chain.push(block);

    // Remove mined transactions from mempool
    const minedTxIds = txsToMine.map(tx => tx.id);
    this.mempool.removeConfirmedTransactions(minedTxIds);

    // Adjust difficulty every 5 blocks
    this.adjustDifficulty();

    console.log(`🎉 Block ${block.index} added to chain! Reward: ${rewardTx.amount} coins`);
    return block;
  }

  /**
   * Dynamically adjusts mining difficulty
   */
  adjustDifficulty() {
    const ADJUSTMENT_INTERVAL = 5;
    const TARGET_BLOCK_TIME = 10000; // 10 seconds in ms

    if (this.chain.length % ADJUSTMENT_INTERVAL === 0 && this.chain.length > 1) {
      const lastAdjustmentBlock = this.chain[this.chain.length - ADJUSTMENT_INTERVAL];
      const latestBlock = this.getLatestBlock();
      const timeExpected = TARGET_BLOCK_TIME * ADJUSTMENT_INTERVAL;
      const timeTaken = latestBlock.timestamp - lastAdjustmentBlock.timestamp;

      if (timeTaken < timeExpected / 2) {
        this.difficulty++;
        console.log(`📈 Difficulty increased to ${this.difficulty}`);
      } else if (timeTaken > timeExpected * 2) {
        this.difficulty = Math.max(1, this.difficulty - 1);
        console.log(`📉 Difficulty decreased to ${this.difficulty}`);
      }
    }
  }

  /**
   * Adds a transaction to the mempool
   */
  addTransaction(transaction) {
    // Validate sender has sufficient balance
    const senderBalance = this.getBalanceOfAddress(transaction.fromAddress);
    if (senderBalance < transaction.amount + transaction.fee) {
      throw new Error(
        `Insufficient balance. Available: ${senderBalance}, Required: ${transaction.amount + transaction.fee}`
      );
    }

    return this.mempool.addTransaction(transaction, ec);
  }

  /**
   * Calculates the balance of an address by scanning the chain
   */
  getBalanceOfAddress(address) {
    let balance = 0;

    for (const block of this.chain) {
      for (const tx of block.transactions) {
        if (tx.fromAddress === address) {
          balance -= tx.amount + tx.fee;
        }
        if (tx.toAddress === address) {
          balance += tx.amount;
        }
      }
    }

    return balance;
  }

  /**
   * Returns all transactions associated with an address
   */
  getTransactionsOfAddress(address) {
    const txs = [];

    for (const block of this.chain) {
      for (const tx of block.transactions) {
        if (tx.fromAddress === address || tx.toAddress === address) {
          txs.push({ ...tx, blockIndex: block.index });
        }
      }
    }

    return txs;
  }

  /**
   * Validates the entire blockchain
   */
  isChainValid() {
    for (let i = 1; i < this.chain.length; i++) {
      const currentBlock = this.chain[i];
      const previousBlock = this.chain[i - 1];

      // Check block transactions are valid
      if (!currentBlock.hasValidTransactions(ec)) {
        console.error(`❌ Block ${i} has invalid transactions`);
        return false;
      }

      // Check block hash integrity
      if (currentBlock.hash !== currentBlock.calculateHash()) {
        console.error(`❌ Block ${i} hash is invalid`);
        return false;
      }

      // Check chain linkage
      if (currentBlock.previousHash !== previousBlock.hash) {
        console.error(`❌ Block ${i} is not linked to previous block`);
        return false;
      }

      // Check proof of work
      const target = Array(currentBlock.difficulty + 1).join('0');
      if (!currentBlock.hash.startsWith(target)) {
        console.error(`❌ Block ${i} does not meet difficulty target`);
        return false;
      }
    }

    return true;
  }

  /**
   * Returns blockchain stats
   */
  getStats() {
    return {
      totalBlocks: this.chain.length,
      difficulty: this.difficulty,
      miningReward: this.miningReward,
      pendingTransactions: this.mempool.getSize(),
      isValid: this.isChainValid(),
      latestBlockHash: this.getLatestBlock().hash,
    };
  }

  getEC() {
    return ec;
  }
}

module.exports = Blockchain;
```
