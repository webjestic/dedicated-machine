```javascript
const crypto = require('crypto');

/**
 * Generate SHA-256 hash
 * @param {string} data
 * @returns {string} hex hash
 */
const sha256 = (data) => {
  return crypto.createHash('sha256').update(data).digest('hex');
};

/**
 * Generate double SHA-256 hash (like Bitcoin)
 * @param {string} data
 * @returns {string} hex hash
 */
const doubleSha256 = (data) => {
  return sha256(sha256(data));
};

module.exports = { sha256, doubleSha256 };
```

### `src/utils/merkle.js`

```javascript
const { sha256 } = require('./hash');

/**
 * Build a Merkle Tree from an array of transaction IDs
 * @param {string[]} txIds
 * @returns {string} Merkle root hash
 */
const buildMerkleRoot = (txIds) => {
  if (!txIds || txIds.length === 0) {
    return sha256('');
  }

  if (txIds.length === 1) {
    return txIds[0];
  }

  let level = [...txIds];

  while (level.length > 1) {
    const nextLevel = [];

    for (let i = 0; i < level.length; i += 2) {
      const left = level[i];
      const right = i + 1 < level.length ? level[i + 1] : level[i]; // Duplicate last if odd
      nextLevel.push(sha256(left + right));
    }

    level = nextLevel;
  }

  return level[0];
};

module.exports = { buildMerkleRoot };
```

### `src/Transaction.js`

```javascript
const { sha256 } = require('./utils/hash');
const { v4: uuidv4 } = require('uuid');
const EC = require('elliptic').ec;

const ec = new EC('secp256k1');

class Transaction {
  /**
   * @param {string} fromAddress - Sender's public key (null for coinbase)
   * @param {string} toAddress   - Recipient's public key
   * @param {number} amount      - Amount to transfer
   * @param {number} fee         - Transaction fee
   */
  constructor(fromAddress, toAddress, amount, fee = 0) {
    this.id = uuidv4();
    this.fromAddress = fromAddress;
    this.toAddress = toAddress;
    this.amount = amount;
    this.fee = fee;
    this.timestamp = Date.now();
    this.signature = null;
  }

  /**
   * Calculate the hash of this transaction
   * @returns {string}
   */
  calculateHash() {
    return sha256(
      this.id +
      this.fromAddress +
      this.toAddress +
      this.amount +
      this.fee +
      this.timestamp
    );
  }

  /**
   * Sign the transaction with a private key
   * @param {object} keyPair - EC key pair
   */
  sign(keyPair) {
    if (keyPair.getPublic('hex') !== this.fromAddress) {
      throw new Error('You cannot sign transactions for other wallets!');
    }

    const hash = this.calculateHash();
    const sig = keyPair.sign(hash, 'base64');
    this.signature = sig.toDER('hex');
  }

  /**
   * Verify the transaction signature
   * @returns {boolean}
   */
  isValid() {
    // Coinbase transaction (mining reward) — no signature needed
    if (this.fromAddress === null) return true;

    if (!this.signature || this.signature.length === 0) {
      throw new Error('No signature in this transaction');
    }

    const publicKey = ec.keyFromPublic(this.fromAddress, 'hex');
    return publicKey.verify(this.calculateHash(), this.signature);
  }

  /**
   * Create a coinbase (mining reward) transaction
   * @param {string} minerAddress
   * @param {number} reward
   * @returns {Transaction}
   */
  static createCoinbase(minerAddress, reward) {
    const tx = new Transaction(null, minerAddress, reward, 0);
    tx.signature = 'COINBASE';
    return tx;
  }

  toJSON() {
    return {
      id: this.id,
      fromAddress: this.fromAddress,
      toAddress: this.toAddress,
      amount: this.amount,
      fee: this.fee,
      timestamp: this.timestamp,
      signature: this.signature,
    };
  }
}

module.exports = Transaction;
```

### `src/Block.js`

```javascript
const { sha256 } = require('./utils/hash');
const { buildMerkleRoot } = require('./utils/merkle');

class Block {
  /**
   * @param {number}        index        - Block height
   * @param {string}        previousHash - Hash of the previous block
   * @param {Transaction[]} transactions - Array of transactions
   * @param {number}        difficulty   - Mining difficulty
   */
  constructor(index, previousHash, transactions, difficulty = 4) {
    this.index = index;
    this.previousHash = previousHash;
    this.timestamp = Date.now();
    this.transactions = transactions;
    this.difficulty = difficulty;
    this.nonce = 0;
    this.merkleRoot = this.calculateMerkleRoot();
    this.hash = this.calculateHash();
  }

  /**
   * Calculate the Merkle root of all transactions
   * @returns {string}
   */
  calculateMerkleRoot() {
    const txIds = this.transactions.map((tx) => tx.id || tx.calculateHash());
    return buildMerkleRoot(txIds);
  }

  /**
   * Calculate the block hash
   * @returns {string}
   */
  calculateHash() {
    return sha256(
      this.index +
      this.previousHash +
      this.timestamp +
      this.merkleRoot +
      this.nonce +
      this.difficulty
    );
  }

  /**
   * Mine the block (Proof of Work)
   * @param {number} difficulty
   */
  mine(difficulty) {
    this.difficulty = difficulty;
    const target = '0'.repeat(difficulty);

    console.log(`⛏  Mining block ${this.index} [difficulty: ${difficulty}]...`);
    const startTime = Date.now();

    while (!this.hash.startsWith(target)) {
      this.nonce++;
      this.hash = this.calculateHash();
    }

    const elapsed = ((Date.now() - startTime) / 1000).toFixed(2);
    console.log(`✅ Block ${this.index} mined in ${elapsed}s | Hash: ${this.hash}`);
  }

  /**
   * Validate all transactions in the block
   * @returns {boolean}
   */
  hasValidTransactions() {
    for (const tx of this.transactions) {
      if (!tx.isValid()) return false;
    }
    return true;
  }

  /**
   * Check if the block hash meets the difficulty target
   * @returns {boolean}
   */
  hasValidHash() {
    const target = '0'.repeat(this.difficulty);
    return (
      this.hash.startsWith(target) &&
      this.hash === this.calculateHash()
    );
  }

  toJSON() {
    return {
      index: this.index,
      previousHash: this.previousHash,
      timestamp: this.timestamp,
      transactions: this.transactions.map((tx) =>
        typeof tx.toJSON === 'function' ? tx.toJSON() : tx
      ),
      merkleRoot: this.merkleRoot,
      difficulty: this.difficulty,
      nonce: this.nonce,
      hash: this.hash,
    };
  }
}

module.exports = Block;
```

### `src/MemPool.js`

```javascript
/**
 * Memory Pool — holds unconfirmed transactions
 */
class MemPool {
  constructor() {
    this.transactions = [];
  }

  /**
   * Add a validated transaction to the pool
   * @param {Transaction} transaction
   */
  addTransaction(transaction) {
    if (!transaction.isValid()) {
      throw new Error('Cannot add invalid transaction to mempool');
    }
    this.transactions.push(transaction);
    console.log(`📥 Transaction ${transaction.id} added to mempool`);
  }

  /**
   * Get the highest-fee transactions up to a limit
   * @param {number} limit
   * @returns {Transaction[]}
   */
  getTopTransactions(limit = 10) {
    return [...this.transactions]
      .sort((a, b) => b.fee - a.fee)
      .slice(0, limit);
  }

  /**
   * Remove confirmed transactions from the pool
   * @param {Transaction[]} confirmedTxs
   */
  removeConfirmed(confirmedTxs) {
    const confirmedIds = new Set(confirmedTxs.map((tx) => tx.id));
    this.transactions = this.transactions.filter(
      (tx) => !confirmedIds.has(tx.id)
    );
  }

  /**
   * Check if a transaction is already in the pool
   * @param {string} txId
   * @returns {boolean}
   */
  has(txId) {
    return this.transactions.some((tx) => tx.id === txId);
  }

  get size() {
    return this.transactions.length;
  }

  clear() {
    this.transactions = [];
  }
}

module.exports = MemPool;
```

```javascript
const EC = require('elliptic').ec;
const { sha256 } = require('./utils/hash');
const Transaction = require('./Transaction');

const ec = new EC('secp256k1');

class Wallet {
  /**
   * Generate a new wallet with a fresh key pair
   */
  constructor() {
    this.keyPair = ec.genKeyPair();
    this.publicKey = this.keyPair.getPublic('hex');
    this.privateKey = this.keyPair.getPrivate('hex');
  }

  /**
   * Restore a wallet from an existing private key
   * @param {string} privateKey - hex private key
   * @returns {Wallet}
   */
  static fromPrivateKey(privateKey) {
    const wallet = new Wallet();
    wallet.keyPair = ec.keyFromPrivate(privateKey, 'hex');
    wallet.publicKey = wallet.keyPair.getPublic('hex');
    wallet.privateKey = privateKey;
    return wallet;
  }

  /**
   * Create and sign a transaction
   * @param {string} toAddress
   * @param {number} amount
   * @param {number} fee
   * @param {Blockchain} blockchain
   * @returns {Transaction}
   */
  createTransaction(toAddress, amount, fee, blockchain) {
    const balance = blockchain.getBalanceOf(this.publicKey);

    if (amount + fee > balance) {
      throw new Error(
        `Insufficient funds. Balance: ${balance}, Required: ${amount + fee}`
      );
    }

    const tx = new Transaction(this.publicKey, toAddress, amount, fee);
    tx.sign(this.keyPair);
    return tx;
  }

  /**
   * Get wallet address (shortened public key hash)
   * @returns {string}
   */
  get address() {
    return sha256(this.publicKey).substring(0, 40);
  }

  toJSON() {
    return {
      publicKey: this.publicKey,
      address: this.address,
    };
  }
}

module.exports = Wallet;
```
