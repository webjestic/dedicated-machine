## 2. Transaction

```javascript
// src/Transaction.js
const CryptoJS = require('crypto-js');
const EC = require('elliptic').ec;
const { v4: uuidv4 } = require('uuid');

const ec = new EC('secp256k1');

class Transaction {
  constructor(fromAddress, toAddress, amount, fee = 0) {
    this.id = uuidv4();
    this.fromAddress = fromAddress; // null = coinbase (mining reward)
    this.toAddress = toAddress;
    this.amount = amount;
    this.fee = fee;
    this.timestamp = Date.now();
    this.signature = null;
  }

  /**
   * Calculate the hash of the transaction
   */
  calculateHash() {
    return CryptoJS.SHA256(
      this.id +
      this.fromAddress +
      this.toAddress +
      this.amount +
      this.fee +
      this.timestamp
    ).toString();
  }

  /**
   * Sign the transaction with the sender's private key
   */
  sign(signingKey) {
    if (signingKey.getPublic('hex') !== this.fromAddress) {
      throw new Error('You cannot sign transactions for other wallets!');
    }

    const txHash = this.calculateHash();
    const sig = signingKey.sign(txHash, 'base64');
    this.signature = sig.toDER('hex');
  }

  /**
   * Validate the transaction
   */
  isValid() {
    // Coinbase transaction (mining reward)
    if (this.fromAddress === null) return true;

    if (!this.signature || this.signature.length === 0) {
      throw new Error('No signature in this transaction');
    }

    if (!this.toAddress) {
      throw new Error('Transaction must include a to address');
    }

    if (this.amount <= 0) {
      throw new Error('Transaction amount must be greater than 0');
    }

    if (this.fee < 0) {
      throw new Error('Transaction fee cannot be negative');
    }

    const publicKey = ec.keyFromPublic(this.fromAddress, 'hex');
    return publicKey.verify(this.calculateHash(), this.signature);
  }

  toJSON() {
    return {
      id: this.id,
      fromAddress: this.fromAddress,
      toAddress: this.toAddress,
      amount: this.amount,
      fee: this.fee,
      timestamp: this.timestamp,
      signature: this.signature,
    };
  }
}

module.exports = { Transaction, ec };
```

## 3. Block

```javascript
// src/Block.js
const CryptoJS = require('crypto-js');

class Block {
  constructor(index, timestamp, transactions, previousHash = '', difficulty = 4) {
    this.index = index;
    this.timestamp = timestamp;
    this.transactions = transactions;
    this.previousHash = previousHash;
    this.nonce = 0;
    this.difficulty = difficulty;
    this.hash = this.calculateHash();
    this.merkleRoot = this.calculateMerkleRoot();
  }

  /**
   * Calculate SHA256 hash of the block
   */
  calculateHash() {
    return CryptoJS.SHA256(
      this.index +
      this.previousHash +
      this.timestamp +
      JSON.stringify(this.transactions) +
      this.nonce +
      this.difficulty
    ).toString();
  }

  /**
   * Calculate Merkle Root of transactions
   */
  calculateMerkleRoot() {
    if (this.transactions.length === 0) return '';

    let hashes = this.transactions.map(tx =>
      CryptoJS.SHA256(JSON.stringify(tx)).toString()
    );

    while (hashes.length > 1) {
      if (hashes.length % 2 !== 0) {
        hashes.push(hashes[hashes.length - 1]); // Duplicate last hash if odd
      }

      const newHashes = [];
      for (let i = 0; i < hashes.length; i += 2) {
        newHashes.push(
          CryptoJS.SHA256(hashes[i] + hashes[i + 1]).toString()
        );
      }
      hashes = newHashes;
    }

    return hashes[0];
  }

  /**
   * Mine the block - Proof of Work
   */
  mineBlock(difficulty) {
    this.difficulty = difficulty;
    const target = Array(difficulty + 1).join('0');
    const startTime = Date.now();

    console.log(`⛏️  Mining block ${this.index}...`);

    while (this.hash.substring(0, difficulty) !== target) {
      this.nonce++;
      this.hash = this.calculateHash();
    }

    const elapsed = ((Date.now() - startTime) / 1000).toFixed(2);
    console.log(
      `✅ Block ${this.index} mined in ${elapsed}s | Nonce: ${this.nonce} | Hash: ${this.hash}`
    );
  }

  /**
   * Validate all transactions in the block
   */
  hasValidTransactions() {
    for (const tx of this.transactions) {
      if (!tx.isValid()) return false;
    }
    return true;
  }

  toJSON() {
    return {
      index: this.index,
      timestamp: this.timestamp,
      transactions: this.transactions.map(tx => tx.toJSON ? tx.toJSON() : tx),
      previousHash: this.previousHash,
      hash: this.hash,
      nonce: this.nonce,
      difficulty: this.difficulty,
      merkleRoot: this.merkleRoot,
    };
  }
}

module.exports = Block;
```

## 4. MemPool (Transaction Pool)

```javascript
// src/MemPool.js

class MemPool {
  constructor() {
    this.pendingTransactions = [];
  }

  /**
   * Add a transaction to the pool
   */
  addTransaction(transaction) {
    if (!transaction.fromAddress || !transaction.toAddress) {
      throw new Error('Transaction must include from and to address');
    }

    if (!transaction.isValid()) {
      throw new Error('Cannot add invalid transaction to pool');
    }

    if (transaction.amount <= 0) {
      throw new Error('Transaction amount should be higher than 0');
    }

    this.pendingTransactions.push(transaction);
    console.log(`📥 Transaction added to mempool: ${transaction.id}`);
    return transaction.id;
  }

  /**
   * Get transactions sorted by fee (highest fee first)
   */
  getTransactionsByFee(limit = 10) {
    return [...this.pendingTransactions]
      .sort((a, b) => b.fee - a.fee)
      .slice(0, limit);
  }

  /**
   * Remove transactions that were included in a block
   */
  removeConfirmedTransactions(confirmedTxIds) {
    this.pendingTransactions = this.pendingTransactions.filter(
      tx => !confirmedTxIds.includes(tx.id)
    );
  }

  /**
   * Get pool size
   */
  size() {
    return this.pendingTransactions.length;
  }

  /**
   * Check if transaction exists in pool
   */
  hasTransaction(txId) {
    return this.pendingTransactions.some(tx => tx.id === txId);
  }

  getAll() {
    return this.pendingTransactions;
  }
}

module.exports = MemPool;
```

// src/Wallet.js
const EC = require('elliptic').ec;
const CryptoJS = require('crypto-js');
const { Transaction } = require('./Transaction');

const ec = new EC('secp256k1');

class Wallet {
  constructor() {
    this.keyPair = ec.genKeyPair();
    this.publicKey = this.keyPair.getPublic('hex');
    this.privateKey = this.keyPair.getPrivate('hex');
  }

  /**
   * Create a wallet from an existing private key
   */
  static fromPrivateKey(privateKey) {
    const wallet = new Wallet();
    wallet.keyPair = ec.keyFromPrivate(privateKey, 'hex');
    wallet.publicKey = wallet.keyPair.getPublic('hex');
    wallet.privateKey = privateKey;
    return wallet;
  }

  /**
   * Create and sign a transaction
   */
  createTransaction(toAddress, amount, fee, blockchain) {
    const balance = blockchain.getBalanceOfAddress(this.publicKey);

    if (balance < amount + fee) {
      throw new Error(
        `Insufficient balance. Available: ${balance}, Required: ${amount + fee}`
      );
    }

    const tx = new Transaction(this.publicKey, toAddress, amount, fee);
    tx.sign(this.keyPair);
    return tx;
  }

  /**
   * Get wallet info (never expose private key in production!)
   */
  getInfo() {
    return {
      publicKey: this.publicKey,
      privateKey: this.privateKey, // ⚠️ Only for demo purposes
    };
  }
}

module.exports = Wallet;

## 6. Blockchain

```javascript
// src/Blockchain.js
const Block = require('./Block');
const { Transaction } = require('./Transaction');
const MemPool = require('./MemPool');

class Blockchain {
  constructor() {
    this.chain = [this.createGenesisBlock()];
    this.difficulty = 3;
    this.miningReward = 50;
    this.memPool = new MemPool();
    this.targetBlockTime = 10000; // 10 seconds in ms
    this.difficultyAdjustmentInterval = 5; // Adjust every 5 blocks
  }

  /**
   * Create the genesis block
   */
  createGenesisBlock() {
    const genesisBlock = new Block(
      0,
      new Date('2024-01-01').getTime(),
      [],
      '0',
      this.difficulty || 3
    );
    genesisBlock.hash = genesisBlock.calculateHash();
    return genesisBlock;
  }

  /**
   * Get the latest block
   */
  getLatestBlock() {
    return this.chain[this.chain.length - 1];
  }

  /**
   * Adjust mining difficulty based on block time
   */
  adjustDifficulty() {
    if (this.chain.length % this.difficultyAdjustmentInterval !== 0) {
      return this.difficulty;
    }

    const lastAdjustmentBlock =
      this.chain[this.chain.length - this.difficultyAdjustmentInterval];
    const latestBlock = this.getLatestBlock();

    const timeExpected =
      this.targetBlockTime * this.difficultyAdjustmentInterval;
    const timeTaken = latestBlock.timestamp - lastAdjustmentBlock.timestamp;

    if (timeTaken < timeExpected / 2) {
      console.log(`📈 Difficulty increased to ${this.difficulty + 1}`);
      return this.difficulty + 1;
    } else if (timeTaken > timeExpected * 2) {
      console.log(`📉 Difficulty decreased to ${Math.max(1, this.difficulty - 1)}`);
      return Math.max(1, this.difficulty - 1);
    }

    return this.difficulty;
  }

  /**
   * Mine pending transactions and add a new block
   */
  minePendingTransactions(miningRewardAddress) {
    // Adjust difficulty
    this.difficulty = this.adjustDifficulty();

    // Create coinbase (reward) transaction
    const rewardTx = new Transaction(
      null,
      miningRewardAddress,
      this.miningReward,
      0
    );

    // Get top transactions from mempool (by fee)
    const transactions = [
      rewardTx,
      ...this.memPool.getTransactionsByFee(10),
    ];

    // Create new block
    const block = new Block(
      this.chain.length,
      Date.now(),
      transactions,
      this.getLatestBlock().hash,
      this.difficulty
    );

    // Mine the block
    block.mineBlock(this.difficulty);

    // Add block to chain
    this.chain.push(block);

    // Remove confirmed transactions from mempool
    const confirmedTxIds = transactions.map(tx => tx.id);
    this.memPool.removeConfirmedTransactions(confirmedTxIds);

    console.log(`🔗 Block ${block.index} added to the chain`);
    return block;
  }

  /**
   * Add a transaction to the mempool
   */
  addTransaction(transaction) {
    if (!transaction.isValid()) {
      throw new Error('Invalid transaction');
    }

    const senderBalance = this.getBalanceOfAddress(transaction.fromAddress);
    if (senderBalance < transaction.amount + transaction.fee) {
      throw new Error(
        `Insufficient balance: ${senderBalance} < ${transaction.amount + transaction.fee}`
      );
    }

    return this.memPool.addTransaction(transaction);
  }

  /**
   * Get balance of an address
   */
  getBalanceOfAddress(address) {
    let balance = 0;

    for (const block of this.chain) {
      for (const tx of block.transactions) {
        const transaction = tx.toJSON ? tx : tx;

        if (transaction.fromAddress === address) {
          balance -= transaction.amount;
          balance -= transaction.fee;
        }

        if (transaction.toAddress === address) {
          balance += transaction.amount;
        }
      }
    }

    return balance;
  }

  /**
   * Get all transactions for an address
   */
  getTransactionsForAddress(address) {
    const txs = [];

    for (const block of this.chain) {
      for (const tx of block.transactions) {
        if (tx.fromAddress === address || tx.toAddress === address) {
          txs.push({
            ...tx.toJSON ? tx.toJSON() : tx,
            blockIndex: block.index,
            blockHash: block.hash,
          });
        }
      }
    }

    return txs;
  }

  /**
   * Validate the entire
