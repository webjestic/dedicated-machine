/**
 * Structured logger with levels and context.
 * In production you'd ship to Datadog/CloudWatch/etc.
 */
class Logger {
  constructor(context) {
    this.context = context;
  }

  _format(level, message, meta = {}) {
    return JSON.stringify({
      timestamp: new Date().toISOString(),
      level,
      context: this.context,
      message,
      ...meta,
    });
  }

  info(message, meta)  { console.log(this._format('INFO',  message, meta)); }
  warn(message, meta)  { console.warn(this._format('WARN',  message, meta)); }
  error(message, meta) { console.error(this._format('ERROR', message, meta)); }
  debug(message, meta) {
    if (process.env.DEBUG) {
      console.log(this._format('DEBUG', message, meta));
    }
  }
}

module.exports = Logger;

## src/crypto/MerkleTree.js

```javascript
const crypto = require('crypto');

/**
 * Binary Merkle Tree over transaction IDs.
 *
 * Properties:
 *  - Odd number of leaves: last leaf is duplicated (Bitcoin convention)
 *  - Any single transaction inclusion can be proven in O(log n)
 *  - Root is deterministic given the same ordered set of transactions
 */
class MerkleTree {
  constructor(transactions) {
    // Leaves are the raw transaction hashes (txid)
    this.leaves = transactions.map(tx =>
      typeof tx === 'string' ? tx : tx.txId
    );
    this.root = this._buildRoot(this.leaves);
  }

  static hash(data) {
    // Double-SHA256 — same as Bitcoin, prevents length-extension attacks
    const first  = crypto.createHash('sha256').update(data).digest();
    return crypto.createHash('sha256').update(first).digest('hex');
  }

  _buildRoot(leaves) {
    if (leaves.length === 0) return MerkleTree.hash('empty');
    if (leaves.length === 1) return leaves[0];

    let level = [...leaves];

    while (level.length > 1) {
      const nextLevel = [];

      for (let i = 0; i < level.length; i += 2) {
        const left  = level[i];
        const right = level[i + 1] ?? level[i]; // duplicate last if odd
        nextLevel.push(MerkleTree.hash(left + right));
      }

      level = nextLevel;
    }

    return level[0];
  }

  /**
   * Generate an inclusion proof for the leaf at `index`.
   * Returns an array of { hash, position } objects.
   * Verifier only needs the root and this proof — O(log n) hashes.
   */
  getProof(index) {
    if (index >= this.leaves.length) throw new Error('Index out of range');

    const proof = [];
    let level   = [...this.leaves];
    let idx     = index;

    while (level.length > 1) {
      const nextLevel = [];

      for (let i = 0; i < level.length; i += 2) {
        const left  = level[i];
        const right = level[i + 1] ?? level[i];

        if (i === idx || i + 1 === idx) {
          const siblingIdx = idx % 2 === 0 ? idx + 1 : idx - 1;
          const sibling    = level[siblingIdx] ?? level[idx];
          proof.push({
            hash:     sibling,
            position: idx % 2 === 0 ? 'right' : 'left',
          });
        }

        nextLevel.push(MerkleTree.hash(left + right));
      }

      idx   = Math.floor(idx / 2);
      level = nextLevel;
    }

    return proof;
  }

  /**
   * Verify an inclusion proof without the full tree.
   */
  static verify(leafHash, proof, root) {
    let computed = leafHash;

    for (const { hash, position } of proof) {
      if (position === 'right') {
        computed = MerkleTree.hash(computed + hash);
      } else {
        computed = MerkleTree.hash(hash + computed);
      }
    }

    return computed === root;
  }
}

module.exports = MerkleTree;
```

## src/crypto/KeyPair.js

```javascript
const crypto    = require('crypto');
const elliptic  = require('elliptic');

const ec = new elliptic.ec('secp256k1'); // Same curve as Bitcoin/Ethereum

/**
 * Wraps secp256k1 key operations.
 *
 * Private key  → 32 bytes of entropy (never leaves the wallet)
 * Public key   → 33-byte compressed point on the curve
 * Address      → RIPEMD160(SHA256(pubkey)) — shorter, collision-resistant
 */
class KeyPair {
  constructor(privateKeyHex) {
    if (privateKeyHex) {
      this.keyPair = ec.keyFromPrivate(privateKeyHex, 'hex');
    } else {
      this.keyPair = ec.genKeyPair();
    }
  }

  getPrivateKey() {
    return this.keyPair.getPrivate('hex');
  }

  getPublicKey() {
    // Compressed: 33 bytes (02/03 prefix + 32-byte X coordinate)
    return this.keyPair.getPublic(true, 'hex');
  }

  /**
   * Derive a wallet address from the public key.
   * Bitcoin-style: Base58Check would be next step for production.
   */
  getAddress() {
    const pubKey = Buffer.from(this.getPublicKey(), 'hex');

    const sha256 = crypto
      .createHash('sha256')
      .update(pubKey)
      .digest();

    // Node's built-in crypto doesn't expose RIPEMD160 in all versions
    // so we use a second SHA256 pass as a stand-in (production: use ripemd160)
    const hash160 = crypto
      .createHash('sha256')
      .update(sha256)
      .digest('hex')
      .slice(0, 40); // 20 bytes → 40 hex chars

    return hash160;
  }

  /**
   * Sign arbitrary data (we sign the hash, not the raw data).
   * Returns DER-encoded signature as hex.
   */
  sign(dataHash) {
    const signature = this.keyPair.sign(dataHash);
    return signature.toDER('hex');
  }

  /**
   * Verify a signature given a public key hex and data hash.
   */
  static verify(publicKeyHex, dataHash, signatureHex) {
    try {
      const key = ec.keyFromPublic(publicKeyHex, 'hex');
      return key.verify(dataHash, signatureHex);
    } catch {
      return false;
    }
  }

  static fromPublicKey(publicKeyHex) {
    const instance = new KeyPair('1'); // throwaway
    instance.keyPair = ec.keyFromPublic(publicKeyHex, 'hex');
    return instance;
  }
}

module.exports = KeyPair;
```

## src/core/Transaction.js

```javascript
const crypto  = require('crypto');
const KeyPair = require('../crypto/KeyPair');

/**
 * UTXO-based transaction model (Bitcoin-style).
 *
 * Each transaction:
 *  - Consumes one or more UTXOs as inputs (each input signed by its owner)
 *  - Produces one or more outputs (new UTXOs locked to recipient addresses)
 *  - Fee = sum(inputs) - sum(outputs) — goes to the miner
 *
 * Coinbase transactions have no inputs and create new coins (block reward).
 */
class Transaction {
  constructor({ inputs = [], outputs = [], timestamp, txId, isCoinbase = false }) {
    this.inputs    = inputs;    // [{ txId, outputIndex, publicKey, signature }]
    this.outputs   = outputs;   // [{ address, amount }]
    this.timestamp = timestamp || Date.now();
    this.isCoinbase = isCoinbase;
    this.txId      = txId || this._computeId();
  }

  _computeId() {
    const data = JSON.stringify({
      inputs:    this.inputs.map(i => ({ txId: i.txId, outputIndex: i.outputIndex })),
      outputs:   this.outputs,
      timestamp: this.timestamp,
      isCoinbase: this.isCoinbase,
    });
    // Double-SHA256 for txid (Bitcoin convention)
    const first  = crypto.createHash('sha256').update(data).digest();
    return crypto.createHash('sha256').update(first).digest('hex');
  }

  /**
   * The hash that input signatures commit to.
   * Covers outputs + the specific input being signed (SIGHASH_ALL simplified).
   * Production: implement full SIGHASH types.
   */
  _sigHash(inputIndex) {
    const data = JSON.stringify({
      inputs: this.inputs.map((inp, i) => ({
        txId:        inp.txId,
        outputIndex: inp.outputIndex,
        // Exclude signature from the hash (it's what we're computing)
        publicKey:   inp.publicKey,
      })),
      outputs:   this.outputs,
      timestamp: this.timestamp,
    });
    return crypto.createHash('sha256').update(data).digest('hex');
  }

  /**
   * Sign a specific input with the provided KeyPair.
   * The caller must own the UTXO referenced by this input.
   */
  signInput(inputIndex, keyPair) {
    const sigHash = this._sigHash(inputIndex);
    this.inputs[inputIndex].signature = keyPair.sign(sigHash);
    this.inputs[inputIndex].publicKey = keyPair.getPublicKey();
    // Recompute txId after signing
    this.txId = this._computeId();
    return this;
  }

  /**
   * Validate all input signatures.
   * Does NOT validate against the UTXO set — that's Blockchain's job.
   */
  verifySignatures() {
    if (this.isCoinbase) return true;

    for (let i = 0; i < this.inputs.length; i++) {
      const input   = this.inputs[i];
      const sigHash = this._sigHash(i);

      if (!input.publicKey || !input.signature) {
        return false;
      }

      if (!KeyPair.verify(input.publicKey, sigHash, input.signature)) {
        return false;
      }
    }

    return true;
  }

  /**
   * Basic structural validation (not UTXO-aware).
   */
  isValid() {
    if (this.isCoinbase) {
      return (
        this.inputs.length  === 0 &&
        this.outputs.length === 1 &&
        this.outputs[0].amount === Transaction.BLOCK_REWARD
      );
    }

    if (this.inputs.length  === 0) return false;
    if (this.outputs.length === 0) return false;

    for (const output of this.outputs) {
      if (typeof output.amount !== 'number' || output.amount <= 0) return false;
      if (!output.address) return false;
    }

    return this.verifySignatures();
  }

  /**
   * Create a coinbase transaction rewarding the miner.
   */
  static createCoinbase(minerAddress, blockHeight) {
    return new Transaction({
      inputs:    [],
      outputs:   [{ address: minerAddress, amount: Transaction.BLOCK_REWARD }],
      isCoinbase: true,
      timestamp: Date.now(),
    });
  }

  toJSON() {
    return {
      txId:      this.txId,
      inputs:    this.inputs,
      outputs:   this.outputs,
      timestamp: this.timestamp,
      isCoinbase: this.isCoinbase,
    };
  }

  static fromJSON(data) {
    return new Transaction(data);
  }
}

// Block reward — in production this halves every N blocks
Transaction.BLOCK_REWARD = 50;

module.exports = Transaction;
```

## src/core/UTXO.js

```javascript
/**
 * Unspent Transaction Output set.
 *
 * The UTXO set is the canonical source of truth for balances.
 * It's an in-memory map: `${txId}:${outputIndex}` → { address, amount }
 *
 * In production: persist to LevelDB/RocksDB with a Merkle Patricia Trie
 * so you can prove the state root in the block header.
 */
class UTXOSet {
  constructor() {
    // key: `${txId}:${outputIndex}`, value: { txId, outputIndex, address, amount }
    this.utxos = new Map();
  }

  /**
   * Apply a confirmed block's transactions to the UTXO set.
   * This is atomic — either all succeed or we throw.
   */
  applyBlock(transactions) {
    // Collect all inputs being spent in this block (for double-spend detection)
    const spentKeys = new Set();

    for (const tx of transactions) {
      if (!tx.isCoinbase) {
        for (const input of tx.inputs) {
          const
