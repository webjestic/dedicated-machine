### `src/crypto/hash.js`

```javascript
'use strict';

/**
 * Canonical hashing utilities.
 *
 * PRODUCTION-GAP addressed: All hashing goes through one module so algorithm
 * migration (e.g., SHA-256 → SHA-3) is a single-point change. Never call
 * crypto.createHash() directly outside this module.
 */

const crypto = require('crypto');

const HASH_ALGORITHM = 'sha256';
const HASH_ENCODING  = 'hex';

/**
 * Hash arbitrary data canonically.
 * Input is JSON-serialized with sorted keys to prevent hash malleability
 * from key ordering differences.
 */
function hashData(data) {
  const canonical = typeof data === 'string'
    ? data
    : canonicalSerialize(data);

  return crypto
    .createHash(HASH_ALGORITHM)
    .update(canonical, 'utf8')
    .digest(HASH_ENCODING);
}

/**
 * Double-SHA256, matching Bitcoin's block/transaction hashing convention.
 * Provides length-extension attack resistance.
 */
function doubleHash(data) {
  const first = hashData(data);
  return crypto
    .createHash(HASH_ALGORITHM)
    .update(first, HASH_ENCODING)
    .digest(HASH_ENCODING);
}

/**
 * Merkle root of an array of hex-string hashes.
 * Returns a 32-byte zero hash for empty arrays.
 */
function merkleRoot(hashes) {
  if (!hashes || hashes.length === 0) {
    return '0'.repeat(64);
  }

  let layer = [...hashes];

  while (layer.length > 1) {
    const next = [];
    for (let i = 0; i < layer.length; i += 2) {
      // Duplicate last element if odd count (standard Merkle convention)
      const left  = layer[i];
      const right = layer[i + 1] ?? layer[i];
      next.push(
        crypto
          .createHash(HASH_ALGORITHM)
          .update(left + right, 'utf8')
          .digest(HASH_ENCODING)
      );
    }
    layer = next;
  }

  return layer[0];
}

/**
 * Serialize an object with sorted keys for canonical representation.
 * This prevents hash malleability from JSON key ordering.
 */
function canonicalSerialize(obj) {
  return JSON.stringify(obj, Object.keys(obj).sort());
}

/**
 * Constant-time comparison to prevent timing attacks on hash comparisons.
 */
function safeEqual(a, b) {
  if (typeof a !== 'string' || typeof b !== 'string') return false;
  if (a.length !== b.length) return false;
  return crypto.timingSafeEqual(Buffer.from(a, 'hex'), Buffer.from(b, 'hex'));
}

module.exports = { hashData, doubleHash, merkleRoot, canonicalSerialize, safeEqual };
```

### `src/crypto/keys.js`

```javascript
'use strict';

/**
 * Key management: generation, signing, verification, and encrypted keystores.
 *
 * PRODUCTION-GAP addressed:
 *   - Keys are never stored in plaintext; keystore uses AES-256-GCM encryption
 *     with a PBKDF2-derived key (600,000 iterations per OWASP 2023 guidance).
 *   - PRODUCTION-GAP flagged: In a real deployment, private keys should live
 *     in an HSM (e.g., AWS CloudHSM, YubiHSM) or at minimum a secrets manager.
 *     This implementation provides encrypted file keystores as a baseline.
 *   - Signatures use secp256k1 (same curve as Bitcoin/Ethereum) for ecosystem
 *     compatibility.
 */

const crypto   = require('crypto');
const elliptic = require('elliptic');
const fs       = require('fs');
const path     = require('path');

const ec = new elliptic.ec('secp256k1');

// PBKDF2 parameters — tune upward on faster hardware
const PBKDF2_ITERATIONS = 600_000;
const PBKDF2_KEYLEN     = 32; // 256 bits for AES-256
const PBKDF2_DIGEST     = 'sha512';
const AES_ALGORITHM     = 'aes-256-gcm';
const SALT_BYTES        = 32;
const IV_BYTES          = 16;

/**
 * Generate a new secp256k1 key pair.
 * Returns { privateKey: hex, publicKey: hex (compressed) }
 */
function generateKeyPair() {
  const keyPair = ec.genKeyPair();
  return {
    privateKey: keyPair.getPrivate('hex'),
    publicKey:  keyPair.getPublic(true, 'hex'), // compressed
  };
}

/**
 * Derive a public key from a private key.
 */
function privateKeyToPublic(privateKeyHex) {
  const keyPair = ec.keyFromPrivate(privateKeyHex, 'hex');
  return keyPair.getPublic(true, 'hex');
}

/**
 * Derive a wallet address from a public key.
 * Address = last 20 bytes of SHA-256(publicKey), hex-encoded, with 0x prefix.
 * PRODUCTION-GAP: Real deployments should use a checksum address format
 * (EIP-55 or similar) to prevent address typo errors.
 */
function publicKeyToAddress(publicKeyHex) {
  const hash = crypto
    .createHash('sha256')
    .update(publicKeyHex, 'hex')
    .digest('hex');
  return '0x' + hash.slice(-40); // last 20 bytes
}

/**
 * Sign data with a private key.
 * Returns DER-encoded signature as hex string.
 * The data is hashed before signing (sign the hash, not the data directly).
 */
function sign(dataHash, privateKeyHex) {
  if (typeof dataHash !== 'string' || dataHash.length !== 64) {
    throw new Error('sign() requires a 32-byte hex hash');
  }
  const keyPair  = ec.keyFromPrivate(privateKeyHex, 'hex');
  const signature = keyPair.sign(dataHash);
  return signature.toDER('hex');
}

/**
 * Verify a signature against a public key and data hash.
 * Returns boolean.
 */
function verify(dataHash, signatureHex, publicKeyHex) {
  try {
    const key = ec.keyFromPublic(publicKeyHex, 'hex');
    return key.verify(dataHash, signatureHex);
  } catch {
    // Any malformed input returns false rather than throwing
    return false;
  }
}

/**
 * Encrypt a private key into a keystore object.
 * The keystore format is intentionally similar to Ethereum's Web3 Secret
 * Storage Definition for tooling compatibility.
 */
function encryptKeystore(privateKeyHex, password) {
  const salt = crypto.randomBytes(SALT_BYTES);
  const iv   = crypto.randomBytes(IV_BYTES);

  const derivedKey = crypto.pbkdf2Sync(
    password,
    salt,
    PBKDF2_ITERATIONS,
    PBKDF2_KEYLEN,
    PBKDF2_DIGEST
  );

  const cipher = crypto.createCipheriv(AES_ALGORITHM, derivedKey, iv);
  const encrypted = Buffer.concat([
    cipher.update(privateKeyHex, 'utf8'),
    cipher.final(),
  ]);
  const authTag = cipher.getAuthTag();

  return {
    version: 1,
    address: publicKeyToAddress(privateKeyToPublic(privateKeyHex)),
    crypto: {
      cipher:     AES_ALGORITHM,
      ciphertext: encrypted.toString('hex'),
      iv:         iv.toString('hex'),
      authTag:    authTag.toString('hex'),
      kdf:        'pbkdf2',
      kdfparams: {
        iterations: PBKDF2_ITERATIONS,
        keylen:     PBKDF2_KEYLEN,
        digest:     PBKDF2_DIGEST,
        salt:       salt.toString('hex'),
      },
    },
  };
}

/**
 * Decrypt a keystore object with a password.
 * Throws on wrong password or tampered ciphertext (GCM auth tag mismatch).
 */
function decryptKeystore(keystore, password) {
  const { kdfparams, iv, ciphertext, authTag } = keystore.crypto;

  const derivedKey = crypto.pbkdf2Sync(
    password,
    Buffer.from(kdfparams.salt, 'hex'),
    kdfparams.iterations,
    kdfparams.keylen,
    kdfparams.digest
  );

  const decipher = crypto.createDecipheriv(
    AES_ALGORITHM,
    derivedKey,
    Buffer.from(iv, 'hex')
  );
  decipher.setAuthTag(Buffer.from(authTag, 'hex'));

  const decrypted = Buffer.concat([
    decipher.update(Buffer.from(ciphertext, 'hex')),
    decipher.final(), // throws if auth tag invalid
  ]);

  return decrypted.toString('utf8');
}

/**
 * Save a keystore to disk.
 */
function saveKeystore(keystore, keystoreDir) {
  fs.mkdirSync(keystoreDir, { recursive: true });
  const filename = `${keystore.address}.json`;
  const filepath = path.join(keystoreDir, filename);
  fs.writeFileSync(filepath, JSON.stringify(keystore, null, 2), { mode: 0o600 });
  return filepath;
}

/**
 * Load a keystore from disk by address.
 */
function loadKeystore(address, keystoreDir) {
  const filename = `${address}.json`;
  const filepath = path.join(keystoreDir, filename);
  const raw = fs.readFileSync(filepath, 'utf8');
  return JSON.parse(raw);
}

module.exports = {
  generateKeyPair,
  privateKeyToPublic,
  publicKeyToAddress,
  sign,
  verify,
  encryptKeystore,
  decryptKeystore,
  saveKeystore,
  loadKeystore,
};
```
