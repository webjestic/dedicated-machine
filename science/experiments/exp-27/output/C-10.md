# Production Blockchain Implementation in Node.js

Let me be direct about what I'm building and what I'm flagging before writing a single line of code.

## Gap Analysis: What the Requirements Don't Say

Before implementation, here are the production gaps I'm identifying explicitly:

**Gaps I will address in this implementation:**
- Persistent storage (requirements say nothing about it — in-memory chains die on restart)
- Transaction mempool management with eviction policy
- Proper cryptographic key management (not toy examples)
- Chain reorganization handling
- Input validation and DoS protection on all public surfaces
- Structured logging with levels (not `console.log`)
- Graceful shutdown handling
- Configuration management (not hardcoded values)
- Fee market (without it, the mempool fills and never clears)

**Gaps I'm flagging but NOT implementing (would require separate systems):**
- **P2P networking**: A real peer discovery and gossip protocol (libp2p, devp2p) is a project unto itself. I'll implement the interface and a stub.
- **Consensus at scale**: I'm implementing Proof of Work with difficulty adjustment. PoS/BFT require validator set management that's out of scope here.
- **Wallet UX**: Key storage, HD wallets, hardware wallet integration — flagged, not built.
- **Smart contract execution**: EVM or WASM runtime — flagged, not built.
- **Archive node vs. full node vs. light client**: I'm building a full node. The others require protocol-level decisions.
- **Formal security audit**: This code is not audited. Do not use in production without one.
- **MEV protection**: Transaction ordering fairness is not addressed.

/**
 * Structured logger. Using pino for production-grade performance.
 * Never use console.log in production blockchain infrastructure —
 * you need log levels, structured fields, and the ability to ship
 * logs to aggregators (Loki, Datadog, etc.).
 */

import pino from 'pino';
import config from '../config/index.js';

const transport = config.logging.pretty
  ? pino.transport({
      target: 'pino-pretty',
      options: {
        colorize: true,
        translateTime: 'SYS:standard',
        ignore: 'pid,hostname',
      },
    })
  : undefined;

const logger = pino(
  {
    level: config.logging.level,
    // Serialize BigInt as strings — JSON.stringify chokes on BigInt
    serializers: {
      ...pino.stdSerializers,
    },
    formatters: {
      level(label) {
        return { level: label };
      },
    },
  },
  transport
);

export default logger;

/**
 * Create a child logger with a fixed component label.
 * Use this in every module: const log = createLogger('Blockchain')
 */
export function createLogger(component) {
  return logger.child({ component });
}

### `src/crypto/Hash.js`

```javascript
/**
 * Cryptographic hashing utilities.
 *
 * Design decisions:
 * - SHA-256 via Node's built-in crypto (native, audited, fast)
 * - Double-SHA-256 for block/transaction hashing (Bitcoin-style)
 *   to defend against length-extension attacks
 * - All inputs normalized to Buffer before hashing to prevent
 *   encoding ambiguity bugs
 */

import { createHash } from 'crypto';

/**
 * Single SHA-256 hash.
 * @param {string|Buffer} data
 * @returns {string} hex-encoded hash
 */
export function sha256(data) {
  const buf = Buffer.isBuffer(data) ? data : Buffer.from(data, 'utf8');
  return createHash('sha256').update(buf).digest('hex');
}

/**
 * Double SHA-256 hash (SHA256(SHA256(data))).
 * Used for all block and transaction IDs.
 * Defends against length-extension attacks on the outer hash.
 * @param {string|Buffer} data
 * @returns {string} hex-encoded hash
 */
export function doubleSha256(data) {
  const buf = Buffer.isBuffer(data) ? data : Buffer.from(data, 'utf8');
  const first = createHash('sha256').update(buf).digest();
  return createHash('sha256').update(first).digest('hex');
}

/**
 * Compute a Merkle root from an array of transaction IDs.
 * Returns the hash of an empty string for an empty array
 * (genesis block / coinbase-only blocks).
 *
 * Production note: Bitcoin uses a specific Merkle tree construction
 * with duplicate-last-element behavior for odd counts. We replicate
 * that here for compatibility, but note it has a known (non-critical)
 * CVE around duplicate transactions — a production system should use
 * a different tree construction.
 *
 * @param {string[]} txIds - array of hex transaction IDs
 * @returns {string} hex Merkle root
 */
export function computeMerkleRoot(txIds) {
  if (txIds.length === 0) {
    return doubleSha256('');
  }

  let level = [...txIds];

  while (level.length > 1) {
    const nextLevel = [];
    for (let i = 0; i < level.length; i += 2) {
      const left = level[i];
      // Duplicate last element if odd count (Bitcoin convention)
      const right = level[i + 1] ?? level[i];
      nextLevel.push(doubleSha256(left + right));
    }
    level = nextLevel;
  }

  return level[0];
}
```

### `src/crypto/KeyPair.js`

```javascript
/**
 * Elliptic curve key pair management.
 *
 * Uses secp256k1 — the same curve as Bitcoin and Ethereum.
 * The 'elliptic' library is the standard Node.js choice here.
 *
 * Production gaps flagged:
 * - Private keys are held in memory as hex strings. A production
 *   system would use a Hardware Security Module (HSM) or at minimum
 *   encrypt keys at rest using a KDF + passphrase.
 * - No BIP-32/BIP-39 HD wallet derivation implemented here.
 * - No key rotation mechanism.
 */

import elliptic from 'elliptic';
import { doubleSha256 } from './Hash.js';
import { createHash } from 'crypto';

const { ec: EC } = elliptic;
const secp256k1 = new EC('secp256k1');

export class KeyPair {
  /**
   * @param {string} [privateKeyHex] - optional existing private key
   */
  constructor(privateKeyHex) {
    if (privateKeyHex) {
      this._keyPair = secp256k1.keyFromPrivate(privateKeyHex, 'hex');
    } else {
      this._keyPair = secp256k1.genKeyPair();
    }
  }

  /**
   * Generate a new random key pair.
   * @returns {KeyPair}
   */
  static generate() {
    return new KeyPair();
  }

  /**
   * Restore a key pair from an existing private key.
   * @param {string} privateKeyHex
   * @returns {KeyPair}
   */
  static fromPrivateKey(privateKeyHex) {
    if (!/^[0-9a-fA-F]{64}$/.test(privateKeyHex)) {
      throw new Error('Invalid private key: must be 64 hex characters');
    }
    return new KeyPair(privateKeyHex);
  }

   * The private key as a hex string.
   * NEVER log this. NEVER send this over the network.

  get privateKey() {
    return this._keyPair.getPrivate('hex').padStart(64, '0');
