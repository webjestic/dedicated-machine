## src/crypto/KeyPair.js

```javascript
'use strict';

/**
 * Cryptographic key pair management using secp256k1 (same curve as Bitcoin/Ethereum).
 * All signing uses DER-encoded signatures. Public keys are compressed (33 bytes).
 */

const { ec: EC } = require('elliptic');
const crypto = require('crypto');

const ec = new EC('secp256k1');

class KeyPair {
  constructor(privateKeyHex = null) {
    this.keyPair = privateKeyHex
      ? ec.keyFromPrivate(privateKeyHex, 'hex')
      : ec.genKeyPair();
  }

  // ─── Getters ───────────────────────────────────────────────────────────────

  get privateKey() {
    return this.keyPair.getPrivate('hex');
  }

  /**
   * Compressed public key (33 bytes / 66 hex chars).
   * This is what gets stored in transactions and used as an address component.
   */
  get publicKey() {
    return this.keyPair.getPublic(true, 'hex'); // true = compressed
  }

  /**
   * Derive a deterministic address from the public key.
   * Mirrors Bitcoin's approach: RIPEMD160(SHA256(pubkey)), then hex-encode.
   */
  get address() {
    return KeyPair.publicKeyToAddress(this.publicKey);
  }

  // ─── Signing ───────────────────────────────────────────────────────────────

  /**
   * Sign arbitrary data. The data is SHA-256 hashed before signing so we
   * always feed a fixed-size digest to ECDSA regardless of input size.
   *
   * @param {string|Buffer} data
   * @returns {string} DER-encoded signature as hex
   */
  sign(data) {
    const hash = KeyPair.hash(data);
    const signature = this.keyPair.sign(hash);
    return signature.toDER('hex');
  }

  // ─── Static helpers ────────────────────────────────────────────────────────

  /**
   * Verify a DER-encoded signature against a public key and original data.
   *
   * @param {string} publicKeyHex  Compressed public key (hex)
   * @param {string} signatureHex  DER signature (hex)
   * @param {string|Buffer} data   Original data that was signed
   * @returns {boolean}
   */
  static verify(publicKeyHex, signatureHex, data) {
    try {
      const key = ec.keyFromPublic(publicKeyHex, 'hex');
      const hash = KeyPair.hash(data);
      return key.verify(hash, signatureHex);
    } catch {
      return false;
    }
  }

  /**
   * SHA-256 hash. Returns a Buffer so elliptic can consume it directly.
   */
  static hash(data) {
    return crypto
      .createHash('sha256')
      .update(typeof data === 'string' ? data : JSON.stringify(data))
      .digest();
  }

  /**
   * Derive address from compressed public key.
   * SHA-256 → RIPEMD-160 → hex string (40 chars).
   */
  static publicKeyToAddress(publicKeyHex) {
    const sha256 = crypto
      .createHash('sha256')
      .update(Buffer.from(publicKeyHex, 'hex'))
      .digest();

    return crypto
      .createHash('ripemd160')
      .update(sha256)
      .digest('hex');
  }

  toJSON() {
    return {
      publicKey: this.publicKey,
      address: this.address,
      // Never serialize the private key in production logs/APIs
    };
  }
}

module.exports = { KeyPair, ec };
```

## src/crypto/MerkleTree.js

```javascript
'use strict';

/**
 * Binary Merkle Tree implementation.
 *
 * Used to produce a single 32-byte root that commits to every transaction
 * in a block. Any single transaction change invalidates the root, making
 * tampering detectable without re-reading all transactions.
 *
 * Odd-length layers duplicate the last node (Bitcoin convention).
 */

const crypto = require('crypto');

class MerkleTree {
  /**
   * @param {string[]} leaves  Array of hex strings (e.g. transaction IDs)
   */
  constructor(leaves = []) {
    this.leaves = leaves.map(MerkleTree.hash);
    this.layers = [this.leaves];
    this._build();
  }

  // ─── Public API ────────────────────────────────────────────────────────────

  get root() {
    if (this.leaves.length === 0) return MerkleTree.hash('');
    return this.layers[this.layers.length - 1][0];
  }

  /**
   * Generate a proof-of-inclusion for the leaf at `index`.
   * Returns an array of { position: 'left'|'right', hash } objects.
   */
  getProof(index) {
    const proof = [];
    let currentIndex = index;

    for (let i = 0; i < this.layers.length - 1; i++) {
      const layer = this.layers[i];
      const isRightNode = currentIndex % 2 === 1;
      const siblingIndex = isRightNode ? currentIndex - 1 : currentIndex + 1;

      if (siblingIndex < layer.length) {
        proof.push({
          position: isRightNode ? 'left' : 'right',
          hash: layer[siblingIndex],
        });
      }

      currentIndex = Math.floor(currentIndex / 2);
    }

    return proof;
  }

  /**
   * Verify a proof-of-inclusion.
   *
   * @param {string}   leafHash  Hash of the leaf to verify
   * @param {object[]} proof     Proof array from getProof()
   * @param {string}   root      Expected Merkle root
   */
  static verify(leafHash, proof, root) {
    let computedHash = leafHash;

    for (const { position, hash } of proof) {
      computedHash = position === 'left'
        ? MerkleTree.hashPair(hash, computedHash)
        : MerkleTree.hashPair(computedHash, hash);
    }

    return computedHash === root;
  }

  // ─── Private ───────────────────────────────────────────────────────────────

  _build() {
    let currentLayer = this.leaves;

    while (currentLayer.length > 1) {
      const nextLayer = [];

      for (let i = 0; i < currentLayer.length; i += 2) {
        const left = currentLayer[i];
        // Duplicate last node if layer has odd length (Bitcoin convention)
        const right = currentLayer[i + 1] ?? currentLayer[i];
        nextLayer.push(MerkleTree.hashPair(left, right));
      }

      this.layers.push(nextLayer);
      currentLayer = nextLayer;
    }
  }

  static hash(data) {
    return crypto.createHash('sha256').update(data).digest('hex');
  }

  static hashPair(left, right) {
    return crypto
      .createHash('sha256')
      .update(left + right)
      .digest('hex');
  }
}

module.exports = MerkleTree;
```

## src/core/Transaction.js

```javascript
'use strict';

/**
 * UTXO-inspired transaction model.
 *
 * Each transaction has:
 *   - inputs:  references to previous outputs being spent
 *   - outputs: new UTXOs being created
 *   - a signature over the canonical transaction data
 *
 * The coinbase transaction (block reward) has no inputs and is created
 * by the miner. It is identified by `isCoinbase: true`.
 */

const crypto = require('crypto');
const { KeyPair } = require('../crypto/KeyPair');

// Mining reward in the smallest unit (satoshi-equivalent)
const MINING_REWARD = 50;

class TransactionInput {
  /**
   * @param {string} txId         ID of the transaction containing the output
   * @param {number} outputIndex  Index of the output in that transaction
   * @param {string} publicKey    Spender's public key (for signature verification)
   * @param {string} signature    Signature over the parent transaction's output data
   */
  constructor(txId, outputIndex, publicKey, signature) {
    this.txId = txId;
    this.outputIndex = outputIndex;
    this.publicKey = publicKey;
    this.signature = signature;
  }
}

class TransactionOutput {
  /**
   * @param {string} address  Recipient address (RIPEMD160(SHA256(pubkey)))
   * @param {number} amount   Value in smallest units
   */
  constructor(address, amount) {
    this.address = address;
    this.amount = amount;
  }
}

class Transaction {
  constructor({ id, inputs, outputs, timestamp, isCoinbase = false } = {}) {
    this.id = id || Transaction.generateId();
    this.inputs = inputs || [];
    this.outputs = outputs || [];
    this.timestamp = timestamp || Date.now();
    this.isCoinbase = isCoinbase;
  }

  // ─── Factory methods ───────────────────────────────────────────────────────

  /**
   * Create a standard (non-coinbase) transaction.
   * The wallet signs the canonical output data to authorize spending.
   *
   * @param {KeyPair}              senderKeyPair
   * @param {string}               recipientAddress
   * @param {number}               amount
   * @param {TransactionOutput[]}  utxos  Unspent outputs belonging to sender
   * @returns {Transaction}
   */
  static create(senderKeyPair, recipientAddress, amount, utxos) {
    // Calculate total available from UTXOs
    const totalAvailable = utxos.reduce((sum, utxo) => sum + utxo.amount, 0);

    if (totalAvailable < amount) {
      throw new Error(
        `Insufficient funds: have ${totalAvailable}, need ${amount}`
      );
    }

    const senderAddress = senderKeyPair.address;
    const outputs = [
      new TransactionOutput(recipientAddress, amount),
    ];

    // Return change to sender if needed
    const change = totalAvailable - amount;
    if (change > 0) {
      outputs.push(new TransactionOutput(senderAddress, change));
    }

    // Sign the canonical output data
    const outputData = Transaction.serializeOutputs(outputs);
    const signature = senderKeyPair.sign(outputData);

    // Build inputs referencing the UTXOs being spent
    // In a full UTXO model each input would reference a specific UTXO.
    // We simplify here by using the first UTXO's txId/index.
    const inputs = utxos.map((utxo, i) =>
      new TransactionInput(
        utxo.txId || 'genesis',
        utxo.outputIndex ?? i,
        senderKeyPair.publicKey,
        signature
      )
    );

    return new Transaction({ inputs, outputs });
  }

  /**
   * Create a coinbase transaction (block reward for the miner).
   *
   * @param {string} minerAddress  Address to receive the reward
   * @returns {Transaction}
   */
  static createCoinbase(minerAddress) {
    const outputs = [new TransactionOutput(minerAddress, MINING_REWARD)];
    return new Transaction({
      inputs: [],
      outputs,
      isCoinbase: true,
    });
  }

  // ─── Validation ────────────────────────────────────────────────────────────

  /**
   * Validate transaction structure and signatures.
   * Does NOT validate against the UTXO set (that's the blockchain's job).
   *
   * @returns {{ valid: boolean, error?: string }}
   */
  validate() {
    // Coinbase transactions are always structurally valid
    if (this.isCoinbase) {
      if (this.outputs.length !== 1) {
        return { valid: false, error: 'Coinbase must have exactly one output' };
      }
      if (this.outputs[0].amount !== MINING_REWARD) {
        return { valid: false, error: `Coinbase reward must be ${MINING_REWARD}` };
      }
      return { valid: true };
    }

    if (this.inputs.length === 0) {
      return { valid: false, error: 'Transaction must have at least one input' };
    }

    if (this.outputs.length === 0) {
      return { valid: false, error: 'Transaction must have at least one output' };
    }

    // Verify all output amounts are positive
    for (const output of this.outputs) {
      if (output.amount <= 0) {
        return { valid: false, error: 'Output amounts must be positive' };
      }
    }

    // Verify signatures on all inputs
    const outputData = Transaction.serializeOutputs(this.outputs);
    for (const input of this.inputs) {
      const valid = KeyPair.verify(input.publicKey, input.signature, outputData);
      if (!valid) {
        return { valid: false, error: `Invalid signature on input from ${input.publicKey}` };
      }
    }

    return { valid: true };
  }

  // ─── Helpers ───────────────────────────────────────────────
