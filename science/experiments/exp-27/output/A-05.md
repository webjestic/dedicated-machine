## `src/Block.js`

```javascript
const CryptoJS = require('crypto-js');

class Block {
  constructor(index, timestamp, transactions, previousHash = '', difficulty = 4) {
    this.index = index;
    this.timestamp = timestamp;
    this.transactions = transactions;
    this.previousHash = previousHash;
    this.difficulty = difficulty;
    this.nonce = 0;
    this.hash = this.calculateHash();
  }

  /**
   * Calculate SHA-256 hash of the block
   */
  calculateHash() {
    return CryptoJS.SHA256(
      this.index +
      this.previousHash +
      this.timestamp +
      JSON.stringify(this.transactions) +
      this.nonce +
      this.difficulty
    ).toString();
  }

  /**
   * Proof-of-Work: Mine block until hash starts with required zeros
   */
  mineBlock() {
    const target = Array(this.difficulty + 1).join('0');

    console.log(`⛏️  Mining block ${this.index} (difficulty: ${this.difficulty})...`);
    const startTime = Date.now();

    while (this.hash.substring(0, this.difficulty) !== target) {
      this.nonce++;
      this.hash = this.calculateHash();
    }

    const elapsed = ((Date.now() - startTime) / 1000).toFixed(2);
    console.log(`✅ Block ${this.index} mined in ${elapsed}s | Hash: ${this.hash}`);
  }

  /**
   * Validate block integrity
   */
  isValid() {
    const target = Array(this.difficulty + 1).join('0');
    return (
      this.hash === this.calculateHash() &&
      this.hash.substring(0, this.difficulty) === target
    );
  }

  toString() {
    return JSON.stringify(this, null, 2);
  }
}

module.exports = Block;
```

## `src/Transaction.js`

```javascript
const CryptoJS = require('crypto-js');
const { ec: EC } = require('elliptic');
const { v4: uuidv4 } = require('uuid');

const ec = new EC('secp256k1'); // Same curve as Bitcoin

class Transaction {
  constructor(fromAddress, toAddress, amount, fee = 0) {
    this.id = uuidv4();
    this.fromAddress = fromAddress; // null for coinbase (mining reward)
    this.toAddress = toAddress;
    this.amount = amount;
    this.fee = fee;
    this.timestamp = Date.now();
    this.signature = null;
  }

  /**
   * Calculate transaction hash
   */
  calculateHash() {
    return CryptoJS.SHA256(
      this.id +
      this.fromAddress +
      this.toAddress +
      this.amount +
      this.fee +
      this.timestamp
    ).toString();
  }

  /**
   * Sign transaction with private key
   */
  sign(signingKey) {
    if (signingKey.getPublic('hex') !== this.fromAddress) {
      throw new Error('❌ Cannot sign transactions for other wallets!');
    }

    if (this.amount <= 0) {
      throw new Error('❌ Transaction amount must be positive!');
    }

    const txHash = this.calculateHash();
    const sig = signingKey.sign(txHash, 'base64');
    this.signature = sig.toDER('hex');
  }

  /**
   * Verify transaction signature
   */
  isValid() {
    // Coinbase/mining reward transactions don't need a signature
    if (this.fromAddress === null) return true;

    if (!this.signature || this.signature.length === 0) {
      throw new Error('❌ No signature found in this transaction!');
    }

    const publicKey = ec.keyFromPublic(this.fromAddress, 'hex');
    return publicKey.verify(this.calculateHash(), this.signature);
  }

  isCoinbase() {
    return this.fromAddress === null;
  }
}

module.exports = { Transaction, ec };
```

## `src/Wallet.js`

```javascript
const { ec: EC } = require('elliptic');
const { Transaction } = require('./Transaction');

const ec = new EC('secp256k1');

class Wallet {
  constructor() {
    this.keyPair = ec.genKeyPair();
    this.publicKey = this.keyPair.getPublic('hex');
    this.privateKey = this.keyPair.getPrivate('hex');
  }

  /**
   * Restore wallet from existing private key
   */
  static fromPrivateKey(privateKey) {
    const wallet = new Wallet();
    wallet.keyPair = ec.keyFromPrivate(privateKey, 'hex');
    wallet.publicKey = wallet.keyPair.getPublic('hex');
    wallet.privateKey = privateKey;
    return wallet;
  }

  /**
   * Create and sign a transaction
   */
  createTransaction(toAddress, amount, fee, blockchain) {
    const balance = blockchain.getBalanceOfAddress(this.publicKey);

    if (amount <= 0) {
      throw new Error('❌ Amount must be greater than 0');
    }

    if (amount + fee > balance) {
      throw new Error(
        `❌ Insufficient funds. Balance: ${balance}, Required: ${amount + fee}`
      );
    }

    const tx = new Transaction(this.publicKey, toAddress, amount, fee);
    tx.sign(this.keyPair);
    return tx;
  }

  getInfo() {
    return {
      publicKey: this.publicKey,
      privateKey: this.privateKey,
    };
  }
}

module.exports = Wallet;
```

## `src/TransactionPool.js`

```javascript
class TransactionPool {
  constructor() {
    this.transactions = [];
  }

  /**
   * Add transaction to pool after validation
   */
  addTransaction(transaction) {
    if (!transaction.isValid()) {
      throw new Error('❌ Invalid transaction signature!');
    }

    if (this.transactions.find(tx => tx.id === transaction.id)) {
      throw new Error('❌ Transaction already in pool!');
    }

    this.transactions.push(transaction);
    console.log(`📥 Transaction added to pool: ${transaction.id}`);
    return transaction;
  }

  /**
   * Get transactions sorted by fee (highest first) for mining
   */
  getTransactionsForMining(limit = 10) {
    return [...this.transactions]
      .sort((a, b) => b.fee - a.fee)
      .slice(0, limit);
  }

  /**
   * Remove confirmed transactions from pool
   */
  clearConfirmedTransactions(confirmedTxIds) {
    const before = this.transactions.length;
    this.transactions = this.transactions.filter(
      tx => !confirmedTxIds.includes(tx.id)
    );
    console.log(`🧹 Cleared ${before - this.transactions.length} confirmed transactions`);
  }

  getSize() {
    return this.transactions.length;
  }

  clear() {
    this.transactions = [];
  }
}

module.exports = TransactionPool;
```

## `src/Blockchain.js`

```javascript
const Block = require('./Block');
const { Transaction } = require('./Transaction');

const MINING_REWARD = 50;
const INITIAL_DIFFICULTY = 3;
const DIFFICULTY_ADJUSTMENT_INTERVAL = 5; // Adjust every 5 blocks
const TARGET_BLOCK_TIME = 10000; // 10 seconds in ms

class Blockchain {
  constructor() {
    this.chain = [this.createGenesisBlock()];
    this.difficulty = INITIAL_DIFFICULTY;
    this.pendingTransactions = [];
    this.miningReward = MINING_REWARD;
  }

  /**
   * Create the first block in the chain
   */
  createGenesisBlock() {
    const genesis = new Block(0, Date.now(), [], '0', INITIAL_DIFFICULTY);
    genesis.mineBlock();
    return genesis;
  }

  getLatestBlock() {
    return this.chain[this.chain.length - 1];
  }

  /**
   * Adjust difficulty based on block mining time
   */
  adjustDifficulty() {
    if (this.chain.length % DIFFICULTY_ADJUSTMENT_INTERVAL !== 0) {
      return this.difficulty;
    }

    const lastAdjustmentBlock =
      this.chain[this.chain.length - DIFFICULTY_ADJUSTMENT_INTERVAL];
    const latestBlock = this.getLatestBlock();

    const timeExpected = TARGET_BLOCK_TIME * DIFFICULTY_ADJUSTMENT_INTERVAL;
    const timeTaken = latestBlock.timestamp - lastAdjustmentBlock.timestamp;

    if (timeTaken < timeExpected / 2) {
      console.log(`🔺 Difficulty increased to ${this.difficulty + 1}`);
      return this.difficulty + 1;
    } else if (timeTaken > timeExpected * 2) {
      const newDifficulty = Math.max(1, this.difficulty - 1);
      console.log(`🔻 Difficulty decreased to ${newDifficulty}`);
      return newDifficulty;
    }

    return this.difficulty;
  }

  /**
   * Add transaction to pending pool
   */
  addTransaction(transaction) {
    if (!transaction.fromAddress && !transaction.toAddress) {
      throw new Error('❌ Transaction must include from and to address');
    }

    if (!transaction.isValid()) {
      throw new Error('❌ Cannot add invalid transaction to chain');
    }

    if (transaction.amount <= 0) {
      throw new Error('❌ Transaction amount should be higher than 0');
    }

    // Check sender balance (skip for coinbase)
    if (!transaction.isCoinbase()) {
      const senderBalance = this.getBalanceOfAddress(transaction.fromAddress);
      if (senderBalance < transaction.amount + transaction.fee) {
        throw new Error(
          `❌ Insufficient balance. Has: ${senderBalance}, Needs: ${transaction.amount + transaction.fee}`
        );
      }
    }

    this.pendingTransactions.push(transaction);
    return transaction;
  }

  /**
   * Mine pending transactions and reward the miner
   */
  minePendingTransactions(miningRewardAddress) {
    // Collect transaction fees
    const totalFees = this.pendingTransactions.reduce(
      (sum, tx) => sum + (tx.fee || 0), 0
    );

    // Create coinbase transaction (mining reward + fees)
    const rewardTx = new Transaction(
      null,
      miningRewardAddress,
      this.miningReward + totalFees
    );

    const transactions = [rewardTx, ...this.pendingTransactions];

    // Adjust difficulty
    this.difficulty = this.adjustDifficulty();

    // Create and mine new block
    const block = new Block(
      this.chain.length,
      Date.now(),
      transactions,
      this.getLatestBlock().hash,
      this.difficulty
    );

    block.mineBlock();

    this.chain.push(block);
    this.pendingTransactions = []; // Clear pending transactions

    console.log(`💰 Mining reward of ${this.miningReward + totalFees} sent to ${miningRewardAddress.substring(0, 20)}...`);

    return block;
  }

  /**
   * Get confirmed balance for an address
   */
  getBalanceOfAddress(address) {
    let balance = 0;

    for (const block of this.chain) {
      for (const tx of block.transactions) {
        if (tx.fromAddress === address) {
          balance -= tx.amount + (tx.fee || 0);
        }
        if (tx.toAddress === address) {
          balance += tx.amount;
        }
      }
    }

    return balance;
  }

  /**
   * Get all transactions for an address
   */
  getTransactionsForAddress(address) {
    const txs = [];

    for (const block of this.chain) {
      for (const tx of block.transactions) {
        if (tx.fromAddress === address || tx.toAddress === address) {
          txs.push({ ...tx, blockIndex: block.index });
        }
      }
    }

    return txs;
  }

  /**
   * Validate entire blockchain
   */
  isChainValid() {
    for (let i = 1; i < this.chain.length; i++) {
      const currentBlock = this.chain[i];
      const previousBlock = this.chain[i - 1];

      // Validate all transactions in block
      for (const tx of currentBlock.transactions) {
        if (!tx.isValid()) {
          console.error(`❌ Invalid transaction in block ${i}`);
          return false;
        }
      }

      // Validate block hash
      if (!currentBlock.isValid()) {
        console.error(`❌ Block ${i} has invalid hash`);
        return false;
      }

      // Validate chain linkage
      if (currentBlock.previousHash !== previousBlock.hash) {
        console.error(`❌ Block ${i} has broken chain link`);
        return false;
      }
    }

    return true;
  }

  /**
   * Replace chain with a longer valid chain (consensus)
   */
  replaceChain(newChain) {
    if (newChain.length <= this.chain.length) {
      console.log('❌ Received chain is not longer than current chain');
      return false;
    }

    // Temporarily replace and validate
    const originalChain = this.chain;
    this.chain = newChain;

    if (!this.isChainValid()) {
      console.log('❌ Received chain is invalid');
      this.chain = originalChain;
      return false;
    }

    console.log('✅ Chain replaced with longer valid chain');
    return true;
  }

  getStats() {
    return {
      height: this.chain.length,
      difficulty: this.difficulty,
      pendingTransactions: this.pendingTransactions.length,
      miningReward: this.miningReward,
      isValid: this.isChainValid(),
    };
  }
}

module.exports = { Blockchain, MINING_REWARD };
```
