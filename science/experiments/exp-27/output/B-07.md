```javascript
// src/crypto/KeyPair.js
'use strict';

const { createHash } = require('crypto');
const EC = require('elliptic').ec;

const ec = new EC('secp256k1');

/**
 * Deterministic SHA-256 hash. Returns hex string.
 */
function sha256(data) {
  return createHash('sha256')
    .update(typeof data === 'string' ? data : JSON.stringify(data))
    .digest('hex');
}

/**
 * Double SHA-256 — same pattern Bitcoin uses to avoid
 * length-extension attacks on block headers.
 */
function hash256(data) {
  return sha256(sha256(data));
}

/**
 * RIPEMD-160(SHA-256(data)) — standard address derivation.
 */
function hash160(data) {
  const sha = createHash('sha256')
    .update(typeof data === 'string' ? Buffer.from(data, 'hex') : data)
    .digest();
  return createHash('ripemd160').update(sha).digest('hex');
}

class KeyPair {
  constructor(privateKeyHex = null) {
    this._ec = ec;
    this._key = privateKeyHex
      ? ec.keyFromPrivate(privateKeyHex, 'hex')
      : ec.genKeyPair();
  }

  get privateKey() {
    return this._key.getPrivate('hex');
  }

  get publicKey() {
    // Compressed 33-byte public key — standard for Bitcoin-style addresses
    return this._key.getPublic(true, 'hex');
  }

  /**
   * Derive a wallet address from the public key.
   * Address = first 40 chars of RIPEMD160(SHA256(pubkey))
   * Prefixed with '0x' for readability.
   */
  get address() {
    return '0x' + hash160(this.publicKey).slice(0, 40);
  }

  /**
   * Sign arbitrary data. Returns DER-encoded signature as hex.
   * We sign the hash of the data, never raw data.
   */
  sign(data) {
    const msgHash = sha256(
      typeof data === 'string' ? data : JSON.stringify(data)
    );
    const signature = this._key.sign(msgHash);
    return signature.toDER('hex');
  }

  /**
   * Verify a signature against a public key (hex) and data.
   */
  static verify(publicKeyHex, data, signatureHex) {
    try {
      const key = ec.keyFromPublic(publicKeyHex, 'hex');
      const msgHash = sha256(
        typeof data === 'string' ? data : JSON.stringify(data)
      );
      return key.verify(msgHash, signatureHex);
    } catch {
      return false;
    }
  }

  toJSON() {
    return {
      privateKey: this.privateKey,
      publicKey: this.publicKey,
      address: this.address,
    };
  }

  static fromPrivateKey(hex) {
    return new KeyPair(hex);
  }
}

module.exports = { KeyPair, sha256, hash256, hash160 };
```

```javascript
// src/crypto/MerkleTree.js
'use strict';

const { sha256 } = require('./KeyPair');

/**
 * Binary Merkle Tree implementation.
 *
 * Properties:
 * - Deterministic root for any ordered set of leaves
 * - O(log n) inclusion proofs
 * - Odd-length layers duplicate the last node (Bitcoin convention)
 */
class MerkleTree {
  constructor(leaves = []) {
    // Leaves are hashes of transaction IDs
    this._leaves = leaves.map((l) => sha256(l));
    this._layers = this._buildLayers(this._leaves);
  }

  _buildLayers(leaves) {
    if (leaves.length === 0) return [[]];

    const layers = [leaves];
    let current = leaves;

    while (current.length > 1) {
      const next = [];
      for (let i = 0; i < current.length; i += 2) {
        const left = current[i];
        // Duplicate last node if odd count — Bitcoin convention
        const right = current[i + 1] || current[i];
        next.push(sha256(left + right));
      }
      layers.push(next);
      current = next;
    }

    return layers;
  }

  get root() {
    if (this._leaves.length === 0) {
      // Empty block gets a deterministic zero-hash root
      return sha256('empty');
    }
    return this._layers[this._layers.length - 1][0];
  }

  /**
   * Generate an inclusion proof for leaf at index.
   * Returns array of { hash, position: 'left'|'right' }
   */
  getProof(index) {
    if (index < 0 || index >= this._leaves.length) {
      throw new Error(`Index ${index} out of bounds`);
    }

    const proof = [];
    let currentIndex = index;

    for (let i = 0; i < this._layers.length - 1; i++) {
      const layer = this._layers[i];
      const isRightNode = currentIndex % 2 === 1;
      const siblingIndex = isRightNode ? currentIndex - 1 : currentIndex + 1;

      if (siblingIndex < layer.length) {
        proof.push({
          hash: layer[siblingIndex],
          position: isRightNode ? 'left' : 'right',
        });
      }

      currentIndex = Math.floor(currentIndex / 2);
    }

    return proof;
  }

  /**
   * Verify an inclusion proof against a known root.
   */
  static verify(leafData, proof, root) {
    let hash = sha256(leafData);

    for (const { hash: siblingHash, position } of proof) {
      if (position === 'left') {
        hash = sha256(siblingHash + hash);
      } else {
        hash = sha256(hash + siblingHash);
      }
    }

    return hash === root;
  }
}

module.exports = { MerkleTree };
```

## Step 3: Transaction Model

```javascript
// src/core/Transaction.js
'use strict';

const { sha256 } = require('../crypto/KeyPair');
const { KeyPair } = require('../crypto/KeyPair');

/**
 * Transaction states
 */
const TxStatus = Object.freeze({
  PENDING: 'PENDING',
  CONFIRMED: 'CONFIRMED',
  REJECTED: 'REJECTED',
});

/**
 * A UTXO-inspired transaction model.
 *
 * Each transaction has:
 * - inputs:  references to previous outputs being spent
 * - outputs: new UTXOs being created
 * - fee:     explicit miner fee (inputs_sum - outputs_sum)
 *
 * Coinbase transactions (block rewards) have no inputs.
 */
class Transaction {
  constructor({ inputs, outputs, timestamp, data } = {}) {
    this.inputs = inputs || [];   // [{ txId, outputIndex, publicKey, signature }]
    this.outputs = outputs || []; // [{ address, amount }]
    this.timestamp = timestamp || Date.now();
    this.data = data || null;     // Optional arbitrary payload (smart contract stub)
    this.id = this._computeId();
  }

  _computeId() {
    // ID commits to all structural fields — NOT signatures
    // This matches Bitcoin's approach: txid is stable pre-signing
    return sha256(
      JSON.stringify({
        inputs: this.inputs.map((i) => ({
          txId: i.txId,
          outputIndex: i.outputIndex,
        })),
        outputs: this.outputs,
        timestamp: this.timestamp,
        data: this.data,
      })
    );
  }

  /**
   * The signing payload for a specific input.
   * Each input signs: txId + outputIndex + all outputs
   * This prevents output tampering after signing.
   */
  _signingPayload(inputIndex) {
    return JSON.stringify({
      txId: this.id,
      inputIndex,
      outputs: this.outputs,
      timestamp: this.timestamp,
    });
  }

  /**
   * Sign a specific input with a KeyPair.
   */
  signInput(inputIndex, keyPair) {
    if (inputIndex >= this.inputs.length) {
      throw new Error(`Input index ${inputIndex} does not exist`);
    }
    const payload = this._signingPayload(inputIndex);
    this.inputs[inputIndex].signature = keyPair.sign(payload);
    this.inputs[inputIndex].publicKey = keyPair.publicKey;
  }

  /**
   * Verify all input signatures.
   */
  verifySignatures() {
    for (let i = 0; i < this.inputs.length; i++) {
      const input = this.inputs[i];
      if (!input.signature || !input.publicKey) {
        return false;
      }
      const payload = this._signingPayload(i);
      if (!KeyPair.verify(input.publicKey, payload, input.signature)) {
        return false;
      }
    }
    return true;
  }

  /**
   * Is this a coinbase transaction (block reward)?
   */
  isCoinbase() {
    return this.inputs.length === 0;
  }

  /**
   * Total output value
   */
  totalOutput() {
    return this.outputs.reduce((sum, o) => sum + o.amount, 0);
  }

  /**
   * Serialize for network/storage
   */
  serialize() {
    return JSON.stringify(this.toJSON());
  }

  toJSON() {
    return {
      id: this.id,
      inputs: this.inputs,
      outputs: this.outputs,
      timestamp: this.timestamp,
      data: this.data,
    };
  }

  static deserialize(raw) {
    const data = typeof raw === 'string' ? JSON.parse(raw) : raw;
    const tx = new Transaction({
      inputs: data.inputs,
      outputs: data.outputs,
      timestamp: data.timestamp,
      data: data.data,
    });
    // Restore the original ID — don't recompute (signatures are already attached)
    tx.id = data.id;
    return tx;
  }

  /**
   * Create a coinbase transaction (miner reward)
   */
  static createCoinbase(minerAddress, reward, blockHeight) {
    return new Transaction({
      inputs: [],
      outputs: [{ address: minerAddress, amount: reward }],
      data: { type: 'coinbase', blockHeight },
    });
  }
}

module.exports = { Transaction, TxStatus };
```

// src/core/Block.js
'use strict';

const { hash256 } = require('../crypto/KeyPair');
const { MerkleTree } = require('../crypto/MerkleTree');
const { Transaction } = require('./Transaction');

class Block {
  constructor({
    height,
    previousHash,
    transactions,
    timestamp,
    nonce,
    difficulty,
    miner,
  } = {}) {
    this.height = height;
    this.previousHash = previousHash;
    this.transactions = transactions || [];
    this.timestamp = timestamp || Date.now();
    this.nonce = nonce || 0;
    this.difficulty = difficulty || 4;
    this.miner = miner || null;

    // Computed fields
    this.merkleRoot = this._computeMerkleRoot();
    this.hash = this._computeHash();
  }

  _computeMerkleRoot() {
    const txIds = this.transactions.map((tx) => tx.id);
    return new MerkleTree(txIds).root;
  }

  /**
   * Block hash commits to the full header.
   * We use double-SHA256 — same as Bitcoin.
   */
  _computeHash() {
    return hash256(this._headerString());
  }

  _headerString() {
    return JSON.stringify({
      height: this.height,
      previousHash: this.previousHash,
      merkleRoot: this.merkleRoot,
      timestamp: this.timestamp,
      nonce: this.nonce,
      difficulty: this.difficulty,
      miner: this.miner,
    });
  }

   * Recompute hash after nonce changes during mining.
   * Mutates in place — only called by the miner.

  rehash() {
    this.hash = this._computeHash();
    return this.hash;
