## `src/utils/crypto.js`

```javascript
const SHA256 = require('crypto-js/sha256');
const EC = require('elliptic').ec;

const ec = new EC('secp256k1');

/**
 * Generate a SHA256 hash from any number of arguments
 */
const generateHash = (...args) => {
  return SHA256(
    args.map(arg => JSON.stringify(arg)).join('')
  ).toString();
};

/**
 * Generate a new key pair (private/public keys)
 */
const generateKeyPair = () => {
  const keyPair = ec.genKeyPair();
  return {
    privateKey: keyPair.getPrivate('hex'),
    publicKey: keyPair.getPublic('hex'),
  };
};

/**
 * Sign data with a private key
 */
const signData = (privateKey, data) => {
  const keyPair = ec.keyFromPrivate(privateKey, 'hex');
  const hash = generateHash(data);
  return keyPair.sign(hash).toDER('hex');
};

/**
 * Verify a signature against a public key
 */
const verifySignature = (publicKey, data, signature) => {
  try {
    const keyPair = ec.keyFromPublic(publicKey, 'hex');
    const hash = generateHash(data);
    return keyPair.verify(hash, signature);
  } catch (err) {
    return false;
  }
};

module.exports = { generateHash, generateKeyPair, signData, verifySignature };
```

## `src/Transaction.js`

```javascript
const { v4: uuidv4 } = require('uuid');
const { generateHash, signData, verifySignature } = require('./utils/crypto');

const TRANSACTION_FEE = 0.001;

class Transaction {
  constructor({ id, from, to, amount, fee, timestamp, signature, type } = {}) {
    this.id = id || uuidv4();
    this.from = from || null;       // null = coinbase (mining reward)
    this.to = to;
    this.amount = amount;
    this.fee = fee || TRANSACTION_FEE;
    this.timestamp = timestamp || Date.now();
    this.signature = signature || null;
    this.type = type || 'transfer'; // 'transfer' | 'coinbase'
  }

  /**
   * Create the data object used for signing/hashing
   */
  get signingData() {
    return {
      id: this.id,
      from: this.from,
      to: this.to,
      amount: this.amount,
      fee: this.fee,
      timestamp: this.timestamp,
      type: this.type,
    };
  }

  /**
   * Sign the transaction with sender's private key
   */
  sign(privateKey) {
    if (this.type === 'coinbase') {
      throw new Error('Cannot sign a coinbase transaction');
    }
    this.signature = signData(privateKey, this.signingData);
    return this;
  }

  /**
   * Validate the transaction
   */
  isValid() {
    // Coinbase transactions don't need a signature
    if (this.type === 'coinbase') {
      return this.to && this.amount > 0;
    }

    if (!this.from || !this.to || !this.signature) {
      return false;
    }

    if (this.amount <= 0) {
      return false;
    }

    if (this.from === this.to) {
      return false;
    }

    return verifySignature(this.from, this.signingData, this.signature);
  }

  /**
   * Create a coinbase (mining reward) transaction
   */
  static createCoinbase(minerAddress, reward) {
    return new Transaction({
      from: null,
      to: minerAddress,
      amount: reward,
      fee: 0,
      type: 'coinbase',
    });
  }

  toJSON() {
    return { ...this };
  }
}

module.exports = { Transaction, TRANSACTION_FEE };
```

## `src/Block.js`

```javascript
const { generateHash } = require('./utils/crypto');

class Block {
  constructor({
    index,
    previousHash,
    transactions,
    timestamp,
    nonce,
    hash,
    difficulty,
    merkleRoot,
  } = {}) {
    this.index = index;
    this.previousHash = previousHash;
    this.transactions = transactions || [];
    this.timestamp = timestamp || Date.now();
    this.nonce = nonce || 0;
    this.difficulty = difficulty || 3;
    this.merkleRoot = merkleRoot || this.calculateMerkleRoot();
    this.hash = hash || this.calculateHash();
  }

  /**
   * Calculate the block hash
   */
  calculateHash() {
    return generateHash(
      this.index,
      this.previousHash,
      this.merkleRoot,
      this.timestamp,
      this.nonce,
      this.difficulty
    );
  }

  /**
   * Build a simple Merkle root from transaction IDs
   */
  calculateMerkleRoot() {
    if (this.transactions.length === 0) return generateHash('');

    let hashes = this.transactions.map(tx =>
      generateHash(tx.id, tx.from, tx.to, tx.amount)
    );

    while (hashes.length > 1) {
      if (hashes.length % 2 !== 0) {
        hashes.push(hashes[hashes.length - 1]); // duplicate last hash
      }
      const newHashes = [];
      for (let i = 0; i < hashes.length; i += 2) {
        newHashes.push(generateHash(hashes[i] + hashes[i + 1]));
      }
      hashes = newHashes;
    }

    return hashes[0];
  }

  /**
   * Proof-of-Work mining
   */
  mine(difficulty) {
    this.difficulty = difficulty;
    const target = '0'.repeat(difficulty);

    console.log(`⛏  Mining block ${this.index} (difficulty: ${difficulty})...`);
    const startTime = Date.now();

    while (!this.hash.startsWith(target)) {
      this.nonce++;
      this.hash = this.calculateHash();
    }

    const elapsed = ((Date.now() - startTime) / 1000).toFixed(2);
    console.log(`✅ Block ${this.index} mined in ${elapsed}s | Hash: ${this.hash}`);

    return this;
  }

  /**
   * Validate the block's integrity
   */
  isValid(previousBlock) {
    // Check hash integrity
    if (this.hash !== this.calculateHash()) {
      return { valid: false, error: 'Invalid block hash' };
    }

    // Check Merkle root
    if (this.merkleRoot !== this.calculateMerkleRoot()) {
      return { valid: false, error: 'Invalid Merkle root' };
    }

    // Check proof of work
    if (!this.hash.startsWith('0'.repeat(this.difficulty))) {
      return { valid: false, error: 'Block does not meet difficulty requirement' };
    }

    // Check link to previous block
    if (previousBlock && this.previousHash !== previousBlock.hash) {
      return { valid: false, error: 'Previous hash mismatch' };
    }

    // Validate all transactions
    for (const tx of this.transactions) {
      if (!tx.isValid()) {
        return { valid: false, error: `Invalid transaction: ${tx.id}` };
      }
    }

    return { valid: true };
  }

  /**
   * Genesis block factory
   */
  static genesis() {
    return new Block({
      index: 0,
      previousHash: '0'.repeat(64),
      transactions: [],
      timestamp: 1700000000000,
      nonce: 0,
      difficulty: 1,
      hash: '0000genesis0000000000000000000000000000000000000000000000000000000',
    });
  }

  toJSON() {
    return { ...this };
  }
}

module.exports = Block;
```

## `src/MemPool.js`

```javascript
const { TRANSACTION_FEE } = require('./Transaction');

class MemPool {
  constructor() {
    this.transactions = new Map(); // id -> Transaction
  }

  /**
   * Add a validated transaction to the pool
   */
  add(transaction) {
    if (!transaction.isValid()) {
      throw new Error(`Transaction ${transaction.id} is invalid`);
    }

    if (this.transactions.has(transaction.id)) {
      throw new Error(`Transaction ${transaction.id} already in mempool`);
    }

    this.transactions.set(transaction.id, transaction);
    console.log(`📥 Transaction ${transaction.id} added to mempool`);
    return true;
  }

  /**
   * Get transactions sorted by fee (highest first), limited by count
   */
  getTopTransactions(limit = 10) {
    return [...this.transactions.values()]
      .sort((a, b) => b.fee - a.fee)
      .slice(0, limit);
  }

  /**
   * Remove confirmed transactions from the pool
   */
  removeConfirmed(transactions) {
    transactions.forEach(tx => this.transactions.delete(tx.id));
  }

  /**
   * Get mempool size
   */
  get size() {
    return this.transactions.size;
  }

  /**
   * Get all transactions as array
   */
  getAll() {
    return [...this.transactions.values()];
  }

  /**
   * Check if a transaction exists in the pool
   */
  has(txId) {
    return this.transactions.has(txId);
  }

  toJSON() {
    return this.getAll();
  }
}

module.exports = MemPool;
```

## `src/Wallet.js`

```javascript
const { generateKeyPair, generateHash } = require('./utils/crypto');
const { Transaction } = require('./Transaction');

class Wallet {
  constructor(privateKey = null) {
    if (privateKey) {
      const EC = require('elliptic').ec;
      const ec = new EC('secp256k1');
      const keyPair = ec.keyFromPrivate(privateKey, 'hex');
      this.privateKey = privateKey;
      this.publicKey = keyPair.getPublic('hex');
    } else {
      const keys = generateKeyPair();
      this.privateKey = keys.privateKey;
      this.publicKey = keys.publicKey;
    }

    // The wallet address is derived from the public key
    this.address = generateHash(this.publicKey).slice(0, 40);
  }

  /**
   * Create and sign a transaction
   */
  createTransaction(to, amount, blockchain) {
    const balance = blockchain.getBalance(this.publicKey);

    if (amount <= 0) {
      throw new Error('Amount must be positive');
    }

    if (balance < amount + Transaction.TRANSACTION_FEE) {
      throw new Error(
        `Insufficient balance. Have: ${balance}, Need: ${amount + 0.001}`
      );
    }

    const tx = new Transaction({
      from: this.publicKey,
      to,
      amount,
    });

    tx.sign(this.privateKey);
    return tx;
  }

  /**
   * Get wallet info (never expose private key in production!)
   */
  toJSON() {
    return {
      address: this.address,
      publicKey: this.publicKey,
      // NOTE: Never expose private key in a real application
    };
  }
}

module.exports = Wallet;
```

   * Get the latest block

    return this.chain[this.chain.length - 1];

   * Get chain length
   */

    return this.chain.length - 1;
  }

   * Add a transaction to the mempool
   */
  addTransaction(transaction) {
    if (!transaction.isValid()) {
      throw new Error('Invalid transaction');
    }

    // Prevent double spending by checking pending transactions
    if (this.isDoubleSpend(transaction)) {
      throw new Error('Double spend detected');
    }

    // Verify sender has enough balance
    if (transaction.type !== 'coinbase') {
      const balance = this.getBalance(transaction.from);
      if (balance < transaction.amount + transaction.fee) {
        throw new Error(
          `Insufficient balance: ${balance} < ${transaction.amount + transaction.fee}`
        );
      }
    }

    return this.memPool.add(transaction);
