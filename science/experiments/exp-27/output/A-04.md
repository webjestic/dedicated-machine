```javascript
const crypto = require('crypto');

/**
 * Creates a SHA-256 hash of the given data.
 * @param {string} data - The data to hash.
 * @returns {string} - The hex-encoded hash.
 */
const sha256 = (data) => {
  return crypto.createHash('sha256').update(data).digest('hex');
};

/**
 * Creates a double SHA-256 hash (like Bitcoin).
 * @param {string} data
 * @returns {string}
 */
const doubleSha256 = (data) => {
  return sha256(sha256(data));
};

module.exports = { sha256, doubleSha256 };
```

## 3. `src/utils/merkle.js`

```javascript
const { sha256 } = require('./hash');

/**
 * Builds a Merkle Tree from an array of transaction IDs.
 * Returns the Merkle Root hash.
 * @param {string[]} txIds - Array of transaction IDs.
 * @returns {string} - The Merkle root hash.
 */
const buildMerkleRoot = (txIds) => {
  if (!txIds || txIds.length === 0) {
    return sha256('');
  }

  if (txIds.length === 1) {
    return txIds[0];
  }

  let level = [...txIds];

  while (level.length > 1) {
    const nextLevel = [];

    // If odd number of elements, duplicate the last one
    if (level.length % 2 !== 0) {
      level.push(level[level.length - 1]);
    }

    for (let i = 0; i < level.length; i += 2) {
      const combined = level[i] + level[i + 1];
      nextLevel.push(sha256(combined));
    }

    level = nextLevel;
  }

  return level[0];
};

module.exports = { buildMerkleRoot };
```

## 4. `src/Transaction.js`

```javascript
const { sha256 } = require('./utils/hash');
const { v4: uuidv4 } = require('uuid');

class Transaction {
  /**
   * @param {string} fromAddress - Sender's public key (null for coinbase).
   * @param {string} toAddress   - Recipient's public key.
   * @param {number} amount      - Amount to transfer.
   * @param {number} fee         - Transaction fee for miners.
   */
  constructor(fromAddress, toAddress, amount, fee = 0) {
    this.id = uuidv4();
    this.fromAddress = fromAddress;
    this.toAddress = toAddress;
    this.amount = amount;
    this.fee = fee;
    this.timestamp = Date.now();
    this.signature = null;
  }

  /**
   * Calculates the hash of the transaction data (used for signing).
   * @returns {string}
   */
  calculateHash() {
    return sha256(
      this.id +
      this.fromAddress +
      this.toAddress +
      this.amount +
      this.fee +
      this.timestamp
    );
  }

  /**
   * Signs the transaction with the sender's private key.
   * @param {object} keyPair - EC key pair from the elliptic library.
   */
  sign(keyPair) {
    const txHash = this.calculateHash();
    const signature = keyPair.sign(txHash, 'base64');
    this.signature = signature.toDER('hex');
  }

  /**
   * Validates the transaction signature and data.
   * @param {object} ec - Elliptic curve instance.
   * @returns {boolean}
   */
  isValid(ec) {
    // Coinbase transactions (mining rewards) have no sender
    if (this.fromAddress === null) {
      return true;
    }

    if (!this.signature || this.signature.length === 0) {
      throw new Error('Transaction has no signature.');
    }

    if (this.amount <= 0) {
      throw new Error('Transaction amount must be positive.');
    }

    if (this.fee < 0) {
      throw new Error('Transaction fee cannot be negative.');
    }

    try {
      const publicKey = ec.keyFromPublic(this.fromAddress, 'hex');
      return publicKey.verify(this.calculateHash(), this.signature);
    } catch (err) {
      throw new Error(`Signature verification failed: ${err.message}`);
    }
  }

  /**
   * Returns a plain object representation.
   * @returns {object}
   */
  toJSON() {
    return {
      id: this.id,
      fromAddress: this.fromAddress,
      toAddress: this.toAddress,
      amount: this.amount,
      fee: this.fee,
      timestamp: this.timestamp,
      signature: this.signature,
    };
  }
}

module.exports = Transaction;
```

## 5. `src/Block.js`

```javascript
const { sha256 } = require('./utils/hash');
const { buildMerkleRoot } = require('./utils/merkle');

class Block {
  /**
   * @param {number}        index        - Block index in the chain.
   * @param {Transaction[]} transactions - List of transactions.
   * @param {string}        previousHash - Hash of the previous block.
   * @param {number}        difficulty   - Mining difficulty.
   */
  constructor(index, transactions, previousHash = '', difficulty = 4) {
    this.index = index;
    this.timestamp = Date.now();
    this.transactions = transactions;
    this.previousHash = previousHash;
    this.difficulty = difficulty;
    this.nonce = 0;
    this.merkleRoot = this.calculateMerkleRoot();
    this.hash = this.calculateHash();
  }

  /**
   * Calculates the Merkle root of all transactions in the block.
   * @returns {string}
   */
  calculateMerkleRoot() {
    const txIds = this.transactions.map((tx) => tx.id);
    return buildMerkleRoot(txIds);
  }

  /**
   * Calculates the SHA-256 hash of the block's contents.
   * @returns {string}
   */
  calculateHash() {
    return sha256(
      this.index +
      this.previousHash +
      this.timestamp +
      this.merkleRoot +
      this.nonce +
      this.difficulty
    );
  }

  /**
   * Mines the block by finding a hash that meets the difficulty target.
   * Uses Proof of Work (PoW).
   */
  mine() {
    const target = '0'.repeat(this.difficulty);
    const startTime = Date.now();

    console.log(`⛏️  Mining block #${this.index} (difficulty: ${this.difficulty})...`);

    while (!this.hash.startsWith(target)) {
      this.nonce++;
      this.hash = this.calculateHash();
    }

    const elapsed = ((Date.now() - startTime) / 1000).toFixed(2);
    console.log(`✅ Block #${this.index} mined in ${elapsed}s | Hash: ${this.hash}`);
  }

  /**
   * Validates all transactions within the block.
   * @param {object} ec - Elliptic curve instance.
   * @returns {boolean}
   */
  hasValidTransactions(ec) {
    for (const tx of this.transactions) {
      if (!tx.isValid(ec)) {
        return false;
      }
    }
    return true;
  }

  /**
   * Returns a plain object representation.
   * @returns {object}
   */
  toJSON() {
    return {
      index: this.index,
      timestamp: this.timestamp,
      transactions: this.transactions.map((tx) => tx.toJSON()),
      previousHash: this.previousHash,
      hash: this.hash,
      merkleRoot: this.merkleRoot,
      nonce: this.nonce,
      difficulty: this.difficulty,
    };
  }
}

module.exports = Block;
```

## 6. `src/MemPool.js`

```javascript
/**
 * MemPool (Memory Pool) - Holds unconfirmed transactions
 * waiting to be included in a block.
 */
class MemPool {
  constructor() {
    this.transactions = [];
  }

  /**
   * Adds a transaction to the pool.
   * @param {Transaction} transaction
   */
  addTransaction(transaction) {
    // Prevent duplicate transactions
    const exists = this.transactions.find((tx) => tx.id === transaction.id);
    if (exists) {
      throw new Error('Transaction already exists in the mempool.');
    }
    this.transactions.push(transaction);
    console.log(`📥 Transaction ${transaction.id} added to mempool.`);
  }

  /**
   * Returns the highest-fee transactions up to a limit.
   * @param {number} limit - Max number of transactions to return.
   * @returns {Transaction[]}
   */
  getTopTransactions(limit = 10) {
    return [...this.transactions]
      .sort((a, b) => b.fee - a.fee)
      .slice(0, limit);
  }

  /**
   * Removes confirmed transactions from the pool.
   * @param {Transaction[]} confirmedTxs
   */
  removeConfirmed(confirmedTxs) {
    const confirmedIds = new Set(confirmedTxs.map((tx) => tx.id));
    this.transactions = this.transactions.filter(
      (tx) => !confirmedIds.has(tx.id)
    );
  }

  /**
   * Returns the number of pending transactions.
   * @returns {number}
   */
  size() {
    return this.transactions.length;
  }

  /**
   * Clears all transactions from the pool.
   */
  clear() {
    this.transactions = [];
  }
}

module.exports = MemPool;
```

## 7. `src/Wallet.js`

```javascript
const EC = require('elliptic').ec;
const { sha256 } = require('./utils/hash');
const Transaction = require('./Transaction');

const ec = new EC('secp256k1'); // Same curve as Bitcoin

class Wallet {
  /**
   * Creates a new wallet with a fresh key pair.
   */
  constructor() {
    this.keyPair = ec.genKeyPair();
    this.publicKey = this.keyPair.getPublic('hex');
    this.privateKey = this.keyPair.getPrivate('hex');
    this.address = this.generateAddress();
  }

  /**
   * Restores a wallet from an existing private key.
   * @param {string} privateKeyHex
   * @returns {Wallet}
   */
  static fromPrivateKey(privateKeyHex) {
    const wallet = new Wallet();
    wallet.keyPair = ec.keyFromPrivate(privateKeyHex, 'hex');
    wallet.publicKey = wallet.keyPair.getPublic('hex');
    wallet.privateKey = privateKeyHex;
    wallet.address = wallet.generateAddress();
    return wallet;
  }

  /**
   * Generates a wallet address from the public key.
   * Simulates Bitcoin-style address generation.
   * @returns {string}
   */
  generateAddress() {
    // In a real implementation, this would use RIPEMD-160 + Base58Check
    return sha256(this.publicKey).substring(0, 40);
  }

  /**
   * Creates and signs a new transaction.
   * @param {string} toAddress  - Recipient's public key.
   * @param {number} amount     - Amount to send.
   * @param {number} fee        - Transaction fee.
   * @returns {Transaction}
   */
  createTransaction(toAddress, amount, fee = 0) {
    if (amount <= 0) {
      throw new Error('Transaction amount must be positive.');
    }

    const tx = new Transaction(this.publicKey, toAddress, amount, fee);
    tx.sign(this.keyPair);

    return tx;
  }

  /**
   * Returns wallet info (never expose private key in production!).
   * @returns {object}
   */
  toJSON() {
    return {
      address: this.address,
      publicKey: this.publicKey,
      // WARNING: Never expose private keys in a real application!
      privateKey: this.privateKey,
    };
  }
}

module.exports = { Wallet, ec };
```
