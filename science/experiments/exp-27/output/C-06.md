# Production Blockchain Implementation in Node.js

Let me be direct about what I'm building and what I'm flagging before writing a line of code.

## Gap Analysis: What Production Requires That You Didn't Ask For

Before implementation, here are the gaps I'm naming explicitly:

| Gap | Risk if Ignored | My Decision |
|-----|----------------|-------------|
| **Persistent storage** | Node restart loses all chain state | Implemented (LevelDB) |
| **P2P networking** | Single node, not a blockchain | Implemented (libp2p/TCP) |
| **Mempool management** | No transaction ordering, DoS vector | Implemented |
| **Fork resolution** | Competing chains corrupt state | Implemented (longest chain rule) |
| **Wallet / key management** | No way to sign transactions | Implemented (secp256k1) |
| **API authentication** | Open RPC = instant exploit | Flagged, basic auth scaffolded |
| **Replay attack protection** | Same tx rebroadcast repeatedly | Implemented (nonce + chainId) |
| **DoS protection** | Malformed blocks crash node | Implemented (input validation) |
| **Pruning / archival strategy** | Disk fills indefinitely | Flagged, not implemented |
| **Finality model** | Clients don't know when to trust a block | Implemented (confirmation depth) |
| **Monitoring / observability** | Silent failures in production | Implemented (structured logging, metrics endpoint) |
| **Graceful shutdown** | Corrupted state on SIGTERM | Implemented |

**Flagged but NOT implemented (require product decisions):**
- Pruning strategy (archive vs. pruned node)
- Slashing / Byzantine fault penalties
- Light client protocol
- Mempool fee market / EIP-1559 style base fee
- Snapshot sync for new nodes
- Production API authentication (JWT/mTLS)

### `src/crypto/wallet.js`

```javascript
// src/crypto/wallet.js
//
// Wallet handles all cryptographic operations:
//   - Key pair generation (secp256k1)
//   - Address derivation (keccak256 of public key, last 20 bytes)
//   - Transaction signing
//   - Signature verification
//
// PRODUCTION GAP: This stores private keys in memory only.
// A production system needs an HSM, encrypted keystore (scrypt/argon2),
// or external signing service (e.g., AWS KMS, Hashicorp Vault).

import { createHash, randomBytes } from 'crypto';
import { ec as EC } from 'elliptic';

const secp256k1 = new EC('secp256k1');

// ---------------------------------------------------------------------------
// Hashing utilities
// ---------------------------------------------------------------------------

export function sha256(data) {
  return createHash('sha256').update(data).digest();
}

export function sha256Hex(data) {
  return createHash('sha256').update(data).digest('hex');
}

/**
 * Simplified keccak256 substitute using double-SHA256.
 *
 * PRODUCTION GAP: Use a proper keccak256 implementation (e.g., the
 * `ethereum-cryptography` package) for Ethereum address compatibility.
 * We avoid the dependency here to keep the implementation self-contained,
 * but this means addresses are NOT Ethereum-compatible.
 */
export function keccak256(data) {
  const first = createHash('sha256').update(data).digest();
  return createHash('sha256').update(first).digest();
}

// ---------------------------------------------------------------------------
// Key management
// ---------------------------------------------------------------------------

export class Wallet {
  constructor(privateKeyHex = null) {
    if (privateKeyHex) {
      this.keyPair = secp256k1.keyFromPrivate(privateKeyHex, 'hex');
    } else {
      // Generate cryptographically secure random key
      const privateKey = randomBytes(32);
      this.keyPair = secp256k1.keyFromPrivate(privateKey);
    }

    this.privateKey = this.keyPair.getPrivate('hex');
    this.publicKey = this.keyPair.getPublic('hex');
    this.address = Wallet.publicKeyToAddress(this.publicKey);
  }

  /**
   * Derive address from public key.
   * Takes the last 20 bytes of keccak256(publicKey) and hex-encodes it.
   */
  static publicKeyToAddress(publicKeyHex) {
    const pubKeyBuffer = Buffer.from(publicKeyHex, 'hex');
    const hash = keccak256(pubKeyBuffer);
    return '0x' + hash.slice(-20).toString('hex');
  }

  /**
   * Sign arbitrary data. Returns { r, s, recoveryParam } as hex strings.
   * We sign the hash of the data, never raw data.
   */
  sign(dataHex) {
    const hash = sha256(Buffer.from(dataHex, 'hex'));
    const signature = this.keyPair.sign(hash);
    return {
      r: signature.r.toString('hex'),
      s: signature.s.toString('hex'),
      v: signature.recoveryParam,
    };
  }

  /**
   * Verify a signature against a public key.
   * Returns true if valid.
   */
  static verify(dataHex, signature, publicKeyHex) {
    try {
      const hash = sha256(Buffer.from(dataHex, 'hex'));
      const key = secp256k1.keyFromPublic(publicKeyHex, 'hex');
      return key.verify(hash, { r: signature.r, s: signature.s });
    } catch {
      return false;
    }
  }

  /**
   * Recover public key from signature and signed data.
   * This allows us to verify address ownership without storing public keys
   * in every transaction (same approach as Ethereum).
   */
  static recoverPublicKey(dataHex, signature) {
    try {
      const hash = sha256(Buffer.from(dataHex, 'hex'));
      const recovered = secp256k1.recoverPubKey(
        hash,
        { r: signature.r, s: signature.s },
        signature.v
      );
      return recovered.encode('hex');
    } catch {
      return null;
    }
  }

  toJSON() {
    return {
      address: this.address,
      publicKey: this.publicKey,
      // Never serialize private key in production logs
    };
  }

  exportKeystore() {
    // PRODUCTION GAP: Encrypt with scrypt/argon2 before writing to disk
    return {
      address: this.address,
      publicKey: this.publicKey,
      privateKey: this.privateKey, // UNSAFE - encrypt this
    };
  }
}

### `src/core/transaction.js`

```javascript
// src/core/transaction.js
//
// Transaction lifecycle:
//   1. Created by sender with nonce, to, value, data, gasPrice
//   2. Signed with sender's private key
//   3. Broadcast to network
//   4. Validated by receiving nodes (signature, nonce, balance)
//   5. Included in a block by a miner
//   6. Finalized after CONFIRMATION_DEPTH blocks

import { sha256Hex, keccak256, Wallet } from '../crypto/wallet.js';
import { Config } from '../../config.js';

export class Transaction {
  constructor({
    from,
    to,
    value,
    data = '',
    nonce,
    gasPrice = Config.MIN_GAS_PRICE,
    gasLimit = BigInt(21000),
    chainId = Config.CHAIN_ID,
    signature = null,
    publicKey = null,
    timestamp = Date.now(),
  }) {
    // Validate required fields immediately on construction
    if (!from) throw new Error('Transaction missing: from');
    if (!to && !data) throw new Error('Transaction must have recipient or data');
    if (nonce === undefined || nonce === null) throw new Error('Transaction missing: nonce');

    this.from = from.toLowerCase();
    this.to = to ? to.toLowerCase() : null; // null = contract creation
    this.value = BigInt(value ?? 0);
    this.data = data;
    this.nonce = Number(nonce);
    this.gasPrice = BigInt(gasPrice);
    this.gasLimit = BigInt(gasLimit);
    this.chainId = chainId;
    this.timestamp = timestamp;
    this.signature = signature;
    this.publicKey = publicKey;

    // Computed fields
    this.id = this._computeId();
  }

  /**
   * The signing payload excludes the signature itself.
   * chainId is included to prevent replay attacks across chains.
   */
  _signingPayload() {
    return JSON.stringify({
      from: this.from,
      to: this.to,
      value: this.value.toString(),
      data: this.data,
      nonce: this.nonce,
      gasPrice: this.gasPrice.toString(),
      gasLimit: this.gasLimit.toString(),
      chainId: this.chainId,
      timestamp: this.timestamp,
    });
  }

  _computeId() {
    const payload = this._signingPayload() + (this.signature ? JSON.stringify(this.signature) : '');
    return sha256Hex(Buffer.from(payload));
  }

  /**
   * Sign this transaction with a wallet.
   * Mutates the transaction in place and recomputes the ID.
   */
  sign(wallet) {
    if (wallet.address.toLowerCase() !== this.from) {
      throw new Error(
        `Wallet address ${wallet.address} does not match transaction from ${this.from}`
      );
    }
    const payloadHex = Buffer.from(this._signingPayload()).toString('hex');
    this.signature = wallet.sign(payloadHex);
    this.publicKey = wallet.publicKey;
    this.id = this._computeId();
    return this;
  }

  /**
   * Validate transaction structure and signature.
   * Does NOT validate nonce or balance (that requires chain state).
   */
  validate() {
    const errors = [];

    // Structure checks
    if (!this.from || !/^0x[0-9a-f]{40}$/.test(this.from)) {
      errors.push('Invalid from address format');
    }
    if (this.to && !/^0x[0-9a-f]{40}$/.test(this.to)) {
      errors.push('Invalid to address format');
    }
    if (this.value < BigInt(0)) {
      errors.push('Negative value');
    }
    if (this.gasPrice < Config.MIN_GAS_PRICE) {
      errors.push(`Gas price below minimum: ${this.gasPrice} < ${Config.MIN_GAS_PRICE}`);
    }
    if (this.nonce < 0 || !Number.isInteger(this.nonce)) {
      errors.push('Invalid nonce');
    }
    if (this.chainId !== Config.CHAIN_ID) {
      errors.push(`Wrong chain ID: expected ${Config.CHAIN_ID}, got ${this.chainId}`);
    }
    if (!this.signature) {
      errors.push('Missing signature');
    }
    if (!this.publicKey) {
      errors.push('Missing public key');
    }

    if (errors.length > 0) {
      return { valid: false, errors };
    }

    // Signature
